14   hiddenimports, binaries, datas = add_qt6_dependencies(__file__)

16   # In PySide 6.7.0, Qt3DRender module added a reference to QtOpenGL type system. The hidden import is required on
17   # Windows, while on macOS and Linux we seem to pick it up automatically due to the corresponding Qt shared library
18   # appearing among binary dependencies. Keep it around on all OSes, though - just in case this ever changes.
19   if pyside6_library_info.version is not None and pyside6_library_info.version >= [6, 7]:
20       hiddenimports += ['PySide6.QtOpenGL']
