143          # Test waitpid() + WIFEXITED -> WEXITSTATUS.
144          # normal return, same as exit(0)
145          cmd = [PYTHON_EXE, "-c", "pass"]

171          p = self.spawn_psproc()
172          if POSIX:
173              # Test waitpid() + WIFSTOPPED and WIFCONTINUED.
174              # Note: if a process is stopped it ignores SIGTERM.
175              p.send_signal(signal.SIGSTOP)

180              self.assertEqual(p.wait(), -signal.SIGTERM)
181              self.assertEqual(p.wait(), -signal.SIGTERM)

184              self.assertRaises(psutil.TimeoutExpired, p.wait, timeout=0.001)
185              p.resume()
186              self.assertRaises(psutil.TimeoutExpired, p.wait, timeout=0.001)
187              p.terminate()
188              self.assertEqual(p.wait(), signal.SIGTERM)
189              self.assertEqual(p.wait(), signal.SIGTERM)

196          self.assertRaises(psutil.TimeoutExpired, grandchild.wait, 0.01)

198          # grandchild will hang until the parent is gone.
199          child.terminate()

201          child_ret = child.wait()
202          grandchild_ret = grandchild.wait()
203          if POSIX:
204              self.assertEqual(child_ret, -signal.SIGTERM)
205              # For processes which are not our children we're supposed
206              # to get None.
207              self.assertEqual(grandchild_ret, None)
208          else:
209              self.assertEqual(child_ret, signal.SIGTERM)
210              self.assertEqual(child_ret, signal.SIGTERM)
