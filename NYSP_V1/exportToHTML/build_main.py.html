85   @isolated.decorate
86   def discover_hook_directories():
87       """ 
88       Discover hook directories via pyinstaller40 entry points. Perform the discovery in an isolated subprocess 
89       to avoid importing the package(s) in the main process. 
90    
91       :return: list of discovered hook directories. 
92       """
93   
94       from traceback import format_exception_only
95       from PyInstaller.log import logger
96       from PyInstaller.compat import importlib_metadata
97   
98       # The “selectable” entry points (via group and name keyword args) were introduced in importlib_metadata 4.6 and
99       # Python 3.10. The compat module ensures we are using a compatible version.
100      entry_points = importlib_metadata.entry_points(group='pyinstaller40', name='hook-dirs')
101  
102      # Ensure that pyinstaller_hooks_contrib comes last so that hooks from packages providing their own take priority.
103      entry_points = sorted(entry_points, key=lambda x: x.module == "_pyinstaller_hooks_contrib.hooks")
104  
105      hook_directories = []
106      for entry_point in entry_points:
107          try:
108              hook_directories.extend(entry_point.load()())
109          except Exception as e:
110              msg = "".join(format_exception_only(type(e), e)).strip()
111              logger.warning("discover_hook_directories: Failed to process hook entry point '%s': %s", entry_point, msg)
112  
113      logger.debug("discover_hook_directories: Hook directories: %s", hook_directories)
114  
115      return hook_directories

118  def find_binary_dependencies(binaries, import_packages):
119      """ 
120      Find dynamic dependencies (linked shared libraries) for the provided list of binaries. 
121   
122      On Windows, this function performs additional pre-processing in an isolated environment in an attempt to handle 
123      dynamic library search path modifications made by packages during their import. The packages from the given list 
124      of collected packages are imported one by one, while keeping track of modifications made by `os.add_dll_directory` 
125      calls and additions to the `PATH`  environment variable. The recorded additional search paths are then passed to 
126      the binary dependency analysis step. 
127   
128      binaries 
129              List of binaries to scan for dynamic dependencies. 
130      import_packages 
131              List of packages to import prior to scanning binaries. 
132   
133      :return: expanded list of binaries and then dependencies. 
134      """
135  
136      # Extra library search paths (used on Windows to resolve DLL paths).
137      extra_libdirs = []
138      if compat.is_win:
139          # Always search `sys.base_prefix`, and search it first. This ensures that we resolve the correct version of
140          # `python3X.dll` and `python3.dll` (a PEP-0384 stable ABI stub that forwards symbols to the fully versioned
141          # `python3X.dll`), regardless of other python installations that might be present in the PATH.
142          extra_libdirs.append(compat.base_prefix)
143  
144          # If `pywin32` is installed, resolve the path to the `pywin32_system32` directory. Most `pywin32` extensions
145          # reference the `pywintypes3X.dll` in there. Based on resolved `pywin32_system32` directory, also add other
146          # `pywin32` directory, in case extensions in different directories reference each other (the ones in the same
147          # directory should already be resolvable due to binary dependency analysis passing the analyzed binary's
148          # location to the `get_imports` function). This allows us to avoid searching all paths in `sys.path`, which
149          # may lead to other corner-case issues (e.g., #5560).
150          pywin32_system32_dir = None
151          try:
152              # Look up the directory by treating it as a namespace package.
153              _, pywin32_system32_dir = get_package_paths('pywin32_system32')
154          except Exception:
155              pass
156  
157          if pywin32_system32_dir:
158              pywin32_base_dir = os.path.dirname(pywin32_system32_dir)
159              extra_libdirs += [
160                  pywin32_system32_dir,  # .../pywin32_system32
161                  # based on pywin32.pth
162                  os.path.join(pywin32_base_dir, 'win32'),  # .../win32
163                  os.path.join(pywin32_base_dir, 'win32', 'lib'),  # .../win32/lib
164                  os.path.join(pywin32_base_dir, 'Pythonwin'),  # .../Pythonwin
165              ]
166  
167      # On Windows, packages' initialization code might register additional DLL search paths, either by modifying the
168      # `PATH` environment variable, or by calling `os.add_dll_directory`. Therefore, we import all collected packages,
169      # and track changes made to the environment.
170      if compat.is_win:
171          # Helper functions to be executed in isolated environment.
172          def setup(suppressed_imports):
173              """ 
174              Prepare environment for change tracking 
175              """
176              import os
177              import sys
178  
179              os._added_dll_directories = []
180              os._original_path_env = os.environ.get('PATH', '')
181  
182              _original_add_dll_directory = os.add_dll_directory
183  
184              def _pyi_add_dll_directory(path):
185                  os._added_dll_directories.append(path)
186                  return _original_add_dll_directory(path)
187  
188              os.add_dll_directory = _pyi_add_dll_directory
189  
190              # Suppress import of specified packages
191              for name in suppressed_imports:
192                  sys.modules[name] = None
193  
194          def import_library(package):
195              """ 
196              Import collected package to set up environment. 
197              """
198              try:
199                  __import__(package)
200              except Exception:
201                  pass
202  
203          def process_search_paths():
204              """ 
205              Obtain lists of added search paths. 
206              """
207              import os
208  
209              # `os.add_dll_directory` might be called with a `pathlib.Path`, which cannot be marhsalled out of the helper
210              # process. So explicitly convert all entries to strings.
211              dll_directories = [str(path) for path in os._added_dll_directories]
212  
213              orig_path = set(os._original_path_env.split(os.pathsep))
214              modified_path = os.environ.get('PATH', '').split(os.pathsep)
215              path_additions = [path for path in modified_path if path and path not in orig_path]
216  
217              return dll_directories, path_additions
218  
219          # Pre-process the list of packages to import.
220          # Check for Qt bindings packages, and put them at the front of the packages list. This ensures that they are
221          # always imported first, which should prevent packages that support multiple bindings (`qtypy`, `pyqtgraph`,
222          # `matplotlib`, etc.) from trying to auto-select bindings.
223          _QT_BINDINGS = ('PySide2', 'PyQt5', 'PySide6', 'PyQt6')
224  
225          qt_packages = []
226          other_packages = []
227          for package in import_packages:
228              if package.startswith(_QT_BINDINGS):
229                  qt_packages.append(package)
230              else:
231                  other_packages.append(package)
232          import_packages = qt_packages + other_packages
233  
234          # Just in case, explicitly suppress imports of Qt bindings that we are *not* collecting - if multiple bindings
235          # are available and some were excluded from our analysis, a package imported here might still try to import an
236          # excluded bindings package (and succeed at doing so).
237          suppressed_imports = [package for package in _QT_BINDINGS if package not in qt_packages]
238  
239          # If we suppressed PySide2 or PySide6, we must also suppress their corresponding shiboken package
240          if "PySide2" in suppressed_imports:
241              suppressed_imports += ["shiboken2"]
242          if "PySide6" in suppressed_imports:
243              suppressed_imports += ["shiboken6"]
244  
245          # Suppress import of `pyqtgraph.canvas`, which is known to crash python interpreter. See #7452 and #8322, as
246          # well as https://github.com/pyqtgraph/pyqtgraph/issues/2838.
247          suppressed_imports += ['pyqtgraph.canvas']
248  
249          # PySimpleGUI 5.x displays a "first-run" dialog when imported for the first time, which blocks the loop below.
250          # This is a problem for building on CI, where the dialog cannot be closed, and where PySimpleGUI runs "for the
251          # first time" every time. See #8396.
252          suppressed_imports += ['PySimpleGUI']
253  
254          # Processing in isolated environment.
255          with isolated.Python() as child:
256              child.call(setup, suppressed_imports)
257              for package in import_packages:
258                  try:
259                      child.call(import_library, package)
260                  except isolated.SubprocessDiedError as e:
261                      # Re-raise as `isolated.SubprocessDiedError` again, to trigger error-handling codepath in
262                      # `isolated.Python.__exit__()`.
263                      raise isolated.SubprocessDiedError(
264                          f"Isolated subprocess crashed while importing package {package!r}! "
265                          f"Package import list: {import_packages!r}"
266                      ) from e
267              added_dll_directories, added_path_directories = child.call(process_search_paths)
268  
269          # Process extra search paths...
270          logger.info("Extra DLL search directories (AddDllDirectory): %r", added_dll_directories)
271          extra_libdirs += added_dll_directories
272  
273          logger.info("Extra DLL search directories (PATH): %r", added_path_directories)
274          extra_libdirs += added_path_directories
275  
276      # Deduplicate search paths
277      # NOTE: `list(set(extra_libdirs))` does not preserve the order of search paths (which matters here), so we need to
278      # de-duplicate using `list(dict.fromkeys(extra_libdirs).keys())` instead.
279      extra_libdirs = list(dict.fromkeys(extra_libdirs).keys())
280  
281      # Search for dependencies of the given binaries
282      return bindepend.binary_dependency_analysis(binaries, search_paths=extra_libdirs)

285  class _ModuleCollectionMode(enum.IntFlag):
286      """ 
287      Module collection mode flags. 
288      """
289      PYZ = enum.auto()  # Collect byte-compiled .pyc into PYZ archive
290      PYC = enum.auto()  # Collect byte-compiled .pyc as external data file
291      PY = enum.auto()  # Collect source .py file as external data file
292  
293  
294  _MODULE_COLLECTION_MODES = {
295      "pyz": _ModuleCollectionMode.PYZ,
296      "pyc": _ModuleCollectionMode.PYC,
297      "py": _ModuleCollectionMode.PY,
298      "pyz+py": _ModuleCollectionMode.PYZ | _ModuleCollectionMode.PY,
299      "py+pyz": _ModuleCollectionMode.PYZ | _ModuleCollectionMode.PY,
300  }
301  
302  
303  def _get_module_collection_mode(mode_dict, name, noarchive=False):
304      """ 
305      Determine the module/package collection mode for the given module name, based on the provided collection 
306      mode settings dictionary. 
307      """
308      # Default mode: collect into PYZ, unless noarchive is enabled. In that case, collect as pyc.
309      mode_flags = _ModuleCollectionMode.PYC if noarchive else _ModuleCollectionMode.PYZ
310  
311      # If we have no collection mode settings, end here and now.
312      if not mode_dict:
313          return mode_flags
314  
315      # Search the parent modules/packages in top-down fashion, and take the last given setting. This ensures that
316      # a setting given for the top-level package is recursively propagated to all its subpackages and submodules,
317      # but also allows individual sub-modules to override the setting again.
318      mode = 'pyz'
319  
320      name_parts = name.split('.')
321      for i in range(len(name_parts)):
322          modlevel = ".".join(name_parts[:i + 1])
323          modlevel_mode = mode_dict.get(modlevel, None)
324          if modlevel_mode is not None:
325              mode = modlevel_mode
326  
327      # Convert mode string to _ModuleCollectionMode flags
328      try:
329          mode_flags = _MODULE_COLLECTION_MODES[mode]
330      except KeyError:
331          raise ValueError(f"Unknown module collection mode for {name!r}: {mode!r}!")
332  
333      # noarchive flag being set means that we need to change _ModuleCollectionMode.PYZ into _ModuleCollectionMode.PYC
334      if noarchive and _ModuleCollectionMode.PYZ in mode_flags:
335          mode_flags ^= _ModuleCollectionMode.PYZ
336          mode_flags |= _ModuleCollectionMode.PYC
337  
338      return mode_flags

388          scripts 
389                  A list of scripts specified as file names. 
390          pathex 
391                  An optional list of paths to be searched before sys.path. 
392          binaries 
393                  An optional list of additional binaries (dlls, etc.) to include.

445                  script = os.path.join(spec_dir, script)

487          self.custom_runtime_hooks = runtime_hooks or []
488  
489          self._input_binaries = []
490          self._input_datas = []
491  
492          self.excludes = excludes or []
493          self.scripts = []
494          self.pure = []

516          # reflect the changes.

518              logger.info("Appending 'binaries' from .spec")
519              self._input_binaries = [(dest_name, src_name, 'BINARY')
520                                      for dest_name, src_name in format_binaries_and_datas(binaries, workingdir=spec_dir)]

526                                   for dest_name, src_name in format_binaries_and_datas(datas, workingdir=spec_dir)]

528  
529          self.__postinit__()

557      def _extend_pathex(self, spec_pathex, scripts):
558          """ 
559          Normalize additional paths where PyInstaller will look for modules and add paths with scripts to the list of 
560          paths. 
561   
562          :param spec_pathex: Additional paths defined defined in .spec file. 
563          :param scripts: Scripts to create executable from. 
564          :return: list of updated paths 
565          """
566          # Based on main supplied script - add top-level modules directory to PYTHONPATH.
567          # Sometimes the main app script is not top-level module but submodule like 'mymodule.mainscript.py'.
568          # In that case PyInstaller will not be able find modules in the directory containing 'mymodule'.
569          # Add this directory to PYTHONPATH so PyInstaller could find it.
570          pathex = []
571          # Add scripts paths first.
572          for script in scripts:
573              logger.debug('script: %s' % script)
574              script_toplevel_dir = get_path_to_toplevel_modules(script)
575              if script_toplevel_dir:
576                  pathex.append(script_toplevel_dir)
577          # Append paths from .spec.
578          if spec_pathex is not None:
579              pathex.extend(spec_pathex)
580          # Normalize paths in pathex and make them absolute.
581          return [absnormpath(p) for p in pathex]
582  
583      def _check_guts(self, data, last_build):
584          if Target._check_guts(self, data, last_build):
585              return True
586          for filename in self.inputs:
587              if mtime(filename) > last_build:
588                  logger.info("Building because %s changed", filename)
589                  return True
590          # Now we know that none of the input parameters and none of the input files has changed. So take the values
591          # that were calculated / analyzed in the last run and store them in `self`. These TOC lists should already
592          # be normalized.
593          self.scripts = data['scripts']
594          self.pure = data['pure']
595          self.binaries = data['binaries']
596          self.zipfiles = data['zipfiles']
597          self.zipped_data = data['zipped_data']
598          self.datas = data['datas']
599  
600          return False
601  
602      def assemble(self):
603          """ 
604          This method is the MAIN method for finding all necessary files to be bundled. 
605          """
606          from PyInstaller.config import CONF
607  
608          logger.info("Running Analysis %s", self.tocbasename)
609          logger.info("Target bytecode optimization level: %d", self.optimize)
610  
611          for m in self.excludes:
612              logger.debug("Excluding module '%s'" % m)
613          self.graph = initialize_modgraph(excludes=self.excludes, user_hook_dirs=self.hookspath)
614  
615          # Initialize `binaries` and `datas` with `_input_binaries` and `_input_datas`. Make sure to copy the lists
616          # to prevent modifications of original lists, which we need to store in original form for guts comparison.
617          self.datas = [entry for entry in self._input_datas]
618          self.binaries = [entry for entry in self._input_binaries]
619  
620          # TODO: find a better place where to put 'base_library.zip' and when to created it.
621          # For Python 3 it is necessary to create file 'base_library.zip' containing core Python modules. In Python 3
622          # some built-in modules are written in pure Python. base_library.zip is a way how to have those modules as
623          # "built-in".
624          libzip_filename = os.path.join(CONF['workpath'], 'base_library.zip')
625          create_py3_base_library(libzip_filename, graph=self.graph)
626          # Bundle base_library.zip as data file.
627          # Data format of TOC item: ('relative_path_in_dist_dir', 'absolute_path_on_disk', 'DATA')
628          self.datas.append((os.path.basename(libzip_filename), libzip_filename, 'DATA'))
629  
630          # Expand sys.path of module graph. The attribute is the set of paths to use for imports: sys.path, plus our
631          # loader, plus other paths from e.g. --path option).
632          self.graph.path = self.pathex + self.graph.path
633  
634          # Scan for legacy namespace packages.
635          self.graph.scan_legacy_namespace_packages()
636  
637          # Search for python shared library, which we need to collect into frozen application.
638          logger.info('Looking for Python shared library...')
639          python_lib = bindepend.get_python_library_path()
640          if python_lib is None:
641              from PyInstaller.exceptions import PythonLibraryNotFoundError
642              raise PythonLibraryNotFoundError()
643          logger.info('Using Python shared library: %s', python_lib)
644          if is_darwin and osxutils.is_framework_bundle_lib(python_lib):
645              # If python library is located in macOS .framework bundle, collect the bundle, and create symbolic link to
646              # top-level directory.
647              src_path = pathlib.PurePath(python_lib)
648              dst_path = pathlib.PurePath(src_path.relative_to(src_path.parent.parent.parent.parent))
649              self.binaries.append((str(dst_path), str(src_path), 'BINARY'))
650              self.binaries.append((os.path.basename(python_lib), str(dst_path), 'SYMLINK'))
651          else:
652              self.binaries.append((os.path.basename(python_lib), python_lib, 'BINARY'))
653  
654          # -- Module graph. --
655          #
656          # Construct the module graph of import relationships between modules required by this user's application. For
657          # each entry point (top-level user-defined Python script), all imports originating from this entry point are
658          # recursively parsed into a subgraph of the module graph. This subgraph is then connected to this graph's root
659          # node, ensuring imported module nodes will be reachable from the root node -- which is is (arbitrarily) chosen
660          # to be the first entry point's node.
661  
662          # List of graph nodes corresponding to program scripts.
663          program_scripts = []
664  
665          # Assume that if the script does not exist, Modulegraph will raise error. Save the graph nodes of each in
666          # sequence.
667          for script in self.inputs:
668              logger.info("Analyzing %s", script)
669              program_scripts.append(self.graph.add_script(script))
670  
671          # Analyze the script's hidden imports (named on the command line)
672          self.graph.add_hiddenimports(self.hiddenimports)
673  
674          # -- Post-graph hooks. --
675          self.graph.process_post_graph_hooks(self)
676  
677          # Update 'binaries' and 'datas' TOC lists with entries collected from hooks.
678          self.binaries += self.graph.make_hook_binaries_toc()
679          self.datas += self.graph.make_hook_datas_toc()
680  
681          # We do not support zipped eggs anymore (PyInstaller v6.0), so `zipped_data` and `zipfiles` are always empty.
682          self.zipped_data = []
683          self.zipfiles = []
684  
685          # -- Automatic binary vs. data reclassification. --
686          #
687          # At this point, `binaries` and `datas` contain  TOC entries supplied by user via input arguments, and by hooks
688          # that were ran during the analysis. Neither source can be fully trusted regarding the DATA vs BINARY
689          # classification (no thanks to our hookutils not being 100% reliable, either!). Therefore, inspect the files and
690          # automatically reclassify them as necessary.
691          #
692          # The proper classification is important especially for collected binaries - to ensure that they undergo binary
693          # dependency analysis and platform-specific binary processing. On macOS, the .app bundle generation code also
694          # depends on files to be properly classified.
695          #
696          # For entries added to `binaries` and `datas` after this point, we trust their typecodes due to the nature of
697          # their origin.
698          combined_toc = normalize_toc(self.datas + self.binaries)
699  
700          logger.info('Performing binary vs. data reclassification (%d entries)', len(combined_toc))
701  
702          self.datas = []
703          self.binaries = []
704  
705          for dest_name, src_name, typecode in combined_toc:
706              # Returns 'BINARY' or 'DATA', or None if file cannot be classified.
707              detected_typecode = bindepend.classify_binary_vs_data(src_name)
708              if detected_typecode is not None:
709                  if detected_typecode != typecode:
710                      logger.debug(
711                          "Reclassifying collected file %r from %s to %s...", src_name, typecode, detected_typecode
712                      )
713                  typecode = detected_typecode
714  
715              # Put back into corresponding TOC list.
716              if typecode in {'BINARY', 'EXTENSION'}:
717                  self.binaries.append((dest_name, src_name, typecode))
718              else:
719                  self.datas.append((dest_name, src_name, typecode))
720  
721          # -- Look for dlls that are imported by Python 'ctypes' module. --
722          # First get code objects of all modules that import 'ctypes'.
723          logger.info('Looking for ctypes DLLs')
724          # dict like: {'module1': code_obj, 'module2': code_obj}
725          ctypes_code_objs = self.graph.get_code_using("ctypes")
726  
727          for name, co in ctypes_code_objs.items():
728              # Get dlls that might be needed by ctypes.
729              logger.debug('Scanning %s for ctypes-based references to shared libraries', name)
730              try:
731                  ctypes_binaries = scan_code_for_ctypes(co)
732                  # As this scan happens after automatic binary-vs-data classification, we need to validate the binaries
733                  # ourselves, just in case.
734                  for dest_name, src_name, typecode in set(ctypes_binaries):
735                      # Allow for `None` in case re-classification is not supported on the given platform.
736                      if bindepend.classify_binary_vs_data(src_name) not in (None, 'BINARY'):
737                          logger.warning("Ignoring %s found via ctypes - not a valid binary!", src_name)
738                          continue
739                      self.binaries.append((dest_name, src_name, typecode))
740              except Exception as ex:
741                  raise RuntimeError(f"Failed to scan the module '{name}'. This is a bug. Please report it.") from ex
742  
743          self.datas.extend((dest, source, "DATA")
744                            for (dest, source) in format_binaries_and_datas(self.graph.metadata_required()))
745  
746          # Analyze run-time hooks.
747          rhtook_scripts = self.graph.analyze_runtime_hooks(self.custom_runtime_hooks)
748  
749          # -- Extract the nodes of the graph as TOCs for further processing. --
750  
751          # Initialize the scripts list: run-time hooks (custom ones, followed by regular ones), followed by program
752          # script(s).
753  
754          # We do not optimize bytecode of run-time hooks.
755          rthook_toc = self.graph.nodes_to_toc(rhtook_scripts)
756  
757          # Override the typecode of program script(s) to include bytecode optimization level.
758          program_toc = self.graph.nodes_to_toc(program_scripts)
759          optim_typecode = {0: 'PYSOURCE', 1: 'PYSOURCE-1', 2: 'PYSOURCE-2'}[self.optimize]
760          program_toc = [(name, src_path, optim_typecode) for name, src_path, typecode in program_toc]
761  
762          self.scripts = rthook_toc + program_toc
763          self.scripts = normalize_toc(self.scripts)  # Should not really contain duplicates, but just in case...
764  
765          # Extend the binaries list with all the Extensions modulegraph has found.
766          self.binaries += self.graph.make_binaries_toc()
767  
768          # Convert extension module names into full filenames, and append suffix. Ensure that extensions that come from
769          # the lib-dynload are collected into _MEIPASS/lib-dynload instead of directly into _MEIPASS.
770          for idx, (dest, source, typecode) in enumerate(self.binaries):
771              if typecode != 'EXTENSION':
772                  continue
773  
774              # Convert to full filename and append suffix
775              dest, source, typecode = add_suffix_to_extension(dest, source, typecode)
776  
777              # Divert into lib-dyload, if necessary (i.e., if file comes from lib-dynload directory) and its destination
778              # path does not already have a directory prefix.
779              src_parent = os.path.basename(os.path.dirname(source))
780              if src_parent == 'lib-dynload' and not os.path.dirname(os.path.normpath(dest)):
781                  dest = os.path.join('lib-dynload', dest)
782  
783              # Update
784              self.binaries[idx] = (dest, source, typecode)
785  
786          # Perform initial normalization of `datas` and `binaries`
787          self.datas = normalize_toc(self.datas)
788          self.binaries = normalize_toc(self.binaries)
789  
790          # Post-process GLib schemas
791          self.datas = compile_glib_schema_files(self.datas, os.path.join(CONF['workpath'], "_pyi_gschema_compilation"))
792          self.datas = normalize_toc(self.datas)
793  
794          # Process the pure-python modules list. Depending on the collection mode, these entries end up either in "pure"
795          # list for collection into the PYZ archive, or in the "datas" list for collection as external data files.
796          assert len(self.pure) == 0
797          pure_pymodules_toc = self.graph.make_pure_toc()
798  
799          # Merge package collection mode settings from .spec file. These are applied last, so they override the
800          # settings previously applied by hooks.
801          self.graph._module_collection_mode.update(self.module_collection_mode)
802          logger.debug("Module collection settings: %r", self.graph._module_collection_mode)
803  
804          # If target bytecode optimization level matches the run-time bytecode optimization level (i.e., of the running
805          # build process), we can re-use the modulegraph's code-object cache.
806          if self.optimize == sys.flags.optimize:
807              logger.debug(
808                  "Target optimization level %d matches run-time optimization level %d - using modulegraph's code-object "
809                  "cache.",
810                  self.optimize,
811                  sys.flags.optimize,
812              )
813              code_cache = self.graph.get_code_objects()
814          else:
815              logger.debug(
816                  "Target optimization level %d differs from run-time optimization level %d - ignoring modulegraph's "
817                  "code-object cache.",
818                  self.optimize,
819                  sys.flags.optimize,
820              )
821              code_cache = None
822  
823          pycs_dir = os.path.join(CONF['workpath'], 'localpycs')
824          optim_level = self.optimize  # We could extend this with per-module settings, similar to `collect_mode`.
825          for name, src_path, typecode in pure_pymodules_toc:
826              assert typecode == 'PYMODULE'
827              collect_mode = _get_module_collection_mode(self.graph._module_collection_mode, name, self.noarchive)
828  
829              # Collect byte-compiled .pyc into PYZ archive. Embed optimization level into typecode.
830              if _ModuleCollectionMode.PYZ in collect_mode:
831                  optim_typecode = {0: 'PYMODULE', 1: 'PYMODULE-1', 2: 'PYMODULE-2'}[optim_level]
832                  self.pure.append((name, src_path, optim_typecode))
833  
834              # Pure namespace packages have no source path, and cannot be collected as external data file.
835              if src_path in (None, '-'):
836                  continue
837  
838              # Collect source .py file as external data file
839              if _ModuleCollectionMode.PY in collect_mode:
840                  dest_path = name.replace('.', os.sep)
841                  # Special case: packages have an implied `__init__` filename that needs to be added.
842                  basename, ext = os.path.splitext(os.path.basename(src_path))
843                  if basename == '__init__':
844                      dest_path += os.sep + '__init__' + ext
845                  else:
846                      dest_path += ext
847                  self.datas.append((dest_path, src_path, "DATA"))
848  
849              # Collect byte-compiled .pyc file as external data file
850              if _ModuleCollectionMode.PYC in collect_mode:
851                  dest_path = name.replace('.', os.sep)
852                  # Special case: packages have an implied `__init__` filename that needs to be added.
853                  basename, ext = os.path.splitext(os.path.basename(src_path))
854                  if basename == '__init__':
855                      dest_path += os.sep + '__init__'
856                  # Append the extension for the compiled result. In python 3.5 (PEP-488) .pyo files were replaced by
857                  # .opt-1.pyc and .opt-2.pyc. However, it seems that for bytecode-only module distribution, we always
858                  # need to use the .pyc extension.
859                  dest_path += '.pyc'
860  
861                  # Compile - use optimization-level-specific sub-directory in local working directory.
862                  obj_path = compile_pymodule(
863                      name,
864                      src_path,
865                      workpath=os.path.join(pycs_dir, str(optim_level)),
866                      optimize=optim_level,
867                      code_cache=code_cache,
868                  )
869  
870                  self.datas.append((dest_path, obj_path, "DATA"))
871  
872          # Normalize list of pure-python modules (these will end up in PYZ archive, so use specific normalization).
873          self.pure = normalize_pyz_toc(self.pure)
874  
875          # Associate the `pure` TOC list instance with code cache in the global `CONF`; this is used by `PYZ` writer
876          # to obtain modules' code from cache instead
877          #
878          # (NOTE: back when `pure` was an instance of `TOC` class, the code object was passed by adding an attribute
879          # to the `pure` itself; now that `pure` is plain `list`, we cannot do that anymore. But the association via
880          # object ID should have the same semantics as the added attribute).
881          from PyInstaller.config import CONF
882          global_code_cache_map = CONF['code_cache']
883          global_code_cache_map[id(self.pure)] = code_cache
884  
885          # Add remaining binary dependencies - analyze Python C-extensions and what DLLs they depend on.
886          #
887          # Up until this point, we did very best not to import the packages into the main process. However, a package
888          # may set up additional library search paths during its import (e.g., by modifying PATH or calling the
889          # add_dll_directory() function on Windows, or modifying LD_LIBRARY_PATH on Linux). In order to reliably
890          # discover dynamic libraries, we therefore require an environment with all packages imported. We achieve that
891          # by gathering list of all collected packages, and spawn an isolated process, in which we first import all
892          # the packages from the list, and then perform search for dynamic libraries.
893          logger.info('Looking for dynamic libraries')
894  
895          collected_packages = self.graph.get_collected_packages()
896          self.binaries.extend(find_binary_dependencies(self.binaries, collected_packages))
897  
898          # Apply work-around for (potential) binaries collected from `pywin32` package...
899          if is_win:
900              self.binaries = postprocess_binaries_toc_pywin32(self.binaries)
901              # With anaconda, we need additional work-around...
902              if is_conda:
903                  self.binaries = postprocess_binaries_toc_pywin32_anaconda(self.binaries)
904  
905          # On linux, check for HMAC files accompanying shared library files and, if available, collect them.
906          # These are present on Fedora and RHEL, and are used in FIPS-enabled configurations to ensure shared
907          # library's file integrity.
908          if is_linux:
909              for dest_name, src_name, typecode in self.binaries:
910                  if typecode not in {'BINARY', 'EXTENSION'}:
911                      continue  # Skip symbolic links
912                  src_lib_path = pathlib.Path(src_name)
913                  src_hmac_path = src_lib_path.with_name(f".{src_lib_path.name}.hmac")
914                  if src_hmac_path.is_file():
915                      dest_hmac_path = pathlib.PurePath(dest_name).with_name(src_hmac_path.name)
916                      self.datas.append((str(dest_hmac_path), str(src_hmac_path), 'DATA'))
917  
918                  # Similarly, look for .chk files that are used by NSS libraries.
919                  src_chk_path = src_lib_path.with_suffix(".chk")
920                  if src_chk_path.is_file():
921                      dest_chk_path = pathlib.PurePath(dest_name).with_name(src_chk_path.name)
922                      self.datas.append((str(dest_chk_path), str(src_chk_path), 'DATA'))
923  
924          # Final normalization of `datas` and `binaries`:
925          #  - normalize both TOCs together (to avoid having duplicates across the lists)
926          #  - process the combined normalized TOC for symlinks
927          #  - split back into `binaries` (BINARY, EXTENSION) and `datas` (everything else)
928          combined_toc = normalize_toc(self.datas + self.binaries)
929          combined_toc = toc_process_symbolic_links(combined_toc)
930  
931          # On macOS, look for binaries collected from .framework bundles, and collect their Info.plist files.
932          if is_darwin:
933              combined_toc += osxutils.collect_files_from_framework_bundles(combined_toc)
934  
935          self.datas = []
936          self.binaries = []
937          for entry in combined_toc:
938              dest_name, src_name, typecode = entry
939              if typecode in {'BINARY', 'EXTENSION'}:
940                  self.binaries.append(entry)
941              else:
942                  self.datas.append(entry)
943  
944          # On macOS, the Finder app seems to litter visited directories with `.DS_Store` files. These cause issues with
945          # codesigning when placed in mixed-content directories, where our .app bundle generator cross-links data files
946          # from `Resources` to `Frameworks` tree, and the `codesign` utility explicitly forbids a `.DS_Store` file to be
947          # a symbolic link.
948          # But there is no reason for `.DS_Store` files to be collected in the first place, so filter them out.
949          if is_darwin:
950              self.datas = [(dest_name, src_name, typecode) for dest_name, src_name, typecode in self.datas
951                            if os.path.basename(src_name) != '.DS_Store']
952  
953          # Write warnings about missing modules.
954          self._write_warnings()
955          # Write debug information about the graph
956          self._write_graph_debug()
957  
958          # On macOS, check the SDK version of the binaries to be collected, and warn when the SDK version is either
959          # invalid or too low. Such binaries will likely refuse to be loaded when hardened runtime is enabled and
960          # while we cannot do anything about it, we can at least warn the user about it.
961          # See: https://developer.apple.com/forums/thread/132526
962          if is_darwin:
963              binaries_with_invalid_sdk = []
964              for dest_name, src_name, typecode in self.binaries:
965                  try:
966                      sdk_version = osxutils.get_macos_sdk_version(src_name)
967                  except Exception:
968                      logger.warning("Failed to query macOS SDK version of %r!", src_name, exc_info=True)
969                      binaries_with_invalid_sdk.append((dest_name, src_name, "unavailable"))
970                      continue
971  
972                  if sdk_version < (10, 9, 0):
973                      binaries_with_invalid_sdk.append((dest_name, src_name, sdk_version))
974              if binaries_with_invalid_sdk:
975                  logger.warning("Found one or more binaries with invalid or incompatible macOS SDK version:")
976                  for dest_name, src_name, sdk_version in binaries_with_invalid_sdk:
977                      logger.warning(" * %r, collected as %r; version: %r", src_name, dest_name, sdk_version)
978                  logger.warning("These binaries will likely cause issues with code-signing and hardened runtime!")
979  
980      def _write_warnings(self):
981          """ 
982          Write warnings about missing modules. Get them from the graph and use the graph to figure out who tried to 
983          import them. 
984          """
985          def dependency_description(name, dep_info):
986              if not dep_info or dep_info == 'direct':
987                  imptype = 0
988              else:
989                  imptype = (dep_info.conditional + 2 * dep_info.function + 4 * dep_info.tryexcept)
990              return '%s (%s)' % (name, IMPORT_TYPES[imptype])
991  
992          from PyInstaller.config import CONF
993          miss_toc = self.graph.make_missing_toc()
994          with open(CONF['warnfile'], 'w', encoding='utf-8') as wf:
995              wf.write(WARNFILE_HEADER)
996              for (n, p, status) in miss_toc:
997                  importers = self.graph.get_importers(n)
998                  print(
999                      status,
1000                     'module named',
1001                     n,
1002                     '- imported by',
1003                     ', '.join(dependency_description(name, data) for name, data in importers),
1004                     file=wf
1005                 )
1006         logger.info("Warnings written to %s", CONF['warnfile'])
1007 
1008     def _write_graph_debug(self):
1009         """ 
1010         Write a xref (in html) and with `--log-level DEBUG` a dot-drawing of the graph. 
1011         """
1012         from PyInstaller.config import CONF
1013         with open(CONF['xref-file'], 'w', encoding='utf-8') as fh:
1014             self.graph.create_xref(fh)
1015             logger.info("Graph cross-reference written to %s", CONF['xref-file'])
1016         if logger.getEffectiveLevel() > logging.DEBUG:
1017             return
1018         # The `DOT language's <https://www.graphviz.org/doc/info/lang.html>`_ default character encoding (see the end
1019         # of the linked page) is UTF-8.
1020         with open(CONF['dot-file'], 'w', encoding='utf-8') as fh:
1021             self.graph.graphreport(fh)
1022             logger.info("Graph drawing written to %s", CONF['dot-file'])

1024     def exclude_system_libraries(self, list_of_exceptions=None):
1025         """ 
1026         This method may be optionally called from the spec file to exclude any system libraries from the list of 
1027         binaries other than those containing the shell-style wildcards in list_of_exceptions. Those that match 
1028         '*python*' or are stored under 'lib-dynload' are always treated as exceptions and not excluded.

1032             entry for entry in self.binaries if _should_include_system_binary(entry, list_of_exceptions or [])
1033         ]

1043 def build(spec, distpath, workpath, clean_build):
1044     """ 
1045     Build the executable according to the created SPEC file. 
1046     """
1047     from PyInstaller.config import CONF
1048 
1049     # Ensure starting tilde in distpath / workpath is expanded into user's home directory. This is to work around for
1050     # tilde not being expanded when using ˙--workpath=~/path/abc` instead of ˙--workpath ~/path/abc` (or when the path
1051     # argument is quoted). See https://github.com/pyinstaller/pyinstaller/issues/696
1052     distpath = os.path.abspath(os.path.expanduser(distpath))
1053     workpath = os.path.abspath(os.path.expanduser(workpath))
1054 
1055     CONF['spec'] = os.path.abspath(spec)
1056     CONF['specpath'], CONF['specnm'] = os.path.split(CONF['spec'])
1057     CONF['specnm'] = os.path.splitext(CONF['specnm'])[0]
1058 
1059     # Add 'specname' to workpath and distpath if they point to PyInstaller homepath.
1060     if os.path.dirname(distpath) == HOMEPATH:
