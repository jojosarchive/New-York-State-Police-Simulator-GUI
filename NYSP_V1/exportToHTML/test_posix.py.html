45   def ps(fmt, pid=None):
46       """Wrapper for calling the ps command with a little bit of cross-platform 
47       support for a narrow range of features. 
48       """
49   
50       cmd = ['ps']
51   
52       if LINUX:
53           cmd.append('--no-headers')
54   
55       if pid is not None:
56           cmd.extend(['-p', str(pid)])
57       else:
58           if SUNOS or AIX:
59               cmd.append('-A')
60           else:
61               cmd.append('ax')
62   
63       if SUNOS:
64           fmt = fmt.replace("start", "stime")
65   
66       cmd.extend(['-o', fmt])
67   
68       output = sh(cmd)
69   
70       output = output.splitlines() if LINUX else output.splitlines()[1:]
71   
72       all_output = []
73       for line in output:
74           line = line.strip()
75   
76           try:
77               line = int(line)
78           except ValueError:
79               pass
80   
81           all_output.append(line)
82   
83       if pid is None:
84           return all_output
85       else:
86           return all_output[0]

89   # ps "-o" field names differ wildly between platforms.
90   # "comm" means "only executable name" but is not available on BSD platforms.
91   # "args" means "command with all its arguments", and is also not available
92   # on BSD platforms.
93   # "command" is like "args" on most platforms, but like "comm" on AIX,
94   # and not available on SUNOS.
95   # so for the executable name we can use "comm" on Solaris and split "command"
96   # on other platforms.
97   # to get the cmdline (with args) we have to use "args" on AIX and
98   # Solaris, and can use "command" on all others.
99   
100  
101  def ps_name(pid):
102      field = "command"
103      if SUNOS:
104          field = "comm"
105      return ps(field, pid).split()[0]
106  
107  
108  def ps_args(pid):
109      field = "command"
110      if AIX or SUNOS:
111          field = "args"
112      out = ps(field, pid)
113      # observed on BSD + Github CI: '/usr/local/bin/python3 -E -O (python3.9)'
114      out = re.sub(r"\(python.*?\)$", "", out)
115      return out.strip()

119      field = "rss"
120      if AIX:
121          field = "rssize"

125  def ps_vsz(pid):
126      field = "vsz"
127      if AIX:
128          field = "vsize"
129      return ps(field, pid)

195      def test_name(self):
196          name_ps = ps_name(self.pid)
197          # remove path if there is any, from the command
198          name_ps = os.path.basename(name_ps).lower()
199          name_psutil = psutil.Process(self.pid).name().lower()
200          # ...because of how we calculate PYTHON_EXE; on MACOS this may
201          # be "pythonX.Y".
202          name_ps = re.sub(r"\d.\d", "", name_ps)
203          name_psutil = re.sub(r"\d.\d", "", name_psutil)
204          # ...may also be "python.X"
205          name_ps = re.sub(r"\d", "", name_ps)
206          name_psutil = re.sub(r"\d", "", name_psutil)
207          self.assertEqual(name_ps, name_psutil)

209      def test_name_long(self):
210          # On UNIX the kernel truncates the name to the first 15
211          # characters. In such a case psutil tries to determine the
212          # full name from the cmdline.
213          name = "long-program-name"
214          cmdline = ["long-program-name-extended", "foo", "bar"]
215          with mock.patch("psutil._psplatform.Process.name", return_value=name):
216              with mock.patch(
217                  "psutil._psplatform.Process.cmdline", return_value=cmdline
218              ):
219                  p = psutil.Process()
220                  self.assertEqual(p.name(), "long-program-name-extended")

247      @unittest.skipIf(MACOS or BSD, 'ps -o start not available')
248      def test_create_time(self):
249          time_ps = ps('start', self.pid)
250          time_psutil = psutil.Process(self.pid).create_time()
251          time_psutil_tstamp = datetime.datetime.fromtimestamp(
252              time_psutil
253          ).strftime("%H:%M:%S")
254          # sometimes ps shows the time rounded up instead of down, so we check
255          # for both possible values
256          round_time_psutil = round(time_psutil)
257          round_time_psutil_tstamp = datetime.datetime.fromtimestamp(
258              round_time_psutil
259          ).strftime("%H:%M:%S")
260          self.assertIn(time_ps, [time_psutil_tstamp, round_time_psutil_tstamp])

262      def test_exe(self):
263          ps_pathname = ps_name(self.pid)
264          psutil_pathname = psutil.Process(self.pid).exe()
265          try:
266              self.assertEqual(ps_pathname, psutil_pathname)
267          except AssertionError:
268              # certain platforms such as BSD are more accurate returning:
269              # "/usr/local/bin/python2.7"
270              # ...instead of:
271              # "/usr/local/bin/python"
272              # We do not want to consider this difference in accuracy
273              # an error.
274              adjusted_ps_pathname = ps_pathname[: len(ps_pathname)]
275              self.assertEqual(ps_pathname, adjusted_ps_pathname)

305      @retry_on_failure()
306      def test_pids(self):
307          # Note: this test might fail if the OS is starting/killing
308          # other processes in the meantime
309          pids_ps = sorted(ps("pid"))
310          pids_psutil = psutil.pids()
311  
312          # on MACOS and OPENBSD ps doesn't show pid 0
313          if MACOS or OPENBSD and 0 not in pids_ps:
314              pids_ps.insert(0, 0)
315  
316          # There will often be one more process in pids_ps for ps itself
317          if len(pids_ps) - len(pids_psutil) > 1:
318              difference = [x for x in pids_psutil if x not in pids_ps] + [
319                  x for x in pids_ps if x not in pids_psutil
320              ]
321              raise self.fail("difference: " + str(difference))

340      # @unittest.skipIf(CI_TESTING and not psutil.users(), "unreliable on CI")
341      @retry_on_failure()
342      def test_users(self):
343          out = sh("who -u")
344          if not out.strip():
345              raise self.skipTest("no users on this system")
346          lines = out.split('\n')
347          users = [x.split()[0] for x in lines]
348          terminals = [x.split()[1] for x in lines]
349          self.assertEqual(len(users), len(psutil.users()))
350          with self.subTest(psutil=psutil.users(), who=out):
351              for idx, u in enumerate(psutil.users()):
352                  self.assertEqual(u.name, users[idx])
353                  self.assertEqual(u.terminal, terminals[idx])
354                  if u.pid is not None:  # None on OpenBSD
355                      psutil.Process(u.pid)

357      @retry_on_failure()
358      def test_users_started(self):
359          out = sh("who -u")
360          if not out.strip():
361              raise self.skipTest("no users on this system")
362          tstamp = None
363          # '2023-04-11 09:31' (Linux)
364          started = re.findall(r"\d\d\d\d-\d\d-\d\d \d\d:\d\d", out)
365          if started:
366              tstamp = "%Y-%m-%d %H:%M"
367          else:
368              # 'Apr 10 22:27' (macOS)
369              started = re.findall(r"[A-Z][a-z][a-z] \d\d \d\d:\d\d", out)
370              if started:
371                  tstamp = "%b %d %H:%M"
372              else:
373                  # 'Apr 10'
374                  started = re.findall(r"[A-Z][a-z][a-z] \d\d", out)
375                  if started:
376                      tstamp = "%b %d"
377                  else:
378                      # 'apr 10' (sunOS)
379                      started = re.findall(r"[a-z][a-z][a-z] \d\d", out)
380                      if started:
381                          tstamp = "%b %d"
382                          started = [x.capitalize() for x in started]
383  
384          if not tstamp:
385              raise unittest.SkipTest(
386                  "cannot interpret tstamp in who output\n%s" % (out)
387              )
388  
389          with self.subTest(psutil=psutil.users(), who=out):
390              for idx, u in enumerate(psutil.users()):
391                  psutil_value = datetime.datetime.fromtimestamp(
392                      u.started
393                  ).strftime(tstamp)
394                  self.assertEqual(psutil_value, started[idx])

438      # AIX can return '-' in df output instead of numbers, e.g. for /proc
439      @unittest.skipIf(AIX, "unreliable on AIX")
440      @retry_on_failure()
441      def test_disk_usage(self):
442          def df(device):
443              try:
444                  out = sh("df -k %s" % device).strip()
445              except RuntimeError as err:
446                  if "device busy" in str(err).lower():
447                      raise self.skipTest("df returned EBUSY")
448                  raise
449              line = out.split('\n')[1]
450              fields = line.split()
451              total = int(fields[1]) * 1024
452              used = int(fields[2]) * 1024
453              free = int(fields[3]) * 1024
454              percent = float(fields[4].replace('%', ''))
455              return (total, used, free, percent)
456  
457          tolerance = 4 * 1024 * 1024  # 4MB
458          for part in psutil.disk_partitions(all=False):
459              usage = psutil.disk_usage(part.mountpoint)
460              try:
461                  total, used, free, percent = df(part.device)
462              except RuntimeError as err:
463                  # see:
464                  # https://travis-ci.org/giampaolo/psutil/jobs/138338464
465                  # https://travis-ci.org/giampaolo/psutil/jobs/138343361
466                  err = str(err).lower()
467                  if (
468                      "no such file or directory" in err
469                      or "raw devices not supported" in err
470                      or "permission denied" in err
471                  ):
472                      continue
473                  raise
474              else:
475                  self.assertAlmostEqual(usage.total, total, delta=tolerance)
476                  self.assertAlmostEqual(usage.used, used, delta=tolerance)
477                  self.assertAlmostEqual(usage.free, free, delta=tolerance)
478                  self.assertAlmostEqual(usage.percent, percent, delta=1)
