40   def _get_paths_for_parent_directory_preservation():
41       """ 
42       Return list of paths that serve as prefixes for parent-directory preservation of collected binaries and/or 
43       shared libraries. If a binary is collected from a location that starts with a path from this list, the relative 
44       directory structure is preserved within the frozen application bundle; otherwise, the binary is collected to the 
45       frozen application's top-level directory. 
46       """
47   
48       # Use only site-packages paths. We have no control over contents of `sys.path`, so using all paths from that may
49       # lead to unintended behavior in corner cases. For example, if `sys.path` contained the drive root (see #7028),
50       # all paths that do not match some other sub-path rooted in that drive will end up recognized as relative to the
51       # drive root. In such case, any DLL collected from `c:\Windows\system32` will be collected into `Windows\system32`
52       # sub-directory; ucrt DLLs collected from MSVC or Windows SDK installed in `c:\Program Files\...` will end up
53       # collected into `Program Files\...` subdirectory; etc.
54       #
55       # On the other hand, the DLL parent directory preservation is primarily aimed at packages installed via PyPI
56       # wheels, which are typically installed into site-packages. Therefore, limiting the directory preservation for
57       # shared libraries collected from site-packages should do the trick, and should be reasonably safe.
58       import site
59   
60       orig_paths = site.getsitepackages()
61       orig_paths.append(site.getusersitepackages())
62   
63       # Explicitly excluded paths. `site.getsitepackages` seems to include `sys.prefix`, which we need to exclude, to
64       # avoid issue swith DLLs in its sub-directories. We need both resolved and unresolved variant to handle cases
65       # where `base_prefix` itself is a symbolic link (e.g., `scoop`-installed python on Windows, see #8023).
66       excluded_paths = {
67           pathlib.Path(sys.base_prefix),
68           pathlib.Path(sys.base_prefix).resolve(),
69           pathlib.Path(sys.prefix),
70           pathlib.Path(sys.prefix).resolve(),
71       }
72   
73       # For each path in orig_paths, append a resolved variant. This helps with linux venv where we need to consider
74       # both `venv/lib/python3.11/site-packages` and `venv/lib/python3.11/site-packages` and `lib64` is a symlink
75       # to `lib`.
76       orig_paths += [pathlib.Path(path).resolve() for path in orig_paths]
77   
78       paths = set()
79       for path in orig_paths:
80           if not path:
81               continue
82           path = pathlib.Path(path)
83           # Filter out non-directories (e.g., /path/to/python3x.zip) or non-existent paths
84           if not path.is_dir():
85               continue
86           # Filter out explicitly excluded paths
87           if path in excluded_paths:
88               continue
89           paths.add(path)
90   
91       # Sort by length (in term of path components) to ensure match against the longest common prefix (for example, match
92       # /path/to/venv/lib/site-packages instead of /path/to/venv when both paths are in site paths).
93       paths = sorted(paths, key=lambda x: len(x.parents), reverse=True)
94   
95       return paths

98   def _select_destination_directory(src_filename, parent_dir_preservation_paths):
99       # Check parent directory preservation paths
100      for parent_dir_preservation_path in parent_dir_preservation_paths:
101          if parent_dir_preservation_path in src_filename.parents:
102              # Collect into corresponding sub-directory.
103              return src_filename.relative_to(parent_dir_preservation_path)
104  
105      # Collect into top-level directory.
106      return src_filename.name

109  def binary_dependency_analysis(binaries, search_paths=None):
110      """ 
111      Perform binary dependency analysis on the given TOC list of collected binaries, by recursively scanning each binary 
112      for linked dependencies (shared library imports). Returns new TOC list that contains both original entries and their 
113      binary dependencies. 
114   
115      Additional search paths for dependencies' full path resolution may be supplied via optional argument. 
116      """
117  
118      # Get all path prefixes for binaries' parent-directory preservation. For binaries collected from packages in (for
119      # example) site-packages directory, we should try to preserve the parent directory structure.
120      parent_dir_preservation_paths = _get_paths_for_parent_directory_preservation()
121  
122      # Keep track of processed binaries and processed dependencies.
123      processed_binaries = set()
124      processed_dependencies = set()
125  
126      # Keep track of unresolved dependencies, in order to defer the missing-library warnings until after everything has
127      # been processed. This allows us to suppress warnings for dependencies that end up being collected anyway; for
128      # details, see the end of this function.
129      missing_dependencies = []
130  
131      # Populate output TOC with input binaries - this also serves as TODO list, as we iterate over it while appending
132      # new entries at the end.
133      output_toc = binaries[:]
134      for dest_name, src_name, typecode in output_toc:
135          # Do not process symbolic links (already present in input TOC list, or added during analysis below).
136          if typecode == 'SYMLINK':
137              continue
138  
139          # Keep track of processed binaries, to avoid unnecessarily repeating analysis of the same file. Use pathlib.Path
140          # to avoid having to worry about case normalization.
141          src_path = pathlib.Path(src_name)
142          if src_path in processed_binaries:
143              continue
144          processed_binaries.add(src_path)
145  
146          logger.debug("Analyzing binary %r", src_name)
147  
148          # Analyze imports (linked dependencies)
149          for dep_name, dep_src_path in get_imports(src_name, search_paths):
150              logger.debug("Processing dependency, name: %r, resolved path: %r", dep_name, dep_src_path)
151  
152              # Skip unresolved dependencies. Defer the missing-library warnings until after binary dependency analysis
153              # is complete.
154              if not dep_src_path:
155                  missing_dependencies.append((dep_name, src_name))
156                  continue
157  
158              # Compare resolved dependency against global inclusion/exclusion rules.
159              if not dylib.include_library(dep_src_path):
160                  logger.debug("Skipping dependency %r due to global exclusion rules.", dep_src_path)
161                  continue
162  
163              dep_src_path = pathlib.Path(dep_src_path)  # Turn into pathlib.Path for subsequent processing
164  
165              # Avoid processing this dependency if we have already processed it.
166              if dep_src_path in processed_dependencies:
167                  logger.debug("Skipping dependency %r due to prior processing.", str(dep_src_path))
168                  continue
169              processed_dependencies.add(dep_src_path)
170  
171              # Try to preserve parent directory structure, if applicable.
172              # NOTE: do not resolve the source path, because on macOS and linux, it may be a versioned .so (e.g.,
173              # libsomething.so.1, pointing at libsomething.so.1.2.3), and we need to collect it under original name!
174              dep_dest_path = _select_destination_directory(dep_src_path, parent_dir_preservation_paths)
175              dep_dest_path = pathlib.PurePath(dep_dest_path)  # Might be a str() if it is just a basename...
176  
177              # If we are collecting library into top-level directory on macOS, check whether it comes from a
178              # .framework bundle. If it does, re-create the .framework bundle in the top-level directory
179              # instead.
180              if compat.is_darwin and dep_dest_path.parent == pathlib.PurePath('.'):
181                  if osxutils.is_framework_bundle_lib(dep_src_path):
182                      # dst_src_path is parent_path/Name.framework/Versions/Current/Name
183                      framework_parent_path = dep_src_path.parent.parent.parent.parent
184                      dep_dest_path = pathlib.PurePath(dep_src_path.relative_to(framework_parent_path))
185  
186              logger.debug("Collecting dependency %r as %r.", str(dep_src_path), str(dep_dest_path))
187              output_toc.append((str(dep_dest_path), str(dep_src_path), 'BINARY'))
188  
189              # On non-Windows, if we are not collecting the binary into application's top-level directory ('.'),
190              # add a symbolic link from top-level directory to the actual location. This is to accommodate
191              # LD_LIBRARY_PATH being set to the top-level application directory on linux (although library search
192              # should be mostly done via rpaths, so this might be redundant) and to accommodate library path
193              # rewriting on macOS, which assumes that the library was collected into top-level directory.
194              if not compat.is_win and dep_dest_path.parent != pathlib.PurePath('.'):
195                  logger.debug("Adding symbolic link from %r to top-level application directory.", str(dep_dest_path))
196                  output_toc.append((str(dep_dest_path.name), str(dep_dest_path), 'SYMLINK'))
197  
198      # Display warnings about missing dependencies
199      seen_binaries = set([
200          os.path.normcase(os.path.basename(src_name)) for dest_name, src_name, typecode in output_toc
201          if typecode != 'SYMLINK'
202      ])
203      for dependency_name, referring_binary in missing_dependencies:
204          # Ignore libraries that we would not collect in the first place.
205          if not dylib.include_library(dependency_name):
206              continue
207          # Apply global warning suppression rules.
208          if not dylib.warn_missing_lib(dependency_name):
209              continue
210          # If the binary with a matching basename happens to be among the discovered binaries, suppress the message as
211          # well. This might happen either because the library was collected by some other mechanism (for example, via
212          # hook, or supplied by the user), or because it was discovered during the analysis of another binary (which,
213          # for example, had properly set run-paths on Linux/macOS or was located next to that other analyzed binary on
214          # Windows).
215          if os.path.normcase(os.path.basename(dependency_name)) in seen_binaries:
216              continue
217          logger.warning("Library not found: could not resolve %r, dependency of %r.", dependency_name, referring_binary)
218  
219      return output_toc

225  def get_imports(filename, search_paths=None):
226      """ 
227      Analyze the given binary file (shared library or executable), and obtain the list of shared libraries it imports 
228      (i.e., link-time dependencies). 
229   
230      Returns set of tuples (name, fullpath). The name component is the referenced name, and on macOS, may not be just 
231      a base name. If the library's full path cannot be resolved, fullpath element is None. 
232   
233      Additional list of search paths may be specified via `search_paths`, to be used as a fall-back when the 
234      platform-specific resolution mechanism fails to resolve a library fullpath. 
235      """
236      if compat.is_win:
237          if str(filename).lower().endswith(".manifest"):
238              return []
239          return _get_imports_pefile(filename, search_paths)
240      elif compat.is_darwin:
241          return _get_imports_macholib(filename, search_paths)
242      else:
243          return _get_imports_ldd(filename, search_paths)

246  def _get_imports_pefile(filename, search_paths):
247      """ 
248      Windows-specific helper for `get_imports`, which uses the `pefile` library to walk through PE header. 
249      """
250      import pefile
251  
252      output = set()
253  
254      # By default, pefile library parses all PE information. We are only interested in the list of dependent dlls.
255      # Performance is improved by reading only needed information. https://code.google.com/p/pefile/wiki/UsageExamples
256      pe = pefile.PE(filename, fast_load=True)
257      pe.parse_data_directories(
258          directories=[
259              pefile.DIRECTORY_ENTRY['IMAGE_DIRECTORY_ENTRY_IMPORT'],
260              pefile.DIRECTORY_ENTRY['IMAGE_DIRECTORY_ENTRY_EXPORT'],
261          ],
262          forwarded_exports_only=True,
263          import_dllnames_only=True,
264      )
265  
266      # If a library has no binary dependencies, pe.DIRECTORY_ENTRY_IMPORT does not exist.
267      for entry in getattr(pe, 'DIRECTORY_ENTRY_IMPORT', []):
268          dll_str = entry.dll.decode('utf-8')
269          output.add(dll_str)
270  
271      # We must also read the exports table to find forwarded symbols:
272      # http://blogs.msdn.com/b/oldnewthing/archive/2006/07/19/671238.aspx
273      exported_symbols = getattr(pe, 'DIRECTORY_ENTRY_EXPORT', None)
274      if exported_symbols:
275          for symbol in exported_symbols.symbols:
276              if symbol.forwarder is not None:
277                  # symbol.forwarder is a bytes object. Convert it to a string.
278                  forwarder = symbol.forwarder.decode('utf-8')
279                  # symbol.forwarder is for example 'KERNEL32.EnterCriticalSection'
280                  dll = forwarder.split('.')[0]
281                  output.add(dll + ".dll")
282  
283      pe.close()
284  
285      # Attempt to resolve full paths to referenced DLLs. Always add the input binary's parent directory to the search
286      # paths.
287      search_paths = [os.path.dirname(filename)] + (search_paths or [])
288      output = {(lib, resolve_library_path(lib, search_paths)) for lib in output}
289  
290      return output

293  def _get_imports_ldd(filename, search_paths):
294      """ 
295      Helper for `get_imports`, which uses `ldd` to analyze shared libraries. Used on Linux and other POSIX-like platforms 
296      (with exception of macOS). 
297      """
298  
299      output = set()
300  
301      # Output of ldd varies between platforms...
302      if compat.is_aix:
303          # Match libs of the form
304          #   'archivelib.a(objectmember.so/.o)'
305          # or
306          #   'sharedlib.so'
307          # Will not match the fake lib '/unix'
308          LDD_PATTERN = re.compile(r"^\s*(((?P<libarchive>(.*\.a))(?P<objectmember>\(.*\)))|((?P<libshared>(.*\.so))))$")
309      elif compat.is_hpux:
310          # Match libs of the form
311          #   'sharedlib.so => full-path-to-lib
312          # e.g.
313          #   'libpython2.7.so =>      /usr/local/lib/hpux32/libpython2.7.so'
314          LDD_PATTERN = re.compile(r"^\s+(.*)\s+=>\s+(.*)$")
315      elif compat.is_solar:
316          # Match libs of the form
317          #   'sharedlib.so => full-path-to-lib
318          # e.g.
319          #   'libpython2.7.so.1.0 => /usr/local/lib/libpython2.7.so.1.0'
320          # Will not match the platform specific libs starting with '/platform'
321          LDD_PATTERN = re.compile(r"^\s+(.*)\s+=>\s+(.*)$")
322      else:
323          LDD_PATTERN = re.compile(r"\s*(.*?)\s+=>\s+(.*?)\s+\(.*\)")
324  
325      p = subprocess.run(
326          ['ldd', filename],
327          stdin=subprocess.DEVNULL,
328          stderr=subprocess.PIPE,
329          stdout=subprocess.PIPE,
330          encoding='utf-8',
331      )
332  
333      ldd_warnings = []
334      for line in p.stderr.splitlines():
335          if not line:
336              continue
337          # Python extensions (including stdlib ones) are not linked against python.so but rely on Python's symbols having
338          # already been loaded into symbol space at runtime. musl's ldd issues a series of harmless warnings to stderr
339          # telling us that those symbols are unfindable. These should be suppressed.
340          elif line.startswith("Error relocating ") and line.endswith(" symbol not found"):
341              continue
342          # Shared libraries should have the executable bits set; however, this is not the case for shared libraries
343          # shipped in PyPI wheels, which cause ldd to emit `ldd: warning: you do not have execution permission for ...`
344          # warnings. Suppress these.
345          elif line.startswith("ldd: warning: you do not have execution permission for "):
346              continue
347          # When `ldd` is ran against a file that is not a dynamic binary (i.e., is not a binary at all, or is a static
348          # binary), it emits a "not a dynamic executable" warning. Suppress it.
349          elif "not a dynamic executable" in line:
350              continue
351          # Propagate any other warnings it might have.
352          ldd_warnings.append(line)
353      if ldd_warnings:
354          logger.warning("ldd warnings for %r:\n%s", filename, "\n".join(ldd_warnings))
355  
356      for line in p.stdout.splitlines():
357          name = None  # Referenced name
358          lib = None  # Resolved library path
359  
360          m = LDD_PATTERN.search(line)
361          if m:
362              if compat.is_aix:
363                  libarchive = m.group('libarchive')
364                  if libarchive:
365                      # We matched an archive lib with a request for a particular embedded shared object.
366                      #   'archivelib.a(objectmember.so/.o)'
367                      lib = libarchive
368                      name = os.path.basename(lib) + m.group('objectmember')
369                  else:
370                      # We matched a stand-alone shared library.
371                      #   'sharedlib.so'
372                      lib = m.group('libshared')
373                      name = os.path.basename(lib)
374              elif compat.is_hpux:
375                  name, lib = m.group(1), m.group(2)
376              else:
377                  name, lib = m.group(1), m.group(2)
378              if name[:10] in ('linux-gate', 'linux-vdso'):
379                  # linux-gate is a fake library which does not exist and should be ignored. See also:
380                  # http://www.trilithium.com/johan/2005/08/linux-gate/
381                  continue
382  
383              if compat.is_cygwin:
384                  # exclude Windows system library
385                  if lib.lower().startswith('/cygdrive/c/windows/system'):
386                      continue
387  
388              # Reset library path if it does not exist
389              if not os.path.exists(lib):
390                  lib = None
391          elif line.endswith("not found"):
392              # On glibc-based linux distributions, missing libraries are marked with name.so => not found
393              tokens = line.split('=>')
394              if len(tokens) != 2:
395                  continue
396              name = tokens[0].strip()
397              lib = None
398          else:
399              # TODO: should we warn about unprocessed lines?
400              continue
401  
402          # Fall back to searching the supplied search paths, if any.
403          if not lib:
404              lib = _resolve_library_path_in_search_paths(
405                  os.path.basename(name),  # Search for basename of the referenced name.
406                  search_paths,
407              )
408  
409          # Normalize the resolved path, to remove any extraneous "../" elements.
410          if lib:
411              lib = os.path.normpath(lib)
412  
413          # Return referenced name as-is instead of computing a basename, to provide additional context when library
414          # cannot be resolved.
415          output.add((name, lib))
416  
417      return output

420  def _get_imports_macholib(filename, search_paths):
421      """ 
422      macOS-specific helper for `get_imports`, which uses `macholib` to analyze library load commands in Mach-O headers. 
423      """
424      from macholib.dyld import dyld_find
425      from macholib.mach_o import LC_RPATH
426      from macholib.MachO import MachO
427  
428      output = set()
429      referenced_libs = set()  # Libraries referenced in Mach-O headers.
430  
431      # Parent directory of the input binary and parent directory of python executable, used to substitute @loader_path
432      # and @executable_path. The MacOS dylib loader (dyld) fully resolves the symbolic links when using @loader_path
433      # and @executable_path references, so we need to do the same using `os.path.realpath`.
434      bin_path = os.path.dirname(os.path.realpath(filename))
435      python_bin_path = os.path.dirname(os.path.realpath(sys.executable))
436  
437      # Walk through Mach-O headers, and collect all referenced libraries.
438      m = MachO(filename)
439      for header in m.headers:
440          for idx, name, lib in header.walkRelocatables():
441              referenced_libs.add(lib)
442  
443      # Find LC_RPATH commands to collect rpaths. macholib does not handle @rpath, so we need to handle run paths
444      # ourselves.
445      run_paths = set()
446      for header in m.headers:
447          for command in header.commands:
448              # A command is a tuple like:
449              #   (<macholib.mach_o.load_command object at 0x>,
450              #    <macholib.mach_o.rpath_command object at 0x>,
451              #    '../lib\x00\x00')
452              cmd_type = command[0].cmd
453              if cmd_type == LC_RPATH:
454                  rpath = command[2].decode('utf-8')
455                  # Remove trailing '\x00' characters. E.g., '../lib\x00\x00'
456                  rpath = rpath.rstrip('\x00')
457                  # If run path starts with @, ensure it starts with either @loader_path or @executable_path. We cannot
458                  # process anything else.
459                  if rpath.startswith("@") and not rpath.startswith(("@executable_path", "@loader_path")):
460                      logger.warning("Unsupported rpath format %r found in binary %r - ignoring...", rpath, filename)
461                      continue
462                  run_paths.add(rpath)
463  
464      # For distributions like Anaconda, all of the dylibs are stored in the lib directory of the Python distribution, not
465      # alongside of the .so's in each module's subdirectory. Usually, libraries using @rpath to reference their
466      # dependencies also set up their run-paths via LC_RPATH commands. However, they are not strictly required to do so,
467      # because run-paths are inherited from the process within which the libraries are loaded. Therefore, if the python
468      # executable uses an LC_RPATH command to set up run-path that resolves the shared lib directory (for example,
469      # `@loader_path/../lib` in case of the Anaconda python), all libraries loaded within the python process are able
470      # to resolve the shared libraries within the environment's shared lib directory without using LC_RPATH commands
471      # themselves.
472      #
473      # Our analysis does not account for inherited run-paths, and we attempt to work around this limitation by
474      # registering the following fall-back run-path.
475      run_paths.add(os.path.join(compat.base_prefix, 'lib'))
476  
477      def _resolve_using_loader_path(lib, bin_path, python_bin_path):
478          # macholib does not support @loader_path, so replace it with @executable_path. Strictly speaking, @loader_path
479          # should be anchored to parent directory of analyzed binary (`bin_path`), while @executable_path should be
480          # anchored to the parent directory of the process' executable. Typically, this would be python executable
481          # (`python_bin_path`), unless we are analyzing a collected 3rd party executable. In that case, `bin_path`
482          # is correct option. So we first try resolving using `bin_path`, and then fall back to `python_bin_path`.
483          # This does not account for transitive run paths of higher-order dependencies, but there is only so much we
484          # can do here...
485          if lib.startswith('@loader_path'):
486              lib = lib.replace('@loader_path', '@executable_path')
487  
488          try:
489              # Try resolving with binary's path first...
490              return dyld_find(lib, executable_path=bin_path)
491          except ValueError:
492              # ... and fall-back to resolving with python executable's path
493              try:
494                  return dyld_find(lib, executable_path=python_bin_path)
495              except ValueError:
496                  return None
497  
498      def _resolve_using_path(lib):
499          try:
500              return dyld_find(lib)
501          except ValueError:
502              return None
503  
504      # Try to resolve full path of the referenced libraries.
505      for referenced_lib in referenced_libs:
506          resolved_lib = None
507  
508          # If path starts with @rpath, we have to handle it ourselves.
509          if referenced_lib.startswith('@rpath'):
510              lib = os.path.join(*referenced_lib.split(os.sep)[1:])  # Remove the @rpath/ prefix
511  
512              # Try all run paths.
513              for run_path in run_paths:
514                  # Join the path.
515                  lib_path = os.path.join(run_path, lib)
516  
517                  if lib_path.startswith(("@executable_path", "@loader_path")):
518                      # Run path starts with @executable_path or @loader_path.
519                      lib_path = _resolve_using_loader_path(lib_path, bin_path, python_bin_path)
520                  else:
521                      # If run path was relative, anchor it to binary's location.
522                      if not os.path.isabs(lib_path):
523                          os.path.join(bin_path, lib_path)
524                      lib_path = _resolve_using_path(lib_path)
525  
526                  if lib_path and os.path.exists(lib_path):
527                      resolved_lib = lib_path
528                      break
529          else:
530              if referenced_lib.startswith(("@executable_path", "@loader_path")):
531                  resolved_lib = _resolve_using_loader_path(referenced_lib, bin_path, python_bin_path)
532              else:
533                  resolved_lib = _resolve_using_path(referenced_lib)
534  
535          # Fall back to searching the supplied search paths, if any.
536          if not resolved_lib:
537              resolved_lib = _resolve_library_path_in_search_paths(
538                  os.path.basename(referenced_lib),  # Search for basename of the referenced name.
539                  search_paths,
540              )
541  
542          # Normalize the resolved path, to remove any extraneous "../" elements.
543          if resolved_lib:
544              resolved_lib = os.path.normpath(resolved_lib)
545  
546          # Return referenced library name as-is instead of computing a basename. Full referenced name carries additional
547          # information that might be useful for the caller to determine how to deal with unresolved library (e.g., ignore
548          # unresolved libraries that are supposed to be located in system-wide directories).
549          output.add((referenced_lib, resolved_lib))
550  
551      return output

557  def resolve_library_path(name, search_paths=None):
558      """ 
559      Given a library name, attempt to resolve full path to that library. The search for library is done via 
560      platform-specific mechanism and fall back to optionally-provided list of search paths. Returns None if library 
561      cannot be resolved. If give library name is already an absolute path, the given path is returned without any 
562      processing. 
563      """
564      # No-op if path is already absolute.
565      if os.path.isabs(name):
566          return name
567  
568      if compat.is_unix:
569          # Use platform-specific helper.
570          fullpath = _resolve_library_path_unix(name)
571          if fullpath:
572              return fullpath
573          # Fall back to searching the supplied search paths, if any
574          return _resolve_library_path_in_search_paths(name, search_paths)
575      elif compat.is_win:
576          # Try the caller-supplied search paths, if any.
577          fullpath = _resolve_library_path_in_search_paths(name, search_paths)
578          if fullpath:
579              return fullpath
580  
581          # Fall back to default Windows search paths, using the PATH environment variable (which should also include
582          # the system paths, such as c:\windows and c:\windows\system32)
583          win_search_paths = [path for path in compat.getenv('PATH', '').split(os.pathsep) if path]
584          return _resolve_library_path_in_search_paths(name, win_search_paths)
585      else:
586          return ctypes.util.find_library(name)
587  
588      return None
589

592  # `resolve_library_path`.
593  findLibrary = resolve_library_path
594  findSystemLibrary = resolve_library_path

597  def _resolve_library_path_in_search_paths(name, search_paths=None):
598      """ 
599      Low-level helper for resolving given library name to full path in given list of search paths. 
600      """
601      for search_path in search_paths or []:
602          fullpath = os.path.join(search_path, name)
603          if not os.path.isfile(fullpath):
604              continue
605  
606          # On Windows, ensure that architecture matches that of running python interpreter.
607          if compat.is_win:
608              try:
609                  dll_machine_type = winutils.get_pe_file_machine_type(fullpath)
610              except Exception:
611                  # A search path might contain a DLL that we cannot analyze; for example, a stub file. Skip over.
612                  continue
613              if dll_machine_type != _exe_machine_type:
614                  continue
615  
616          return os.path.normpath(fullpath)
617  
618      return None

621  def _resolve_library_path_unix(name):
622      """ 
623      UNIX-specific helper for resolving library path. 
624   
625      Emulates the algorithm used by dlopen. `name` must include the prefix, e.g., ``libpython2.4.so``. 
626      """
627      assert compat.is_unix, "Current implementation for Unix only (Linux, Solaris, AIX, FreeBSD)"
628  
629      if name.endswith('.so') or '.so.' in name:
630          # We have been given full library name that includes suffix. Use `_resolve_library_path_in_search_paths` to find
631          # the exact match.
632          lib_search_func = _resolve_library_path_in_search_paths
633      else:
634          # We have been given a library name without suffix. Use `_which_library` as search function, which will try to
635          # find library with matching basename.
636          lib_search_func = _which_library
637  
638      # Look in the LD_LIBRARY_PATH according to platform.
639      if compat.is_aix:
640          lp = compat.getenv('LIBPATH', '')
641      elif compat.is_darwin:
642          lp = compat.getenv('DYLD_LIBRARY_PATH', '')
643      else:
644          lp = compat.getenv('LD_LIBRARY_PATH', '')
645      lib = lib_search_func(name, filter(None, lp.split(os.pathsep)))
646  
647      # Look in /etc/ld.so.cache
648      # Solaris does not have /sbin/ldconfig. Just check if this file exists.
649      if lib is None:
650          utils.load_ldconfig_cache()
651          lib = utils.LDCONFIG_CACHE.get(name)
652          if lib:
653              assert os.path.isfile(lib)
654  
655      # Look in the known safe paths.
656      if lib is None:
657          # Architecture independent locations.
658          paths = ['/lib', '/usr/lib']
659          # Architecture dependent locations.
660          if compat.architecture == '32bit':
661              paths.extend(['/lib32', '/usr/lib32'])
662          else:
663              paths.extend(['/lib64', '/usr/lib64'])
664          # Machine dependent locations.
665          if compat.machine == 'intel':
666              if compat.architecture == '32bit':
667                  paths.extend(['/usr/lib/i386-linux-gnu'])
668              else:
669                  paths.extend(['/usr/lib/x86_64-linux-gnu'])
670  
671          # On Debian/Ubuntu /usr/bin/python is linked statically with libpython. Newer Debian/Ubuntu with multiarch
672          # support puts the libpythonX.Y.so in paths like /usr/lib/i386-linux-gnu/. Try to query the arch-specific
673          # sub-directory, if available.
674          arch_subdir = sysconfig.get_config_var('multiarchsubdir')
675          if arch_subdir:
676              arch_subdir = os.path.basename(arch_subdir)
677              paths.append(os.path.join('/usr/lib', arch_subdir))
678          else:
679              logger.debug('Multiarch directory not detected.')
680  
681          # Termux (a Ubuntu like subsystem for Android) has an additional libraries directory.
682          if os.path.isdir('/data/data/com.termux/files/usr/lib'):
683              paths.append('/data/data/com.termux/files/usr/lib')
684  
685          if compat.is_aix:
686              paths.append('/opt/freeware/lib')
687          elif compat.is_hpux:
688              if compat.architecture == '32bit':
689                  paths.append('/usr/local/lib/hpux32')
690              else:
691                  paths.append('/usr/local/lib/hpux64')
692          elif compat.is_freebsd or compat.is_openbsd:
693              paths.append('/usr/local/lib')
694          lib = lib_search_func(name, paths)
695  
696      return lib

699  def _which_library(name, dirs):
700      """ 
701      Search for a shared library in a list of directories. 
702   
703      Args: 
704          name: 
705              The library name including the `lib` prefix but excluding any `.so` suffix. 
706          dirs: 
707              An iterable of folders to search in. 
708      Returns: 
709          The path to the library if found or None otherwise. 
710   
711      """
712      matcher = _library_matcher(name)
713      for path in filter(os.path.exists, dirs):
714          for _path in os.listdir(path):
715              if matcher(_path):
716                  return os.path.join(path, _path)
717

719  def _library_matcher(name):
720      """ 
721      Create a callable that matches libraries if **name** is a valid library prefix for input library full names. 
722      """
723      return re.compile(name + r"[0-9]*\.").match

729  def get_python_library_path():
730      """ 
731      Find dynamic Python library that will be bundled with frozen executable. 
732   
733      NOTE: This is a fallback option when the Python executable is likely statically linked with the Python library and 
734            we need to search more for it. For example, this is the case on Debian/Ubuntu. 
735   
736      Return  full path to Python dynamic library or None when not found. 
737   
738      We need to know name of the Python dynamic library for the bootloader. Bootloader has to know what library to 
739      load and not try to guess. 
740   
741      Some linux distributions (e.g. debian-based) statically link the Python executable to the libpython, 
742      so bindepend does not include it in its output. In this situation let's try to find it. 
743   
744      Custom Mac OS builds could possibly also have non-framework style libraries, so this method also checks for that 
745      variant as well. 
746      """
747      def _find_lib_in_libdirs(*libdirs):
748          for libdir in libdirs:
749              for name in compat.PYDYLIB_NAMES:
750                  full_path = os.path.join(libdir, name)
751                  if not os.path.exists(full_path):
752                      continue
753                  # Resolve potential symbolic links to achieve consistent results with linker-based search; e.g., on
754                  # POSIX systems, linker resolves unversioned library names (python3.X.so) to versioned ones
755                  # (libpython3.X.so.1.0) due to former being symbolic linkes to the latter. See #6831.
756                  full_path = os.path.realpath(full_path)
757                  if not os.path.exists(full_path):
758                      continue
759                  return full_path
760          return None
761  
762      # If this is Microsoft App Store Python, check the compat.base_path first. While compat.python_executable resolves
763      # to actual python.exe file, the latter contains a relative library reference that we fail to properly resolve.
764      if compat.is_ms_app_store:
765          python_libname = _find_lib_in_libdirs(compat.base_prefix)
766          if python_libname:
767              return python_libname
768  
769      # Try to get Python library name from the Python executable. It assumes that Python library is not statically
770      # linked.
771      imported_libraries = get_imports(compat.python_executable)  # (name, fullpath) tuples
772      for _, lib_path in imported_libraries:
773          if lib_path is None:
774              continue  # Skip unresolved imports
775          for name in compat.PYDYLIB_NAMES:
776              if os.path.normcase(os.path.basename(lib_path)) == name:
777                  # Python library found. Return absolute path to it.
778                  return lib_path
779  
780      # Python library NOT found. Resume searching using alternative methods.
781  
782      # Work around for python venv having VERSION.dll rather than pythonXY.dll
783      if compat.is_win and any([os.path.normcase(lib_name) == 'version.dll' for lib_name, _ in imported_libraries]):
784          pydll = 'python%d%d.dll' % sys.version_info[:2]
785          return resolve_library_path(pydll, [os.path.dirname(compat.python_executable)])
786  
787      # Applies only to non Windows platforms and conda.
788  
789      if compat.is_conda:
790          # Conda needs to be the first here since it overrules the operating system specific paths.
791          python_libname = _find_lib_in_libdirs(os.path.join(compat.base_prefix, 'lib'))
792          if python_libname:
793              return python_libname
794  
795      elif compat.is_unix:
796          for name in compat.PYDYLIB_NAMES:
797              python_libname = findLibrary(name)
798              if python_libname:
799                  return python_libname
800  
801      if compat.is_darwin or compat.is_linux:
802          # On MacPython, Analysis.assemble is able to find the libpython with no additional help, asking for
803          # sys.executable dependencies. However, this fails on system python, because the shared library is not listed as
804          # a dependency of the binary (most probably it is opened at runtime using some dlopen trickery). This happens on
805          # Mac OS when Python is compiled as Framework.
806          # Linux using pyenv is similarly linked so that sys.executable dependencies does not yield libpython.so.
807  
808          # Python compiled as Framework contains same values in sys.prefix and exec_prefix. That is why we can use just
809          # sys.prefix. In virtualenv, PyInstaller is not able to find Python library. We need special care for this case.
810          python_libname = _find_lib_in_libdirs(
811              compat.base_prefix,
812              os.path.join(compat.base_prefix, 'lib'),
813          )
814          if python_libname:
815              return python_libname
816  
817      # Python library NOT found. Return None and let the caller deal with this.
818      return None

824  def classify_binary_vs_data(filename):
825      """ 
826      Classify the given file as either BINARY or a DATA, using appropriate platform-specific method. Returns 'BINARY' 
827      or 'DATA' string depending on the determined file type, or None if classification cannot be performed (non-existing 
828      file, missing tool, and other errors during classification). 
829      """
830  
831      # We cannot classify non-existent files.
832      if not os.path.isfile(filename):
833          return None
834  
835      # Use platform-specific implementation.
836      return _classify_binary_vs_data(filename)
837  
838  
839  if compat.is_linux:
840  
841      def _classify_binary_vs_data(filename):
842          # First check for ELF signature, in order to avoid calling `objdump` on every data file, which can be costly.
843          try:
844              with open(filename, 'rb') as fp:
845                  sig = fp.read(4)
846          except Exception:
847              return None
848  
849          if sig != b"\x7FELF":
850              return "DATA"
851  
852          # Verify the binary by checking if `objdump` recognizes the file. The preceding ELF signature check should
853          # ensure that this is an ELF file, while this check should ensure that it is a valid ELF file. In the future,
854          # we could try checking that the architecture matches the running platform.
855          cmd_args = ['objdump', '-a', filename]
856          try:
857              p = subprocess.run(
858                  cmd_args,
859                  stdout=subprocess.PIPE,
860                  stderr=subprocess.PIPE,
861                  stdin=subprocess.DEVNULL,
862                  encoding='utf8',
863              )
864          except Exception:
865              return None  # Failed to run `objdump` or `objdump` unavailable.
866  
867          return 'BINARY' if p.returncode == 0 else 'DATA'
