12   import os
13   import sys
14   
15   # Installing `pyproj` Conda packages requires to set `PROJ_LIB`
16   
17   is_win = sys.platform.startswith('win')
18   if is_win:
19   
20       proj_data = os.path.join(sys._MEIPASS, 'Library', 'share', 'proj')
21   
22   else:
23       proj_data = os.path.join(sys._MEIPASS, 'share', 'proj')
24   
25   if os.path.exists(proj_data):
26       os.environ['PROJ_LIB'] = proj_data
