discovery.py 
1    """Automatic discovery of Python modules and packages (for inclusion in the 
2    distribution) and other config values. 
3     
4    For the purposes of this module, the following nomenclature is used: 
5     
6    - "src-layout": a directory representing a Python project that contains a "src" 
7      folder. Everything under the "src" folder is meant to be included in the 
8      distribution when packaging the project. Example:: 
9     
10       . 
11       ├── tox.ini 
12       ├── pyproject.toml 
13       └── src/ 
14           └── mypkg/ 
15               ├── __init__.py 
16               ├── mymodule.py 
17               └── my_data_file.txt 
18    
19   - "flat-layout": a Python project that does not use "src-layout" but instead 
20     have a directory under the project root for each package:: 
21    
22       . 
23       ├── tox.ini 
24       ├── pyproject.toml 
25       └── mypkg/ 
26           ├── __init__.py 
27           ├── mymodule.py 
28           └── my_data_file.txt 
29    
30   - "single-module": a project that contains a single Python script direct under 
31     the project root (no directory used):: 
32    
33       . 
34       ├── tox.ini 
35       ├── pyproject.toml 
36       └── mymodule.py 
37    
38   """

75   class _Filter:
76       """ 
77       Given a list of patterns, create a callable that will be true only if 
78       the input matches at least one of the patterns. 
79       """
80   
81       def __init__(self, *patterns: str):
82           self._patterns = dict.fromkeys(patterns)
83   
84       def __call__(self, item: str) -> bool:
85           return any(fnmatchcase(item, pat) for pat in self._patterns)
86   
87       def __contains__(self, item: str) -> bool:
88           return item in self._patterns

91   class _Finder:
92       """Base class that exposes functionality for module/package finders"""
93   
94       ALWAYS_EXCLUDE: Tuple[str, ...] = ()
95       DEFAULT_EXCLUDE: Tuple[str, ...] = ()
96   
97       @classmethod
98       def find(
99           cls,
100          where: StrPath = '.',
101          exclude: Iterable[str] = (),
102          include: Iterable[str] = ('*',),
103      ) -> List[str]:
104          """Return a list of all Python items (packages or modules, depending on 
105          the finder implementation) found within directory 'where'. 
106   
107          'where' is the root directory which will be searched. 
108          It should be supplied as a "cross-platform" (i.e. URL-style) path; 
109          it will be converted to the appropriate local path syntax. 
110   
111          'exclude' is a sequence of names to exclude; '*' can be used 
112          as a wildcard in the names. 
113          When finding packages, 'foo.*' will exclude all subpackages of 'foo' 
114          (but not 'foo' itself). 
115   
116          'include' is a sequence of names to include. 
117          If it's specified, only the named items will be included. 
118          If it's not specified, all found items will be included. 
119          'include' can contain shell style wildcard patterns just like 
120          'exclude'. 
121          """
122  
123          exclude = exclude or cls.DEFAULT_EXCLUDE
124          return list(
125              cls._find_iter(
126                  convert_path(str(where)),
127                  _Filter(*cls.ALWAYS_EXCLUDE, *exclude),
128                  _Filter(*include),
129              )
130          )
131  
132      @classmethod
133      def _find_iter(cls, where: StrPath, exclude: _Filter, include: _Filter) -> StrIter:
134          raise NotImplementedError

137  class PackageFinder(_Finder):
138      """ 
139      Generate a list of all Python packages found within a directory 
140      """
141  
142      ALWAYS_EXCLUDE = ("ez_setup", "*__pycache__")
143  
144      @classmethod
145      def _find_iter(cls, where: StrPath, exclude: _Filter, include: _Filter) -> StrIter:
146          """ 
147          All the packages found in 'where' that pass the 'include' filter, but 
148          not the 'exclude' filter. 
149          """
150          for root, dirs, files in os.walk(str(where), followlinks=True):
151              # Copy dirs to iterate over it, then empty dirs.
152              all_dirs = dirs[:]
153              dirs[:] = []
154  
155              for dir in all_dirs:
156                  full_path = os.path.join(root, dir)
157                  rel_path = os.path.relpath(full_path, where)
158                  package = rel_path.replace(os.path.sep, '.')
159  
160                  # Skip directory trees that are not valid packages
161                  if '.' in dir or not cls._looks_like_package(full_path, package):
162                      continue
163  
164                  # Should this package be included?
165                  if include(package) and not exclude(package):
166                      yield package
167  
168                  # Early pruning if there is nothing else to be scanned
169                  if f"{package}*" in exclude or f"{package}.*" in exclude:
170                      continue
171  
172                  # Keep searching subdirectories, as there may be more packages
173                  # down there, even if the parent was excluded.
174                  dirs.append(dir)
175  
176      @staticmethod
177      def _looks_like_package(path: StrPath, _package_name: str) -> bool:
178          """Does a directory look like a package?"""
179          return os.path.isfile(os.path.join(path, '__init__.py'))

188  class ModuleFinder(_Finder):
189      """Find isolated Python modules. 
190      This function will **not** recurse subdirectories. 
191      """
192  
193      @classmethod
194      def _find_iter(cls, where: StrPath, exclude: _Filter, include: _Filter) -> StrIter:
195          for file in glob(os.path.join(where, "*.py")):
196              module, _ext = os.path.splitext(os.path.basename(file))
197  
198              if not cls._looks_like_module(module):
199                  continue
200  
201              if include(module) and not exclude(module):
202                  yield module
203  
204      _looks_like_module = staticmethod(_valid_name)

294  def _find_packages_within(root_pkg: str, pkg_dir: StrPath) -> List[str]:
295      nested = PEP420PackageFinder.find(pkg_dir)
296      return [root_pkg] + [".".join((root_pkg, n)) for n in nested]

299  class ConfigDiscovery:
300      """Fill-in metadata and options that can be automatically derived 
301      (from other metadata/options, the file system or conventions) 
302      """
303  
304      def __init__(self, distribution: "Distribution"):
305          self.dist = distribution
306          self._called = False
307          self._disabled = False
308          self._skip_ext_modules = False
309  
310      def _disable(self):
311          """Internal API to disable automatic discovery"""
312          self._disabled = True
313  
314      def _ignore_ext_modules(self):
315          """Internal API to disregard ext_modules. 
316   
317          Normally auto-discovery would not be triggered if ``ext_modules`` are set 
318          (this is done for backward compatibility with existing packages relying on 
319          ``setup.py`` or ``setup.cfg``). However, ``setuptools`` can call this function 
320          to ignore given ``ext_modules`` and proceed with the auto-discovery if 
321          ``packages`` and ``py_modules`` are not given (e.g. when using pyproject.toml 
322          metadata). 
323          """
324          self._skip_ext_modules = True
325  
326      @property
327      def _root_dir(self) -> StrPath:
328          # The best is to wait until `src_root` is set in dist, before using _root_dir.
329          return self.dist.src_root or os.curdir
330  
331      @property
332      def _package_dir(self) -> Dict[str, str]:
333          if self.dist.package_dir is None:
334              return {}
335          return self.dist.package_dir
336  
337      def __call__(self, force=False, name=True, ignore_ext_modules=False):
338          """Automatically discover missing configuration fields 
339          and modifies the given ``distribution`` object in-place. 
340   
341          Note that by default this will only have an effect the first time the 
342          ``ConfigDiscovery`` object is called. 
343   
344          To repeatedly invoke automatic discovery (e.g. when the project 
345          directory changes), please use ``force=True`` (or create a new 
346          ``ConfigDiscovery`` instance). 
347          """
348          if force is False and (self._called or self._disabled):
349              # Avoid overhead of multiple calls
350              return
351  
352          self._analyse_package_layout(ignore_ext_modules)
353          if name:
354              self.analyse_name()  # depends on ``packages`` and ``py_modules``
355  
356          self._called = True
357  
358      def _explicitly_specified(self, ignore_ext_modules: bool) -> bool:
359          """``True`` if the user has specified some form of package/module listing"""
360          ignore_ext_modules = ignore_ext_modules or self._skip_ext_modules
361          ext_modules = not (self.dist.ext_modules is None or ignore_ext_modules)
362          return (
363              self.dist.packages is not None
364              or self.dist.py_modules is not None
365              or ext_modules
366              or hasattr(self.dist, "configuration")
367              and self.dist.configuration
368              # ^ Some projects use numpy.distutils.misc_util.Configuration
369          )
370  
371      def _analyse_package_layout(self, ignore_ext_modules: bool) -> bool:
372          if self._explicitly_specified(ignore_ext_modules):
373              # For backward compatibility, just try to find modules/packages
374              # when nothing is given
375              return True
376  
377          log.debug(
378              "No `packages` or `py_modules` configuration, performing "
379              "automatic discovery."
380          )
381  
382          return (
383              self._analyse_explicit_layout()
384              or self._analyse_src_layout()
385              # flat-layout is the trickiest for discovery so it should be last
386              or self._analyse_flat_layout()
387          )
388  
389      def _analyse_explicit_layout(self) -> bool:
390          """The user can explicitly give a package layout via ``package_dir``"""
391          package_dir = self._package_dir.copy()  # don't modify directly
392          package_dir.pop("", None)  # This falls under the "src-layout" umbrella
393          root_dir = self._root_dir
394  
395          if not package_dir:
396              return False
397  
398          log.debug(f"`explicit-layout` detected -- analysing {package_dir}")
399          pkgs = chain_iter(
400              _find_packages_within(pkg, os.path.join(root_dir, parent_dir))
401              for pkg, parent_dir in package_dir.items()
402          )
403          self.dist.packages = list(pkgs)
404          log.debug(f"discovered packages -- {self.dist.packages}")
405          return True
406  
407      def _analyse_src_layout(self) -> bool:
408          """Try to find all packages or modules under the ``src`` directory 
409          (or anything pointed by ``package_dir[""]``). 
410   
411          The "src-layout" is relatively safe for automatic discovery. 
412          We assume that everything within is meant to be included in the 
413          distribution. 
414   
415          If ``package_dir[""]`` is not given, but the ``src`` directory exists, 
416          this function will set ``package_dir[""] = "src"``. 
417          """
418          package_dir = self._package_dir
419          src_dir = os.path.join(self._root_dir, package_dir.get("", "src"))
420          if not os.path.isdir(src_dir):
421              return False
422  
423          log.debug(f"`src-layout` detected -- analysing {src_dir}")
424          package_dir.setdefault("", os.path.basename(src_dir))
425          self.dist.package_dir = package_dir  # persist eventual modifications
426          self.dist.packages = PEP420PackageFinder.find(src_dir)
427          self.dist.py_modules = ModuleFinder.find(src_dir)
428          log.debug(f"discovered packages -- {self.dist.packages}")
429          log.debug(f"discovered py_modules -- {self.dist.py_modules}")
430          return True
431  
432      def _analyse_flat_layout(self) -> bool:
433          """Try to find all packages and modules under the project root. 
434   
435          Since the ``flat-layout`` is more dangerous in terms of accidentally including 
436          extra files/directories, this function is more conservative and will raise an 
437          error if multiple packages or modules are found. 
438   
439          This assumes that multi-package dists are uncommon and refuse to support that 
440          use case in order to be able to prevent unintended errors. 
441          """
442          log.debug(f"`flat-layout` detected -- analysing {self._root_dir}")
443          return self._analyse_flat_packages() or self._analyse_flat_modules()
444  
445      def _analyse_flat_packages(self) -> bool:
446          self.dist.packages = FlatLayoutPackageFinder.find(self._root_dir)
447          top_level = remove_nested_packages(remove_stubs(self.dist.packages))
448          log.debug(f"discovered packages -- {self.dist.packages}")
449          self._ensure_no_accidental_inclusion(top_level, "packages")
450          return bool(top_level)
451  
452      def _analyse_flat_modules(self) -> bool:
453          self.dist.py_modules = FlatLayoutModuleFinder.find(self._root_dir)
454          log.debug(f"discovered py_modules -- {self.dist.py_modules}")
455          self._ensure_no_accidental_inclusion(self.dist.py_modules, "modules")
456          return bool(self.dist.py_modules)
457  
458      def _ensure_no_accidental_inclusion(self, detected: List[str], kind: str):
459          if len(detected) > 1:
460              from inspect import cleandoc
461  
462              from setuptools.errors import PackageDiscoveryError
463  
464              msg = f"""Multiple top-level {kind} discovered in a flat-layout: {detected}.
465  
466              To avoid accidental inclusion of unwanted files or directories,
467              setuptools will not proceed with this build.
468  
469              If you are trying to create a single distribution with multiple {kind}
470              on purpose, you should not rely on automatic discovery.
471              Instead, consider the following options:
472  
473              1. set up custom discovery (`find` directive with `include` or `exclude`)
474              2. use a `src-layout`
475              3. explicitly set `py_modules` or `packages` with a list of names
476  
477              To find more information, look for "package discovery" on setuptools docs.
478              """
479              raise PackageDiscoveryError(cleandoc(msg))

481      def analyse_name(self):
482          """The packages/modules are the essential contribution of the author. 
483          Therefore the name of the distribution can be derived from them. 
484          """
485          if self.dist.metadata.name or self.dist.name:
486              # get_name() is not reliable (can return "UNKNOWN")
487              return
488  
489          log.debug("No `name` configuration, performing automatic discovery")
490  
491          name = (
492              self._find_name_single_package_or_module()
493              or self._find_name_from_packages()
494          )
495          if name:
496              self.dist.metadata.name = name
497  
498      def _find_name_single_package_or_module(self) -> Optional[str]:
499          """Exactly one module or package"""
500          for field in ('packages', 'py_modules'):
501              items = getattr(self.dist, field, None) or []
502              if items and len(items) == 1:
503                  log.debug(f"Single module/package detected, name: {items[0]}")
504                  return items[0]
505  
506          return None
507  
508      def _find_name_from_packages(self) -> Optional[str]:
509          """Try to find the root package that is not a PEP 420 namespace"""
510          if not self.dist.packages:
511              return None
512  
513          packages = remove_stubs(sorted(self.dist.packages, key=len))
514          package_dir = self.dist.package_dir or {}
515  
516          parent_pkg = find_parent_package(packages, package_dir, self._root_dir)
517          if parent_pkg:
518              log.debug(f"Common parent package detected, name: {parent_pkg}")
519              return parent_pkg
520  
521          log.warn("No parent package detected, impossible to derive `name`")
522          return None

525  def remove_nested_packages(packages: List[str]) -> List[str]:
526      """Remove nested packages from a list of packages. 
527   
528      >>> remove_nested_packages(["a", "a.b1", "a.b2", "a.b1.c1"]) 
529      ['a'] 
530      >>> remove_nested_packages(["a", "b", "c.d", "c.d.e.f", "g.h", "a.a1"]) 
531      ['a', 'b', 'c.d', 'g.h'] 
532      """
533      pkgs = sorted(packages, key=len)
534      top_level = pkgs[:]
535      size = len(pkgs)
536      for i, name in enumerate(reversed(pkgs)):
537          if any(name.startswith(f"{other}.") for other in top_level):
538              top_level.pop(size - i - 1)
539  
540      return top_level

543  def remove_stubs(packages: List[str]) -> List[str]:
544      """Remove type stubs (:pep:`561`) from a list of packages. 
545   
546      >>> remove_stubs(["a", "a.b", "a-stubs", "a-stubs.b.c", "b", "c-stubs"]) 
547      ['a', 'a.b', 'b'] 
548      """
549      return [pkg for pkg in packages if not pkg.split(".")[0].endswith("-stubs")]

552  def find_parent_package(
553      packages: List[str], package_dir: Mapping[str, str], root_dir: StrPath
554  ) -> Optional[str]:
555      """Find the parent package that is not a namespace."""
556      packages = sorted(packages, key=len)
557      common_ancestors = []
558      for i, name in enumerate(packages):
559          if not all(n.startswith(f"{name}.") for n in packages[i + 1 :]):
560              # Since packages are sorted by length, this condition is able
561              # to find a list of all common ancestors.
562              # When there is divergence (e.g. multiple root packages)
563              # the list will be empty
564              break
565          common_ancestors.append(name)
566  
567      for name in common_ancestors:
568          pkg_path = find_package_path(name, package_dir, root_dir)
569          init = os.path.join(pkg_path, "__init__.py")
570          if os.path.isfile(init):
571              return name
572  
573      return None

576  def find_package_path(
577      name: str, package_dir: Mapping[str, str], root_dir: StrPath
578  ) -> str:
579      """Given a package name, return the path where it should be found on 
580      disk, considering the ``package_dir`` option. 
581   
582      >>> path = find_package_path("my.pkg", {"": "root/is/nested"}, ".") 
583      >>> path.replace(os.sep, "/") 
584      './root/is/nested/my/pkg' 
585   
586      >>> path = find_package_path("my.pkg", {"my": "root/is/nested"}, ".") 
587      >>> path.replace(os.sep, "/") 
588      './root/is/nested/pkg' 
589   
590      >>> path = find_package_path("my.pkg", {"my.pkg": "root/is/nested"}, ".") 
591      >>> path.replace(os.sep, "/") 
592      './root/is/nested' 
593   
594      >>> path = find_package_path("other.pkg", {"my.pkg": "root/is/nested"}, ".") 
595      >>> path.replace(os.sep, "/") 
596      './other/pkg' 
597      """
598      parts = name.split(".")
599      for i in range(len(parts), 0, -1):
600          # Look backwards, the most specific package_dir first
601          partial_name = ".".join(parts[:i])
602          if partial_name in package_dir:
603              parent = package_dir[partial_name]
604              return os.path.join(root_dir, parent, *parts[i:])
605  
606      parent = package_dir.get("") or ""
607      return os.path.join(root_dir, *parent.split("/"), *parts)

610  def construct_package_dir(packages: List[str], package_path: StrPath) -> Dict[str, str]:
611      parent_pkgs = remove_nested_packages(packages)
612      prefix = Path(package_path).parts
613      return {pkg: "/".join([*prefix, *pkg.split(".")]) for pkg in parent_pkgs}
