1    # -*- coding: utf-8 -*-
2    """ 
3    Tools for searching bytecode for key statements that indicate the need for additional resources, such as data files 
4    and package metadata. 
5     
6    By *bytecode* I mean the ``code`` object given by ``compile()``, accessible from the ``__code__`` attribute of any 
7    non-builtin function or, in PyInstallerLand, the ``PyiModuleGraph.node("some.module").code`` attribute. The best 
8    guide for bytecode format I have found is the disassembler reference: https://docs.python.org/3/library/dis.html 
9     
10   This parser implementation aims to combine the flexibility and speed of regex with the clarity of the output of 
11   ``dis.dis(code)``. It has not achieved the 2nd, but C'est la vie... 
12    
13   The biggest clarity killer here is the ``EXTENDED_ARG`` opcode which can appear almost anywhere and therefore needs 
14   to be tiptoed around at every step. If this code needs to expand significantly, I would recommend an upgrade to a 
15   regex-based grammar parsing library such as Reparse. This way, little steps like unpacking ``EXTENDED_ARGS`` can be 
16   defined once then simply referenced forming a nice hierarchy rather than copied everywhere its needed. 
17   """
18

27   # Python 3.11 introduced specialized opcodes that are not covered by opcode.opmap (and equivalent dis.opmap), but dis

29   opmap = getattr(dis, '_all_opmap', dis.opmap)

32   def _instruction_to_regex(x: str):
33       """ 
34       Get a regex-escaped opcode byte from its human readable name. 
35       """
36       return re.escape(bytes([opmap[x]]))

39   def bytecode_regex(pattern: bytes, flags=re.VERBOSE | re.DOTALL):
40       """ 
41       A regex-powered Python bytecode matcher. 
42    
43       ``bytecode_regex`` provides a very thin wrapper around :func:`re.compile`. 
44    
45         * Any opcode names wrapped in backticks are substituted for their corresponding opcode bytes. 
46         * Patterns are compiled in VERBOSE mode by default so that whitespace and comments may be used. 
47    
48       This aims to mirror the output of :func:`dis.dis`, which is far more readable than looking at raw byte strings. 
49       """
50       assert isinstance(pattern, bytes)
51   
52       # Replace anything wrapped in backticks with regex-escaped opcodes.
53       pattern = re.sub(
54           rb"`(\w+)`",
55           lambda m: _instruction_to_regex(m[1].decode()),
56           pattern,
57       )
58       return re.compile(pattern, flags=flags)

61   def finditer(pattern: Pattern, string: bytes):
62       """ 
63       Call ``pattern.finditer(string)``, but remove any matches beginning on an odd byte (i.e., matches where 
64       match.start() is not a multiple of 2). 
65    
66       This should be used to avoid false positive matches where a bytecode pair's argument is mistaken for an opcode. 
67       """
68       assert isinstance(string, bytes)
69       string = _cleanup_bytecode_string(string)
70       matches = pattern.finditer(string)
71       while True:
72           for match in matches:
73               if match.start() % 2 == 0:
74                   # All is good. This match starts on an OPCODE.
75                   yield match
76               else:
77                   # This match has started on an odd byte, meaning that it is a false positive and should be skipped.
78                   # There is a very slim chance that a genuine match overlaps this one and, because re.finditer() does not
79                   # allow overlapping matches, it would be lost. To avoid that, restart the regex scan, starting at the
80                   # next even byte.
81                   matches = pattern.finditer(string, match.start() + 1)
82                   break
83           else:
84               break

87   # Opcodes involved in function calls with constant arguments. The differences between python versions are handled by
88   # variables below, which are then used to construct the _call_function_bytecode regex.
89   # NOTE1: the _OPCODES_* entries are typically used in (non-capturing) groups that match the opcode plus an arbitrary
90   # argument. But because the entries themselves may contain more than on opcode (with OR operator between them), they
91   # themselves need to be enclosed in another (non-capturing) group. E.g., "(?:(?:_OPCODES_FUNCTION_GLOBAL).)".
92   # NOTE2: _OPCODES_EXTENDED_ARG2 is an exception, as it is used as a list of opcodes to exclude, i.e.,
93   # "[^_OPCODES_EXTENDED_ARG2]". Therefore, multiple opcodes are not separated by the OR operator.
94   if not compat.is_py311:
95       # Python 3.7 introduced two new function-related opcodes, LOAD_METHOD and CALL_METHOD
96       _OPCODES_EXTENDED_ARG = rb"`EXTENDED_ARG`"
97       _OPCODES_EXTENDED_ARG2 = _OPCODES_EXTENDED_ARG
98       _OPCODES_FUNCTION_GLOBAL = rb"`LOAD_NAME`|`LOAD_GLOBAL`|`LOAD_FAST`"
99       _OPCODES_FUNCTION_LOAD = rb"`LOAD_ATTR`|`LOAD_METHOD`"
100      _OPCODES_FUNCTION_ARGS = rb"`LOAD_CONST`"
101      _OPCODES_FUNCTION_CALL = rb"`CALL_FUNCTION`|`CALL_METHOD`|`CALL_FUNCTION_EX`"
102  
103      def _cleanup_bytecode_string(bytecode):
104          return bytecode  # Nothing to do here
105  elif not compat.is_py312:
106      # Python 3.11 removed CALL_FUNCTION and CALL_METHOD, and replaced them with PRECALL + CALL instruction sequence.
107      # As both PRECALL and CALL have the same parameter (the argument count), we need to match only up to the PRECALL.
108      # The CALL_FUNCTION_EX is still present.
109      # From Python 3.11b1 on, there is an EXTENDED_ARG_QUICK specialization opcode present.
110      _OPCODES_EXTENDED_ARG = rb"`EXTENDED_ARG`|`EXTENDED_ARG_QUICK`"
111      _OPCODES_EXTENDED_ARG2 = rb"`EXTENDED_ARG``EXTENDED_ARG_QUICK`"  # Special case; see note above the if/else block!
112      _OPCODES_FUNCTION_GLOBAL = rb"`LOAD_NAME`|`LOAD_GLOBAL`|`LOAD_FAST`"
113      _OPCODES_FUNCTION_LOAD = rb"`LOAD_ATTR`|`LOAD_METHOD`"
114      _OPCODES_FUNCTION_ARGS = rb"`LOAD_CONST`"
115      _OPCODES_FUNCTION_CALL = rb"`PRECALL`|`CALL_FUNCTION_EX`"
116  
117      # Starting with python 3.11, the bytecode is peppered with CACHE instructions (which dis module conveniently hides
118      # unless show_caches=True is used). Dealing with these CACHE instructions in regex rules is going to render them
119      # unreadable, so instead we pre-process the bytecode and filter the offending opcodes out.
120      _cache_instruction_filter = bytecode_regex(rb"(`CACHE`.)|(..)")
121  
122      def _cleanup_bytecode_string(bytecode):
123          return _cache_instruction_filter.sub(rb"\2", bytecode)
124  
125  else:
126      # Python 3.12 merged EXTENDED_ARG_QUICK back in to EXTENDED_ARG, and LOAD_METHOD in to LOAD_ATTR
127      # PRECALL is no longer a valid key
128      _OPCODES_EXTENDED_ARG = rb"`EXTENDED_ARG`"
129      _OPCODES_EXTENDED_ARG2 = _OPCODES_EXTENDED_ARG
130      _OPCODES_FUNCTION_GLOBAL = rb"`LOAD_NAME`|`LOAD_GLOBAL`|`LOAD_FAST`"
131      _OPCODES_FUNCTION_LOAD = rb"`LOAD_ATTR`"
132      _OPCODES_FUNCTION_ARGS = rb"`LOAD_CONST`"
133      _OPCODES_FUNCTION_CALL = rb"`CALL`|`CALL_FUNCTION_EX`"
134  
135      _cache_instruction_filter = bytecode_regex(rb"(`CACHE`.)|(..)")
136  
137      def _cleanup_bytecode_string(bytecode):
138          return _cache_instruction_filter.sub(rb"\2", bytecode)

141  # language=PythonVerboseRegExp
142  _call_function_bytecode = bytecode_regex(
143      rb""" 
144      # Matches `global_function('some', 'constant', 'arguments')`. 
145   
146      # Load the global function. In code with >256 of names, this may require extended name references. 
147      ( 
148       (?:(?:""" + _OPCODES_EXTENDED_ARG + rb""").)* 
149       (?:(?:""" + _OPCODES_FUNCTION_GLOBAL + rb""").) 
150      ) 
151   
152      # For foo.bar.whizz(), the above is the 'foo', below is the 'bar.whizz' (one opcode per name component, each 
153      # possibly preceded by name reference extension). 
154      ( 
155       (?: 
156         (?:(?:""" + _OPCODES_EXTENDED_ARG + rb""").)* 
157         (?:""" + _OPCODES_FUNCTION_LOAD + rb"""). 
158       )* 
159      ) 
160   
161      # Load however many arguments it takes. These (for now) must all be constants. 
162      # Again, code with >256 constants may need extended enumeration. 
163      ( 
164        (?: 
165          (?:(?:""" + _OPCODES_EXTENDED_ARG + rb""").)* 
166          (?:""" + _OPCODES_FUNCTION_ARGS + rb"""). 
167        )* 
168      ) 
169   
170      # Call the function. If opcode is CALL_FUNCTION_EX, the parameter are flags. For other opcodes, the parameter 
171      # is the argument count (which may be > 256). 
172      ( 
173        (?:(?:""" + _OPCODES_EXTENDED_ARG + rb""").)* 
174        (?:""" + _OPCODES_FUNCTION_CALL + rb"""). 
175      ) 
176  """
177  )

180  _extended_arg_bytecode = bytecode_regex(
181      rb"""( 
182      # Arbitrary number of EXTENDED_ARG pairs. 
183      (?:(?:""" + _OPCODES_EXTENDED_ARG + rb""").)* 
184   
185      # Followed by some other instruction (usually a LOAD). 
186      [^""" + _OPCODES_EXTENDED_ARG2 + rb"""]. 
187  )"""
188  )

191  def extended_arguments(extended_args: bytes):
192      """ 
193      Unpack the (extended) integer used to reference names or constants. 
194   
195      The input should be a bytecode snippet of the following form:: 
196   
197          EXTENDED_ARG    ?      # Repeated 0-4 times. 
198          LOAD_xxx        ?      # Any of LOAD_NAME/LOAD_CONST/LOAD_METHOD/... 
199   
200      Each ? byte combined together gives the number we want. 
201      """
202      return int.from_bytes(extended_args[1::2], "big")

205  def load(raw: bytes, code: CodeType) -> str:
206      """ 
207      Parse an (extended) LOAD_xxx instruction. 
208      """
209      # Get the enumeration.
210      index = extended_arguments(raw)
211  
212      # Work out what that enumeration was for (constant/local var/global var).
213  
214      # If the last instruction byte is a LOAD_FAST:
215      if raw[-2] == opmap["LOAD_FAST"]:
216          # Then this is a local variable.
217          return code.co_varnames[index]
218      # Or if it is a LOAD_CONST:
219      if raw[-2] == opmap["LOAD_CONST"]:
220          # Then this is a literal.
221          return code.co_consts[index]
222      # Otherwise, it is a global name.
223      if compat.is_py311 and raw[-2] == opmap["LOAD_GLOBAL"]:
224          # In python 3.11, namei>>1 is pushed on stack...
225          return code.co_names[index >> 1]
226      if compat.is_py312 and raw[-2] == opmap["LOAD_ATTR"]:
227          # In python 3.12, namei>>1 is pushed on stack...
228          return code.co_names[index >> 1]
229  
230      return code.co_names[index]

233  def loads(raw: bytes, code: CodeType) -> list:
234      """ 
235      Parse multiple consecutive LOAD_xxx instructions. Or load() in a for loop. 
236   
237      May be used to unpack a function's parameters or nested attributes ``(foo.bar.pop.whack)``. 
238      """
239      return [load(i, code) for i in _extended_arg_bytecode.findall(raw)]

242  def function_calls(code: CodeType) -> list:
243      """ 
244      Scan a code object for all function calls on constant arguments. 
245      """
246      match: re.Match
247      out = []
248  
249      for match in finditer(_call_function_bytecode, code.co_code):
250          function_root, methods, args, function_call = match.groups()
251  
252          # For foo():
253          #   `function_root` contains 'foo' and `methods` is empty.
254          # For foo.bar.whizz():
255          #   `function_root` contains 'foo' and `methods` contains the rest.
256          function_root = load(function_root, code)
257          methods = loads(methods, code)
258          function = ".".join([function_root] + methods)
259  
260          args = loads(args, code)
261          if function_call[0] == opmap['CALL_FUNCTION_EX']:
262              flags = extended_arguments(function_call)
263              if flags != 0:
264                  # Keyword arguments present. Unhandled at the moment.
265                  continue
266              # In calls with const arguments, args contains a single
267              # tuple with all values.
268              if len(args) != 1 or not isinstance(args[0], tuple):
269                  continue
270              args = list(args[0])
271          else:
272              arg_count = extended_arguments(function_call)
273  
274              if arg_count != len(args):
275                  # This happens if there are variable or keyword arguments. Bail out in either case.
276                  continue
277  
278          out.append((function, args))
279  
280      return out

283  def search_recursively(search: callable, code: CodeType, _memo=None) -> dict:
284      """ 
285      Apply a search function to a code object, recursing into child code objects (function definitions). 
286      """
287      if _memo is None:
288          _memo = {}
289      if code not in _memo:
290          _memo[code] = search(code)
291          for const in code.co_consts:
292              if isinstance(const, CodeType):
293                  search_recursively(search, const, _memo)
294      return _memo
295  
296  
297  def recursive_function_calls(code: CodeType) -> dict:
298      """ 
299      Scan a code object for function calls on constant arguments, recursing into function definitions and bodies of 
300      comprehension loops. 
301      """
302      return search_recursively(function_calls, code)

305  def any_alias(full_name: str):
306      """List possible aliases of a fully qualified Python name. 
307   
308          >>> list(any_alias("foo.bar.wizz")) 
309          ['foo.bar.wizz', 'bar.wizz', 'wizz'] 
310   
311      This crudely allows us to capture uses of wizz() under any of 
312      :: 
313          import foo 
314          foo.bar.wizz() 
315      :: 
316          from foo import bar 
317          bar.wizz() 
318      :: 
319          from foo.bar import wizz 
320          wizz() 
321   
322      However, it will fail for any form of aliases and quite likely find false matches. 
323      """
324      parts = full_name.split('.')
325      while parts:
326          yield ".".join(parts)
327          parts = parts[1:]
328
