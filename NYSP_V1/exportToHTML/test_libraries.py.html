32   @importorskip('fiona')
33   def test_fiona_transform(pyi_builder):
34       # Test that fiona in frozen application has access to its projections database. If projection data is unavailable,
35       # the transform becomes an identity transform.
36       pyi_builder.test_source(
37           """ 
38           from fiona.transform import transform_geom 
39           from fiona.crs import from_epsg 
40    
41           eiffel_tower = { 
42               'type': 'Point', 
43               'coordinates': (2.294694, 48.858093), 
44           } 
45    
46           crs_source = from_epsg(4326)  # WGS84 
47           crs_target = from_epsg(25831)  # ETRS89 / UTM zone 31N 
48    
49           transformed = transform_geom(crs_source, crs_target, eiffel_tower) 
50           print(f"Transformed point: {transformed}") 
51    
52           # Expected coordinates: obtained by manually running this program unfrozen 
53           EXPECTED_COORDINATES = (448265.9146792292, 5411920.651338793) 
54           EPS = 1e-6 
55    
56           delta = [abs(value - expected) for value, expected in zip(transformed["coordinates"], EXPECTED_COORDINATES)] 
57           print(f"Delta: {delta}") 
58           assert all([value < EPS for value in delta]), f"Delta {delta} exceeds threshold!" 
59           """
60       )

1037             has_gui = False
1038     elif backend == "NONE":
