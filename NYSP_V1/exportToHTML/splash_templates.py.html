11   """ 
12   Templates for the splash screen tcl script. 
13   """
14   from PyInstaller.compat import is_cygwin, is_darwin, is_win
15   
16   ipc_script = r""" 
17   proc _ipc_server {channel clientaddr clientport} { 
18       # This function is called if a new client connects to 
19       # the server. This creates a channel, which calls 
20       # _ipc_caller if data was send through the connection 
21       set client_name [format <%s:%d> $clientaddr $clientport] 
22    
23       chan configure $channel \ 
24           -buffering none \ 
25           -encoding utf-8 \ 
26           -eofchar \x04 \ 
27           -translation cr 
28       chan event $channel readable [list _ipc_caller $channel $client_name] 
29   } 
30    
31   proc _ipc_caller {channel client_name} { 
32       # This function is called if a command was sent through 
33       # the tcp connection. The current implementation supports 
34       # two commands: update_text and exit, although exit 
35       # is implemented to be called if the connection gets 
36       # closed (from python) or the character 0x04 was received 
37       chan gets $channel cmd 
38    
39       if {[chan eof $channel]} { 
40           # This is entered if either the connection was closed 
41           # or the char 0x04 was send 
42           chan close $channel 
43           exit 
44    
45       } elseif {![chan blocked $channel]} { 
46           # RPC methods 
47    
48           # update_text command 
49           if {[string match "update_text*" $cmd]} { 
50               global status_text 
51               set first [expr {[string first "(" $cmd] + 1}] 
52               set last [expr {[string last ")" $cmd] - 1}] 
53    
54               set status_text [string range $cmd $first $last] 
55           } 
56           # Implement other procedures here 
57       } 
58   } 
59    
60   # By setting the port to 0 the os will assign a free port 
61   set server_socket [socket -server _ipc_server -myaddr localhost 0] 
62   set server_port [fconfigure $server_socket -sockname] 
63    
64   # This environment variable is shared between the python and the tcl 
65   # interpreter and publishes the port the tcp server socket is available 
66   set env(_PYIBoot_SPLASH) [lindex $server_port 2] 
67   """
68   
69   image_script = r""" 
70   # The variable $_image_data, which holds the data for the splash 
71   # image is created by the bootloader. 
72   image create photo splash_image 
73   splash_image put $_image_data 
74   # delete the variable, because the image now holds the data 
75   unset _image_data 
76    
77   proc canvas_text_update {canvas tag _var - -}  { 
78       # This function is rigged to be called if the a variable 
79       # status_text gets changed. This updates the text on 
80       # the canvas 
81       upvar $_var var 
82       $canvas itemconfigure $tag -text $var 
83   } 
84   """
85   
86   splash_canvas_setup = r""" 
87   package require Tk 
88    
89   set image_width [image width splash_image] 
90   set image_height [image height splash_image] 
91   set display_width [winfo screenwidth .] 
92   set display_height [winfo screenheight .] 
93    
94   set x_position [expr {int(0.5*($display_width - $image_width))}] 
95   set y_position [expr {int(0.5*($display_height - $image_height))}] 
96    
97   # Toplevel frame in which all widgets should be positioned 
98   frame .root 
99    
100  # Configure the canvas on which the splash 
101  # screen will be drawn 
102  canvas .root.canvas \ 
103      -width $image_width \ 
104      -height $image_height \ 
105      -borderwidth 0 \ 
106      -highlightthickness 0 
107   
108  # Draw the image into the canvas, filling it. 
109  .root.canvas create image \ 
110      [expr {$image_width / 2}] \ 
111      [expr {$image_height / 2}] \ 
112      -image splash_image 
113  """
114  
115  splash_canvas_text = r""" 
116  # Create a text on the canvas, which tracks the local 
117  # variable status_text. status_text is changed via C to 
118  # update the progress on the splash screen. 
119  # We cannot use the default label, because it has a 
120  # default background, which cannot be turned transparent 
121  .root.canvas create text \ 
122          %(pad_x)d \ 
123          %(pad_y)d \ 
124          -fill %(color)s \ 
125          -justify center \ 
126          -font myFont \ 
127          -tag vartext \ 
128          -anchor sw 
129  trace variable status_text w \ 
130      [list canvas_text_update .root.canvas vartext] 
131  set status_text "%(default_text)s" 
132  """
133  
134  splash_canvas_default_font = r""" 
135  font create myFont {*}[font actual TkDefaultFont] 
136  font configure myFont -size %(font_size)d 
137  """
138  
139  splash_canvas_custom_font = r""" 
140  font create myFont -family %(font)s -size %(font_size)d 
141  """
142  
143  if is_win or is_cygwin:
144      transparent_setup = r""" 
145  # If the image is transparent, the background will be filled 
146  # with magenta. The magenta background is later replaced with transparency. 
147  # Here is the limitation of this implementation, that only 
148  # sharp transparent image corners are possible 
149  wm attributes . -transparentcolor magenta 
150  .root.canvas configure -background magenta 
151  """
152  
153  elif is_darwin:
154      # This is untested, but should work following: https://stackoverflow.com/a/44296157/5869139
155      transparent_setup = r""" 
156  wm attributes . -transparent 1 
157  . configure -background systemTransparent 
158  .root.canvas configure -background systemTransparent 
159  """
160  
161  else:
162      # For Linux there is no common way to create a transparent window
163      transparent_setup = r""
164  
165  pack_widgets = r""" 
166  # Position all widgets in the window 
167  pack .root 
168  grid .root.canvas   -column 0 -row 0 -columnspan 1 -rowspan 2 
169  """
170  
171  # Enable always-on-top behavior, by setting overrideredirect and the topmost attribute.
172  position_window_on_top = r""" 
173  # Set position and mode of the window - always-on-top behavior 
174  wm overrideredirect . 1 
175  wm geometry         . +${x_position}+${y_position} 
176  wm attributes       . -topmost 1 
177  """
178  
179  # Disable always-on-top behavior
180  if is_win or is_cygwin or is_darwin:
181      # On Windows, we disable the always-on-top behavior while still setting overrideredirect
182      # (to disable window decorations), but set topmost attribute to 0.
183      position_window = r""" 
184  # Set position and mode of the window 
185  wm overrideredirect . 1 
186  wm geometry         . +${x_position}+${y_position} 
187  wm attributes       . -topmost 0 
188  """
189  else:
190      # On Linux, we must not use overrideredirect; instead, we set X11-specific type attribute to splash,
191      # which lets the window manager to properly handle the splash screen (without window decorations
192      # but allowing other windows to be brought to front).
193      position_window = r""" 
194  # Set position and mode of the window 
195  wm geometry         . +${x_position}+${y_position} 
196  wm attributes       . -type splash 
197  """
198  
199  raise_window = r""" 
200  raise . 
201  """
202  
203  
204  def build_script(text_options=None, always_on_top=False):
205      """ 
206      This function builds the tcl script for the splash screen. 
207      """
208      # Order is important!
209      script = [
210          ipc_script,
211          image_script,
212          splash_canvas_setup,
213      ]
214  
215      if text_options:
216          # If the default font is used we need a different syntax
217          if text_options['font'] == "TkDefaultFont":
218              script.append(splash_canvas_default_font % text_options)
219          else:
220              script.append(splash_canvas_custom_font % text_options)
221          script.append(splash_canvas_text % text_options)
222  
223      script.append(transparent_setup)
224  
225      script.append(pack_widgets)
226      script.append(position_window_on_top if always_on_top else position_window)
227      script.append(raise_window)
228  
229      return '\n'.join(script)
