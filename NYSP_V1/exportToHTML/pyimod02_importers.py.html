47   def _decode_source(source_bytes):
48       """ 
49       Decode bytes representing source code and return the string. Universal newline support is used in the decoding. 
50       Based on CPython's implementation of the same functionality: 
51       https://github.com/python/cpython/blob/3.9/Lib/importlib/_bootstrap_external.py#L679-L688 
52       """
53       # Local import to avoid including `tokenize` and its dependencies in `base_library.zip`
54       from tokenize import detect_encoding
55       source_bytes_readline = io.BytesIO(source_bytes).readline
56       encoding = detect_encoding(source_bytes_readline)
57       newline_decoder = io.IncrementalNewlineDecoder(decoder=None, translate=True)
58       return newline_decoder.decode(source_bytes.decode(encoding[0]))

72   class PyiFrozenImporter:
73       """ 
74       Load bytecode of Python modules from the executable created by PyInstaller. 
75    
76       Python bytecode is zipped and appended to the executable. 
77    
78       NOTE: PYZ format cannot be replaced by zipimport module. 
79    
80       The problem is that we have no control over zipimport; for instance, it does not work if the zip file is embedded 
81       into a PKG that is appended to an executable, like we create in one-file mode. 
82    
83       This used to be PEP-302 finder and loader class for the ``sys.meta_path`` hook. A PEP-302 finder requires method 
84       find_module() to return loader class with method load_module(). However, both of these methods were deprecated in 
85       python 3.4 by PEP-451 (see below). Therefore, this class now provides only optional extensions to the PEP-302 
86       importer protocol. 
87    
88       This is also a PEP-451 finder and loader class for the ModuleSpec type import system. A PEP-451 finder requires 
89       method find_spec(), a PEP-451 loader requires methods exec_module(), load_module() and (optionally) create_module(). 
90       All these methods are implemented in this one class. 
91       """
92       def __init__(self):
93           """ 
94           Load, unzip and initialize the Zip archive bundled with the executable. 
95           """
96           # Examine all items in sys.path and the one like /path/executable_name?117568 is the correct executable with
97           # the bundled zip archive. Use this value for the ZlibArchiveReader class, and remove this item from sys.path.
98           # It was needed only for PyiFrozenImporter class. Wrong path from sys.path raises an ArchiveReadError exception.
99           for pyz_filepath in sys.path:
100              try:
101                  # Unzip zip archive bundled with the executable.
102                  self._pyz_archive = ZlibArchiveReader(pyz_filepath, check_pymagic=True)
103  
104                  # As no exception was raised, we can assume that ZlibArchiveReader was successfully loaded.
105                  # Let's remove 'pyz_filepath' from sys.path.
106                  trace("# PyInstaller: PyiFrozenImporter(%s)", pyz_filepath)
107                  sys.path.remove(pyz_filepath)
108                  break
109              except IOError:
110                  # Item from sys.path is not ZlibArchiveReader; let's try next one.
111                  continue
112              except ArchiveReadError:
113                  # Item from sys.path is not ZlibArchiveReader; let's try next one.
114                  continue
115          else:
116              # sys.path does not contain the filename of the executable with the bundled zip archive. Raise import error.
117              raise ImportError("Cannot load frozen modules.")
118  
119          # Some runtime hooks might need access to the list of available frozen modules. Make them accessible as a set().
120          self.toc = set(self._pyz_archive.toc.keys())
121  
122          # Some runtime hooks might need to traverse available frozen package/module hierarchy to simulate filesystem.
123          # Such traversals can be efficiently implemented using a prefix tree (trie), whose computation we defer
124          # until first access.
125          self._lock = _thread.RLock()
126          self._toc_tree = None
127  
128      @property
129      def toc_tree(self):
130          with self._lock:
131              if self._toc_tree is None:
132                  self._toc_tree = self._build_pyz_prefix_tree()
133              return self._toc_tree
134  
135      # Helper for computing PYZ prefix tree
136      def _build_pyz_prefix_tree(self):
137          tree = dict()
138          for entry_name in self.toc:
139              name_components = entry_name.split('.')
140              current = tree
141              if self._pyz_archive.is_package(entry_name):  # self.is_package() without unnecessary checks
142                  # Package; create new dictionary node for its modules
143                  for name_component in name_components:
144                      current = current.setdefault(name_component, {})
145              else:
146                  # Module; create the leaf node (empty string)
147                  for name_component in name_components[:-1]:
148                      current = current.setdefault(name_component, {})
149                  current[name_components[-1]] = ''
150          return tree

154          if fullname in self.toc:
155              try:
156                  return self._pyz_archive.is_pep420_namespace_package(fullname)
157              except Exception as e:
158                  raise ImportError(f'PyiFrozenImporter cannot handle module {fullname!r}') from e

173      def get_code(self, fullname):
174          """ 
175          Get the code object associated with the module. 
176   
177          ImportError should be raised if module not found. 
178          """
179          try:
180              if fullname == '__main__':
181                  # Special handling for __main__ module; the bootloader should store code object to _pyi_main_co
182                  # attribute of the module.
183                  return sys.modules['__main__']._pyi_main_co
184  
185              # extract() returns None if fullname is not in the archive, and the subsequent subscription attempt raises
186              # exception, which is turned into ImportError.
187              return self._pyz_archive.extract(fullname)
188          except Exception as e:
189              raise ImportError(f'PyiFrozenImporter cannot handle module {fullname!r}') from e

191      def get_source(self, fullname):
192          """ 
193          Method should return the source code for the module as a string. 
194          But frozen modules does not contain source code. 
195   
196          Return None, unless the corresponding source file was explicitly collected to the filesystem. 
197          """
198          if fullname in self.toc:
199              # Try loading the .py file from the filesystem (only for collected modules)
200              if self.is_package(fullname):
201                  fullname += '.__init__'
202              filename = os.path.join(SYS_PREFIX, fullname.replace('.', os.sep) + '.py')
203              try:
204                  # Read in binary mode, then decode
205                  with open(filename, 'rb') as fp:
206                      source_bytes = fp.read()
207                  return _decode_source(source_bytes)
208              except FileNotFoundError:
209                  pass
210              return None
211          else:
212              # ImportError should be raised if module not found.
213              raise ImportError('No module named ' + fullname)

215      def get_data(self, path):
216          """ 
217          Returns the data as a string, or raises IOError if the file was not found. The data is always returned as if 
218          "binary" mode was used. 
219   
220          The 'path' argument is a path that can be constructed by munging module.__file__ (or pkg.__path__ items). 
221   
222          This assumes that the file in question was collected into frozen application bundle as a file, and is available 
223          on the filesystem. Older versions of PyInstaller also supported data embedded in the PYZ archive, but that has 
224          been deprecated in v6. 
225          """
226          # Try to fetch the data from the filesystem. Since __file__ attribute works properly, just try to open the file
227          # and read it.
228          with open(path, 'rb') as fp:
229              return fp.read()

231      def get_filename(self, fullname):
232          """ 
233          This method should return the value that __file__ would be set to if the named module was loaded. If the module 
234          is not found, an ImportError should be raised. 
235          """
236          # The absolute absolute path to the executable is taken from sys.prefix. In onefile mode it points to the temp
237          # directory where files are unpacked by PyInstaller. Then, append the appropriate suffix (__init__.pyc for a
238          # package, or just .pyc for a module).
239          # Method is_package() will raise ImportError if module not found.
240          if self.is_package(fullname):
241              filename = os.path.join(SYS_PREFIX, fullname.replace('.', os.path.sep), '__init__.pyc')
242          else:
243              filename = os.path.join(SYS_PREFIX, fullname.replace('.', os.path.sep) + '.pyc')
244          return filename

246      def find_spec(self, fullname, path=None, target=None):
247          """ 
248          PEP-451 finder.find_spec() method for the ``sys.meta_path`` hook. 
249   
250          fullname     fully qualified name of the module 
251          path         None for a top-level module, or package.__path__ for 
252                       submodules or subpackages. 
253          target       unused by this Finder 
254   
255          Finders are still responsible for identifying, and typically creating, the loader that should be used to load a 
256          module. That loader will now be stored in the module spec returned by find_spec() rather than returned directly. 
257          As is currently the case without the PEP-452, if a loader would be costly to create, that loader can be designed 
258          to defer the cost until later. 
259   
260          Finders must return ModuleSpec objects when find_spec() is called. This new method replaces find_module() and 
261          find_loader() (in the PathEntryFinder case). If a loader does not have find_spec(), find_module() and 
262          find_loader() are used instead, for backward-compatibility. 
263          """
264          entry_name = None  # None means - no module found in this importer.
265  
266          # Try to handle module.__path__ modifications by the modules themselves. This needs to be done first in
267          # order to support module overrides in alternative locations while we also have the original module
268          # available at non-override location.
269          if path is not None:
270              # Reverse the fake __path__ we added to the package module into a dotted module name, and add the tail
271              # module from fullname onto that to synthesize a new fullname.
272              modname = fullname.rsplit('.')[-1]
273  
274              for p in path:
275                  if not p.startswith(SYS_PREFIX):
276                      continue
277                  p = p[SYS_PREFIXLEN:]
278                  parts = p.split(os.sep)
279                  if not parts:
280                      continue
281                  if not parts[0]:
282                      parts = parts[1:]
283                  parts.append(modname)
284                  entry_name = ".".join(parts)
285                  if entry_name in self.toc:
286                      trace("import %s as %s # PyInstaller PYZ (__path__ override: %s)", entry_name, fullname, p)
287                      break
288              else:
289                  entry_name = None
290  
291          if entry_name is None:
292              # Either there was no path override, or the module was not available there. Check the fully qualified name
293              # of the module directly.
294              if fullname in self.toc:
295                  entry_name = fullname
296                  trace("import %s # PyInstaller PYZ", fullname)
297  
298          if entry_name is None:
299              trace("# %s not found in PYZ", fullname)
300              return None
301  
302          if self._is_pep420_namespace_package(entry_name):
303              from importlib._bootstrap_external import _NamespacePath
304              # PEP-420 namespace package; as per PEP 451, we need to return a spec with "loader" set to None
305              # (a.k.a. not set)
306              spec = _frozen_importlib.ModuleSpec(fullname, None, is_package=True)
307              # Set submodule_search_locations, which seems to fill the __path__ attribute.
308              # This needs to be an instance of `importlib._bootstrap_external._NamespacePath` for `importlib.resources`
309              # to work correctly with the namespace package; otherwise `importlib.resources.files()` throws an
310              # `ValueError('Invalid path')` due to this check:
311              # https://github.com/python/cpython/blob/v3.11.5/Lib/importlib/resources/readers.py#L109-L110
312              spec.submodule_search_locations = _NamespacePath(
313                  entry_name,
314                  [os.path.dirname(self.get_filename(entry_name))],
315                  # The `path_finder` argument must be a callable with two arguments (`name` and `path`) that
316                  # returns the spec - so we need to bind our `find_spec` via lambda.
317                  lambda name, path: self.find_spec(name, path),
318              )
319              return spec
320  
321          # origin has to be the filename
322          origin = self.get_filename(entry_name)
323          is_pkg = self.is_package(entry_name)
324  
325          spec = _frozen_importlib.ModuleSpec(
326              fullname,
327              self,
328              is_package=is_pkg,
329              origin=origin,
330              # Provide the entry_name (name of module entry in the PYZ) for the loader to use during loading.
331              loader_state=PyiFrozenImporterState(entry_name)
332          )
333  
334          # Make the import machinery set __file__.
335          # PEP 451 says: "has_location" is true if the module is locatable. In that case the spec's origin is used
336          # as the location and __file__ is set to spec.origin. If additional location information is required
337          # (e.g., zipimport), that information may be stored in spec.loader_state.
338          spec.has_location = True
339  
340          # Set submodule_search_locations for packages. Seems to be required for importlib_resources from 3.2.0;
341          # see issue #5395.
342          if is_pkg:
343              spec.submodule_search_locations = [os.path.dirname(self.get_filename(entry_name))]
344  
345          return spec

347      def create_module(self, spec):
348          """ 
349          PEP-451 loader.create_module() method for the ``sys.meta_path`` hook. 
350   
351          Loaders may also implement create_module() that will return a new module to exec. It may return None to indicate 
352          that the default module creation code should be used. One use case, though atypical, for create_module() is to 
353          provide a module that is a subclass of the builtin module type. Most loaders will not need to implement 
354          create_module(). 
355   
356          create_module() should properly handle the case where it is called more than once for the same spec/module. This 
357          may include returning None or raising ImportError. 
358          """
359          # Contrary to what is defined in PEP-451, this method is not optional. We want the default results, so we simply
360          # return None (which is handled for su my the import machinery).
361          # See https://bugs.python.org/issue23014 for more information.
362          return None

364      def exec_module(self, module):
365          """ 
366          PEP-451 loader.exec_module() method for the ``sys.meta_path`` hook. 
367   
368          Loaders will have a new method, exec_module(). Its only job is to "exec" the module and consequently populate 
369          the module's namespace. It is not responsible for creating or preparing the module object, nor for any cleanup 
370          afterward. It has no return value. exec_module() will be used during both loading and reloading. 
371   
372          exec_module() should properly handle the case where it is called more than once. For some kinds of modules this 
373          may mean raising ImportError every time after the first time the method is called. This is particularly relevant 
374          for reloading, where some kinds of modules do not support in-place reloading. 
375          """
376          spec = module.__spec__
377  
378          if isinstance(spec.loader_state, PyiFrozenImporterState):
379              # Use the module name stored in the `loader_state`, which was set by our `find_spec()` implementation.
380              # This is necessary to properly resolve aliased modules; for example, `module.__spec__.name` contains
381              # `pkg_resources.extern.jaraco.text`, but the original name stored in `loader_state`, which we need
382              # to use for code look-up, is `pkg_resources._vendor.jaraco.text`.
383              module_name = spec.loader_state.pyz_entry_name
384          elif isinstance(spec.loader_state, dict):
385              # This seems to happen when `importlib.util.LazyLoader` is used, and our original `loader_state` is lost.
386              # We could use `spec.name` and hope for the best, but that will likely fail with aliased modules (see
387              # the comment in the branch above for an example).
388              #
389              # So try to reconstruct the original module name from the `origin` - which is essentially the reverse of
390              # our `get_filename()` implementation.
391              assert spec.origin.startswith(SYS_PREFIX)
392              module_name = spec.origin[SYS_PREFIXLEN:].replace(os.sep, '.')
393              if module_name.endswith('.pyc'):
394                  module_name = module_name[:-4]
395              if module_name.endswith('.__init__'):
396                  module_name = module_name[:-9]
397          else:
398              raise RuntimeError(f"Module's spec contains loader_state of incompatible type: {type(spec.loader_state)}")
399  
400          bytecode = self.get_code(module_name)
401          if bytecode is None:
402              raise RuntimeError(f"Failed to retrieve bytecode for {spec.name!r}!")
403  
404          # Set by the import machinery
405          assert hasattr(module, '__file__')
406  
407          # If `submodule_search_locations` is not None, this is a package; set __path__.
408          if spec.submodule_search_locations is not None:
409              # Since PYTHONHOME is set in bootloader, 'sys.prefix' points to the correct path where PyInstaller should
410              # find bundled dynamic libraries. In one-file mode it points to the tmp directory where bundled files are
411              # extracted at execution time.
412              #
413              # __path__ cannot be empty list because 'wx' module prepends something to it. It cannot contain value
414              # 'sys.prefix' because 'xml.etree.cElementTree' fails otherwise.
415              #
416              # Set __path__ to point to 'sys.prefix/package/subpackage'.
417              module.__path__ = [os.path.dirname(module.__file__)]
418  
419          exec(bytecode, module.__dict__)

428  class PyiFrozenResourceReader:
429      """ 
430      Resource reader for importlib.resources / importlib_resources support. 
431   
432      Supports only on-disk resources, which should cover the typical use cases, i.e., the access to data files; 
433      PyInstaller collects data files onto filesystem, and as of v6.0.0, the embedded PYZ archive is guaranteed 
434      to contain only .pyc modules. 
435   
436      When listing resources, source .py files will not be listed as they are not collected by default. Similarly, 
437      sub-directories that contained only .py files are not reconstructed on filesystem, so they will not be listed, 
438      either. If access to .py files is required for whatever reason, they need to be explicitly collected as data files 
439      anyway, which will place them on filesystem and make them appear as resources. 
440   
441      For on-disk resources, we *must* return path compatible with pathlib.Path() in order to avoid copy to a temporary 
442      file, which might break under some circumstances, e.g., metpy with importlib_resources back-port, due to: 
443      https://github.com/Unidata/MetPy/blob/a3424de66a44bf3a92b0dcacf4dff82ad7b86712/src/metpy/plots/wx_symbols.py#L24-L25 
444      (importlib_resources tries to use 'fonts/wx_symbols.ttf' as a temporary filename suffix, which fails as it contains 
445      a separator). 
446   
447      Furthermore, some packages expect files() to return either pathlib.Path or zipfile.Path, e.g., 
448      https://github.com/tensorflow/datasets/blob/master/tensorflow_datasets/core/utils/resource_utils.py#L81-L97 
449      This makes implementation of mixed support for on-disk and embedded resources using importlib.abc.Traversable 
450      protocol rather difficult. 
451   
452      So in order to maximize compatibility with unfrozen behavior, the below implementation is basically equivalent of 
453      importlib.readers.FileReader from python 3.10: 
454        https://github.com/python/cpython/blob/839d7893943782ee803536a47f1d4de160314f85/Lib/importlib/readers.py#L11 
455      and its underlying classes, importlib.abc.TraversableResources and importlib.abc.ResourceReader: 
456        https://github.com/python/cpython/blob/839d7893943782ee803536a47f1d4de160314f85/Lib/importlib/abc.py#L422 
457        https://github.com/python/cpython/blob/839d7893943782ee803536a47f1d4de160314f85/Lib/importlib/abc.py#L312 
458      """
459      def __init__(self, importer, name):
460          # Local import to avoid including `pathlib` and its dependencies in `base_library.zip`
461          from pathlib import Path
462          self.importer = importer
463          self.path = Path(sys._MEIPASS).joinpath(*name.split('.'))
464  
465      def open_resource(self, resource):
466          return self.files().joinpath(resource).open('rb')
467  
468      def resource_path(self, resource):
469          return str(self.path.joinpath(resource))
470  
471      def is_resource(self, path):
472          return self.files().joinpath(path).is_file()
473  
474      def contents(self):
475          return (item.name for item in self.files().iterdir())
476  
477      def files(self):
478          return self.path

481  def install():
482      """ 
483      Install PyiFrozenImporter class into the import machinery. 
484   
485      This function installs the PyiFrozenImporter class into the import machinery of the running process. The importer 
486      is added to sys.meta_path. It could be added to sys.path_hooks, but sys.meta_path is processed by Python before 
487      looking at sys.path! 
488   
489      The order of processing import hooks in sys.meta_path: 
490   
491      1. built-in modules 
492      2. modules from the bundled ZIP archive 
493      3. C extension modules 
494      4. Modules from sys.path 
495      """
496      # Ensure Python looks in the bundled zip archive for modules before any other places.
497      importer = PyiFrozenImporter()
498      sys.meta_path.append(importer)
499  
500      # On Windows there is importer _frozen_importlib.WindowsRegistryFinder that looks for Python modules in Windows
501      # registry. The frozen executable should not look for anything in the Windows registry. Remove this importer
502      # from sys.meta_path.
503      for item in sys.meta_path:
504          if hasattr(item, '__name__') and item.__name__ == 'WindowsRegistryFinder':
505              sys.meta_path.remove(item)
506              break
507      # _frozen_importlib.PathFinder is also able to handle Python C extensions. However, PyInstaller needs its own
508      # importer as it uses extension names like 'module.submodle.so' (instead of paths). As of Python 3.7.0b2, there
509      # are several PathFinder instances (and duplicate ones) on sys.meta_path. This propobly is a bug, see
510      # https://bugs.python.org/issue33128. Thus we need to move all of them to the end, and eliminate the duplicates.
511      path_finders = []
512      for item in reversed(sys.meta_path):
513          if getattr(item, '__name__', None) == 'PathFinder':
514              sys.meta_path.remove(item)
515              if item not in path_finders:
516                  path_finders.append(item)
517      sys.meta_path.extend(reversed(path_finders))
518      # TODO: do we need _frozen_importlib.FrozenImporter in Python 3? Could it be also removed?
519  
520      # Set the FrozenImporter as loader for __main__, in order for python to treat __main__ as a module instead of
521      # a built-in.
522      try:
523          sys.modules['__main__'].__loader__ = importer
524      except Exception:
525          pass
526  
527      # Apply hack for python >= 3.11 and its frozen stdlib modules.
528      if sys.version_info >= (3, 11):
529          _fixup_frozen_stdlib()
530  
531  
532  # A hack for python >= 3.11 and its frozen stdlib modules. Unless `sys._stdlib_dir` is set, these modules end up
533  # missing __file__ attribute, which causes problems with 3rd party code. At the time of writing, python interpreter
534  # configuration API does not allow us to influence `sys._stdlib_dir` - it always resets it to `None`. Therefore,
535  # we manually set the path, and fix __file__ attribute on modules.
536  def _fixup_frozen_stdlib():
537      import _imp  # built-in
538  
539      # If sys._stdlib_dir is None or empty, override it with sys._MEIPASS
540      if not sys._stdlib_dir:
541          try:
542              sys._stdlib_dir = sys._MEIPASS
543          except AttributeError:
544              pass
545  
546      # The sys._stdlib_dir set above should affect newly-imported python-frozen modules. However, most of them have
547      # been already imported during python initialization and our bootstrap, so we need to retroactively fix their
548      # __file__ attribute.
549      for module_name, module in sys.modules.items():
550          if not _imp.is_frozen(module_name):
551              continue
552  
553          is_pkg = _imp.is_frozen_package(module_name)
554  
555          # Determine "real" name from __spec__.loader_state.
556          loader_state = module.__spec__.loader_state
557  
558          orig_name = loader_state.origname
559          if is_pkg:
560              orig_name += '.__init__'
561  
562          # We set suffix to .pyc to be consistent with out PyiFrozenImporter.
563          filename = os.path.join(sys._MEIPASS, *orig_name.split('.')) + '.pyc'
564  
565          # Fixup the __file__ attribute
566          if not hasattr(module, '__file__'):
567              try:
568                  module.__file__ = filename
569              except AttributeError:
570                  pass
571  
572          # Fixup the loader_state.filename
573          # Except for _frozen_importlib (importlib._bootstrap), whose loader_state.filename appears to be left at
574          # None in python.
575          if loader_state.filename is None and orig_name != 'importlib._bootstrap':
576              loader_state.filename = filename
