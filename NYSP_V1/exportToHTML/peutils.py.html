20   class SignatureDatabase(object):
21       """This class loads and keeps a parsed PEiD signature database. 
22    
23       Usage: 
24    
25           sig_db = SignatureDatabase('/path/to/signature/file') 
26    
27       and/or 
28    
29           sig_db = SignatureDatabase() 
30           sig_db.load('/path/to/signature/file') 
31    
32       Signature databases can be combined by performing multiple loads. 
33    
34       The filename parameter can be a URL too. In that case the 
35       signature database will be downloaded from that location. 
36       """
37   
38       def __init__(self, filename=None, data=None):
39   
40           # RegExp to match a signature block
41           #
42           self.parse_sig = re.compile(
43               "\[(.*?)\]\s+?signature\s*=\s*(.*?)(\s+\?\?)*\s*ep_only\s*=\s*(\w+)(?:\s*section_start_only\s*=\s*(\w+)|)",
44               re.S,
45           )
46   
47           # Signature information
48           #
49           # Signatures are stored as trees using dictionaries
50           # The keys are the byte values while the values for
51           # each key are either:
52           #
53           # - Other dictionaries of the same form for further
54           #   bytes in the signature
55           #
56           # - A dictionary with a string as a key (packer name)
57           #   and None as value to indicate a full signature
58           #
59           self.signature_tree_eponly_true = dict()
60           self.signature_count_eponly_true = 0
61           self.signature_tree_eponly_false = dict()
62           self.signature_count_eponly_false = 0
63           self.signature_tree_section_start = dict()
64           self.signature_count_section_start = 0
65   
66           # The depth (length) of the longest signature
67           #
68           self.max_depth = 0
69   
70           self.__load(filename=filename, data=data)

72       def generate_section_signatures(self, pe, name, sig_length=512):
73           """Generates signatures for all the sections in a PE file. 
74    
75           If the section contains any data a signature will be created 
76           for it. The signature name will be a combination of the 
77           parameter 'name' and the section number and its name. 
78           """
79   
80           section_signatures = list()
81   
82           for idx, section in enumerate(pe.sections):
83   
84               if section.SizeOfRawData < sig_length:
85                   continue
86   
87               # offset = pe.get_offset_from_rva(section.VirtualAddress)
88               offset = section.PointerToRawData
89   
90               sig_name = "%s Section(%d/%d,%s)" % (
91                   name,
92                   idx + 1,
93                   len(pe.sections),
94                   "".join([c for c in section.Name if c in string.printable]),
95               )
96   
97               section_signatures.append(
98                   self.__generate_signature(
99                       pe,
100                      offset,
101                      sig_name,
102                      ep_only=False,
103                      section_start_only=True,
104                      sig_length=sig_length,
105                  )
106              )
107  
108          return "\n".join(section_signatures) + "\n"

110      def generate_ep_signature(self, pe, name, sig_length=512):
111          """Generate signatures for the entry point of a PE file. 
112   
113          Creates a signature whose name will be the parameter 'name' 
114          and the section number and its name. 
115          """
116  
117          offset = pe.get_offset_from_rva(pe.OPTIONAL_HEADER.AddressOfEntryPoint)
118  
119          return self.__generate_signature(
120              pe, offset, name, ep_only=True, sig_length=sig_length
121          )

123      def __generate_signature(
124          self, pe, offset, name, ep_only=False, section_start_only=False, sig_length=512
125      ):
126  
127          data = pe.__data__[offset : offset + sig_length]
128  
129          signature_bytes = " ".join(["%02x" % ord(c) for c in data])
130  
131          if ep_only == True:
132              ep_only = "true"
133          else:
134              ep_only = "false"
135  
136          if section_start_only == True:
137              section_start_only = "true"
138          else:
139              section_start_only = "false"
140  
141          signature = "[%s]\nsignature = %s\nep_only = %s\nsection_start_only = %s\n" % (
142              name,
143              signature_bytes,
144              ep_only,
145              section_start_only,
146          )
147  
148          return signature

150      def match(self, pe, ep_only=True, section_start_only=False):
151          """Matches and returns the exact match(es). 
152   
153          If ep_only is True the result will be a string with 
154          the packer name. Otherwise it will be a list of the 
155          form (file_offset, packer_name) specifying where 
156          in the file the signature was found. 
157          """
158  
159          matches = self.__match(pe, ep_only, section_start_only)
160  
161          # The last match (the most precise) from the
162          # list of matches (if any) is returned
163          #
164          if matches:
165              if ep_only == False:
166                  # Get the most exact match for each list of matches
167                  # at a given offset
168                  #
169                  return [(match[0], match[1][-1]) for match in matches]
170  
171              return matches[1][-1]
172  
173          return None

175      def match_all(self, pe, ep_only=True, section_start_only=False):
176          """Matches and returns all the likely matches."""
177  
178          matches = self.__match(pe, ep_only, section_start_only)
179  
180          if matches:
181              if ep_only == False:
182                  # Get the most exact match for each list of matches
183                  # at a given offset
184                  #
185                  return matches
186  
187              return matches[1]
188  
189          return None

191      def __match(self, pe, ep_only, section_start_only):
192  
193          # Load the corresponding set of signatures
194          # Either the one for ep_only equal to True or
195          # to False
196          #
197          if section_start_only is True:
198  
199              # Fetch the data of the executable as it'd
200              # look once loaded in memory
201              #
202              try:
203                  data = pe.__data__
204              except Exception as excp:
205                  raise
206  
207              # Load the corresponding tree of signatures
208              #
209              signatures = self.signature_tree_section_start
210  
211              # Set the starting address to start scanning from
212              #
213              scan_addresses = [section.PointerToRawData for section in pe.sections]
214  
215          elif ep_only is True:
216  
217              # Fetch the data of the executable as it'd
218              # look once loaded in memory
219              #
220              try:
221                  data = pe.get_memory_mapped_image()
222              except Exception as excp:
223                  raise
224  
225              # Load the corresponding tree of signatures
226              #
227              signatures = self.signature_tree_eponly_true
228  
229              # Fetch the entry point of the PE file and the data
230              # at the entry point
231              #
232              ep = pe.OPTIONAL_HEADER.AddressOfEntryPoint
233  
234              # Set the starting address to start scanning from
235              #
236              scan_addresses = [ep]
237  
238          else:
239  
240              data = pe.__data__
241  
242              signatures = self.signature_tree_eponly_false
243  
244              scan_addresses = range(len(data))
245  
246          # For each start address, check if any signature matches
247          #
248          matches = []
249          for idx in scan_addresses:
250              result = self.__match_signature_tree(
251                  signatures, data[idx : idx + self.max_depth]
252              )
253              if result:
254                  matches.append((idx, result))
255  
256          # Return only the matched items found at the entry point if
257          # ep_only is True (matches will have only one element in that
258          # case)
259          #
260          if ep_only is True:
261              if matches:
262                  return matches[0]
263  
264          return matches

266      def match_data(self, code_data, ep_only=True, section_start_only=False):
267  
268          data = code_data
269          scan_addresses = [0]
270  
271          # Load the corresponding set of signatures
272          # Either the one for ep_only equal to True or
273          # to False
274          #
275          if section_start_only is True:
276  
277              # Load the corresponding tree of signatures
278              #
279              signatures = self.signature_tree_section_start
280  
281              # Set the starting address to start scanning from
282              #
283  
284          elif ep_only is True:
285  
286              # Load the corresponding tree of signatures
287              #
288              signatures = self.signature_tree_eponly_true
289  
290          # For each start address, check if any signature matches

292          matches = []
293          for idx in scan_addresses:
294              result = self.__match_signature_tree(

310      def __match_signature_tree(self, signature_tree, data, depth=0):
311          """Recursive function to find matches along the signature tree. 
312   
313          signature_tree  is the part of the tree left to walk 
314          data    is the data being checked against the signature tree 
315          depth   keeps track of how far we have gone down the tree 
316          """
317  
318          matched_names = list()
319          match = signature_tree
320  
321          # Walk the bytes in the data and match them
322          # against the signature
323          #
324          for idx, byte in enumerate([b if isinstance(b, int) else ord(b) for b in data]):
325  
326              # If the tree is exhausted...
327              #
328              if match is None:
329                  break
330  
331              # Get the next byte in the tree
332              #
333              match_next = match.get(byte, None)
334  
335              # If None is among the values for the key
336              # it means that a signature in the database
337              # ends here and that there's an exact match.
338              #
339              if None in list(match.values()):
340                  # idx represent how deep we are in the tree
341                  #
342                  # names = [idx+depth]
343                  names = list()
344  
345                  # For each of the item pairs we check
346                  # if it has an element other than None,
347                  # if not then we have an exact signature
348                  #
349                  for item in list(match.items()):
350                      if item[1] is None:
351                          names.append(item[0])
352                  matched_names.append(names)
353  
354              # If a wildcard is found keep scanning the signature
355              # ignoring the byte.
356              #
357              if "??" in match:
358                  match_tree_alternate = match.get("??", None)
359                  data_remaining = data[idx + 1 :]
360                  if data_remaining:
361                      matched_names.extend(
362                          self.__match_signature_tree(
363                              match_tree_alternate, data_remaining, idx + depth + 1
364                          )
365                      )
366  
367              match = match_next
368  
369          # If we have any more packer name in the end of the signature tree
370          # add them to the matches
371          #
372          if match is not None and None in list(match.values()):
373              # names = [idx + depth + 1]
374              names = list()
375              for item in list(match.items()):
376                  if item[1] is None:
377                      names.append(item[0])
378              matched_names.append(names)
379  
380          return matched_names

390      def __load(self, filename=None, data=None):
391  
392          if filename is not None:
393              # If the path does not exist, attempt to open a URL
394              #
395              if not os.path.exists(filename):
396                  try:
397                      sig_f = urllib.request.urlopen(filename)
398                      sig_data = sig_f.read()
399                      sig_f.close()
400                  except IOError:
401                      # Let this be raised back to the user...
402                      raise
403              else:
404                  # Get the data for a file
405                  #
406                  try:
407                      sig_f = open(filename, "rt")
408                      sig_data = sig_f.read()
409                      sig_f.close()
410                  except IOError:
411                      # Let this be raised back to the user...
412                      raise
413          else:
414              sig_data = data
415  
416          # If the file/URL could not be read or no "raw" data
417          # was provided there's nothing else to do
418          #
419          if not sig_data:
420              return
421  
422          # Helper function to parse the signature bytes
423          #
424          def to_byte(value):
425              if "?" in value:
426                  return value
427              return int(value, 16)
428  
429          # Parse all the signatures in the file
430          #
431          matches = self.parse_sig.findall(sig_data)
432  
433          # For each signature, get the details and load it into the
434          # signature tree
435          #
436          for (
437              packer_name,
438              signature,
439              superfluous_wildcards,
440              ep_only,
441              section_start_only,
442          ) in matches:
443  
444              ep_only = ep_only.strip().lower()
445  
446              signature = signature.replace("\\n", "").strip()
447  
448              signature_bytes = [to_byte(b) for b in signature.split()]
449  
450              if ep_only == "true":
451                  ep_only = True
452              else:
453                  ep_only = False
454  
455              if section_start_only == "true":
456                  section_start_only = True
457              else:
458                  section_start_only = False
459  
460              depth = 0
461  
462              if section_start_only is True:
463  
464                  tree = self.signature_tree_section_start
465                  self.signature_count_section_start += 1
466  
467              else:
468                  if ep_only is True:
469                      tree = self.signature_tree_eponly_true
470                      self.signature_count_eponly_true += 1
471                  else:
472                      tree = self.signature_tree_eponly_false
473                      self.signature_count_eponly_false += 1
474  
475              for idx, byte in enumerate(signature_bytes):
476  
477                  if idx + 1 == len(signature_bytes):
478  
479                      tree[byte] = tree.get(byte, dict())
480                      tree[byte][packer_name] = None
481  
482                  else:
483  
484                      tree[byte] = tree.get(byte, dict())
485  
486                  tree = tree[byte]
487                  depth += 1
488  
489              if depth > self.max_depth:
490                  self.max_depth = depth

498  def is_suspicious(pe):
499      """ 
500      unusual locations of import tables 
501      non recognized section names 
502      presence of long ASCII strings 
503      """
504  
505      relocations_overlap_entry_point = False
506      sequential_relocs = 0
507  
508      # If relocation data is found and the entries go over the entry point, and also are very
509      # continuous or point outside section's boundaries => it might imply that an obfuscation
510      # trick is being used or the relocations are corrupt (maybe intentionally)
511      #
512      if hasattr(pe, "DIRECTORY_ENTRY_BASERELOC"):
513          for base_reloc in pe.DIRECTORY_ENTRY_BASERELOC:
514              last_reloc_rva = None
515              for reloc in base_reloc.entries:
516                  if reloc.rva <= pe.OPTIONAL_HEADER.AddressOfEntryPoint <= reloc.rva + 4:
517                      relocations_overlap_entry_point = True
518  
519                  if (
520                      last_reloc_rva is not None
521                      and last_reloc_rva <= reloc.rva <= last_reloc_rva + 4
522                  ):
523                      sequential_relocs += 1
524  
525                  last_reloc_rva = reloc.rva
526  
527      # If import tables or strings exist (are pointed to) to within the header or in the area
528      # between the PE header and the first section that's suspicious
529      #
530      # IMPLEMENT

532      warnings_while_parsing = False
533      # If we have warnings, that's suspicious, some of those will be because of out-of-ordinary
534      # values are found in the PE header fields

536      # (parsing problems, special section characteristics i.e. W & X, uncommon values of fields,
537      # unusual entrypoint, suspicious imports)

545      # might help to discard cases of legitimate installer or compressed data)

552  def is_probably_packed(pe):
553      """Returns True is there is a high likelihood that a file is packed or contains compressed data. 
554   
555      The sections of the PE file will be analyzed, if enough sections 
556      look like containing compressed data and the data makes 
557      up for more than 20% of the total file size, the function will 
558      return True. 
559      """
560  
561      # Calculate the length of the data up to the end of the last section in the
562      # file. Overlay data won't be taken into account
563      #
564      total_pe_data_length = len(pe.trim())
565      # Assume that the file is packed when no data is available
566      if not total_pe_data_length:
567          return True
568      has_significant_amount_of_compressed_data = False
569  
570      # If some of the sections have high entropy and they make for more than 20% of the file's size
571      # it's assumed that it could be an installer or a packed file
572  
573      total_compressed_data = 0
574      for section in pe.sections:
575          s_entropy = section.get_entropy()
576          s_length = len(section.get_data())
577          # The value of 7.4 is empirical, based on looking at a few files packed
578          # by different packers
579          if s_entropy > 7.4:
580              total_compressed_data += s_length
581  
582      if (total_compressed_data / total_pe_data_length) > 0.2:
583          has_significant_amount_of_compressed_data = True
584  
585      return has_significant_amount_of_compressed_data
