80   def parse_bdist_wininst(name):
81       """Return (base,pyversion) or (None,None) for possible .exe name"""
82   
83       lower = name.lower()
84       base, py_ver, plat = None, None, None
85   
86       if lower.endswith('.exe'):
87           if lower.endswith('.win32.exe'):
88               base = name[:-10]
89               plat = 'win32'
90           elif lower.startswith('.win32-py', -16):
91               py_ver = name[-7:-4]
92               base = name[:-16]
93               plat = 'win32'
94           elif lower.endswith('.win-amd64.exe'):
95               base = name[:-14]
96               plat = 'win-amd64'
97           elif lower.startswith('.win-amd64-py', -20):
98               py_ver = name[-7:-4]
99               base = name[:-20]
100              plat = 'win-amd64'
101      return base, py_ver, plat

104  def egg_info_for_url(url):
105      parts = urllib.parse.urlparse(url)
106      scheme, server, path, parameters, query, fragment = parts
107      base = urllib.parse.unquote(path.split('/')[-1])
108      if server == 'sourceforge.net' and base == 'download':  # XXX Yuck
109          base = urllib.parse.unquote(path.split('/')[-2])
110      if '#' in base:
111          base, fragment = base.split('#', 1)
112      return base, fragment

115  def distros_for_url(url, metadata=None):
116      """Yield egg or source distribution objects that might be found at a URL"""
117      base, fragment = egg_info_for_url(url)
118      yield from distros_for_location(url, base, metadata)
119      if fragment:
120          match = EGG_FRAGMENT.match(fragment)
121          if match:
122              yield from interpret_distro_name(
123                  url, match.group(1), metadata, precedence=CHECKOUT_DIST
124              )

127  def distros_for_location(location, basename, metadata=None):
128      """Yield egg or source distribution objects based on basename"""
129      if basename.endswith('.egg.zip'):
130          basename = basename[:-4]  # strip the .zip
131      if basename.endswith('.egg') and '-' in basename:
132          # only one, unambiguous interpretation
133          return [Distribution.from_location(location, basename, metadata)]
134      if basename.endswith('.whl') and '-' in basename:
135          wheel = Wheel(basename)
136          if not wheel.is_compatible():
137              return []
138          return [
139              Distribution(
140                  location=location,
141                  project_name=wheel.project_name,
142                  version=wheel.version,
143                  # Increase priority over eggs.
144                  precedence=EGG_DIST + 1,
145              )
146          ]
147      if basename.endswith('.exe'):
148          win_base, py_ver, platform = parse_bdist_wininst(basename)
149          if win_base is not None:
150              return interpret_distro_name(
151                  location, win_base, metadata, py_ver, BINARY_DIST, platform
152              )
153      # Try source distro extensions (.zip, .tgz, etc.)
154      #
155      for ext in EXTENSIONS:
156          if basename.endswith(ext):
157              basename = basename[: -len(ext)]
158              return interpret_distro_name(location, basename, metadata)
159      return []  # no extension matched

169  def interpret_distro_name(
170      location, basename, metadata, py_version=None, precedence=SOURCE_DIST, platform=None
171  ):
172      """Generate the interpretation of a source distro name 
173   
174      Note: if `location` is a filesystem filename, you should call 
175      ``pkg_resources.normalize_path()`` on it before passing it to this 
176      routine! 
177      """
178  
179      parts = basename.split('-')
180      if not py_version and any(re.match(r'py\d\.\d$', p) for p in parts[2:]):
181          # it is a bdist_dumb, not an sdist -- bail out
182          return
183  
184      # find the pivot (p) that splits the name from the version.
185      # infer the version as the first item that has a digit.
186      for p in range(len(parts)):
187          if parts[p][:1].isdigit():
188              break
189      else:
190          p = len(parts)
191  
192      yield Distribution(
193          location,
194          metadata,
195          '-'.join(parts[:p]),
196          '-'.join(parts[p:]),
197          py_version=py_version,
198          precedence=precedence,
199          platform=platform,
200      )

203  def unique_values(func):
204      """ 
205      Wrap a function returning an iterable such that the resulting iterable 
206      only ever yields unique items. 
207      """
208  
209      @wraps(func)
210      def wrapper(*args, **kwargs):
211          return unique_everseen(func(*args, **kwargs))
212  
213      return wrapper

222  @unique_values
223  def find_external_links(url, page):
224      """Find rel="homepage" and rel="download" links in `page`, yielding URLs"""
225  
226      for match in REL.finditer(page):
227          tag, rel = match.groups()
228          rels = set(map(str.strip, rel.lower().split(',')))
229          if 'homepage' in rels or 'download' in rels:
230              for match in HREF.finditer(tag):
231                  yield urllib.parse.urljoin(url, htmldecode(match.group(1)))
232  
233      for tag in ("<th>Home Page", "<th>Download URL"):
234          pos = page.find(tag)
235          if pos != -1:
236              match = HREF.search(page, pos)
237              if match:
238                  yield urllib.parse.urljoin(url, htmldecode(match.group(1)))

266  class HashChecker(ContentChecker):
267      pattern = re.compile(
268          r'(?P<hash_name>sha1|sha224|sha384|sha256|sha512|md5)='
269          r'(?P<expected>[a-f0-9]+)'
270      )
271  
272      def __init__(self, hash_name, expected):
273          self.hash_name = hash_name
274          self.hash = hashlib.new(hash_name)
275          self.expected = expected
276  
277      @classmethod
278      def from_url(cls, url):
279          "Construct a (possibly null) ContentChecker from a URL"
280          fragment = urllib.parse.urlparse(url)[-1]
281          if not fragment:
282              return ContentChecker()
283          match = cls.pattern.search(fragment)
284          if not match:
285              return ContentChecker()
286          return cls(**match.groupdict())
287  
288      def feed(self, block):
289          self.hash.update(block)
290  
291      def is_valid(self):
292          return self.hash.hexdigest() == self.expected
293  
294      def report(self, reporter, template):
295          msg = template % self.hash_name
296          return reporter(msg)

302      def __init__(
303          self,
304          index_url="https://pypi.org/simple/",
305          hosts=('*',),
306          ca_bundle=None,
307          verify_ssl=True,
308          *args,
309          **kw,
310      ):
311          super().__init__(*args, **kw)
312          self.index_url = index_url + "/"[: not index_url.endswith('/')]
313          self.scanned_urls = {}
314          self.fetched_urls = {}
315          self.package_pages = {}
316          self.allows = re.compile('|'.join(map(translate, hosts))).match
317          self.to_scan = []

320      def add(self, dist):
321          # ignore invalid versions
322          try:
323              parse_version(dist.version)
324          except Exception:

326          return super().add(dist)

328      # FIXME: 'PackageIndex.process_url' is too complex (14)
329      def process_url(self, url, retrieve=False):  # noqa: C901
330          """Evaluate a URL as a possible download, and maybe retrieve it"""
331          if url in self.scanned_urls and not retrieve:
332              return
333          self.scanned_urls[url] = True
334          if not URL_SCHEME(url):
335              self.process_filename(url)
336              return
337          else:
338              dists = list(distros_for_url(url))
339              if dists:
340                  if not self.url_ok(url):
341                      return
342                  self.debug("Found link: %s", url)
343  
344          if dists or not retrieve or url in self.fetched_urls:
345              list(map(self.add, dists))
346              return  # don't need the actual page
347  
348          if not self.url_ok(url):
349              self.fetched_urls[url] = True
350              return
351  
352          self.info("Reading %s", url)
353          self.fetched_urls[url] = True  # prevent multiple fetch attempts
354          tmpl = "Download error on %s: %%s -- Some packages may not be found!"
355          f = self.open_url(url, tmpl % url)
356          if f is None:
357              return
358          if isinstance(f, urllib.error.HTTPError) and f.code == 401:
359              self.info("Authentication error: %s" % f.msg)
360          self.fetched_urls[f.url] = True
361          if 'html' not in f.headers.get('content-type', '').lower():
362              f.close()  # not html, we can't process it
363              return
364  
365          base = f.url  # handle redirects
366          page = f.read()
367          if not isinstance(page, str):
368              # In Python 3 and got bytes but want str.
369              if isinstance(f, urllib.error.HTTPError):
370                  # Errors have no charset, assume latin1:
371                  charset = 'latin-1'
372              else:
373                  charset = f.headers.get_param('charset') or 'latin-1'
374              page = page.decode(charset, "ignore")
375          f.close()
376          for match in HREF.finditer(page):
377              link = urllib.parse.urljoin(base, htmldecode(match.group(1)))
378              self.process_url(link)
379          if url.startswith(self.index_url) and getattr(f, 'code', None) != 404:
380              page = self.process_index(url, page)
381  
382      def process_filename(self, fn, nested=False):
383          # process filenames or directories
384          if not os.path.exists(fn):
385              self.warn("Not found: %s", fn)
386              return
387  
388          if os.path.isdir(fn) and not nested:
389              path = os.path.realpath(fn)
390              for item in os.listdir(path):
391                  self.process_filename(os.path.join(path, item), True)
392  
393          dists = distros_for_filename(fn)
394          if dists:
395              self.debug("Found: %s", fn)
396              list(map(self.add, dists))

398      def url_ok(self, url, fatal=False):
399          s = URL_SCHEME(url)
400          is_file = s and s.group(1).lower() == 'file'
401          if is_file or self.allows(urllib.parse.urlparse(url)[1]):
402              return True
403          msg = (
404              "\nNote: Bypassing %s (disallowed host; see "
405              "https://setuptools.pypa.io/en/latest/deprecated/"
406              "easy_install.html#restricting-downloads-with-allow-hosts for details).\n"
407          )
408          if fatal:
409              raise DistutilsError(msg % url)
410          else:
411              self.warn(msg, url)
412              return False

414      def scan_egg_links(self, search_path):
415          dirs = filter(os.path.isdir, search_path)
416          egg_links = (
417              (path, entry)
418              for path in dirs
419              for entry in os.listdir(path)
420              if entry.endswith('.egg-link')
421          )
422          list(itertools.starmap(self.scan_egg_link, egg_links))

424      def scan_egg_link(self, path, entry):
425          content = _read_utf8_with_fallback(os.path.join(path, entry))
426          # filter non-empty lines
427          lines = list(filter(None, map(str.strip, content.splitlines())))
428  
429          if len(lines) != 2:
430              # format is not recognized; punt
431              return
432  
433          egg_path, setup_path = lines
434  
435          for dist in find_distributions(os.path.join(path, egg_path)):
436              dist.location = os.path.join(path, *lines)
437              dist.precedence = SOURCE_DIST
438              self.add(dist)

440      def _scan(self, link):
441          # Process a URL to see if it's for a package page
442          NO_MATCH_SENTINEL = None, None
443          if not link.startswith(self.index_url):
444              return NO_MATCH_SENTINEL
445  
446          parts = list(map(urllib.parse.unquote, link[len(self.index_url) :].split('/')))
447          if len(parts) != 2 or '#' in parts[1]:
448              return NO_MATCH_SENTINEL
449  
450          # it's a package page, sanitize and index it
451          pkg = safe_name(parts[0])
452          ver = safe_version(parts[1])
453          self.package_pages.setdefault(pkg.lower(), {})[link] = True
454          return to_filename(pkg), to_filename(ver)
455  
456      def process_index(self, url, page):
457          """Process the contents of a PyPI page"""
458  
459          # process an index page into the package-page index
460          for match in HREF.finditer(page):
461              try:
462                  self._scan(urllib.parse.urljoin(url, htmldecode(match.group(1))))
463              except ValueError:
464                  pass
465  
466          pkg, ver = self._scan(url)  # ensure this page is in the page index
467          if not pkg:
468              return ""  # no sense double-scanning non-package pages
469  
470          # process individual package page
471          for new_url in find_external_links(url, page):
472              # Process the found URL
473              base, frag = egg_info_for_url(new_url)
474              if base.endswith('.py') and not frag:
475                  if ver:
476                      new_url += '#egg=%s-%s' % (pkg, ver)
477                  else:
478                      self.need_version_info(url)
479              self.scan_url(new_url)
480  
481          return PYPI_MD5.sub(
482              lambda m: '<a href="%s#md5=%s">%s</a>' % m.group(1, 3, 2), page
483          )

485      def need_version_info(self, url):
486          self.scan_all(
487              "Page at %s links to .py file(s) without version info; an index "
488              "scan is required.",
489              url,

492      def scan_all(self, msg=None, *args):
493          if self.index_url not in self.fetched_urls:
494              if msg:
495                  self.warn(msg, *args)
496              self.info("Scanning index of all packages (this may take a while)")
497          self.scan_url(self.index_url)

499      def find_packages(self, requirement):
500          self.scan_url(self.index_url + requirement.unsafe_name + '/')
501  
502          if not self.package_pages.get(requirement.key):
503              # Fall back to safe version of the name
504              self.scan_url(self.index_url + requirement.project_name + '/')
505  
506          if not self.package_pages.get(requirement.key):
507              # We couldn't find the target package, so search the index page too
508              self.not_found_in_index(requirement)
509  
510          for url in list(self.package_pages.get(requirement.key, ())):
511              # scan each page that might be related to the desired package
512              self.scan_url(url)
513  
514      def obtain(self, requirement, installer=None):
515          self.prescan()
516          self.find_packages(requirement)
517          for dist in self[requirement.key]:
518              if dist in requirement:
519                  return dist
520              self.debug("%s does not match %s", requirement, dist)
521          return super().obtain(requirement, installer)

523      def check_hash(self, checker, filename, tfp):
524          """ 
525          checker is a ContentChecker 
526          """
527          checker.report(self.debug, "Validating %%s checksum for %s" % filename)
528          if not checker.is_valid():
529              tfp.close()
530              os.unlink(filename)
531              raise DistutilsError(
532                  "%s validation failed for %s; "
533                  "possible download problem?"
534                  % (checker.hash.name, os.path.basename(filename))
535              )

537      def add_find_links(self, urls):
538          """Add `urls` to the list that will be prescanned for searches"""
539          for url in urls:
540              if (
541                  self.to_scan is None  # if we have already "gone online"
542                  or not URL_SCHEME(url)  # or it's a local file/directory
543                  or url.startswith('file:')
544                  or list(distros_for_url(url))  # or a direct package link
545              ):
546                  # then go ahead and process it now
547                  self.scan_url(url)
548              else:
549                  # otherwise, defer retrieval till later
550                  self.to_scan.append(url)

552      def prescan(self):
553          """Scan urls scheduled for prescanning (e.g. --find-links)"""
554          if self.to_scan:
555              list(map(self.scan_url, self.to_scan))
556          self.to_scan = None  # from now on, go ahead and process immediately

558      def not_found_in_index(self, requirement):
559          if self[requirement.key]:  # we've seen at least one distro
560              meth, msg = self.info, "Couldn't retrieve index page for %r"
561          else:  # no distros seen for this name, might be misspelled
562              meth, msg = (
563                  self.warn,
564                  "Couldn't find index page for %r (maybe misspelled?)",
565              )
566          meth(msg, requirement.unsafe_name)
567          self.scan_all()

569      def download(self, spec, tmpdir):
570          """Locate and/or download `spec` to `tmpdir`, returning a local path 
571   
572          `spec` may be a ``Requirement`` object, or a string containing a URL, 
573          an existing local filename, or a project/version requirement spec 
574          (i.e. the string form of a ``Requirement`` object).  If it is the URL 
575          of a .py file with an unambiguous ``#egg=name-version`` tag (i.e., one 
576          that escapes ``-`` as ``_`` throughout), a trivial ``setup.py`` is 
577          automatically created alongside the downloaded file. 
578   
579          If `spec` is a ``Requirement`` object or a string containing a 
580          project/version requirement spec, this method returns the location of 
581          a matching distribution (possibly after downloading it to `tmpdir`). 
582          If `spec` is a locally existing file or directory name, it is simply 
583          returned unchanged.  If `spec` is a URL, it is downloaded to a subpath 
584          of `tmpdir`, and the local filename is returned.  Various errors may be 
585          raised if a problem occurs during downloading. 
586          """
587          if not isinstance(spec, Requirement):
588              scheme = URL_SCHEME(spec)
589              if scheme:
590                  # It's a url, download it to tmpdir
591                  found = self._download_url(spec, tmpdir)
592                  base, fragment = egg_info_for_url(spec)
593                  if base.endswith('.py'):
594                      found = self.gen_setup(found, fragment, tmpdir)
595                  return found
596              elif os.path.exists(spec):
597                  # Existing file or directory, just return it
598                  return spec
599              else:
600                  spec = parse_requirement_arg(spec)
601          return getattr(self.fetch_distribution(spec, tmpdir), 'location', None)

603      def fetch_distribution(  # noqa: C901  # is too complex (14)  # FIXME
604          self,
605          requirement,
606          tmpdir,
607          force_scan=False,
608          source=False,
609          develop_ok=False,
610          local_index=None,
611      ):
612          """Obtain a distribution suitable for fulfilling `requirement` 
613   
614          `requirement` must be a ``pkg_resources.Requirement`` instance. 
615          If necessary, or if the `force_scan` flag is set, the requirement is 
616          searched for in the (online) package index as well as the locally 
617          installed packages.  If a distribution matching `requirement` is found, 
618          the returned distribution's ``location`` is the value you would have 
619          gotten from calling the ``download()`` method with the matching 
620          distribution's URL or filename.  If no matching distribution is found, 
621          ``None`` is returned. 
622   
623          If the `source` flag is set, only source distributions and source 
624          checkout links will be considered.  Unless the `develop_ok` flag is 
625          set, development and system eggs (i.e., those using the ``.egg-info`` 
626          format) will be ignored. 
627          """
628          # process a Requirement
629          self.info("Searching for %s", requirement)
630          skipped = {}
631          dist = None
632  
633          def find(req, env=None):
634              if env is None:
635                  env = self
636              # Find a matching distribution; may be called more than once
637  
638              for dist in env[req.key]:
639                  if dist.precedence == DEVELOP_DIST and not develop_ok:
640                      if dist not in skipped:
641                          self.warn(
642                              "Skipping development or system egg: %s",
643                              dist,
644                          )
645                          skipped[dist] = 1
646                      continue
647  
648                  test = dist in req and (dist.precedence <= SOURCE_DIST or not source)
649                  if test:
650                      loc = self.download(dist.location, tmpdir)
651                      dist.download_location = loc
652                      if os.path.exists(dist.download_location):
653                          return dist
654  
655              return None
656  
657          if force_scan:
658              self.prescan()
659              self.find_packages(requirement)
660              dist = find(requirement)
661  
662          if not dist and local_index is not None:
663              dist = find(requirement, local_index)
664  
665          if dist is None:
666              if self.to_scan is not None:
667                  self.prescan()
668              dist = find(requirement)

675              self.warn(

677                  (source and "a source distribution of " or ""),
678                  requirement,

680              return None
681          else:

683              return dist.clone(location=dist.download_location)

698      def gen_setup(self, filename, fragment, tmpdir):
699          match = EGG_FRAGMENT.match(fragment)
700          dists = (
701              match
702              and [
703                  d
704                  for d in interpret_distro_name(filename, match.group(1), None)
705                  if d.version
706              ]
707              or []
708          )
709  
710          if len(dists) == 1:  # unambiguous ``#egg`` fragment
711              basename = os.path.basename(filename)
712  
713              # Make sure the file has been downloaded to the temp dir.
714              if os.path.dirname(filename) != tmpdir:
715                  dst = os.path.join(tmpdir, basename)
716                  if not (os.path.exists(dst) and os.path.samefile(filename, dst)):
717                      shutil.copy2(filename, dst)
718                      filename = dst
719  
720              with open(os.path.join(tmpdir, 'setup.py'), 'w', encoding="utf-8") as file:
721                  file.write(
722                      "from setuptools import setup\n"
723                      "setup(name=%r, version=%r, py_modules=[%r])\n"
724                      % (
725                          dists[0].project_name,
726                          dists[0].version,
727                          os.path.splitext(basename)[0],
728                      )
729                  )
730              return filename
731  
732          elif match:
733              raise DistutilsError(
734                  "Can't unambiguously interpret project/version identifier %r; "
735                  "any dashes in the name or version should be escaped using "
736                  "underscores. %r" % (fragment, dists)
737              )
738          else:
739              raise DistutilsError(
740                  "Can't process plain .py files without an '#egg=name-version'"
741                  " suffix to enable automatic setup script generation."

746      def _download_to(self, url, filename):
747          self.info("Downloading %s", url)
748          # Download the file
749          fp = None
750          try:
751              checker = HashChecker.from_url(url)
752              fp = self.open_url(url)
753              if isinstance(fp, urllib.error.HTTPError):
754                  raise DistutilsError(
755                      "Can't download %s: %s %s" % (url, fp.code, fp.msg)
756                  )
757              headers = fp.info()
758              blocknum = 0
759              bs = self.dl_blocksize
760              size = -1
761              if "content-length" in headers:
762                  # Some servers return multiple Content-Length headers :(
763                  sizes = headers.get_all('Content-Length')
764                  size = max(map(int, sizes))
765                  self.reporthook(url, filename, blocknum, bs, size)
766              with open(filename, 'wb') as tfp:
767                  while True:
768                      block = fp.read(bs)
769                      if block:
770                          checker.feed(block)
771                          tfp.write(block)
772                          blocknum += 1
773                          self.reporthook(url, filename, blocknum, bs, size)
774                      else:
775                          break
776                  self.check_hash(checker, filename, tfp)
777              return headers
778          finally:
779              if fp:
780                  fp.close()

821          # Determine download filename
822          #
823          name, fragment = egg_info_for_url(url)
824          if name:
825              while '..' in name:
826                  name = name.replace('..', '.').replace('\\', '_')
827          else:
828              name = "__downloaded__"  # default if URL has no path contents

830          if name.endswith('.egg.zip'):
831              name = name[:-4]  # strip the extra .zip before download

837      @staticmethod
838      def _resolve_vcs(url):
839          """ 
840          >>> rvcs = PackageIndex._resolve_vcs 
841          >>> rvcs('git+http://foo/bar') 
842          'git' 
843          >>> rvcs('hg+https://foo/bar') 
844          'hg' 
845          >>> rvcs('git:myhost') 
846          'git' 
847          >>> rvcs('hg:myhost') 
848          >>> rvcs('http://foo/bar') 
849          """
850          scheme = urllib.parse.urlsplit(url).scheme
851          pre, sep, post = scheme.partition('+')
852          # svn and git have their own protocol; hg does not
853          allowed = set(['svn', 'git'] + ['hg'] * bool(sep))
854          return next(iter({pre} & allowed), None)

856      def _download_vcs(self, url, spec_filename):
857          vcs = self._resolve_vcs(url)
858          if not vcs:
859              return
860          if vcs == 'svn':
861              raise DistutilsError(
862                  f"Invalid config, SVN download is not supported: {url}"
863              )
864  
865          filename, _, _ = spec_filename.partition('#')
866          url, rev = self._vcs_split_rev_from_url(url)
867  
868          self.info(f"Doing {vcs} clone from {url} to {filename}")
869          subprocess.check_call([vcs, 'clone', '--quiet', url, filename])
870  
871          co_commands = dict(
872              git=[vcs, '-C', filename, 'checkout', '--quiet', rev],
873              hg=[vcs, '--cwd', filename, 'up', '-C', '-r', rev, '-q'],
874          )
875          if rev is not None:
876              self.info(f"Checking out {rev}")
877              subprocess.check_call(co_commands[vcs])
878  
879          return filename

903      @staticmethod
904      def _vcs_split_rev_from_url(url):
905          """ 
906          Given a possible VCS URL, return a clean URL and resolved revision if any. 
907   
908          >>> vsrfu = PackageIndex._vcs_split_rev_from_url 
909          >>> vsrfu('git+https://github.com/pypa/setuptools@v69.0.0#egg-info=setuptools') 
910          ('https://github.com/pypa/setuptools', 'v69.0.0') 
911          >>> vsrfu('git+https://github.com/pypa/setuptools#egg-info=setuptools') 
912          ('https://github.com/pypa/setuptools', None) 
913          >>> vsrfu('http://foo/bar') 
914          ('http://foo/bar', None) 
915          """
916          parts = urllib.parse.urlsplit(url)
917  
918          clean_scheme = parts.scheme.split('+', 1)[-1]
919  
920          # Some fragment identification fails
921          no_fragment_path, _, _ = parts.path.partition('#')
922  
923          pre, sep, post = no_fragment_path.rpartition('@')
924          clean_path, rev = (pre, post) if sep else (post, None)
925  
926          resolved = parts._replace(
927              scheme=clean_scheme,
928              path=clean_path,
929              # discard the fragment
930              fragment='',
931          ).geturl()
932  
933          return resolved, rev

982  def _encode_auth(auth):
983      """ 
984      Encode auth from a URL suitable for an HTTP header. 
985      >>> str(_encode_auth('username%3Apassword')) 
986      'dXNlcm5hbWU6cGFzc3dvcmQ=' 
987   
988      Long auth strings should not cause a newline to be inserted. 
989      >>> long_auth = 'username:' + 'password'*10 
990      >>> chr(10) in str(_encode_auth(long_auth)) 
991      False 
992      """
993      auth_s = urllib.parse.unquote(auth)
994      # convert to bytes
995      auth_bytes = auth_s.encode()
996      encoded_bytes = base64.b64encode(auth_bytes)
997      # convert back to a string
998      encoded = encoded_bytes.decode()
999      # strip the trailing carriage return
1000     return encoded.replace('\n', '')

1060 def open_with_auth(url, opener=urllib.request.urlopen):
1061     """Open a urllib2 request, handling HTTP authentication"""
1062 
1063     parsed = urllib.parse.urlparse(url)
1064     scheme, netloc, path, params, query, frag = parsed
1065 
1066     # Double scheme does not raise on macOS as revealed by a
1067     # failing test. We would expect "nonnumeric port". Refs #20.
1068     if netloc.endswith(':'):
1069         raise http.client.InvalidURL("nonnumeric port: ''")
1070 
1071     if scheme in ('http', 'https'):
1072         auth, address = _splituser(netloc)
1073     else:
1074         auth = None
1075 
1076     if not auth:
1077         cred = PyPIConfig().find_credential(url)
1078         if cred:
1079             auth = str(cred)
1080             info = cred.username, url
1081             log.info('Authenticating as %s for %s (from .pypirc)', *info)
1082 
1083     if auth:
1084         auth = "Basic " + _encode_auth(auth)
1085         parts = scheme, address, path, params, query, frag
1086         new_url = urllib.parse.urlunparse(parts)
1087         request = urllib.request.Request(new_url)
1088         request.add_header("Authorization", auth)
1089     else:
1090         request = urllib.request.Request(url)
1091 
1092     request.add_header('User-Agent', user_agent)
1093     fp = opener(request)
1094 
1095     if auth:
1096         # Put authentication info back into request URL if same host,
1097         # so that links found on the page will work
1098         s2, h2, path2, param2, query2, frag2 = urllib.parse.urlparse(fp.url)
1099         if s2 == scheme and h2 == address:
1100             parts = s2, netloc, path2, param2, query2, frag2
1101             fp.url = urllib.parse.urlunparse(parts)
1102 
1103     return fp

1122 def local_open(url):
1123     """Read a local path, with special support for directories"""
1124     scheme, server, path, param, query, frag = urllib.parse.urlparse(url)
1125     filename = urllib.request.url2pathname(path)
1126     if os.path.isfile(filename):
1127         return urllib.request.urlopen(url)
1128     elif path.endswith('/') and os.path.isdir(filename):
1129         files = []
1130         for f in os.listdir(filename):
1131             filepath = os.path.join(filename, f)
1132             if f == 'index.html':
1133                 body = _read_utf8_with_fallback(filepath)
1134                 break
1135             elif os.path.isdir(filepath):
1136                 f += '/'
1137             files.append('<a href="{name}">{name}</a>'.format(name=f))
1138         else:
1139             tmpl = (
1140                 "<html><head><title>{url}</title>" "</head><body>{files}</body></html>"
1141             )
1142             body = tmpl.format(url=url, files='\n'.join(files))
1143         status, message = 200, "OK"
1144     else:
1145         status, message, body = 404, "Path not found", "Not found"
1146 
1147     headers = {'content-type': 'text/html'}
1148     body_stream = io.StringIO(body)
1149     return urllib.error.HTTPError(url, status, message, headers, body_stream)
