26   VERSION_PATTERN = r""" 
27       v? 
28       (?: 
29           (?:(?P<epoch>[0-9]+)!)?                           # epoch 
30           (?P<release>[0-9]+(?:\.[0-9]+)*)                  # release segment 
31           (?P<pre>                                          # pre-release 
32               [-_\.]? 
33               (?P<pre_l>alpha|a|beta|b|preview|pre|c|rc) 
34               [-_\.]? 
35               (?P<pre_n>[0-9]+)? 
36           )? 
37           (?P<post>                                         # post release 
38               (?:-(?P<post_n1>[0-9]+)) 
39               | 
40               (?: 
41                   [-_\.]? 
42                   (?P<post_l>post|rev|r) 
43                   [-_\.]? 
44                   (?P<post_n2>[0-9]+)? 
45               ) 
46           )? 
47           (?P<dev>                                          # dev release 
48               [-_\.]? 
49               (?P<dev_l>dev) 
50               [-_\.]? 
51               (?P<dev_n>[0-9]+)? 
52           )? 
53       ) 
54       (?:\+(?P<local>[a-z0-9]+(?:[-_\.][a-z0-9]+)*))?       # local version 
55   """
56   
57   VERSION_REGEX = re.compile(r"^\s*" + VERSION_PATTERN + r"\s*$", re.X | re.I)
58

60   def pep440(version: str) -> bool:
61       """See :ref:`PyPA's version specification <pypa:version-specifiers>` 
62       (initially introduced in :pep:`440`). 
63       """
64       return VERSION_REGEX.match(version) is not None

109  def pep508_versionspec(value: str) -> bool:
110      """Expression that can be used to specify/lock versions (including ranges) 
111      See ``versionspec`` in :ref:`PyPA's dependency specifiers 
112      <pypa:dependency-specifiers>` (initially introduced in :pep:`508`). 
113      """
114      if any(c in value for c in (";", "]", "@")):
115          # In PEP 508:
116          # conditional markers, extras and URL specs are not included in the
117          # versionspec
118          return False
119      # Let's pretend we have a dependency called `requirement` with the given
120      # version spec, then we can reuse the pep508 function for validation:
121      return pep508(f"requirement{value}")

128  def pep517_backend_reference(value: str) -> bool:
129      """See PyPA's specification for defining build-backend references 
130      introduced in :pep:`517#source-trees`. 
131   
132      This is similar to an entry-point reference (e.g., ``package.module:object``). 
133      """
134      module, _, obj = value.partition(":")
135      identifiers = (i.strip() for i in _chain(module.split("."), obj.split(".")))
136      return all(python_identifier(i) for i in identifiers if i)

143  def _download_classifiers() -> str:
144      import ssl
145      from email.message import Message
146      from urllib.request import urlopen
147  
148      url = "https://pypi.org/pypi?:action=list_classifiers"
149      context = ssl.create_default_context()
150      with urlopen(url, context=context) as response:  # noqa: S310 (audit URLs)
151          headers = Message()
152          headers["content_type"] = response.getheader("content-type", "text/plain")
153          return response.read().decode(headers.get_param("charset", "utf-8"))  # type: ignore[no-any-return]

156  class _TroveClassifier:
157      """The ``trove_classifiers`` package is the official way of validating classifiers, 
158      however this package might not be always available. 
159      As a workaround we can still download a list from PyPI. 
160      We also don't want to be over strict about it, so simply skipping silently is an 
161      option (classifiers will be validated anyway during the upload to PyPI). 
162      """
163  
164      downloaded: typing.Union[None, "Literal[False]", typing.Set[str]]
165  
166      def __init__(self) -> None:
167          self.downloaded = None
168          self._skip_download = False
169          # None => not cached yet
170          # False => cache not available
171          self.__name__ = "trove_classifier"  # Emulate a public function
172  
173      def _disable_download(self) -> None:
174          # This is a private API. Only setuptools has the consent of using it.
175          self._skip_download = True
176  
177      def __call__(self, value: str) -> bool:
178          if self.downloaded is False or self._skip_download is True:
179              return True
180  
181          if os.getenv("NO_NETWORK") or os.getenv("VALIDATE_PYPROJECT_NO_NETWORK"):
182              self.downloaded = False
183              msg = (
184                  "Install ``trove-classifiers`` to ensure proper validation. "
185                  "Skipping download of classifiers list from PyPI (NO_NETWORK)."
186              )
187              _logger.debug(msg)
188              return True
189  
190          if self.downloaded is None:
191              msg = (
192                  "Install ``trove-classifiers`` to ensure proper validation. "
193                  "Meanwhile a list of classifiers will be downloaded from PyPI."
194              )
195              _logger.debug(msg)
196              try:
197                  self.downloaded = set(_download_classifiers().splitlines())
198              except Exception:
199                  self.downloaded = False
200                  _logger.debug("Problem with download, skipping validation")
201                  return True
202  
203          return value in self.downloaded or value.lower().startswith("private ::")
204

221  def pep561_stub_name(value: str) -> bool:
222      """Name of a directory containing type stubs. 
223      It must follow the name scheme ``<package>-stubs`` as defined in 
224      :pep:`561#stub-only-packages`. 
225      """
226      top, *children = value.split(".")
227      if not top.endswith("-stubs"):
228          return False
229      return python_module_name(".".join([top[: -len("-stubs")], *children]))

236  def url(value: str) -> bool:
237      """Valid URL (validation uses :obj:`urllib.parse`). 
238      For maximum compatibility please make sure to include a ``scheme`` prefix 
239      in your URL (e.g. ``http://``). 
240      """
241      from urllib.parse import urlparse
242  
243      try:
244          parts = urlparse(value)
245          if not parts.scheme:
246              _logger.warning(
247                  "For maximum compatibility please make sure to include a "
248                  "`scheme` prefix in your URL (e.g. 'http://'). "
249                  f"Given value: {value}"
250              )
251              if not (value.startswith("/") or value.startswith("\\") or "@" in value):
252                  parts = urlparse(f"http://{value}")
253  
254          return bool(parts.scheme and parts.netloc)
255      except Exception:
256          return False

268  def python_identifier(value: str) -> bool:
269      """Can be used as identifier in Python. 
270      (Validation uses :obj:`str.isidentifier`). 
271      """
272      return value.isidentifier()

275  def python_qualified_identifier(value: str) -> bool:
276      """ 
277      Python "dotted identifier", i.e. a sequence of :obj:`python_identifier` 
278      concatenated with ``"."`` (e.g.: ``package.module.submodule``). 
279      """
280      if value.startswith(".") or value.endswith("."):
281          return False
282      return all(python_identifier(m) for m in value.split("."))

286      """Module name that can be used in an ``import``-statement in Python. 
287      See :obj:`python_qualified_identifier`.

299  def python_entrypoint_name(value: str) -> bool:
300      """See ``Data model > name`` in the :ref:`PyPA's entry-points specification 
301      <pypa:entry-points>`. 
302      """
303      if not ENTRYPOINT_REGEX.match(value):
304          return False
305      if not RECOMMEDED_ENTRYPOINT_REGEX.match(value):
306          msg = f"Entry point `{value}` does not follow recommended pattern: "
307          msg += RECOMMEDED_ENTRYPOINT_PATTERN
308          _logger.warning(msg)
309      return True

312  def python_entrypoint_reference(value: str) -> bool:
313      """Reference to a Python object using in the format:: 
314   
315          importable.module:object.attr 
316   
317      See ``Data model >object reference`` in the :ref:`PyPA's entry-points specification 
318      <pypa:entry-points>`. 
319      """
320      module, _, rest = value.partition(":")
321      if "[" in rest:
322          obj, _, extras_ = rest.partition("[")
323          if extras_.strip()[-1] != "]":
324              return False
325          extras = (x.strip() for x in extras_.strip(string.whitespace + "[]").split(","))
326          if not all(pep508_identifier(e) for e in extras):
327              return False
328          _logger.warning(f"`{value}` - using extras for entry points is not recommended")
329      else:
330          obj = rest
331  
332      module_parts = module.split(".")
333      identifiers = _chain(module_parts, obj.split(".")) if rest else module_parts
334      return all(python_identifier(i.strip()) for i in identifiers)

338      r"""Unsigned 8-bit integer (:math:`0 \leq x < 2^8`)"""
339      return 0 <= value < 2**8
