33   def escape_win_filepath(path):
34       # escape all \ with another \ after using normpath to clean up the path
35       return os.path.normpath(path).replace('\\', '\\\\')

38   def make_path_spec_relative(filename, spec_dir):
39       """ 
40       Make the filename relative to the directory containing .spec file if filename is relative and not absolute. 
41       Otherwise keep filename untouched. 
42       """
43       if os.path.isabs(filename):
44           return filename
45       else:
46           filename = os.path.abspath(filename)
47           # Make it relative.
48           filename = os.path.relpath(filename, start=spec_dir)
49           return filename

58   class SourceDestAction(argparse.Action):
59       """ 
60       A command line option which takes multiple source:dest pairs. 
61       """
62       def __init__(self, *args, default=None, metavar=None, **kwargs):
63           super().__init__(*args, default=[], metavar='SOURCE:DEST', **kwargs)
64   
65       def __call__(self, parser, namespace, value, option_string=None):
66           try:
67               # Find the only separator that isn't a Windows drive.
68               separator, = (m for m in re.finditer(rf"(^\w:[/\\])|[:{os.pathsep}]", value) if not m[1])
69           except ValueError:
70               # Split into SRC and DEST failed, wrong syntax
71               raise argparse.ArgumentError(self, f'Wrong syntax, should be {self.option_strings[0]}=SOURCE:DEST')
72           src = value[:separator.start()]
73           dest = value[separator.end():]
74           if not src or not dest:
75               # Syntax was correct, but one or both of SRC and DEST was not given
76               raise argparse.ArgumentError(self, "You have to specify both SOURCE and DEST")
77   
78           # argparse is not particularly smart with copy by reference typed defaults. If the current list is the default,
79           # replace it before modifying it to avoid changing the default.
80           if getattr(namespace, self.dest) is self.default:
81               setattr(namespace, self.dest, [])
82           getattr(namespace, self.dest).append((src, dest))

85   def make_variable_path(filename, conversions=path_conversions):
86       if not os.path.isabs(filename):
87           # os.path.commonpath can not compare relative and absolute paths, and if filename is not absolute, none of the
88           # paths in conversions will match anyway.
89           return None, filename
90       for (from_path, to_name) in conversions:
91           assert os.path.abspath(from_path) == from_path, ("path '%s' should already be absolute" % from_path)
92           try:
93               common_path = os.path.commonpath([filename, from_path])
94           except ValueError:
95               # Per https://docs.python.org/3/library/os.path.html#os.path.commonpath, this raises ValueError in several
96               # cases which prevent computing a common path.
97               common_path = None
98           if common_path == from_path:
99               rest = filename[len(from_path):]
100              if rest.startswith(('\\', '/')):
101                  rest = rest[1:]
102              return to_name, rest
103      return None, filename

136  # An object used in place of a "path string", which knows how to repr() itself using variable names instead of
137  # hard-coded paths.
138  class Path:
139      def __init__(self, *parts):
140          self.path = os.path.join(*parts)
141          self.variable_prefix = self.filename_suffix = None
142  
143      def __repr__(self):
144          if self.filename_suffix is None:
145              self.variable_prefix, self.filename_suffix = make_variable_path(self.path)
146          if self.variable_prefix is None:
147              return repr(self.path)
148          return "os.path.join(" + self.variable_prefix + "," + repr(self.filename_suffix) + ")"

151  # An object used to construct extra preamble for the spec file, in order to accommodate extra collect_*() calls from the
152  # command-line
153  class Preamble:
154      def __init__(
155          self, datas, binaries, hiddenimports, collect_data, collect_binaries, collect_submodules, collect_all,
156          copy_metadata, recursive_copy_metadata
157      ):
158          # Initialize with literal values - will be switched to preamble variable name later, if necessary
159          self.binaries = binaries or []
160          self.hiddenimports = hiddenimports or []
161          self.datas = datas or []
162          # Preamble content
163          self.content = []
164  
165          # Import statements
166          if collect_data:
167              self._add_hookutil_import('collect_data_files')
168          if collect_binaries:
169              self._add_hookutil_import('collect_dynamic_libs')
170          if collect_submodules:
171              self._add_hookutil_import('collect_submodules')
172          if collect_all:
173              self._add_hookutil_import('collect_all')
174          if copy_metadata or recursive_copy_metadata:
175              self._add_hookutil_import('copy_metadata')
176          if self.content:
177              self.content += ['']  # empty line to separate the section
178          # Variables
179          if collect_data or copy_metadata or collect_all or recursive_copy_metadata:
180              self._add_var('datas', self.datas)
181              self.datas = 'datas'  # switch to variable
182          if collect_binaries or collect_all:
183              self._add_var('binaries', self.binaries)
184              self.binaries = 'binaries'  # switch to variable
185          if collect_submodules or collect_all:
186              self._add_var('hiddenimports', self.hiddenimports)
187              self.hiddenimports = 'hiddenimports'  # switch to variable
188          # Content - collect_data_files
189          for entry in collect_data:
190              self._add_collect_data(entry)
191          # Content - copy_metadata
192          for entry in copy_metadata:
193              self._add_copy_metadata(entry)
194          # Content - copy_metadata(..., recursive=True)
195          for entry in recursive_copy_metadata:
196              self._add_recursive_copy_metadata(entry)
197          # Content - collect_binaries
198          for entry in collect_binaries:
199              self._add_collect_binaries(entry)
200          # Content - collect_submodules
201          for entry in collect_submodules:
202              self._add_collect_submodules(entry)
203          # Content - collect_all
204          for entry in collect_all:
205              self._add_collect_all(entry)
206          # Merge
207          if self.content and self.content[-1] != '':
208              self.content += ['']  # empty line
209          self.content = '\n'.join(self.content)
210  
211      def _add_hookutil_import(self, name):
212          self.content += ['from PyInstaller.utils.hooks import {0}'.format(name)]
213  
214      def _add_var(self, name, initial_value):
215          self.content += ['{0} = {1}'.format(name, initial_value)]
216  
217      def _add_collect_data(self, name):
218          self.content += ['datas += collect_data_files(\'{0}\')'.format(name)]
219  
220      def _add_copy_metadata(self, name):
221          self.content += ['datas += copy_metadata(\'{0}\')'.format(name)]
222  
223      def _add_recursive_copy_metadata(self, name):
224          self.content += ['datas += copy_metadata(\'{0}\', recursive=True)'.format(name)]
225  
226      def _add_collect_binaries(self, name):
227          self.content += ['binaries += collect_dynamic_libs(\'{0}\')'.format(name)]
228  
229      def _add_collect_submodules(self, name):
230          self.content += ['hiddenimports += collect_submodules(\'{0}\')'.format(name)]
231  
232      def _add_collect_all(self, name):
233          self.content += [
234              'tmp_ret = collect_all(\'{0}\')'.format(name),
235              'datas += tmp_ret[0]; binaries += tmp_ret[1]; hiddenimports += tmp_ret[2]'
236          ]

622          '--codesign-identity',
623          dest='codesign_identity',
624          metavar='IDENTITY',
625          default=None,
626          help="Code signing identity (macOS only). Use the provided identity to sign collected binaries and generated "
627          "executable. If signing identity is not provided, ad-hoc signing is performed instead.",
628      )
629

631          '--osx-entitlements-file',
632          dest='entitlements_file',
633          metavar='FILENAME',
634          default=None,
635          help="Entitlements file to use when code-signing the collected binaries (macOS only).",
636      )
637

639      g.add_argument(

641          dest="runtime_tmpdir",
642          metavar="PATH",
643          help="Where to extract libraries and support files in `onefile` mode. If this option is given, the bootloader "

645          "here. Please use this option only if you know what you are doing. Note that on POSIX systems, PyInstaller's "
646          "bootloader does NOT perform shell-style environment variable expansion on the given path string. Therefore, "
647          "using environment variables (e.g., ``~`` or ``$HOME``) in path will NOT work.",
648      )

651          action="store_true",
652          default=False,
653          help="Tell the bootloader to ignore signals rather than forwarding them to the child process. Useful in "
654          "situations where for example a supervisor process signals both the bootloader and the child (e.g., via a "

659  def main(
660      scripts,
661      name=None,
662      onefile=False,
663      console=True,
664      debug=[],
665      python_options=[],
666      strip=False,
667      noupx=False,
668      upx_exclude=None,
669      runtime_tmpdir=None,
670      contents_directory=None,
671      pathex=[],
672      version_file=None,
673      specpath=None,
674      bootloader_ignore_signals=False,
675      disable_windowed_traceback=False,
676      datas=[],
677      binaries=[],
678      icon_file=None,
679      manifest=None,
680      resources=[],
681      bundle_identifier=None,
682      hiddenimports=[],
683      hookspath=[],
684      runtime_hooks=[],
685      excludes=[],
686      uac_admin=False,
687      uac_uiaccess=False,
688      collect_submodules=[],
689      collect_binaries=[],
690      collect_data=[],
691      collect_all=[],
692      copy_metadata=[],
693      splash=None,
694      recursive_copy_metadata=[],
695      target_arch=None,
696      codesign_identity=None,
697      entitlements_file=None,
698      argv_emulation=False,
699      hide_console=None,
700      optimize=None,
701      **_kwargs
702  ):
703      # Default values for onefile and console when not explicitly specified on command-line (indicated by None)
704      if onefile is None:
705          onefile = False
706  
707      if console is None:
708          console = True
709  
710      # If appname is not specified - use the basename of the main script as name.
711      if name is None:
712          name = os.path.splitext(os.path.basename(scripts[0]))[0]
713  
714      # If specpath not specified - use default value - current working directory.
715      if specpath is None:
716          specpath = DEFAULT_SPECPATH
717      else:
718          # Expand starting tilde into user's home directory, as a work-around for tilde not being expanded by shell when
719          # using --specpath=~/path/abc` instead of --specpath ~/path/abc` (or when the path argument is quoted).
720          specpath = os.path.expanduser(specpath)
721      # If cwd is the root directory of PyInstaller, generate the .spec file in ./appname/ subdirectory.
722      if specpath == HOMEPATH:
723          specpath = os.path.join(HOMEPATH, name)
724      # Create directory tree if missing.
725      if not os.path.exists(specpath):
726          os.makedirs(specpath)
727  
728      # Handle additional EXE options.
729      exe_options = ''
730      if version_file:
731          exe_options += "\n    version='%s'," % escape_win_filepath(version_file)
732      if uac_admin:
733          exe_options += "\n    uac_admin=True,"
734      if uac_uiaccess:
735          exe_options += "\n    uac_uiaccess=True,"
736      if icon_file:
737          # Icon file for Windows.
738          # On Windows, the default icon is embedded in the bootloader executable.
739          if icon_file[0] == 'NONE':
740              exe_options += "\n    icon='NONE',"
741          else:
742              exe_options += "\n    icon=[%s]," % ','.join("'%s'" % escape_win_filepath(ic) for ic in icon_file)
743          # Icon file for Mac OS.
744          # We need to encapsulate it into apostrofes.
745          icon_file = "'%s'" % icon_file[0]
746      else:
747          # On Mac OS, the default icon has to be copied into the .app bundle.
748          # The the text value 'None' means - use default icon.
749          icon_file = 'None'
750      if contents_directory:
751          exe_options += "\n    contents_directory='%s'," % (contents_directory or "_internal")
752      if hide_console:
753          exe_options += "\n    hide_console='%s'," % hide_console
754  
755      if bundle_identifier:
756          # We need to encapsulate it into apostrofes.
757          bundle_identifier = "'%s'" % bundle_identifier
758  
759      if manifest:
760          if "<" in manifest:
761              # Assume XML string
762              exe_options += "\n    manifest='%s'," % manifest.replace("'", "\\'")
763          else:
764              # Assume filename
765              exe_options += "\n    manifest='%s'," % escape_win_filepath(manifest)
766      if resources:
767          resources = list(map(escape_win_filepath, resources))
768          exe_options += "\n    resources=%s," % repr(resources)
769  
770      hiddenimports = hiddenimports or []
771      upx_exclude = upx_exclude or []
772  
773      # If file extension of the first script is '.pyw', force --windowed option.
774      if is_win and os.path.splitext(scripts[0])[-1] == '.pyw':
775          console = False
776  
777      # If script paths are relative, make them relative to the directory containing .spec file.
778      scripts = [make_path_spec_relative(x, specpath) for x in scripts]
779      # With absolute paths replace prefix with variable HOMEPATH.
780      scripts = list(map(Path, scripts))
781  
782      # Translate the default of ``debug=None`` to an empty list.
783      if debug is None:
784          debug = []
785      # Translate the ``all`` option.
786      if DEBUG_ALL_CHOICE[0] in debug:
787          debug = DEBUG_ARGUMENT_CHOICES
788  
789      # Create preamble (for collect_*() calls)
790      preamble = Preamble(
791          datas, binaries, hiddenimports, collect_data, collect_binaries, collect_submodules, collect_all, copy_metadata,
792          recursive_copy_metadata
793      )
794  
795      if splash:
796          splash_init = splashtmpl % {'splash_image': splash}
797          splash_binaries = "\n    splash.binaries,"
798          splash_target = "\n    splash,"
799      else:
800          splash_init = splash_binaries = splash_target = ""
801  
802      # Infer byte-code optimization level.
803      opt_level = sum([opt == 'O' for opt in python_options])
804      if opt_level > 2:
805          logger.warning(
806              "The switch '--python-option O' has been specified %d times - it should be specified at most twice!",
807              opt_level,
808          )
809          opt_level = 2
810  
811      if optimize is None:
812          if opt_level == 0:
813              # Infer from running python process
814              optimize = sys.flags.optimize
815          else:
816              # Infer from `--python-option O` switch(es).
817              optimize = opt_level
818      elif optimize != opt_level and opt_level != 0:
819          logger.warning(
820              "Mismatch between optimization level passed via --optimize switch (%d) and number of '--python-option O' "
821              "switches (%d)!",
822              optimize,
823              opt_level,
824          )
825  
826      if optimize >= 0:
827          # Ensure OPTIONs passed to bootloader match the optimization settings.
828          python_options += max(0, optimize - opt_level) * ['O']
829  
830      # Create OPTIONs array
831      if 'imports' in debug and 'v' not in python_options:
832          python_options.append('v')
833      python_options_array = [(opt, None, 'OPTION') for opt in python_options]
834  
835      d = {
836          'scripts': scripts,
837          'pathex': pathex or [],
838          'binaries': preamble.binaries,
839          'datas': preamble.datas,
840          'hiddenimports': preamble.hiddenimports,
841          'preamble': preamble.content,
842          'name': name,
843          'noarchive': 'noarchive' in debug,
844          'optimize': optimize,
845          'options': python_options_array,
846          'debug_bootloader': 'bootloader' in debug,
847          'bootloader_ignore_signals': bootloader_ignore_signals,
848          'strip': strip,
849          'upx': not noupx,
850          'upx_exclude': upx_exclude,
851          'runtime_tmpdir': runtime_tmpdir,
852          'exe_options': exe_options,
853          # Directory with additional custom import hooks.
854          'hookspath': hookspath,
855          # List with custom runtime hook files.
856          'runtime_hooks': runtime_hooks or [],
857          # List of modules/packages to ignore.
858          'excludes': excludes or [],
859          # only Windows and Mac OS distinguish windowed and console apps
860          'console': console,
861          'disable_windowed_traceback': disable_windowed_traceback,
862          # Icon filename. Only Mac OS uses this item.
863          'icon': icon_file,
864          # .app bundle identifier. Only OSX uses this item.
865          'bundle_identifier': bundle_identifier,
866          # argv emulation (macOS only)
867          'argv_emulation': argv_emulation,
868          # Target architecture (macOS only)
869          'target_arch': target_arch,
870          # Code signing identity (macOS only)
871          'codesign_identity': codesign_identity,
872          # Entitlements file (macOS only)
873          'entitlements_file': entitlements_file,
874          # splash screen
875          'splash_init': splash_init,
876          'splash_target': splash_target,
877          'splash_binaries': splash_binaries,
878      }
879  
880      # Write down .spec file to filesystem.
881      specfnm = os.path.join(specpath, name + '.spec')
882      with open(specfnm, 'w', encoding='utf-8') as specfile:
883          if onefile:
884              specfile.write(onefiletmplt % d)
885              # For Mac OS create .app bundle.
886              if is_darwin and not console:
887                  specfile.write(bundleexetmplt % d)
888          else:
889              specfile.write(onedirtmplt % d)
890              # For Mac OS create .app bundle.
891              if is_darwin and not console:
892                  specfile.write(bundletmplt % d)
893  
894      return specfnm
