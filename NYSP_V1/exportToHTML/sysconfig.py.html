45   def _is_python_source_dir(d):
46       """ 
47       Return True if the target directory appears to point to an 
48       un-installed Python. 
49       """
50       modules = pathlib.Path(d).joinpath('Modules')
51       return any(modules.joinpath(fn).is_file() for fn in ('Setup', 'Setup.local'))

57   def _is_parent(dir_a, dir_b):
58       """ 
59       Return True if a is a parent of b. 
60       """
61       return os.path.normcase(dir_a).startswith(os.path.normcase(dir_b))

103  def get_python_version():
104      """Return a string containing the major and minor Python version, 
105      leaving off the patchlevel.  Sample return values could be '1.5' 
106      or '2.2'. 
107      """
108      return '%d.%d' % sys.version_info[:2]

111  def get_python_inc(plat_specific=0, prefix=None):
112      """Return the directory containing installed Python header files. 
113   
114      If 'plat_specific' is false (the default), this is the path to the 
115      non-platform-specific header files, i.e. Python.h and so on; 
116      otherwise, this is the path to platform-specific header files 
117      (namely pyconfig.h). 
118   
119      If 'prefix' is supplied, use it instead of sys.base_prefix or 
120      sys.base_exec_prefix -- i.e., ignore 'plat_specific'. 
121      """
122      default_prefix = BASE_EXEC_PREFIX if plat_specific else BASE_PREFIX
123      resolved_prefix = prefix if prefix is not None else default_prefix
124      try:
125          getter = globals()[f'_get_python_inc_{os.name}']
126      except KeyError:
127          raise DistutilsPlatformError(
128              "I don't know where Python installs its C header files "
129              "on platform '%s'" % os.name
130          )
131      return getter(resolved_prefix, prefix, plat_specific)

142  def _get_python_inc_posix(prefix, spec_prefix, plat_specific):
143      if IS_PYPY and sys.version_info < (3, 8):
144          return os.path.join(prefix, 'include')
145      return (
146          _get_python_inc_posix_python(plat_specific)
147          or _extant(_get_python_inc_from_config(plat_specific, spec_prefix))
148          or _get_python_inc_posix_prefix(prefix)

152  def _get_python_inc_posix_python(plat_specific):
153      """ 
154      Assume the executable is in the build directory. The 
155      pyconfig.h file should be in the same directory. Since 
156      the build directory may not be the source directory, 
157      use "srcdir" from the makefile to find the "Include" 
158      directory. 
159      """
160      if not python_build:
161          return
162      if plat_specific:
163          return _sys_home or project_base
164      incdir = os.path.join(get_config_var('srcdir'), 'Include')
165      return os.path.normpath(incdir)

168  def _get_python_inc_from_config(plat_specific, spec_prefix):
169      """ 
170      If no prefix was explicitly specified, provide the include 
171      directory from the config vars. Useful when 
172      cross-compiling, since the config vars may come from 
173      the host 
174      platform Python installation, while the current Python 
175      executable is from the build platform installation. 
176   
177      >>> monkeypatch = getfixture('monkeypatch') 
178      >>> gpifc = _get_python_inc_from_config 
179      >>> monkeypatch.setitem(gpifc.__globals__, 'get_config_var', str.lower) 
180      >>> gpifc(False, '/usr/bin/') 
181      >>> gpifc(False, '') 
182      >>> gpifc(False, None) 
183      'includepy' 
184      >>> gpifc(True, None) 
185      'confincludepy' 
186      """
187      if spec_prefix is None:
188          return get_config_var('CONF' * plat_specific + 'INCLUDEPY')

192      implementation = 'pypy' if IS_PYPY else 'python'
193      python_dir = implementation + get_python_version() + build_flags
194      return os.path.join(prefix, "include", python_dir)

216  def get_python_lib(plat_specific=0, standard_lib=0, prefix=None):
217      """Return the directory containing the Python library (standard or 
218      site additions). 
219   
220      If 'plat_specific' is true, return the directory containing 
221      platform-specific modules, i.e. any module from a non-pure-Python 
222      module distribution; otherwise, return the platform-shared library 
223      directory.  If 'standard_lib' is true, return the directory 
224      containing standard Python library modules; otherwise, return the 
225      directory for site-specific modules. 
226   
227      If 'prefix' is supplied, use it instead of sys.base_prefix or 
228      sys.base_exec_prefix -- i.e., ignore 'plat_specific'. 
229      """
230  
231      if IS_PYPY and sys.version_info < (3, 8):
232          # PyPy-specific schema
233          if prefix is None:
234              prefix = PREFIX
235          if standard_lib:
236              return os.path.join(prefix, "lib-python", sys.version[0])
237          return os.path.join(prefix, 'site-packages')
238  
239      early_prefix = prefix
240  
241      if prefix is None:
242          if standard_lib:
243              prefix = plat_specific and BASE_EXEC_PREFIX or BASE_PREFIX
244          else:
245              prefix = plat_specific and EXEC_PREFIX or PREFIX
246  
247      if os.name == "posix":
248          if plat_specific or standard_lib:
249              # Platform-specific modules (any module from a non-pure-Python
250              # module distribution) or standard Python library modules.
251              libdir = getattr(sys, "platlibdir", "lib")
252          else:
253              # Pure Python
254              libdir = "lib"
255          implementation = 'pypy' if IS_PYPY else 'python'
256          libpython = os.path.join(prefix, libdir, implementation + get_python_version())
257          return _posix_lib(standard_lib, libpython, early_prefix, prefix)
258      elif os.name == "nt":
259          if standard_lib:
260              return os.path.join(prefix, "Lib")
261          else:
262              return os.path.join(prefix, "Lib", "site-packages")
263      else:
264          raise DistutilsPlatformError(
265              "I don't know where Python installs its library "
266              "on platform '%s'" % os.name
267          )

272      """ 
273      Perform first-time customization of compiler-related 
274      config vars on macOS. Use after a compiler is known 
275      to be needed. This customization exists primarily to support Pythons 
276      from binary installers. The kind and paths to build tools on 
277      the user system may vary significantly from the system 
278      that Python itself was built on.  Also the user OS 
279      version and build tools may not support the same set 
280      of CPU architectures for universal builds. 
281      """

283      sys.platform == "darwin" and __import__('_osx_support').customize_compiler(
284          get_config_vars()
285      )

288  def customize_compiler(compiler):  # noqa: C901
289      """Do any platform-specific customization of a CCompiler instance. 
290   
291      Mainly needed on Unix, so we can plug in the information that 
292      varies across Unices and is stored in Python's Makefile. 
293      """
294      if compiler.compiler_type == "unix":
295          _customize_macos()
296  
297          (
298              cc,
299              cxx,
300              cflags,
301              ccshared,
302              ldshared,
303              shlib_suffix,
304              ar,
305              ar_flags,
306          ) = get_config_vars(
307              'CC',
308              'CXX',
309              'CFLAGS',
310              'CCSHARED',
311              'LDSHARED',
312              'SHLIB_SUFFIX',
313              'AR',
314              'ARFLAGS',
315          )
316  
317          if 'CC' in os.environ:
318              newcc = os.environ['CC']
319              if 'LDSHARED' not in os.environ and ldshared.startswith(cc):
320                  # If CC is overridden, use that as the default
321                  #       command for LDSHARED as well
322                  ldshared = newcc + ldshared[len(cc) :]
323              cc = newcc
324          if 'CXX' in os.environ:
325              cxx = os.environ['CXX']
326          if 'LDSHARED' in os.environ:
327              ldshared = os.environ['LDSHARED']
328          if 'CPP' in os.environ:
329              cpp = os.environ['CPP']
330          else:
331              cpp = cc + " -E"  # not always
332          if 'LDFLAGS' in os.environ:
333              ldshared = ldshared + ' ' + os.environ['LDFLAGS']
334          if 'CFLAGS' in os.environ:
335              cflags = cflags + ' ' + os.environ['CFLAGS']
336              ldshared = ldshared + ' ' + os.environ['CFLAGS']
337          if 'CPPFLAGS' in os.environ:
338              cpp = cpp + ' ' + os.environ['CPPFLAGS']
339              cflags = cflags + ' ' + os.environ['CPPFLAGS']
340              ldshared = ldshared + ' ' + os.environ['CPPFLAGS']
341          if 'AR' in os.environ:
342              ar = os.environ['AR']
343          if 'ARFLAGS' in os.environ:
344              archiver = ar + ' ' + os.environ['ARFLAGS']
345          else:
346              archiver = ar + ' ' + ar_flags
347  
348          cc_cmd = cc + ' ' + cflags
349          compiler.set_executables(
350              preprocessor=cpp,
351              compiler=cc_cmd,
352              compiler_so=cc_cmd + ' ' + ccshared,
353              compiler_cxx=cxx,
354              linker_so=ldshared,
355              linker_exe=cc,
356              archiver=archiver,
357          )
358  
359          if 'RANLIB' in os.environ and compiler.executables.get('ranlib', None):
360              compiler.set_executables(ranlib=os.environ['RANLIB'])
361  
362          compiler.shared_lib_extension = shlib_suffix

385  # Regexes needed for parsing Makefile (and similar syntaxes,
386  # like old-style Setup files).
387  _variable_rx = re.compile(r"([a-zA-Z][a-zA-Z0-9_]+)\s*=\s*(.*)")
388  _findvar1_rx = re.compile(r"\$\(([A-Za-z][A-Za-z0-9_]*)\)")
389  _findvar2_rx = re.compile(r"\${([A-Za-z][A-Za-z0-9_]*)}")
390

392  def parse_makefile(fn, g=None):  # noqa: C901
393      """Parse a Makefile-style file. 
394   
395      A dictionary containing name/value pairs is returned.  If an 
396      optional dictionary is passed in as the second argument, it is 
397      used instead of a new dictionary. 
398      """
399      from distutils.text_file import TextFile
400  
401      fp = TextFile(
402          fn, strip_comments=1, skip_blanks=1, join_lines=1, errors="surrogateescape"
403      )
404  
405      if g is None:
406          g = {}
407      done = {}
408      notdone = {}
409  
410      while True:
411          line = fp.readline()
412          if line is None:  # eof
413              break
414          m = _variable_rx.match(line)
415          if m:
416              n, v = m.group(1, 2)
417              v = v.strip()
418              # `$$' is a literal `$' in make
419              tmpv = v.replace('$$', '')
420  
421              if "$" in tmpv:
422                  notdone[n] = v
423              else:
424                  try:
425                      v = int(v)
426                  except ValueError:
427                      # insert literal `$'
428                      done[n] = v.replace('$$', '$')
429                  else:
430                      done[n] = v
431  
432      # Variables with a 'PY_' prefix in the makefile. These need to
433      # be made available without that prefix through sysconfig.
434      # Special care is needed to ensure that variable expansion works, even
435      # if the expansion uses the name without a prefix.
436      renamed_variables = ('CFLAGS', 'LDFLAGS', 'CPPFLAGS')
437  
438      # do variable interpolation here
439      while notdone:
440          for name in list(notdone):
441              value = notdone[name]
442              m = _findvar1_rx.search(value) or _findvar2_rx.search(value)
443              if m:
444                  n = m.group(1)
445                  found = True
446                  if n in done:
447                      item = str(done[n])
448                  elif n in notdone:
449                      # get it on a subsequent round
450                      found = False
451                  elif n in os.environ:
452                      # do it like make: fall back to environment
453                      item = os.environ[n]
454  
455                  elif n in renamed_variables:
456                      if name.startswith('PY_') and name[3:] in renamed_variables:
457                          item = ""
458  
459                      elif 'PY_' + n in notdone:
460                          found = False
461  
462                      else:
463                          item = str(done['PY_' + n])
464                  else:
465                      done[n] = item = ""
466                  if found:
467                      after = value[m.end() :]
468                      value = value[: m.start()] + item + after
469                      if "$" in after:
470                          notdone[name] = value
471                      else:
472                          try:
473                              value = int(value)
474                          except ValueError:
475                              done[name] = value.strip()
476                          else:
477                              done[name] = value
478                          del notdone[name]
479  
480                          if name.startswith('PY_') and name[3:] in renamed_variables:
481                              name = name[3:]
482                              if name not in done:
483                                  done[name] = value
484              else:
485                  # bogus variable reference; just drop it since we can't deal
486                  del notdone[name]
487  
488      fp.close()
489  
490      # strip spurious spaces
491      for k, v in done.items():
492          if isinstance(v, str):
493              done[k] = v.strip()
494  
495      # save the results in the global dictionary
496      g.update(done)
497      return g

500  def expand_makefile_vars(s, vars):
501      """Expand Makefile-style variables -- "${foo}" or "$(foo)" -- in 
502      'string' according to 'vars' (a dictionary mapping variable names to 
503      values).  Variables not present in 'vars' are silently expanded to the 
504      empty string.  The variable values in 'vars' should not contain further 
505      variable expansions; if 'vars' is the output of 'parse_makefile()', 
506      you're fine.  Returns a variable-expanded version of 's'. 
507      """
508  
509      # This algorithm does multiple expansion, so if vars['foo'] contains
510      # "${bar}", it will expand ${foo} to ${bar}, and then expand
511      # ${bar}... and so forth.  This is fine as long as 'vars' comes from
512      # 'parse_makefile()', which takes care of such expansions eagerly,
513      # according to make's variable expansion semantics.
514  
515      while True:
516          m = _findvar1_rx.search(s) or _findvar2_rx.search(s)
517          if m:
518              (beg, end) = m.span()
519              s = s[0:beg] + vars.get(m.group(1)) + s[end:]
520          else:
521              break
522      return s

528  def get_config_vars(*args):
529      """With no arguments, return a dictionary of all configuration 
530      variables relevant for the current platform.  Generally this includes 
531      everything needed to build extensions and install both pure modules and 
532      extensions.  On Unix, this means every variable defined in Python's 
533      installed Makefile; on Windows it's a much smaller set. 
534   
535      With arguments, return a list of values that result from looking up 
536      each argument in the configuration variable dictionary. 
537      """
538      global _config_vars
539      if _config_vars is None:
540          _config_vars = sysconfig.get_config_vars().copy()
541          py39compat.add_ext_suffix(_config_vars)
542  
543      return [_config_vars.get(name) for name in args] if args else _config_vars

546  def get_config_var(name):
547      """Return the value of a single variable using the dictionary 
548      returned by 'get_config_vars()'.  Equivalent to 
549      get_config_vars().get(name) 
550      """
551      if name == 'SO':
552          import warnings
553  
554          warnings.warn('SO is deprecated, use EXT_SUFFIX', DeprecationWarning, 2)
555      return get_config_vars().get(name)
