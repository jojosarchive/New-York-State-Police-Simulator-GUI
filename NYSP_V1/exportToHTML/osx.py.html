33   class BUNDLE(Target):
34       def __init__(self, *args, **kwargs):
35           from PyInstaller.config import CONF
36   
37           # BUNDLE only has a sense under Mac OS, it's a noop on other platforms
38           if not is_darwin:
39               return
40   
41           # Get a path to a .icns icon for the app bundle.
42           self.icon = kwargs.get('icon')
43           if not self.icon:
44               # --icon not specified; use the default in the pyinstaller folder
45               self.icon = os.path.join(
46                   os.path.dirname(os.path.dirname(__file__)), 'bootloader', 'images', 'icon-windowed.icns'
47               )
48           else:
49               # User gave an --icon=path. If it is relative, make it relative to the spec file location.
50               if not os.path.isabs(self.icon):
51                   self.icon = os.path.join(CONF['specpath'], self.icon)
52   
53           super().__init__()
54   
55           # .app bundle is created in DISTPATH.
56           self.name = kwargs.get('name', None)
57           base_name = os.path.basename(self.name)
58           self.name = os.path.join(CONF['distpath'], base_name)
59   
60           self.appname = os.path.splitext(base_name)[0]
61           # Ensure version is a string, even if user accidentally passed an int or a float.
62           # Having a `CFBundleShortVersionString` entry of non-string type in `Info.plist` causes the .app bundle to
63           # crash at start (#4466).
64           self.version = str(kwargs.get("version", "0.0.0"))
65           self.toc = []
66           self.strip = False
67           self.upx = False
68           self.console = True
69           self.target_arch = None
70           self.codesign_identity = None
71           self.entitlements_file = None
72   
73           # .app bundle identifier for Code Signing
74           self.bundle_identifier = kwargs.get('bundle_identifier')
75           if not self.bundle_identifier:
76               # Fallback to appname.
77               self.bundle_identifier = self.appname
78   
79           self.info_plist = kwargs.get('info_plist', None)
80   
81           for arg in args:
82               # Valid arguments: EXE object, COLLECT object, and TOC-like iterables
83               if isinstance(arg, EXE):
84                   # Add EXE as an entry to the TOC, and merge its dependencies TOC
85                   self.toc.append((os.path.basename(arg.name), arg.name, 'EXECUTABLE'))
86                   self.toc.extend(arg.dependencies)
87                   # Inherit settings
88                   self.strip = arg.strip
89                   self.upx = arg.upx
90                   self.upx_exclude = arg.upx_exclude
91                   self.console = arg.console
92                   self.target_arch = arg.target_arch
93                   self.codesign_identity = arg.codesign_identity
94                   self.entitlements_file = arg.entitlements_file
95               elif isinstance(arg, COLLECT):
96                   # Merge the TOC
97                   self.toc.extend(arg.toc)
98                   # Inherit settings
99                   self.strip = arg.strip_binaries
100                  self.upx = arg.upx_binaries
101                  self.upx_exclude = arg.upx_exclude
102                  self.console = arg.console
103                  self.target_arch = arg.target_arch
104                  self.codesign_identity = arg.codesign_identity
105                  self.entitlements_file = arg.entitlements_file
106              elif miscutils.is_iterable(arg):
107                  # TOC-like iterable
108                  self.toc.extend(arg)
109              else:
110                  raise TypeError(f"Invalid argument type for BUNDLE: {type(arg)!r}")
111  
112          # Infer the executable name from the first EXECUTABLE entry in the TOC; it might have come from the COLLECT
113          # (as opposed to the stand-alone EXE).
114          for dest_name, src_name, typecode in self.toc:
115              if typecode == "EXECUTABLE":
116                  self.exename = src_name
117                  break
118          else:
119              raise ValueError("No EXECUTABLE entry found in the TOC!")
120  
121          # Normalize TOC
122          self.toc = normalize_toc(self.toc)
123  
124          self.__postinit__()
125

135      # Helper for determining whether the given file belongs to a .framework bundle or not. If it does, it returns
136      # the path to the top-level .framework bundle directory; otherwise, returns None.
137      @staticmethod
138      def _is_framework_file(dest_path):
139          for parent in dest_path.parents:
140              if parent.name.endswith('.framework'):
141                  return parent
142          return None

144      # Helper that computes relative cross-link path between link's location and target, assuming they are both
145      # rooted in the `Contents` directory of a macOS .app bundle.
146      @staticmethod
147      def _compute_relative_crosslink(crosslink_location, crosslink_target):
148          # We could take symlink_location and symlink_target as they are (relative to parent of the `Contents`
149          # directory), but that would introduce an unnecessary `../Contents` part. So instead, we take both paths
150          # relative to the `Contents` directory.
151          return os.path.join(
152              *['..' for level in pathlib.PurePath(crosslink_location).relative_to('Contents').parent.parts],
153              pathlib.PurePath(crosslink_target).relative_to('Contents'),
154          )

156      # This method takes the original (input) TOC and processes it into final TOC, based on which the `assemble` method
157      # performs its file collection. The TOC processing here represents the core of our efforts to generate an .app
158      # bundle that is compatible with Apple's code-signing requirements.
159      #
160      # For in-depth details on the code-signing, see Apple's `Technical Note TN2206: macOS Code Signing In Depth` at
161      # https://developer.apple.com/library/archive/technotes/tn2206/_index.html
162      #
163      # The requirements, framed from PyInstaller's perspective, can be summarized as follows:
164      #
165      # 1. The `Contents/MacOS` directory is expected to contain only the program executable and (binary) code (= dylibs
166      #    and nested .framework bundles). Alternatively, the dylibs and .framework bundles can be also placed into
167      #    `Contents/Frameworks` directory (where same rules apply as for `Contents/MacOS`, so the remainder of this
168      #    text refers to the two inter-changeably, unless explicitly noted otherwise). The code in `Contents/MacOS`
169      #    is expected to be signed, and the `codesign` utility will recursively sign all found code when using `--deep`
170      #    option to sign the .app bundle.
171      #
172      # 2. All non-code files should be be placed in `Contents/Resources`, so they become sealed (data) resources;
173      #    i.e., their signature data is recorded in `Contents/_CodeSignature/CodeResources`. (As a side note,
174      #    it seems that signature information for data/resources in `Contents/Resources` is kept nder `file` key in
175      #    the `CodeResources` file, while the information for contents in `Contents/MacOS` is kept under `file2` key).
176      #
177      # 3. The directories in `Contents/MacOS` may not contain dots (`.`) in their names, except for the nested
178      #    .framework bundle directories. The directories in `Contents/Resources` have no such restrictions.
179      #
180      # 4. There may not be any content in the top level of a bundle. In other words, if a bundle has a `Contents`
181      #    or a `Versions` directory at its top level, there may be no other files or directories alongside them. The
182      #    sole exception is that alongside ˙Versions˙, there may be symlinks to files and directories in
183      #    `Versions/Current`. This rule is important for nested .framework bundles that we collect from python packages.
184      #
185      # Next, let us consider the consequences of violating each of the above requirements:
186      #
187      # 1. Code signing machinery can directly store signature only in Mach-O binaries and nested .framework bundles; if
188      #    a data file is placed in `Contents/MacOS`, the signature is stored in the file's extended attributes. If the
189      #    extended attributes are lost, the program's signature will be broken. Many file transfer techniques (e.g., a
190      #    zip file) do not preserve extended attributes, nor are they preserved when uploading to the Mac App Store.
191      #
192      # 2. Putting code (a dylib or a .framework bundle) into `Contents/Resources` causes it to be treated as a resource;
193      #    the outer signature (i.e., of the whole .app bundle) does not know that this nested content is actually a code.
194      #    Consequently, signing the bundle with ˙codesign --deep` will NOT sign binaries placed in the
195      #    `Contents/Resources`, which may result in missing signatures when .app bundle is verified for notarization.
196      #    This might be worked around by signing each binary separately, and then signing the whole bundle (without the
197      #    `--deep˙ option), but requires the user to keep track of the offending binaries.
198      #
199      # 3. If a directory in `Contents/MacOS` contains a dot in the name, code-signing the bundle fails with
200      #    ˙bundle format unrecognized, invalid, or unsuitable` due to code signing machinery treating directory as a
201      #    nested .framework bundle directory.
202      #
203      # 4. If nested .framework bundle is malformed, the signing of the .app bundle might succeed, but subsequent
204      #    verification will fail, for example with `embedded framework contains modified or invalid version` (as observed
205      #    with .framework bundles shipped by contemporary PyQt/PySide PyPI wheels).
206      #
207      # The above requirements are unfortunately often at odds with the structure of python packages:
208      #
209      # * In general, python packages are mixed-content directories, where binaries and data files may be expected to
210      #   be found next to each other.
211      #
212      #   For example, `opencv-python` provides a custom loader script that requires the package to be collected in the
213      #   source-only form by PyInstaller (i.e., the python modules and scripts collected as source .py files). At the
214      #   same time, it expects the .py loader script to be able to find the binary extension next to itself.
215      #
216      #   Another example of mixed-mode directories are Qt QML components' sub-directories, which contain both the
217      #   component's plugin (a binary) and associated meta files (data files).
218      #
219      # * In python world, the directories often contain dots in their names.
220      #
221      #   Dots are often used for private directories containing binaries that are shipped with a package. For example,
222      #   `numpy/.dylibs`, `scipy/.dylibs`, etc.
223      #
224      #   Qt QML components may also contain a dot in their name; couple of examples from `PySide2` package:
225      #   `PySide2/Qt/qml/QtQuick.2`, ˙PySide2/Qt/qml/QtQuick/Controls.2˙, ˙PySide2/Qt/qml/QtQuick/Particles.2˙, etc.
226      #
227      #   The packages' metadata directories also invariably contain dots in the name due to version (for example,
228      #   `numpy-1.24.3.dist-info`).
229      #
230      # In the light of all above, PyInstaller attempts to strictly place all files to their mandated location
231      # (`Contents/MacOS` or `Contents/Frameworks` vs `Contents/Resources`). To preserve the illusion of mixed-content
232      # directories, the content is cross-linked from one directory to the other. Specifically:
233      #
234      # * All entries with DATA typecode are assumed to be data files, and are always placed in corresponding directory
235      #   structure rooted in `Contents/Resources`.
236      #
237      # * All entries with BINARY or EXTENSION typecode are always placed in corresponding directory structure rooted in
238      #   `Contents/Frameworks`.
239      #
240      # * All entries with EXECUTABLE are placed in `Contents/MacOS` directory.
241      #
242      # * For the purposes of relocation, nested .framework bundles are treated as a single BINARY entity; i.e., the
243      #   whole .bundle directory is placed in corresponding directory structure rooted in `Contents/Frameworks` (even
244      #   though some of its contents, such as `Info.plist` file, are actually data files).
245      #
246      # * Top-level data files and binaries are always cross-linked to the other directory. For example, given a data file
247      #   `data_file.txt` that was collected into `Contents/Resources`, we create a symbolic link called
248      #   `Contents/MacOS/data_file.txt` that points to `../Resources/data_file.txt`.
249      #
250      # * The executable itself, while placed in `Contents/MacOS`, are cross-linked into both `Contents/Framworks` and
251      #   `Contents/Resources`.
252      #
253      # * The stand-alone PKG entries (used with onefile builds that side-load the PKG archive) are treated as data files
254      #   and collected into `Contents/Resources`, but cross-linked only into `Contents/MacOS` directory (because they
255      #   must appear to be next to the program executable). This is the only entry type that is cross-linked into the
256      #   `Contents/MacOS` directory and also the only data-like entry type that is not cross-linked into the
257      #   `Contents/Frameworks` directory.
258      #
259      # * For files in sub-directories, the cross-linking behavior depends on the type of directory:
260      #
261      #    * A data-only directory is created in directory structure rooted in `Contents/Resources`, and cross-linked
262      #      into directory structure rooted in `Contents/Frameworks` at directory level (i.e., we link the whole
263      #      directory instead of individual files).
264      #
265      #      This largely saves us from having to deal with dots in the names of collected metadata directories, which
266      #      are examples of data-only directories.
267      #
268      #    * A binary-only directory is created in directory structure rooted in `Contents/Frameworks`, and cross-linked
269      #      into `Contents/Resources` at directory level.
270      #
271      #    * A mixed-content directory is created in both directory structures. Files are placed into corresponding
272      #      directory structure based on their type, and cross-linked into other directory structure at file level.
273      #
274      #    * This rule is applied recursively; for example, a data-only sub-directory in a mixed-content directory is
275      #      cross-linked at directory level, while adjacent binary and data files are cross-linked at file level.
276      #
277      # * To work around the issue with dots in the names of directories in `Contents/Frameworks` (applicable to
278      #   binary-only or mixed-content directories), such directories are created with modified name (the dot replaced
279      #   with a pre-defined pattern). Next to the modified directory, a symbolic link with original name is created,
280      #   pointing to the directory with modified name. With mixed-content directories, this modification is performed
281      #   only on the `Contents/Frameworks` side; the corresponding directory in `Contents/Resources` can be created
282      #   directly, without name modification and symbolic link.
283      #
284      # * If a symbolic link needs to be created in a mixed-content directory due to a SYMLINK entry from the original
285      #   TOC (i.e., a "collected" symlink originating from analysis, as opposed to the cross-linking mechanism described
286      #   above), the link is created in both directory structures, each pointing to the resource in its corresponding
287      #   directory structure (with one such resource being an actual file, and the other being a cross-link to the file).
288      #
289      # Final remarks:
290      #
291      # NOTE: the relocation mechanism is codified by tests in `tests/functional/test_macos_bundle_structure.py`.
292      #
293      # NOTE: by placing binaries and nested .framework entries into `Contents/Frameworks` instead of `Contents/MacOS`,
294      # we have effectively relocated the `sys._MEIPASS` directory from the `Contents/MacOS` (= the parent directory of
295      # the program executable) into `Contents/Frameworks`. This requires the PyInstaller's bootloader to detect that it
296      # is running in the app-bundle mode (e.g., by checking if program executable's parent directory is `Contents/NacOS`)
297      # and adjust the path accordingly.
298      #
299      # NOTE: the implemented relocation mechanism depends on the input TOC containing properly classified entries
300      # w.r.t. BINARY vs DATA. So hooks and .spec files triggering collection of binaries as datas (and vice versa) will
301      # result in incorrect placement of those files in the generated .app bundle. However, this is *not* the proper place
302      # to address such issues; if necessary, automatic (re)classification should be added to analysis process, to ensure
303      # that BUNDLE (as well as other build targets) receive correctly classified TOC.
304      #
305      # NOTE: similar to the previous note, the relocation mechanism is also not the proper place to enforce compliant
306      # structure of the nested .framework bundles. Instead, this is handled by the analysis process, using the
307      # `PyInstaller.utils.osx.collect_files_from_framework_bundles` helper function. So the input TOC that BUNDLE
308      # receives should already contain entries that reconstruct compliant nested .framework bundles.
309      def _process_bundle_toc(self, toc):
310          bundle_toc = []
311  
312          # Step 1: inspect the directory layout and classify the directories according to their contents.
313          directory_types = dict()
314  
315          _MIXED_DIR_TYPE = 'MIXED-DIR'
316          _DATA_DIR_TYPE = 'DATA-DIR'
317          _BINARY_DIR_TYPE = 'BINARY-DIR'
318          _FRAMEWORK_DIR_TYPE = 'FRAMEWORK-DIR'
319  
320          _TOP_LEVEL_DIR = pathlib.PurePath('.')
321  
322          for dest_name, src_name, typecode in toc:
323              dest_path = pathlib.PurePath(dest_name)
324  
325              framework_dir = self._is_framework_file(dest_path)
326              if framework_dir:
327                  # Mark the framework directory as FRAMEWORK-DIR.
328                  directory_types[framework_dir] = _FRAMEWORK_DIR_TYPE
329                  # Treat the framework directory as BINARY file when classifying parent directories.
330                  typecode = 'BINARY'
331                  parent_dirs = framework_dir.parents
332              else:
333                  parent_dirs = dest_path.parents
334                  # Treat BINARY and EXTENSION as BINARY to simplify further processing.
335                  if typecode == 'EXTENSION':
336                      typecode = 'BINARY'
337  
338              # (Re)classify parent directories
339              for parent_dir in parent_dirs:
340                  # Skip the top-level `.` dir. This is also the only directory that can contain EXECUTABLE and PKG
341                  # entries, so we do not have to worry about.
342                  if parent_dir == _TOP_LEVEL_DIR:
343                      continue
344  
345                  directory_type = _BINARY_DIR_TYPE if typecode == 'BINARY' else _DATA_DIR_TYPE  # default
346                  directory_type = directory_types.get(parent_dir, directory_type)
347  
348                  if directory_type == _DATA_DIR_TYPE and typecode == 'BINARY':
349                      directory_type = _MIXED_DIR_TYPE
350                  if directory_type == _BINARY_DIR_TYPE and typecode == 'DATA':
351                      directory_type = _MIXED_DIR_TYPE
352  
353                  directory_types[parent_dir] = directory_type
354  
355          logger.debug("Directory classification: %r", directory_types)
356  
357          # Step 2: process the obtained directory structure and create symlink entries for directories that need to be
358          # cross-linked. Such directories are data-only and binary-only directories (and framework directories) that are
359          # located either in the top-level directory (have no parent) or in a mixed-content directory.
360          for directory_path, directory_type in directory_types.items():
361              # Cross-linking at directory level applies only to data-only and binary-only directories (as well as
362              # framework directories).
363              if directory_type == _MIXED_DIR_TYPE:
364                  continue
365  
366              # The parent needs to be either top-level directory or a mixed-content directory. Otherwise, the parent
367              # (or one of its ancestors) will get cross-linked, and we do not need the link here.
368              parent_dir = directory_path.parent
369              requires_crosslink = parent_dir == _TOP_LEVEL_DIR or directory_types.get(parent_dir) == _MIXED_DIR_TYPE
370              if not requires_crosslink:
371                  continue
372  
373              logger.debug("Cross-linking directory %r of type %r", directory_path, directory_type)
374  
375              # Data-only directories are created in `Contents/Resources`, needs to be cross-linked into `Contents/MacOS`.
376              # Vice versa for binary-only or framework directories. The directory creation is handled implicitly, when we
377              # create parent directory structure for collected files.
378              if directory_type == _DATA_DIR_TYPE:
379                  symlink_src = os.path.join('Contents/Resources', directory_path)
380                  symlink_dest = os.path.join('Contents/Frameworks', directory_path)
381              else:
382                  symlink_src = os.path.join('Contents/Frameworks', directory_path)
383                  symlink_dest = os.path.join('Contents/Resources', directory_path)
384              symlink_ref = self._compute_relative_crosslink(symlink_dest, symlink_src)
385  
386              bundle_toc.append((symlink_dest, symlink_ref, 'SYMLINK'))
387  
388          # Step 3: first part of the work-around for directories that are located in `Contents/Frameworks` but contain a
389          # dot in their name. As per `codesign` rules, the only directories in `Contents/Frameworks` that are allowed to
390          # contain a dot in their name are .framework bundle directories. So we replace the dot with a custom character
391          # sequence (stored in global `DOT_REPLACEMENT` variable), and create a symbolic with original name pointing to
392          # the modified name. This is the best we can do with code-sign requirements vs. python community showing their
393          # packages' dylibs into `.dylib` subdirectories, or Qt storing their Qml components in directories named
394          # `QtQuick.2`, `QtQuick/Controls.2`, `QtQuick/Particles.2`, `QtQuick/Templates.2`, etc.
395          #
396          # In this step, we only prepare symlink entries that link the original directory name (with dot) to the modified
397          # one (with dot replaced). The parent paths for collected files are modified in later step(s).
398          for directory_path, directory_type in directory_types.items():
399              # .framework bundle directories contain a dot in the name, but are allowed that.
400              if directory_type == _FRAMEWORK_DIR_TYPE:
401                  continue
402  
403              # Data-only directories are fully located in `Contents/Resources` and cross-linked to `Contents/Frameworks`
404              # at directory level, so they are also allowed a dot in their name.
405              if directory_type == _DATA_DIR_TYPE:
406                  continue
407  
408              # Apply the work-around, if necessary...
409              if '.' not in directory_path.name:
410                  continue
411  
412              logger.debug(
413                  "Creating symlink to work around the dot in the name of directory %r (%s)...", str(directory_path),
414                  directory_type
415              )
416  
417              # Create a SYMLINK entry, but only for this level. In case of nested directories with dots in names, the
418              # symlinks for ancestors will be created by corresponding loop iteration.
419              bundle_toc.append((
420                  os.path.join('Contents/Frameworks', directory_path),
421                  directory_path.name.replace('.', DOT_REPLACEMENT),
422                  'SYMLINK',
423              ))
424  
425          # Step 4: process the entries for collected files, and decide whether they should be placed into
426          # `Contents/MacOS`, `Contents/Frameworks`, or `Contents/Resources`, and whether they should be cross-linked into
427          # other directories.
428          for orig_dest_name, src_name, typecode in toc:
429              orig_dest_path = pathlib.PurePath(orig_dest_name)
430  
431              # Special handling for EXECUTABLE and PKG entries
432              if typecode == 'EXECUTABLE':
433                  # Place into `Contents/MacOS`, ...
434                  file_dest = os.path.join('Contents/MacOS', orig_dest_name)
435                  bundle_toc.append((file_dest, src_name, typecode))
436                  # ... and do nothing else. We explicitly avoid cross-linking the executable to `Contents/Frameworks` and
437                  # `Contents/Resources`, because it should be not necessary (the executable's location should be
438                  # discovered via `sys.executable`) and to prevent issues when executable name collides with name of a
439                  # package from which we collect either binaries or data files (or both); see #7314.
440                  continue
441              elif typecode == 'PKG':
442                  # Place into `Contents/Resources` ...
443                  file_dest = os.path.join('Contents/Resources', orig_dest_name)
444                  bundle_toc.append((file_dest, src_name, typecode))
445                  # ... and cross-link only into `Contents/MacOS`.
446                  # This is used only in `onefile` mode, where there is actually no other content to distribute among the
447                  # `Contents/Resources` and `Contents/Frameworks` directories, so cross-linking into the latter makes
448                  # little sense.
449                  symlink_dest = os.path.join('Contents/MacOS', orig_dest_name)
450                  symlink_ref = self._compute_relative_crosslink(symlink_dest, file_dest)
451                  bundle_toc.append((symlink_dest, symlink_ref, 'SYMLINK'))
452                  continue
453  
454              # Standard data vs binary processing...
455  
456              # Determine file location based on its type.
457              if self._is_framework_file(orig_dest_path):
458                  # File from a framework bundle; put into `Contents/Frameworks`, but never cross-link the file itself.
459                  # The whole .framework bundle directory will be linked as necessary by the directory cross-linking
460                  # mechanism.
461                  file_base_dir = 'Contents/Frameworks'
462                  crosslink_base_dir = None
463              elif typecode == 'DATA':
464                  # Data file; relocate to `Contents/Resources` and cross-link it back into `Contents/Frameworks`.
465                  file_base_dir = 'Contents/Resources'
466                  crosslink_base_dir = 'Contents/Frameworks'
467              else:
468                  # Binary; put into `Contents/Frameworks` and cross-link it into `Contents/Resources`.
469                  file_base_dir = 'Contents/Frameworks'
470                  crosslink_base_dir = 'Contents/Resources'
471  
472              # Determine if we need to cross-link the file. We need to do this for top-level files (the ones without
473              # parent directories), and for files whose parent directories are mixed-content directories.
474              requires_crosslink = False
475              if crosslink_base_dir is not None:
476                  parent_dir = orig_dest_path.parent
477                  requires_crosslink = parent_dir == _TOP_LEVEL_DIR or directory_types.get(parent_dir) == _MIXED_DIR_TYPE
478  
479              # Special handling for SYMLINK entries in original TOC; if we need to cross-link a symlink entry, we create
480              # it in both locations, and have each point to the (relative) resource in the same directory (so one of the
481              # targets will likely be a file, and the other will be a symlink due to cross-linking).
482              if typecode == 'SYMLINK' and requires_crosslink:
483                  bundle_toc.append((os.path.join(file_base_dir, orig_dest_name), src_name, typecode))
484                  bundle_toc.append((os.path.join(crosslink_base_dir, orig_dest_name), src_name, typecode))
485                  continue
486  
487              # The file itself.
488              file_dest = os.path.join(file_base_dir, orig_dest_name)
489              bundle_toc.append((file_dest, src_name, typecode))
490  
491              # Symlink for cross-linking
492              if requires_crosslink:
493                  symlink_dest = os.path.join(crosslink_base_dir, orig_dest_name)
494                  symlink_ref = self._compute_relative_crosslink(symlink_dest, file_dest)
495                  bundle_toc.append((symlink_dest, symlink_ref, 'SYMLINK'))
496  
497          # Step 5: sanitize all destination paths in the new TOC, to ensure that paths that are rooted in
498          # `Contents/Frameworks` do not contain directories with dots in their names. Doing this as a post-processing
499          # step keeps code simple and clean and ensures that this step is applied to files, symlinks that originate from
500          # cross-linking files, and symlinks that originate from cross-linking directories. This in turn ensures that
501          # all directory hierarchies created during the actual file collection have sanitized names, and that collection
502          # outcome does not depend on the order of entries in the TOC.
503          sanitized_toc = []
504          for dest_name, src_name, typecode in bundle_toc:
505              dest_path = pathlib.PurePath(dest_name)
506  
507              # Paths rooted in Contents/Resources do not require sanitizing.
508              if dest_path.parts[0] == 'Contents' and dest_path.parts[1] == 'Resources':
509                  sanitized_toc.append((dest_name, src_name, typecode))
510                  continue
511  
512              # Special handling for files from .framework bundle directories; sanitize only parent path of the .framework
513              # directory.
514              framework_path = self._is_framework_file(dest_path)
515              if framework_path:
516                  parent_path = framework_path.parent
517                  remaining_path = dest_path.relative_to(parent_path)
518              else:
519                  parent_path = dest_path.parent
520                  remaining_path = dest_path.name
521  
522              sanitized_dest_path = pathlib.PurePath(
523                  *parent_path.parts[:2],  # Contents/Frameworks
524                  *[part.replace('.', DOT_REPLACEMENT) for part in parent_path.parts[2:]],
525                  remaining_path,
526              )
527              sanitized_dest_name = str(sanitized_dest_path)
528  
529              if sanitized_dest_path != dest_path:
530                  logger.debug("Sanitizing dest path: %r -> %r", dest_name, sanitized_dest_name)
531  
532              sanitized_toc.append((sanitized_dest_name, src_name, typecode))
533  
534          bundle_toc = sanitized_toc
535  
536          # Normalize and sort the TOC for easier inspection
537          bundle_toc = sorted(normalize_toc(bundle_toc))
538  
539          return bundle_toc

541      def assemble(self):
542          from PyInstaller.config import CONF
543  
544          if _check_path_overlap(self.name) and os.path.isdir(self.name):
545              _rmtree(self.name)
546  
547          logger.info("Building BUNDLE %s", self.tocbasename)
548  
549          # Create a minimal Mac bundle structure.
550          os.makedirs(os.path.join(self.name, "Contents", "MacOS"))
551          os.makedirs(os.path.join(self.name, "Contents", "Resources"))
552          os.makedirs(os.path.join(self.name, "Contents", "Frameworks"))
553  
554          # Makes sure the icon exists and attempts to convert to the proper format if applicable
555          self.icon = normalize_icon_type(self.icon, ("icns",), "icns", CONF["workpath"])
556  
557          # Ensure icon path is absolute
558          self.icon = os.path.abspath(self.icon)
559  
560          # Copy icns icon to Resources directory.
561          shutil.copyfile(self.icon, os.path.join(self.name, 'Contents', 'Resources', os.path.basename(self.icon)))
562  
563          # Key/values for a minimal Info.plist file
564          info_plist_dict = {
565              "CFBundleDisplayName": self.appname,
566              "CFBundleName": self.appname,
567  
568              # Required by 'codesign' utility.
569              # The value for CFBundleIdentifier is used as the default unique name of your program for Code Signing
570              # purposes. It even identifies the APP for access to restricted OS X areas like Keychain.
571              #
572              # The identifier used for signing must be globally unique. The usual form for this identifier is a
573              # hierarchical name in reverse DNS notation, starting with the toplevel domain, followed by the company
574              # name, followed by the department within the company, and ending with the product name. Usually in the
575              # form: com.mycompany.department.appname
576              # CLI option --osx-bundle-identifier sets this value.
577              "CFBundleIdentifier": self.bundle_identifier,
578              "CFBundleExecutable": os.path.basename(self.exename),
579              "CFBundleIconFile": os.path.basename(self.icon),
580              "CFBundleInfoDictionaryVersion": "6.0",
581              "CFBundlePackageType": "APPL",
582              "CFBundleShortVersionString": self.version,
583          }
584  
585          # Set some default values. But they still can be overwritten by the user.
586          if self.console:
587              # Setting EXE console=True implies LSBackgroundOnly=True.
588              info_plist_dict['LSBackgroundOnly'] = True
589          else:
590              # Let's use high resolution by default.
591              info_plist_dict['NSHighResolutionCapable'] = True
592  
593          # Merge info_plist settings from spec file
594          if isinstance(self.info_plist, dict) and self.info_plist:
595              info_plist_dict.update(self.info_plist)
596  
597          plist_filename = os.path.join(self.name, "Contents", "Info.plist")
598          with open(plist_filename, "wb") as plist_fh:
599              plistlib.dump(info_plist_dict, plist_fh)
600  
601          # Pre-process the TOC into its final BUNDLE-compatible form.
602          bundle_toc = self._process_bundle_toc(self.toc)
603  
604          # Perform the actual collection.
605          CONTENTS_FRAMEWORKS_PATH = pathlib.PurePath('Contents/Frameworks')
606          for dest_name, src_name, typecode in bundle_toc:
607              # Create parent directory structure, if necessary
608              dest_path = os.path.join(self.name, dest_name)  # Absolute destination path
609              dest_dir = os.path.dirname(dest_path)
610              try:
611                  os.makedirs(dest_dir, exist_ok=True)
612              except FileExistsError:
613                  raise SystemExit(
614                      f"Pyinstaller needs to create a directory at {dest_dir!r}, "
615                      "but there already exists a file at that path!"
616                  )
617              # Copy extensions and binaries from cache. This ensures that these files undergo additional binary
618              # processing - have paths to linked libraries rewritten (relative to `@rpath`) and have rpath set to the
619              # top-level directory (relative to `@loader_path`, i.e., the file's location). The "top-level" directory
620              # in this case corresponds to `Contents/MacOS` (where `sys._MEIPASS` also points), so we need to pass
621              # the cache retrieval function the *original* destination path (which is without preceding
622              # `Contents/MacOS`).
623              if typecode in ('EXTENSION', 'BINARY'):
624                  orig_dest_name = str(pathlib.PurePath(dest_name).relative_to(CONTENTS_FRAMEWORKS_PATH))
625                  src_name = process_collected_binary(
626                      src_name,
627                      orig_dest_name,
628                      use_strip=self.strip,
629                      use_upx=self.upx,
630                      upx_exclude=self.upx_exclude,
631                      target_arch=self.target_arch,
632                      codesign_identity=self.codesign_identity,
633                      entitlements_file=self.entitlements_file,
634                      strict_arch_validation=(typecode == 'EXTENSION'),
635                  )
636              if typecode == 'SYMLINK':
637                  os.symlink(src_name, dest_path)  # Create link at dest_path, pointing at (relative) src_name
638              else:
639                  # BUNDLE does not support MERGE-based multipackage
640                  assert typecode != 'DEPENDENCY', "MERGE DEPENDENCY entries are not supported in BUNDLE!"
641  
642                  # At this point, `src_name` should be a valid file.
643                  if not os.path.isfile(src_name):
644                      raise ValueError(f"Resource {src_name!r} is not a valid file!")
645                  # If strict collection mode is enabled, the destination should not exist yet.
646                  if strict_collect_mode and os.path.exists(dest_path):
647                      raise ValueError(
648                          f"Attempting to collect a duplicated file into BUNDLE: {dest_name} (type: {typecode})"
649                      )
650                  # Use `shutil.copyfile` to copy file with default permissions. We do not attempt to preserve original
651                  # permissions nor metadata, as they might be too restrictive and cause issues either during subsequent
652                  # re-build attempts or when trying to move the application bundle. For binaries (and data files with
653                  # executable bit set), we manually set the executable bits after copying the file.
654                  shutil.copyfile(src_name, dest_path)
655              if (
656                  typecode in ('EXTENSION', 'BINARY', 'EXECUTABLE')
657                  or (typecode == 'DATA' and os.access(src_name, os.X_OK))
658              ):
659                  os.chmod(dest_path, 0o755)
660  
661          # Sign the bundle
662          logger.info('Signing the BUNDLE...')
663          try:
664              osxutils.sign_binary(self.name, self.codesign_identity, self.entitlements_file, deep=True)
665          except Exception as e:
666              # Display a warning or re-raise the error, depending on the environment-variable setting.
667              if os.environ.get("PYINSTALLER_STRICT_BUNDLE_CODESIGN_ERROR", "0") == "0":
668                  logger.warning("Error while signing the bundle: %s", e)
669                  logger.warning("You will need to sign the bundle manually!")
670              else:
671                  raise RuntimeError("Failed to codesign the bundle!") from e
672  
673          logger.info("Building BUNDLE %s completed successfully.", self.tocbasename)
674  
675          # Optionally verify bundle's signature. This is primarily intended for our CI.
676          if os.environ.get("PYINSTALLER_VERIFY_BUNDLE_SIGNATURE", "0") != "0":
677              logger.info("Verifying signature for BUNDLE %s...", self.name)
678              self.verify_bundle_signature(self.name)
679              logger.info("BUNDLE verification complete!")

681      @staticmethod
682      def verify_bundle_signature(bundle_dir):
683          # First, verify the bundle signature using codesign.
684          cmd_args = ['codesign', '--verify', '--all-architectures', '--deep', '--strict', bundle_dir]
685          p = subprocess.run(cmd_args, stdout=subprocess.PIPE, stderr=subprocess.STDOUT, encoding='utf8')
686          if p.returncode:
687              raise SystemError(
688                  f"codesign command ({cmd_args}) failed with error code {p.returncode}!\noutput: {p.stdout}"
689              )
690  
691          # Ensure that code-signing information is *NOT* embedded in the files' extended attributes.
692          #
693          # This happens when files other than binaries are present in `Contents/MacOS` or `Contents/Frameworks`
694          # directory; as the signature cannot be embedded within the file itself (contrary to binaries with
695          # `LC_CODE_SIGNATURE` section in their header), it ends up stores in the file's extended attributes. However,
696          # if such bundle is transferred using a method that does not support extended attributes (for example, a zip
697          # file), the signatures on these files are lost, and the signature of the bundle as a whole becomes invalid.
698          # This is the primary reason why we need to relocate non-binaries into `Contents/Resources` - the signatures
699          # for files in that directory end up stored in `Contents/_CodeSignature/CodeResources` file.
700          #
701          # This check therefore aims to ensure that all files have been properly relocated to their corresponding
702          # locations w.r.t. the code-signing requirements.
703  
704          try:
705              import xattr
706          except ModuleNotFoundError:
707              logger.info("xattr package not available; skipping verification of extended attributes!")
708              return
709  
710          CODESIGN_ATTRS = (
711              "com.apple.cs.CodeDirectory",
712              "com.apple.cs.CodeRequirements",
713              "com.apple.cs.CodeRequirements-1",
714              "com.apple.cs.CodeSignature",
715          )
716  
717          for entry in pathlib.Path(bundle_dir).rglob("*"):
718              if not entry.is_file():
719                  continue
720  
721              file_attrs = xattr.listxattr(entry)
722              if any([codesign_attr in file_attrs for codesign_attr in CODESIGN_ATTRS]):
723                  raise ValueError(f"Code-sign attributes found in extended attributes of {str(entry)!r}!")
