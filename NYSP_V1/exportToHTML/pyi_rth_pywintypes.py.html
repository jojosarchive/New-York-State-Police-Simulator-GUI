12   # Unfortunately, __import_pywin32_system_module__ from pywintypes module assumes that in a frozen application, the
13   # pythoncom3X.dll and pywintypes3X.dll that are normally found in site-packages/pywin32_system32, are located
14   # directly in the sys.path, without bothering to check first if they are actually available in the standard location.
15   # This obviously runs afoul of our attempts at preserving the directory layout and placing them in the pywin32_system32
16   # sub-directory instead of the top-level application directory. So as a work-around, add the sub-directory to sys.path
17   # to keep pywintypes happy...
18   import sys
19   import os
20   
21   pywin32_system32_path = os.path.join(sys._MEIPASS, 'pywin32_system32')
22   if os.path.isdir(pywin32_system32_path) and pywin32_system32_path not in sys.path:
23       sys.path.append(pywin32_system32_path)
24   del pywin32_system32_path
