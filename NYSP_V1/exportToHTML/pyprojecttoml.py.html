32   def load_file(filepath: StrPath) -> dict:
33       from ..compat.py310 import tomllib
34   
35       with open(filepath, "rb") as file:
36           return tomllib.load(file)

39   def validate(config: dict, filepath: StrPath) -> bool:
40       from . import _validate_pyproject as validator
41   
42       trove_classifier = validator.FORMAT_FUNCTIONS.get("trove-classifier")
43       if hasattr(trove_classifier, "_disable_download"):
44           # Improve reproducibility by default. See issue 31 for validate-pyproject.
45           trove_classifier._disable_download()  # type: ignore
46   
47       try:
48           return validator.validate(config)
49       except validator.ValidationError as ex:
50           summary = f"configuration error: {ex.summary}"
51           if ex.name.strip("`") != "project":
52               # Probably it is just a field missing/misnamed, not worthy the verbosity...
53               _logger.debug(summary)
54               _logger.debug(ex.details)
55   
56           error = f"invalid pyproject.toml config: {ex.name}."
57           raise ValueError(f"{error}\n{summary}") from None

60   def apply_configuration(
61       dist: "Distribution",
62       filepath: StrPath,
63       ignore_option_errors=False,
64   ) -> "Distribution":
65       """Apply the configuration from a ``pyproject.toml`` file into an existing 
66       distribution object. 
67       """
68       config = read_configuration(filepath, True, ignore_option_errors, dist)
69       return _apply(dist, config, filepath)

72   def read_configuration(
73       filepath: StrPath,
74       expand=True,
75       ignore_option_errors=False,
76       dist: Optional["Distribution"] = None,
77   ):
78       """Read given configuration file and returns options from it as a dict. 
79    
80       :param str|unicode filepath: Path to configuration file in the ``pyproject.toml`` 
81           format. 
82    
83       :param bool expand: Whether to expand directives and other computed values 
84           (i.e. post-process the given configuration) 
85    
86       :param bool ignore_option_errors: Whether to silently ignore 
87           options, values of which could not be resolved (e.g. due to exceptions 
88           in directives such as file:, attr:, etc.). 
89           If False exceptions are propagated as expected. 
90    
91       :param Distribution|None: Distribution object to which the configuration refers. 
92           If not given a dummy object will be created and discarded after the 
93           configuration is read. This is used for auto-discovery of packages and in the 
94           case a dynamic configuration (e.g. ``attr`` or ``cmdclass``) is expanded. 
95           When ``expand=False`` this object is simply ignored. 
96    
97       :rtype: dict 
98       """
99       filepath = os.path.abspath(filepath)
100  
101      if not os.path.isfile(filepath):
102          raise FileError(f"Configuration file {filepath!r} does not exist.")
103  
104      asdict = load_file(filepath) or {}
105      project_table = asdict.get("project", {})
106      tool_table = asdict.get("tool", {})
107      setuptools_table = tool_table.get("setuptools", {})
108      if not asdict or not (project_table or setuptools_table):
109          return {}  # User is not using pyproject to configure setuptools
110  
111      if "setuptools" in asdict.get("tools", {}):
112          # let the user know they probably have a typo in their metadata
113          _ToolsTypoInMetadata.emit()
114  
115      if "distutils" in tool_table:
116          _ExperimentalConfiguration.emit(subject="[tool.distutils]")
117  
118      # There is an overall sense in the community that making include_package_data=True
119      # the default would be an improvement.
120      # `ini2toml` backfills include_package_data=False when nothing is explicitly given,
121      # therefore setting a default here is backwards compatible.
122      if dist and getattr(dist, "include_package_data", None) is not None:
123          setuptools_table.setdefault("include-package-data", dist.include_package_data)
124      else:
125          setuptools_table.setdefault("include-package-data", True)
126      # Persist changes:
127      asdict["tool"] = tool_table
128      tool_table["setuptools"] = setuptools_table
129  
130      with _ignore_errors(ignore_option_errors):
131          # Don't complain about unrelated errors (e.g. tools not using the "tool" table)
132          subset = {"project": project_table, "tool": {"setuptools": setuptools_table}}
133          validate(subset, filepath)
134  
135      if expand:
136          root_dir = os.path.dirname(filepath)
137          return expand_configuration(asdict, root_dir, ignore_option_errors, dist)
138  
139      return asdict

142  def expand_configuration(
143      config: dict,
144      root_dir: Optional[StrPath] = None,
145      ignore_option_errors: bool = False,
146      dist: Optional["Distribution"] = None,
147  ) -> dict:
148      """Given a configuration with unresolved fields (e.g. dynamic, cmdclass, ...) 
149      find their final values. 
150   
151      :param dict config: Dict containing the configuration for the distribution 
152      :param str root_dir: Top-level directory for the distribution/project 
153          (the same directory where ``pyproject.toml`` is place) 
154      :param bool ignore_option_errors: see :func:`read_configuration` 
155      :param Distribution|None: Distribution object to which the configuration refers. 
156          If not given a dummy object will be created and discarded after the 
157          configuration is read. Used in the case a dynamic configuration 
158          (e.g. ``attr`` or ``cmdclass``). 
159   
160      :rtype: dict 
161      """
162      return _ConfigExpander(config, root_dir, ignore_option_errors, dist).expand()

165  class _ConfigExpander:
166      def __init__(
167          self,
168          config: dict,
169          root_dir: Optional[StrPath] = None,
170          ignore_option_errors: bool = False,
171          dist: Optional["Distribution"] = None,
172      ):
173          self.config = config
174          self.root_dir = root_dir or os.getcwd()
175          self.project_cfg = config.get("project", {})
176          self.dynamic = self.project_cfg.get("dynamic", [])
177          self.setuptools_cfg = config.get("tool", {}).get("setuptools", {})
178          self.dynamic_cfg = self.setuptools_cfg.get("dynamic", {})
179          self.ignore_option_errors = ignore_option_errors
180          self._dist = dist
181          self._referenced_files: Set[str] = set()
182  
183      def _ensure_dist(self) -> "Distribution":
184          from setuptools.dist import Distribution
185  
186          attrs = {"src_root": self.root_dir, "name": self.project_cfg.get("name", None)}
187          return self._dist or Distribution(attrs)
188  
189      def _process_field(self, container: dict, field: str, fn: Callable):
190          if field in container:
191              with _ignore_errors(self.ignore_option_errors):
192                  container[field] = fn(container[field])
193  
194      def _canonic_package_data(self, field="package-data"):
195          package_data = self.setuptools_cfg.get(field, {})
196          return _expand.canonic_package_data(package_data)
197  
198      def expand(self):
199          self._expand_packages()
200          self._canonic_package_data()
201          self._canonic_package_data("exclude-package-data")
202  
203          # A distribution object is required for discovering the correct package_dir
204          dist = self._ensure_dist()
205          ctx = _EnsurePackagesDiscovered(dist, self.project_cfg, self.setuptools_cfg)
206          with ctx as ensure_discovered:
207              package_dir = ensure_discovered.package_dir
208              self._expand_data_files()
209              self._expand_cmdclass(package_dir)
210              self._expand_all_dynamic(dist, package_dir)
211  
212          dist._referenced_files.update(self._referenced_files)
213          return self.config
214  
215      def _expand_packages(self):
216          packages = self.setuptools_cfg.get("packages")
217          if packages is None or isinstance(packages, (list, tuple)):
218              return
219  
220          find = packages.get("find")
221          if isinstance(find, dict):
222              find["root_dir"] = self.root_dir
223              find["fill_package_dir"] = self.setuptools_cfg.setdefault("package-dir", {})
224              with _ignore_errors(self.ignore_option_errors):
225                  self.setuptools_cfg["packages"] = _expand.find_packages(**find)
226  
227      def _expand_data_files(self):
228          data_files = partial(_expand.canonic_data_files, root_dir=self.root_dir)
229          self._process_field(self.setuptools_cfg, "data-files", data_files)
230  
231      def _expand_cmdclass(self, package_dir: Mapping[str, str]):
232          root_dir = self.root_dir
233          cmdclass = partial(_expand.cmdclass, package_dir=package_dir, root_dir=root_dir)
234          self._process_field(self.setuptools_cfg, "cmdclass", cmdclass)
235  
236      def _expand_all_dynamic(self, dist: "Distribution", package_dir: Mapping[str, str]):
237          special = (  # need special handling
238              "version",
239              "readme",
240              "entry-points",
241              "scripts",
242              "gui-scripts",
243              "classifiers",
244              "dependencies",
245              "optional-dependencies",
246          )
247          # `_obtain` functions are assumed to raise appropriate exceptions/warnings.
248          obtained_dynamic = {
249              field: self._obtain(dist, field, package_dir)
250              for field in self.dynamic
251              if field not in special
252          }
253          obtained_dynamic.update(
254              self._obtain_entry_points(dist, package_dir) or {},
255              version=self._obtain_version(dist, package_dir),
256              readme=self._obtain_readme(dist),
257              classifiers=self._obtain_classifiers(dist),
258              dependencies=self._obtain_dependencies(dist),
259              optional_dependencies=self._obtain_optional_dependencies(dist),
260          )
261          # `None` indicates there is nothing in `tool.setuptools.dynamic` but the value
262          # might have already been set by setup.py/extensions, so avoid overwriting.
263          updates = {k: v for k, v in obtained_dynamic.items() if v is not None}
264          self.project_cfg.update(updates)
265  
266      def _ensure_previously_set(self, dist: "Distribution", field: str):
267          previous = _PREVIOUSLY_DEFINED[field](dist)
268          if previous is None and not self.ignore_option_errors:
269              msg = (
270                  f"No configuration found for dynamic {field!r}.\n"
271                  "Some dynamic fields need to be specified via `tool.setuptools.dynamic`"
272                  "\nothers must be specified via the equivalent attribute in `setup.py`."
273              )
274              raise InvalidConfigError(msg)
275  
276      def _expand_directive(
277          self, specifier: str, directive, package_dir: Mapping[str, str]
278      ):
279          from setuptools.extern.more_itertools import always_iterable
280  
281          with _ignore_errors(self.ignore_option_errors):
282              root_dir = self.root_dir
283              if "file" in directive:
284                  self._referenced_files.update(always_iterable(directive["file"]))
285                  return _expand.read_files(directive["file"], root_dir)
286              if "attr" in directive:
287                  return _expand.read_attr(directive["attr"], package_dir, root_dir)
288              raise ValueError(f"invalid `{specifier}`: {directive!r}")
289          return None
290  
291      def _obtain(self, dist: "Distribution", field: str, package_dir: Mapping[str, str]):
292          if field in self.dynamic_cfg:
293              return self._expand_directive(
294                  f"tool.setuptools.dynamic.{field}",
295                  self.dynamic_cfg[field],
296                  package_dir,
297              )
298          self._ensure_previously_set(dist, field)
299          return None
300  
301      def _obtain_version(self, dist: "Distribution", package_dir: Mapping[str, str]):
302          # Since plugins can set version, let's silently skip if it cannot be obtained
303          if "version" in self.dynamic and "version" in self.dynamic_cfg:
304              return _expand.version(self._obtain(dist, "version", package_dir))
305          return None
306  
307      def _obtain_readme(self, dist: "Distribution") -> Optional[Dict[str, str]]:
308          if "readme" not in self.dynamic:
309              return None
310  
311          dynamic_cfg = self.dynamic_cfg
312          if "readme" in dynamic_cfg:
313              return {
314                  "text": self._obtain(dist, "readme", {}),
315                  "content-type": dynamic_cfg["readme"].get("content-type", "text/x-rst"),
316              }
317  
318          self._ensure_previously_set(dist, "readme")
319          return None
320  
321      def _obtain_entry_points(
322          self, dist: "Distribution", package_dir: Mapping[str, str]
323      ) -> Optional[Dict[str, dict]]:
324          fields = ("entry-points", "scripts", "gui-scripts")
325          if not any(field in self.dynamic for field in fields):
326              return None
327  
328          text = self._obtain(dist, "entry-points", package_dir)
329          if text is None:
330              return None
331  
332          groups = _expand.entry_points(text)
333          expanded = {"entry-points": groups}
334  
335          def _set_scripts(field: str, group: str):
336              if group in groups:
337                  value = groups.pop(group)
338                  if field not in self.dynamic:
339                      raise InvalidConfigError(_MissingDynamic.details(field, value))
340                  expanded[field] = value
341  
342          _set_scripts("scripts", "console_scripts")
343          _set_scripts("gui-scripts", "gui_scripts")
344  
345          return expanded
346  
347      def _obtain_classifiers(self, dist: "Distribution"):
348          if "classifiers" in self.dynamic:
349              value = self._obtain(dist, "classifiers", {})
350              if value:
351                  return value.splitlines()
352          return None
353  
354      def _obtain_dependencies(self, dist: "Distribution"):
355          if "dependencies" in self.dynamic:
356              value = self._obtain(dist, "dependencies", {})
357              if value:
358                  return _parse_requirements_list(value)
359          return None
360  
361      def _obtain_optional_dependencies(self, dist: "Distribution"):
362          if "optional-dependencies" not in self.dynamic:
363              return None
364          if "optional-dependencies" in self.dynamic_cfg:
365              optional_dependencies_map = self.dynamic_cfg["optional-dependencies"]
366              assert isinstance(optional_dependencies_map, dict)
367              return {
368                  group: _parse_requirements_list(
369                      self._expand_directive(
370                          f"tool.setuptools.dynamic.optional-dependencies.{group}",
371                          directive,
372                          {},
373                      )
374                  )
375                  for group, directive in optional_dependencies_map.items()
376              }
377          self._ensure_previously_set(dist, "optional-dependencies")
378          return None
379  
380  
381  def _parse_requirements_list(value):
382      return [
383          line
384          for line in value.splitlines()
385          if line.strip() and not line.strip().startswith("#")
386      ]

389  @contextmanager
390  def _ignore_errors(ignore_option_errors: bool):
391      if not ignore_option_errors:
392          yield
393          return
394  
395      try:
396          yield
397      except Exception as ex:
398          _logger.debug(f"ignored error: {ex.__class__.__name__} - {ex}")

401  class _EnsurePackagesDiscovered(_expand.EnsurePackagesDiscovered):
402      def __init__(
403          self, distribution: "Distribution", project_cfg: dict, setuptools_cfg: dict
404      ):
405          super().__init__(distribution)
406          self._project_cfg = project_cfg
407          self._setuptools_cfg = setuptools_cfg
408  
409      def __enter__(self) -> "Self":
410          """When entering the context, the values of ``packages``, ``py_modules`` and 
411          ``package_dir`` that are missing in ``dist`` are copied from ``setuptools_cfg``. 
412          """
413          dist, cfg = self._dist, self._setuptools_cfg
414          package_dir: Dict[str, str] = cfg.setdefault("package-dir", {})
415          package_dir.update(dist.package_dir or {})
416          dist.package_dir = package_dir  # needs to be the same object
417  
418          dist.set_defaults._ignore_ext_modules()  # pyproject.toml-specific behaviour
419  
420          # Set `name`, `py_modules` and `packages` in dist to short-circuit
421          # auto-discovery, but avoid overwriting empty lists purposefully set by users.
422          if dist.metadata.name is None:
423              dist.metadata.name = self._project_cfg.get("name")
424          if dist.py_modules is None:
425              dist.py_modules = cfg.get("py-modules")
426          if dist.packages is None:
427              dist.packages = cfg.get("packages")
428  
429          return super().__enter__()
430  
431      def __exit__(self, exc_type, exc_value, traceback):
432          """When exiting the context, if values of ``packages``, ``py_modules`` and 
433          ``package_dir`` are missing in ``setuptools_cfg``, copy from ``dist``. 
434          """
435          # If anything was discovered set them back, so they count in the final config.
436          self._setuptools_cfg.setdefault("packages", self._dist.packages)
437          self._setuptools_cfg.setdefault("py-modules", self._dist.py_modules)
438          return super().__exit__(exc_type, exc_value, traceback)
