32   # Regex to define acceptable Distutils command names.  This is not *quite*
33   # the same as a Python NAME -- I don't allow leading underscores.  The fact
34   # that they're very similar is no coincidence; the default naming scheme is
35   # to look for a Python module named after the command.
36   command_re = re.compile(r'^[a-zA-Z]([a-zA-Z0-9_]*)$')

39   def _ensure_list(value, fieldname):
40       if isinstance(value, str):
41           # a string containing comma separated values is okay.  It will
42           # be converted to a list by Distribution.finalize_options().
43           pass
44       elif not isinstance(value, list):
45           # passing a tuple or an iterator perhaps, warn and convert
46           typename = type(value).__name__
47           msg = "Warning: '{fieldname}' should be a list, got type '{typename}'"
48           msg = msg.format(**locals())
49           log.warning(msg)
50           value = list(value)
51       return value

54   class Distribution:
55       """The core of the Distutils.  Most of the work hiding behind 'setup' 
56       is really done within a Distribution instance, which farms the work out 
57       to the Distutils commands specified on the command line. 
58    
59       Setup scripts will almost never instantiate Distribution directly, 
60       unless the 'setup()' function is totally inadequate to their needs. 
61       However, it is conceivable that a setup script might wish to subclass 
62       Distribution for some specialized purpose, and then pass the subclass 
63       to 'setup()' as the 'distclass' keyword argument.  If so, it is 
64       necessary to respect the expectations that 'setup' has of Distribution. 
65       See the code for 'setup()', in core.py, for details. 
66       """
67   
68       # 'global_options' describes the command-line options that may be
69       # supplied to the setup script prior to any actual commands.
70       # Eg. "./setup.py -n" or "./setup.py --quiet" both take advantage of
71       # these global options.  This list should be kept to a bare minimum,
72       # since every global option is also valid as a command option -- and we
73       # don't want to pollute the commands with too many options that they
74       # have minimal control over.
75       # The fourth entry for verbose means that it can be repeated.
76       global_options = [
77           ('verbose', 'v', "run verbosely (default)", 1),
78           ('quiet', 'q', "run quietly (turns verbosity off)"),
79           ('dry-run', 'n', "don't actually do anything"),
80           ('help', 'h', "show detailed help message"),
81           ('no-user-cfg', None, 'ignore pydistutils.cfg in your home directory'),
82       ]
83   
84       # 'common_usage' is a short (2-3 line) string describing the common
85       # usage of the setup script.
86       common_usage = """\ 
87   Common commands: (see '--help-commands' for more) 
88    
89     setup.py build      will build the package underneath 'build/' 
90     setup.py install    will install the package 
91   """

93       # options that are not propagated to the commands
94       display_options = [
95           ('help-commands', None, "list all available commands"),
96           ('name', None, "print package name"),
97           ('version', 'V', "print package version"),
98           ('fullname', None, "print <package name>-<version>"),
99           ('author', None, "print the author's name"),
100          ('author-email', None, "print the author's email address"),
101          ('maintainer', None, "print the maintainer's name"),
102          ('maintainer-email', None, "print the maintainer's email address"),
103          ('contact', None, "print the maintainer's name if known, else the author's"),
104          (

128      def __init__(self, attrs=None):  # noqa: C901
129          """Construct a new Distribution instance: initialize all the 
130          attributes of a Distribution, and then use 'attrs' (a dictionary 
131          mapping attribute names to values) to assign some of those 
132          attributes their "real" values.  (Any attributes not mentioned in 
133          'attrs' will be assigned to some null value: 0, None, an empty list 
134          or dictionary, etc.)  Most importantly, initialize the 
135          'command_obj' attribute to the empty dictionary; this will be 
136          filled in with real command objects by 'parse_command_line()'. 
137          """
138  
139          # Default values for our command-line options
140          self.verbose = 1
141          self.dry_run = 0
142          self.help = 0
143          for attr in self.display_option_names:
144              setattr(self, attr, 0)
145  
146          # Store the distribution meta-data (name, version, author, and so
147          # forth) in a separate object -- we're getting to have enough
148          # information here (and enough command-line options) that it's
149          # worth it.  Also delegate 'get_XXX()' methods to the 'metadata'
150          # object in a sneaky and underhanded (but efficient!) way.
151          self.metadata = DistributionMetadata()
152          for basename in self.metadata._METHOD_BASENAMES:
153              method_name = "get_" + basename
154              setattr(self, method_name, getattr(self.metadata, method_name))
155  
156          # 'cmdclass' maps command names to class objects, so we
157          # can 1) quickly figure out which class to instantiate when
158          # we need to create a new command object, and 2) have a way
159          # for the setup script to override command classes
160          self.cmdclass = {}
161  
162          # 'command_packages' is a list of packages in which commands
163          # are searched for.  The factory for command 'foo' is expected
164          # to be named 'foo' in the module 'foo' in one of the packages
165          # named here.  This list is searched from the left; an error
166          # is raised if no named package provides the command being
167          # searched for.  (Always access using get_command_packages().)
168          self.command_packages = None
169  
170          # 'script_name' and 'script_args' are usually set to sys.argv[0]
171          # and sys.argv[1:], but they can be overridden when the caller is
172          # not necessarily a setup script run from the command-line.
173          self.script_name = None
174          self.script_args = None
175  
176          # 'command_options' is where we store command options between
177          # parsing them (from config files, the command-line, etc.) and when
178          # they are actually needed -- ie. when the command in question is
179          # instantiated.  It is a dictionary of dictionaries of 2-tuples:
180          #   command_options = { command_name : { option : (source, value) } }
181          self.command_options = {}
182  
183          # 'dist_files' is the list of (command, pyversion, file) that
184          # have been created by any dist commands run so far. This is
185          # filled regardless of whether the run is dry or not. pyversion
186          # gives sysconfig.get_python_version() if the dist file is
187          # specific to a Python version, 'any' if it is good for all
188          # Python versions on the target platform, and '' for a source
189          # file. pyversion should not be used to specify minimum or
190          # maximum required Python versions; use the metainfo for that
191          # instead.
192          self.dist_files = []
193  
194          # These options are really the business of various commands, rather
195          # than of the Distribution itself.  We provide aliases for them in
196          # Distribution as a convenience to the developer.
197          self.packages = None
198          self.package_data = {}
199          self.package_dir = None
200          self.py_modules = None
201          self.libraries = None
202          self.headers = None
203          self.ext_modules = None
204          self.ext_package = None
205          self.include_dirs = None
206          self.extra_path = None
207          self.scripts = None
208          self.data_files = None
209          self.password = ''
210  
211          # And now initialize bookkeeping stuff that can't be supplied by
212          # the caller at all.  'command_obj' maps command names to
213          # Command instances -- that's how we enforce that every command
214          # class is a singleton.
215          self.command_obj = {}
216  
217          # 'have_run' maps command names to boolean values; it keeps track
218          # of whether we have actually run a particular command, to make it
219          # cheap to "run" a command whenever we think we might need to -- if
220          # it's already been done, no need for expensive filesystem
221          # operations, we just check the 'have_run' dictionary and carry on.
222          # It's only safe to query 'have_run' for a command class that has
223          # been instantiated -- a false value will be inserted when the
224          # command object is created, and replaced with a true value when
225          # the command is successfully run.  Thus it's probably best to use
226          # '.get()' rather than a straight lookup.
227          self.have_run = {}
228  
229          # Now we'll use the attrs dictionary (ultimately, keyword args from
230          # the setup script) to possibly override any or all of these
231          # distribution options.
232  
233          if attrs:
234              # Pull out the set of command options and work on them
235              # specifically.  Note that this order guarantees that aliased
236              # command options will override any supplied redundantly
237              # through the general options dictionary.
238              options = attrs.get('options')
239              if options is not None:
240                  del attrs['options']
241                  for command, cmd_options in options.items():
242                      opt_dict = self.get_option_dict(command)
243                      for opt, val in cmd_options.items():
244                          opt_dict[opt] = ("setup script", val)
245  
246              if 'licence' in attrs:
247                  attrs['license'] = attrs['licence']
248                  del attrs['licence']
249                  msg = "'licence' distribution option is deprecated; use 'license'"
250                  if warnings is not None:
251                      warnings.warn(msg)
252                  else:
253                      sys.stderr.write(msg + "\n")
254  
255              # Now work on the rest of the attributes.  Any attribute that's
256              # not already defined is invalid!
257              for key, val in attrs.items():
258                  if hasattr(self.metadata, "set_" + key):
259                      getattr(self.metadata, "set_" + key)(val)
260                  elif hasattr(self.metadata, key):
261                      setattr(self.metadata, key, val)
262                  elif hasattr(self, key):
263                      setattr(self, key, val)
264                  else:
265                      msg = "Unknown distribution option: %s" % repr(key)
266                      warnings.warn(msg)
267  
268          # no-user-cfg is handled before other command line args
269          # because other args override the config files, and this
270          # one is needed before we can load the config files.
271          # If attrs['script_args'] wasn't passed, assume false.
272          #
273          # This also make sure we just look at the global options
274          self.want_user_cfg = True
275  
276          if self.script_args is not None:
277              for arg in self.script_args:
278                  if not arg.startswith('-'):
279                      break
280                  if arg == '--no-user-cfg':
281                      self.want_user_cfg = False
282                      break
283  
284          self.finalize_options()
285  
286      def get_option_dict(self, command):
287          """Get the option dictionary for a given command.  If that 
288          command's option dictionary hasn't been created yet, then create it 
289          and return the new dictionary; otherwise, return the existing 
290          option dictionary. 
291          """
292          dict = self.command_options.get(command)
293          if dict is None:
294              dict = self.command_options[command] = {}
295          return dict

297      def dump_option_dicts(self, header=None, commands=None, indent=""):
298          from pprint import pformat
299  
300          if commands is None:  # dump all command option dicts
301              commands = sorted(self.command_options.keys())
302  
303          if header is not None:
304              self.announce(indent + header)
305              indent = indent + "  "
306  
307          if not commands:
308              self.announce(indent + "no commands known yet")
309              return
310  
311          for cmd_name in commands:
312              opt_dict = self.command_options.get(cmd_name)
313              if opt_dict is None:
314                  self.announce(indent + "no option dict for '%s' command" % cmd_name)
315              else:
316                  self.announce(indent + "option dict for '%s' command:" % cmd_name)
317                  out = pformat(opt_dict)
318                  for line in out.split('\n'):
319                      self.announce(indent + "  " + line)

323      def find_config_files(self):
324          """Find as many configuration files as should be processed for this 
325          platform, and return a list of filenames in the order in which they 
326          should be parsed.  The filenames returned are guaranteed to exist 
327          (modulo nasty race conditions). 
328   
329          There are multiple possible config files: 
330          - distutils.cfg in the Distutils installation directory (i.e. 
331            where the top-level Distutils __inst__.py file lives) 
332          - a file in the user's home directory named .pydistutils.cfg 
333            on Unix and pydistutils.cfg on Windows/Mac; may be disabled 
334            with the ``--no-user-cfg`` option 
335          - setup.cfg in the current directory 
336          - a file named by an environment variable 
337          """
338          check_environ()
339          files = [str(path) for path in self._gen_paths() if os.path.isfile(path)]
340  
341          if DEBUG:
342              self.announce("using config files: %s" % ', '.join(files))
343  
344          return files

346      def _gen_paths(self):
347          # The system-wide Distutils config file
348          sys_dir = pathlib.Path(sys.modules['distutils'].__file__).parent
349          yield sys_dir / "distutils.cfg"
350  
351          # The per-user config file
352          prefix = '.' * (os.name == 'posix')
353          filename = prefix + 'pydistutils.cfg'
354          if self.want_user_cfg:
355              yield pathlib.Path('~').expanduser() / filename
356  
357          # All platforms support local setup.cfg
358          yield pathlib.Path('setup.cfg')
359  
360          # Additional config indicated in the environment
361          with contextlib.suppress(TypeError):
362              yield pathlib.Path(os.getenv("DIST_EXTRA_CONFIG"))

364      def parse_config_files(self, filenames=None):  # noqa: C901
365          from configparser import ConfigParser
366  
367          # Ignore install directory options if we have a venv
368          if sys.prefix != sys.base_prefix:
369              ignore_options = [
370                  'install-base',
371                  'install-platbase',
372                  'install-lib',
373                  'install-platlib',
374                  'install-purelib',
375                  'install-headers',
376                  'install-scripts',
377                  'install-data',
378                  'prefix',
379                  'exec-prefix',
380                  'home',
381                  'user',
382                  'root',
383              ]
384          else:
385              ignore_options = []
386  
387          ignore_options = frozenset(ignore_options)
388  
389          if filenames is None:
390              filenames = self.find_config_files()
391  
392          if DEBUG:
393              self.announce("Distribution.parse_config_files():")
394  
395          parser = ConfigParser()
396          for filename in filenames:
397              if DEBUG:
398                  self.announce("  reading %s" % filename)
399              parser.read(filename, encoding='utf-8')
400              for section in parser.sections():
401                  options = parser.options(section)
402                  opt_dict = self.get_option_dict(section)
403  
404                  for opt in options:
405                      if opt != '__name__' and opt not in ignore_options:
406                          val = parser.get(section, opt)
407                          opt = opt.replace('-', '_')
408                          opt_dict[opt] = (filename, val)
409  
410              # Make the ConfigParser forget everything (so we retain
411              # the original filenames that options come from)
412              parser.__init__()
413  
414          # If there was a "global" section in the config file, use it
415          # to set Distribution options.
416  
417          if 'global' in self.command_options:
418              for opt, (_src, val) in self.command_options['global'].items():
419                  alias = self.negative_opt.get(opt)
420                  try:
421                      if alias:
422                          setattr(self, alias, not strtobool(val))
423                      elif opt in ('verbose', 'dry_run'):  # ugh!
424                          setattr(self, opt, strtobool(val))
425                      else:
426                          setattr(self, opt, val)
427                  except ValueError as msg:
428                      raise DistutilsOptionError(msg)

432      def parse_command_line(self):
433          """Parse the setup script's command line, taken from the 
434          'script_args' instance attribute (which defaults to 'sys.argv[1:]' 
435          -- see 'setup()' in core.py).  This list is first processed for 
436          "global options" -- options that set attributes of the Distribution 
437          instance.  Then, it is alternately scanned for Distutils commands 
438          and options for that command.  Each new command terminates the 
439          options for the previous command.  The allowed options for a 
440          command are determined by the 'user_options' attribute of the 
441          command class -- thus, we have to be able to load command classes 
442          in order to parse the command line.  Any error in that 'options' 
443          attribute raises DistutilsGetoptError; any error on the 
444          command-line raises DistutilsArgError.  If no Distutils commands 
445          were found on the command line, raises DistutilsArgError.  Return 
446          true if command-line was successfully parsed and we should carry 
447          on with executing commands; false if no errors but we shouldn't 
448          execute commands (currently, this only happens if user asks for 
449          help). 
450          """
451          #
452          # We now have enough information to show the Macintosh dialog
453          # that allows the user to interactively specify the "command line".
454          #
455          toplevel_options = self._get_toplevel_options()
456  
457          # We have to parse the command line a bit at a time -- global
458          # options, then the first command, then its options, and so on --
459          # because each command will be handled by a different class, and
460          # the options that are valid for a particular class aren't known
461          # until we have loaded the command class, which doesn't happen
462          # until we know what the command is.
463  
464          self.commands = []
465          parser = FancyGetopt(toplevel_options + self.display_options)
466          parser.set_negative_aliases(self.negative_opt)
467          parser.set_aliases({'licence': 'license'})
468          args = parser.getopt(args=self.script_args, object=self)
469          option_order = parser.get_option_order()
470          logging.getLogger().setLevel(logging.WARN - 10 * self.verbose)
471  
472          # for display options we return immediately
473          if self.handle_display_options(option_order):
474              return
475          while args:
476              args = self._parse_command_opts(parser, args)
477              if args is None:  # user asked for help (and got it)
478                  return
479  
480          # Handle the cases of --help as a "global" option, ie.
481          # "setup.py --help" and "setup.py --help command ...".  For the
482          # former, we show global options (--verbose, --dry-run, etc.)
483          # and display-only options (--name, --version, etc.); for the
484          # latter, we omit the display-only options and show help for
485          # each command listed on the command line.
486          if self.help:
487              self._show_help(
488                  parser, display_options=len(self.commands) == 0, commands=self.commands
489              )
490              return
491  
492          # Oops, no commands found -- an end-user error
493          if not self.commands:
494              raise DistutilsArgError("no commands supplied")
495  
496          # All is well: return true
497          return True
498  
499      def _get_toplevel_options(self):
500          """Return the non-display options recognized at the top level. 
501   
502          This includes options that are recognized *only* at the top 
503          level as well as options recognized for commands. 
504          """
505          return self.global_options + [
506              (
507                  "command-packages=",
508                  None,
509                  "list of packages that provide distutils commands",
510              ),
511          ]

513      def _parse_command_opts(self, parser, args):  # noqa: C901
514          """Parse the command-line options for a single command. 
515          'parser' must be a FancyGetopt instance; 'args' must be the list 
516          of arguments, starting with the current command (whose options 
517          we are about to parse).  Returns a new version of 'args' with 
518          the next command at the front of the list; will be the empty 
519          list if there are no more commands on the command line.  Returns 
520          None if the user asked for help on this command. 
521          """
522          # late import because of mutual dependence between these modules
523          from distutils.cmd import Command
524  
525          # Pull the current command from the head of the command line
526          command = args[0]
527          if not command_re.match(command):
528              raise SystemExit("invalid command name '%s'" % command)
529          self.commands.append(command)
530  
531          # Dig up the command class that implements this command, so we
532          # 1) know that it's a valid command, and 2) know which options
533          # it takes.
534          try:
535              cmd_class = self.get_command_class(command)
536          except DistutilsModuleError as msg:
537              raise DistutilsArgError(msg)
538  
539          # Require that the command class be derived from Command -- want
540          # to be sure that the basic "command" interface is implemented.
541          if not issubclass(cmd_class, Command):
542              raise DistutilsClassError(
543                  "command class %s must subclass Command" % cmd_class
544              )
545  
546          # Also make sure that the command object provides a list of its
547          # known options.
548          if not (
549              hasattr(cmd_class, 'user_options')
550              and isinstance(cmd_class.user_options, list)
551          ):
552              msg = (
553                  "command class %s must provide "
554                  "'user_options' attribute (a list of tuples)"
555              )
556              raise DistutilsClassError(msg % cmd_class)
557  
558          # If the command class has a list of negative alias options,
559          # merge it in with the global negative aliases.
560          negative_opt = self.negative_opt
561          if hasattr(cmd_class, 'negative_opt'):
562              negative_opt = negative_opt.copy()
563              negative_opt.update(cmd_class.negative_opt)
564  
565          # Check for help_options in command class.  They have a different
566          # format (tuple of four) so we need to preprocess them here.
567          if hasattr(cmd_class, 'help_options') and isinstance(
568              cmd_class.help_options, list
569          ):
570              help_options = fix_help_options(cmd_class.help_options)
571          else:
572              help_options = []
573  
574          # All commands support the global options too, just by adding
575          # in 'global_options'.
576          parser.set_option_table(
577              self.global_options + cmd_class.user_options + help_options
578          )
579          parser.set_negative_aliases(negative_opt)
580          (args, opts) = parser.getopt(args[1:])
581          if hasattr(opts, 'help') and opts.help:
582              self._show_help(parser, display_options=0, commands=[cmd_class])
583              return
584  
585          if hasattr(cmd_class, 'help_options') and isinstance(
586              cmd_class.help_options, list
587          ):
588              help_option_found = 0
589              for help_option, _short, _desc, func in cmd_class.help_options:
590                  if hasattr(opts, parser.get_attr_name(help_option)):
591                      help_option_found = 1
592                      if callable(func):
593                          func()
594                      else:
595                          raise DistutilsClassError(
596                              f"invalid help function {func!r} for help option '{help_option}': "
597                              "must be a callable object (function, etc.)"
598                          )
599  
600              if help_option_found:
601                  return
602  
603          # Put the options from the command-line into their official
604          # holding pen, the 'command_options' dictionary.
605          opt_dict = self.get_option_dict(command)
606          for name, value in vars(opts).items():
607              opt_dict[name] = ("command line", value)
608  
609          return args

611      def finalize_options(self):
612          """Set final values for all the options on the Distribution 
613          instance, analogous to the .finalize_options() method of Command 
614          objects. 
615          """
616          for attr in ('keywords', 'platforms'):
617              value = getattr(self.metadata, attr)
618              if value is None:
619                  continue
620              if isinstance(value, str):
621                  value = [elm.strip() for elm in value.split(',')]
622                  setattr(self.metadata, attr, value)

624      def _show_help(
625          self, parser, global_options=1, display_options=1, commands: Iterable = ()
626      ):
627          """Show help for the setup script command-line in the form of 
628          several lists of command-line options.  'parser' should be a 
629          FancyGetopt instance; do not expect it to be returned in the 
630          same state, as its option table will be reset to make it 
631          generate the correct help text. 
632   
633          If 'global_options' is true, lists the global options: 
634          --verbose, --dry-run, etc.  If 'display_options' is true, lists 
635          the "display-only" options: --name, --version, etc.  Finally, 
636          lists per-command help for every command name or command class 
637          in 'commands'. 
638          """
639          # late import because of mutual dependence between these modules
640          from distutils.cmd import Command
641          from distutils.core import gen_usage
642  
643          if global_options:
644              if display_options:
645                  options = self._get_toplevel_options()
646              else:
647                  options = self.global_options
648              parser.set_option_table(options)
649              parser.print_help(self.common_usage + "\nGlobal options:")
650              print()
651  
652          if display_options:
653              parser.set_option_table(self.display_options)
654              parser.print_help(
655                  "Information display options (just display "
656                  + "information, ignore any commands)"
657              )
658              print()
659  
660          for command in self.commands:
661              if isinstance(command, type) and issubclass(command, Command):
662                  klass = command
663              else:
664                  klass = self.get_command_class(command)
665              if hasattr(klass, 'help_options') and isinstance(klass.help_options, list):
666                  parser.set_option_table(
667                      klass.user_options + fix_help_options(klass.help_options)
668                  )
669              else:
670                  parser.set_option_table(klass.user_options)
671              parser.print_help("Options for '%s' command:" % klass.__name__)
672              print()
673  
674          print(gen_usage(self.script_name))

676      def handle_display_options(self, option_order):
677          """If there were any non-global "display-only" options 
678          (--help-commands or the metadata display options) on the command 
679          line, display the requested info and return true; else return 
680          false. 
681          """
682          from distutils.core import gen_usage
683  
684          # User just wants a list of commands -- we'll print it out and stop
685          # processing now (ie. if they ran "setup --help-commands foo bar",
686          # we ignore "foo bar").
687          if self.help_commands:
688              self.print_commands()
689              print()
690              print(gen_usage(self.script_name))
691              return 1
692  
693          # If user supplied any of the "display metadata" options, then
694          # display that metadata in the order in which the user supplied the
695          # metadata options.
696          any_display_options = 0
697          is_display_option = {}
698          for option in self.display_options:
699              is_display_option[option[0]] = 1
700  
701          for opt, val in option_order:
702              if val and is_display_option.get(opt):
703                  opt = translate_longopt(opt)
704                  value = getattr(self.metadata, "get_" + opt)()
705                  if opt in ('keywords', 'platforms'):
706                      print(','.join(value))
707                  elif opt in ('classifiers', 'provides', 'requires', 'obsoletes'):
708                      print('\n'.join(value))
709                  else:
710                      print(value)
711                  any_display_options = 1
712  
713          return any_display_options

732      def print_commands(self):
733          """Print out a help message listing all available commands with a 
734          description of each.  The list is divided into "standard commands" 
735          (listed in distutils.command.__all__) and "extra commands" 
736          (mentioned in self.cmdclass, but not a standard command).  The 
737          descriptions come from the command class attribute 
738          'description'. 
739          """
740          import distutils.command
741  
742          std_commands = distutils.command.__all__
743          is_std = {}
744          for cmd in std_commands:
745              is_std[cmd] = 1
746  
747          extra_commands = []
748          for cmd in self.cmdclass.keys():
749              if not is_std.get(cmd):
750                  extra_commands.append(cmd)
751  
752          max_length = 0
753          for cmd in std_commands + extra_commands:
754              if len(cmd) > max_length:
755                  max_length = len(cmd)
756  
757          self.print_command_list(std_commands, "Standard commands", max_length)
758          if extra_commands:
759              print()
760              self.print_command_list(extra_commands, "Extra commands", max_length)

798          """Return a list of packages from which commands are loaded."""

809      def get_command_class(self, command):
810          """Return the class that implements the Distutils command named by 
811          'command'.  First we check the 'cmdclass' dictionary; if the 
812          command is mentioned there, we fetch the class object from the 
813          dictionary and return it.  Otherwise we load the command module 
814          ("distutils.command." + command) and fetch the command class from 
815          the module.  The loaded class is also stored in 'cmdclass' 
816          to speed future calls to 'get_command_class()'. 
817   
818          Raises DistutilsModuleError if the expected module could not be 
819          found, or if that module does not define the expected class. 
820          """
821          klass = self.cmdclass.get(command)
822          if klass:
823              return klass
824  
825          for pkgname in self.get_command_packages():
826              module_name = f"{pkgname}.{command}"
827              klass_name = command
828  
829              try:
830                  __import__(module_name)
831                  module = sys.modules[module_name]
832              except ImportError:
833                  continue
834  
835              try:
836                  klass = getattr(module, klass_name)
837              except AttributeError:
838                  raise DistutilsModuleError(
839                      f"invalid command '{command}' (no class '{klass_name}' in module '{module_name}')"
840                  )
841  
842              self.cmdclass[command] = klass
843              return klass
844  
845          raise DistutilsModuleError("invalid command '%s'" % command)

847      def get_command_obj(self, command, create=1):
848          """Return the command object for 'command'.  Normally this object 
849          is cached on a previous call to 'get_command_obj()'; if no command 
850          object for 'command' is in the cache, then we either create and 
851          return it (if 'create' is true) or return None. 
852          """
853          cmd_obj = self.command_obj.get(command)
854          if not cmd_obj and create:
855              if DEBUG:
856                  self.announce(
857                      "Distribution.get_command_obj(): "
858                      "creating '%s' command object" % command
859                  )
860  
861              klass = self.get_command_class(command)
862              cmd_obj = self.command_obj[command] = klass(self)
863              self.have_run[command] = 0
864  
865              # Set any options that were supplied in config files
866              # or on the command line.  (NB. support for error
867              # reporting is lame here: any errors aren't reported
868              # until 'finalize_options()' is called, which means
869              # we won't report the source of the error.)
870              options = self.command_options.get(command)
871              if options:
872                  self._set_command_options(cmd_obj, options)
873  
874          return cmd_obj

876      def _set_command_options(self, command_obj, option_dict=None):  # noqa: C901
877          """Set the options for 'command_obj' from 'option_dict'.  Basically 
878          this means copying elements of a dictionary ('option_dict') to 
879          attributes of an instance ('command'). 
880   
881          'command_obj' must be a Command instance.  If 'option_dict' is not 
882          supplied, uses the standard option dictionary for this command 
883          (from 'self.command_options'). 
884          """
885          command_name = command_obj.get_command_name()
886          if option_dict is None:
887              option_dict = self.get_option_dict(command_name)
888  
889          if DEBUG:
890              self.announce("  setting options for '%s' command:" % command_name)
891          for option, (source, value) in option_dict.items():
892              if DEBUG:
893                  self.announce(f"    {option} = {value} (from {source})")
894              try:
895                  bool_opts = [translate_longopt(o) for o in command_obj.boolean_options]
896              except AttributeError:
897                  bool_opts = []
898              try:
899                  neg_opt = command_obj.negative_opt
900              except AttributeError:
901                  neg_opt = {}
902  
903              try:
904                  is_string = isinstance(value, str)
905                  if option in neg_opt and is_string:
906                      setattr(command_obj, neg_opt[option], not strtobool(value))
907                  elif option in bool_opts and is_string:
908                      setattr(command_obj, option, strtobool(value))
909                  elif hasattr(command_obj, option):
910                      setattr(command_obj, option, value)
911                  else:
912                      raise DistutilsOptionError(
913                          f"error in {source}: command '{command_name}' has no such option '{option}'"
914                      )
915              except ValueError as msg:
916                  raise DistutilsOptionError(msg)

918      def reinitialize_command(self, command, reinit_subcommands=0):
919          """Reinitializes a command to the state it was in when first 
920          returned by 'get_command_obj()': ie., initialized but not yet 
921          finalized.  This provides the opportunity to sneak option 
922          values in programmatically, overriding or supplementing 
923          user-supplied values from the config files and command line. 
924          You'll have to re-finalize the command object (by calling 
925          'finalize_options()' or 'ensure_finalized()') before using it for 
926          real. 
927   
928          'command' should be a command name (string) or command object.  If 
929          'reinit_subcommands' is true, also reinitializes the command's 
930          sub-commands, as declared by the 'sub_commands' class attribute (if 
931          it has one).  See the "install" command for an example.  Only 
932          reinitializes the sub-commands that actually matter, ie. those 
933          whose test predicates return true. 
934   
935          Returns the reinitialized command object. 
936          """
937          from distutils.cmd import Command
938  
939          if not isinstance(command, Command):
940              command_name = command
941              command = self.get_command_obj(command_name)
942          else:
943              command_name = command.get_command_name()
944  
945          if not command.finalized:
946              return command
947          command.initialize_options()
948          command.finalized = 0
949          self.have_run[command_name] = 0
950          self._set_command_options(command)
951  
952          if reinit_subcommands:
953              for sub in command.get_sub_commands():
954                  self.reinitialize_command(sub, reinit_subcommands)
955  
956          return command

973      def run_command(self, command):
974          """Do whatever it takes to run a command (including nothing at all, 
975          if the command has already been run).  Specifically: if we have 
976          already created and run the command named by 'command', return 
977          silently without doing anything.  If the command named by 'command' 
978          doesn't even have a command object yet, create one.  Then invoke 
979          'run()' on that command object (or an existing one). 
980          """
981          # Already been here, done that? then return silently.
982          if self.have_run.get(command):
983              return
984  
985          log.info("running %s", command)
986          cmd_obj = self.get_command_obj(command)
987          cmd_obj.ensure_finalized()
988          cmd_obj.run()
989          self.have_run[command] = 1
