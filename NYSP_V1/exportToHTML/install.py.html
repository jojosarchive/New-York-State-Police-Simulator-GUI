96   def _load_sysconfig_schemes():
97       with contextlib.suppress(AttributeError):
98           return {
99               scheme: sysconfig.get_paths(scheme, expand=False)
100              for scheme in sysconfig.get_scheme_names()
101          }

104  def _load_schemes():
105      """ 
106      Extend default schemes with schemes from sysconfig. 
107      """
108  
109      sysconfig_schemes = _load_sysconfig_schemes() or {}
110  
111      return {
112          scheme: {
113              **INSTALL_SCHEMES.get(scheme, {}),
114              **sysconfig_schemes.get(scheme, {}),
115          }
116          for scheme in set(itertools.chain(INSTALL_SCHEMES, sysconfig_schemes))
117      }

127  def _select_scheme(ob, name):
128      scheme = _inject_headers(name, _load_scheme(_resolve_scheme(name)))
129      vars(ob).update(_remove_set(ob, _scheme_attrs(scheme)))

132  def _remove_set(ob, attrs):
133      """ 
134      Include only attrs that are None in ob. 
135      """
136      return {key: value for key, value in attrs.items() if getattr(ob, key) is None}

139  def _resolve_scheme(name):
140      os_name, sep, key = name.partition('_')
141      try:
142          resolved = sysconfig.get_preferred_scheme(key)
143      except Exception:
144          resolved = fw.scheme(_pypy_hack(name))
145      return resolved
146  
147  
148  def _load_scheme(name):
149      return _load_schemes()[name]

152  def _inject_headers(name, scheme):
153      """ 
154      Given a scheme name and the resolved scheme, 
155      if the scheme does not include headers, resolve 
156      the fallback scheme for the name and use headers 
157      from it. pypa/distutils#88 
158      """
159      # Bypass the preferred scheme, which may not
160      # have defined headers.
161      fallback = _load_scheme(_pypy_hack(name))
162      scheme.setdefault('headers', fallback['headers'])
163      return scheme

166  def _scheme_attrs(scheme):
167      """Resolve install directories by applying the install schemes."""
168      return {f'install_{key}': scheme[key] for key in SCHEME_KEYS}

171  def _pypy_hack(name):
172      PY37 = sys.version_info < (3, 8)
173      old_pypy = hasattr(sys, 'pypy_version_info') and PY37
174      prefix = not name.endswith(('_user', '_home'))
175      pypy_name = 'pypy' + '_nt' * (os.name == 'nt')
176      return pypy_name if old_pypy and prefix else name

254      def initialize_options(self):
255          """Initializes options."""
256          # High-level options: these select both an installation base
257          # and scheme.
258          self.prefix = None
259          self.exec_prefix = None
260          self.home = None
261          self.user = 0
262  
263          # These select only the installation base; it's up to the user to
264          # specify the installation scheme (currently, that means supplying
265          # the --install-{platlib,purelib,scripts,data} options).
266          self.install_base = None
267          self.install_platbase = None
268          self.root = None
269  
270          # These options are the actual installation directories; if not
271          # supplied by the user, they are filled in using the installation
272          # scheme implied by prefix/exec-prefix/home and the contents of
273          # that installation scheme.
274          self.install_purelib = None  # for pure module distributions
275          self.install_platlib = None  # non-pure (dists w/ extensions)
276          self.install_headers = None  # for C/C++ headers
277          self.install_lib = None  # set to either purelib or platlib
278          self.install_scripts = None
279          self.install_data = None
280          self.install_userbase = USER_BASE
281          self.install_usersite = USER_SITE
282  
283          self.compile = None
284          self.optimize = None
285  
286          # Deprecated
287          # These two are for putting non-packagized distributions into their
288          # own directory and creating a .pth file if it makes sense.
289          # 'extra_path' comes from the setup file; 'install_path_file' can
290          # be turned off if it makes no sense to install a .pth file.  (But
291          # better to install it uselessly than to guess wrong and not
292          # install it when it's necessary and would be used!)  Currently,
293          # 'install_path_file' is always true unless some outsider meddles
294          # with it.
295          self.extra_path = None
296          self.install_path_file = 1
297  
298          # 'force' forces installation, even if target files are not
299          # out-of-date.  'skip_build' skips running the "build" command,
300          # handy if you know it's not necessary.  'warn_dir' (which is *not*
301          # a user option, it's just there so the bdist_* commands can turn
302          # it off) determines whether we warn about installing to a
303          # directory not in sys.path.
304          self.force = 0
305          self.skip_build = 0
306          self.warn_dir = 1
307  
308          # These are only here as a conduit from the 'build' command to the
309          # 'install_*' commands that do the real work.  ('build_base' isn't
310          # actually used anywhere, but it might be useful in future.)  They
311          # are not user options, because if the user told the install
312          # command where the build directory is, that wouldn't affect the
313          # build command.
314          self.build_base = None
315          self.build_lib = None
316  
317          # Not defined yet because we don't know anything about
318          # documentation yet.
319          # self.install_man = None
320          # self.install_html = None
321          # self.install_info = None
322  
323          self.record = None

325      # -- Option finalizing methods -------------------------------------
326      # (This is rather more involved than for most commands,
327      # because this is where the policy for installing third-
328      # party Python modules on various platforms given a wide
329      # array of user input is decided.  Yes, it's quite complex!)
330  
331      def finalize_options(self):  # noqa: C901
332          """Finalizes options."""
333          # This method (and its helpers, like 'finalize_unix()',
334          # 'finalize_other()', and 'select_scheme()') is where the default
335          # installation directories for modules, extension modules, and
336          # anything else we care to install from a Python module
337          # distribution.  Thus, this code makes a pretty important policy
338          # statement about how third-party stuff is added to a Python
339          # installation!  Note that the actual work of installation is done
340          # by the relatively simple 'install_*' commands; they just take
341          # their orders from the installation directory options determined
342          # here.
343  
344          # Check for errors/inconsistencies in the options; first, stuff
345          # that's wrong on any platform.
346  
347          if (self.prefix or self.exec_prefix or self.home) and (
348              self.install_base or self.install_platbase
349          ):
350              raise DistutilsOptionError(
351                  "must supply either prefix/exec-prefix/home or "
352                  + "install-base/install-platbase -- not both"
353              )
354  
355          if self.home and (self.prefix or self.exec_prefix):
356              raise DistutilsOptionError(
357                  "must supply either home or prefix/exec-prefix -- not both"
358              )
359  
360          if self.user and (
361              self.prefix
362              or self.exec_prefix
363              or self.home
364              or self.install_base
365              or self.install_platbase
366          ):
367              raise DistutilsOptionError(
368                  "can't combine user with prefix, "
369                  "exec_prefix/home, or install_(plat)base"
370              )
371  
372          # Next, stuff that's wrong (or dubious) only on certain platforms.
373          if os.name != "posix":
374              if self.exec_prefix:
375                  self.warn("exec-prefix option ignored on this platform")
376                  self.exec_prefix = None
377  
378          # Now the interesting logic -- so interesting that we farm it out
379          # to other methods.  The goal of these methods is to set the final
380          # values for the install_{lib,scripts,data,...}  options, using as
381          # input a heady brew of prefix, exec_prefix, home, install_base,
382          # install_platbase, user-supplied versions of
383          # install_{purelib,platlib,lib,scripts,data,...}, and the
384          # install schemes.  Phew!
385  
386          self.dump_dirs("pre-finalize_{unix,other}")
387  
388          if os.name == 'posix':
389              self.finalize_unix()
390          else:
391              self.finalize_other()
392  
393          self.dump_dirs("post-finalize_{unix,other}()")
394  
395          # Expand configuration variables, tilde, etc. in self.install_base
396          # and self.install_platbase -- that way, we can use $base or
397          # $platbase in the other installation directories and not worry
398          # about needing recursive variable expansion (shudder).
399  
400          py_version = sys.version.split()[0]
401          (prefix, exec_prefix) = get_config_vars('prefix', 'exec_prefix')
402          try:
403              abiflags = sys.abiflags
404          except AttributeError:
405              # sys.abiflags may not be defined on all platforms.
406              abiflags = ''
407          local_vars = {
408              'dist_name': self.distribution.get_name(),
409              'dist_version': self.distribution.get_version(),
410              'dist_fullname': self.distribution.get_fullname(),
411              'py_version': py_version,
412              'py_version_short': '%d.%d' % sys.version_info[:2],
413              'py_version_nodot': '%d%d' % sys.version_info[:2],
414              'sys_prefix': prefix,
415              'prefix': prefix,
416              'sys_exec_prefix': exec_prefix,
417              'exec_prefix': exec_prefix,
418              'abiflags': abiflags,
419              'platlibdir': getattr(sys, 'platlibdir', 'lib'),
420              'implementation_lower': _get_implementation().lower(),
421              'implementation': _get_implementation(),
422          }
423  
424          # vars for compatibility on older Pythons
425          compat_vars = dict(
426              # Python 3.9 and earlier
427              py_version_nodot_plat=getattr(sys, 'winver', '').replace('.', ''),
428          )
429  
430          if HAS_USER_SITE:
431              local_vars['userbase'] = self.install_userbase
432              local_vars['usersite'] = self.install_usersite
433  
434          self.config_vars = _collections.DictStack([
435              fw.vars(),
436              compat_vars,
437              sysconfig.get_config_vars(),
438              local_vars,
439          ])
440  
441          self.expand_basedirs()
442  
443          self.dump_dirs("post-expand_basedirs()")
444  
445          # Now define config vars for the base directories so we can expand
446          # everything else.
447          local_vars['base'] = self.install_base
448          local_vars['platbase'] = self.install_platbase
449  
450          if DEBUG:
451              from pprint import pprint
452  
453              print("config vars:")
454              pprint(dict(self.config_vars))
455  
456          # Expand "~" and configuration variables in the installation
457          # directories.
458          self.expand_dirs()
459  
460          self.dump_dirs("post-expand_dirs()")
461  
462          # Create directories in the home dir:
463          if self.user:
464              self.create_home_path()
465  
466          # Pick the actual directory to install all modules to: either
467          # install_purelib or install_platlib, depending on whether this
468          # module distribution is pure or not.  Of course, if the user
469          # already specified install_lib, use their selection.
470          if self.install_lib is None:
471              if self.distribution.has_ext_modules():  # has extensions: non-pure
472                  self.install_lib = self.install_platlib
473              else:
474                  self.install_lib = self.install_purelib
475  
476          # Convert directories from Unix /-separated syntax to the local
477          # convention.
478          self.convert_paths(
479              'lib',
480              'purelib',
481              'platlib',
482              'scripts',
483              'data',
484              'headers',
485              'userbase',
486              'usersite',
487          )
488  
489          # Deprecated
490          # Well, we're not actually fully completely finalized yet: we still
491          # have to deal with 'extra_path', which is the hack for allowing
492          # non-packagized module distributions (hello, Numerical Python!) to
493          # get their own directories.
494          self.handle_extra_path()
495          self.install_libbase = self.install_lib  # needed for .pth file
496          self.install_lib = os.path.join(self.install_lib, self.extra_dirs)
497  
498          # If a new root directory was supplied, make all the installation
499          # dirs relative to it.
500          if self.root is not None:
501              self.change_roots(
502                  'libbase', 'lib', 'purelib', 'platlib', 'scripts', 'data', 'headers'
503              )
504  
505          self.dump_dirs("after prepending root")
506  
507          # Find out the build directories, ie. where to install from.
508          self.set_undefined_options(
509              'build', ('build_base', 'build_base'), ('build_lib', 'build_lib')
510          )
511  
512          # Punt on doc directories for now -- after all, we're punting on
513          # documentation completely!
514  
515      def dump_dirs(self, msg):
516          """Dumps the list of user options."""
517          if not DEBUG:
518              return
519          from ..fancy_getopt import longopt_xlate
520  
521          log.debug(msg + ":")
522          for opt in self.user_options:
523              opt_name = opt[0]
524              if opt_name[-1] == "=":
525                  opt_name = opt_name[0:-1]
526              if opt_name in self.negative_opt:
527                  opt_name = self.negative_opt[opt_name]
528                  opt_name = opt_name.translate(longopt_xlate)
529                  val = not getattr(self, opt_name)
530              else:
531                  opt_name = opt_name.translate(longopt_xlate)
532                  val = getattr(self, opt_name)
533              log.debug("  %s: %s", opt_name, val)
534  
535      def finalize_unix(self):
536          """Finalizes options for posix platforms."""
537          if self.install_base is not None or self.install_platbase is not None:
538              incomplete_scheme = (
539                  (
540                      self.install_lib is None
541                      and self.install_purelib is None
542                      and self.install_platlib is None
543                  )
544                  or self.install_headers is None
545                  or self.install_scripts is None
546                  or self.install_data is None
547              )
548              if incomplete_scheme:
549                  raise DistutilsOptionError(
550                      "install-base or install-platbase supplied, but "
551                      "installation scheme is incomplete"
552                  )
553              return
554  
555          if self.user:
556              if self.install_userbase is None:
557                  raise DistutilsPlatformError("User base directory is not specified")
558              self.install_base = self.install_platbase = self.install_userbase
559              self.select_scheme("posix_user")
560          elif self.home is not None:
561              self.install_base = self.install_platbase = self.home
562              self.select_scheme("posix_home")
563          else:
564              if self.prefix is None:
565                  if self.exec_prefix is not None:
566                      raise DistutilsOptionError(
567                          "must not supply exec-prefix without prefix"
568                      )
569  
570                  # Allow Fedora to add components to the prefix
571                  _prefix_addition = getattr(sysconfig, '_prefix_addition', "")
572  
573                  self.prefix = os.path.normpath(sys.prefix) + _prefix_addition
574                  self.exec_prefix = os.path.normpath(sys.exec_prefix) + _prefix_addition
575  
576              else:
577                  if self.exec_prefix is None:
578                      self.exec_prefix = self.prefix
579  
580              self.install_base = self.prefix
581              self.install_platbase = self.exec_prefix
582              self.select_scheme("posix_prefix")
583  
584      def finalize_other(self):
585          """Finalizes options for non-posix platforms"""
586          if self.user:
587              if self.install_userbase is None:
588                  raise DistutilsPlatformError("User base directory is not specified")
589              self.install_base = self.install_platbase = self.install_userbase
590              self.select_scheme(os.name + "_user")
591          elif self.home is not None:
592              self.install_base = self.install_platbase = self.home
593              self.select_scheme("posix_home")
594          else:
595              if self.prefix is None:
596                  self.prefix = os.path.normpath(sys.prefix)
597  
598              self.install_base = self.install_platbase = self.prefix
599              try:
600                  self.select_scheme(os.name)
601              except KeyError:
602                  raise DistutilsPlatformError(
603                      "I don't know how to install stuff on '%s'" % os.name
604                  )

606      def select_scheme(self, name):
607          _select_scheme(self, name)

609      def _expand_attrs(self, attrs):
610          for attr in attrs:
611              val = getattr(self, attr)
612              if val is not None:
613                  if os.name in ('posix', 'nt'):
614                      val = os.path.expanduser(val)
615                  val = subst_vars(val, self.config_vars)
616                  setattr(self, attr, val)
617  
618      def expand_basedirs(self):
619          """Calls `os.path.expanduser` on install_base, install_platbase and 
620          root."""
621          self._expand_attrs(['install_base', 'install_platbase', 'root'])
622  
623      def expand_dirs(self):
624          """Calls `os.path.expanduser` on install dirs."""
625          self._expand_attrs([
626              'install_purelib',
627              'install_platlib',
628              'install_lib',
629              'install_headers',
630              'install_scripts',
631              'install_data',
632          ])
633  
634      def convert_paths(self, *names):
635          """Call `convert_path` over `names`."""
636          for name in names:
637              attr = "install_" + name
638              setattr(self, attr, convert_path(getattr(self, attr)))

640      def handle_extra_path(self):
641          """Set `path_file` and `extra_dirs` using `extra_path`."""
642          if self.extra_path is None:
643              self.extra_path = self.distribution.extra_path
644  
645          if self.extra_path is not None:
646              log.warning(
647                  "Distribution option extra_path is deprecated. "
648                  "See issue27919 for details."
649              )
650              if isinstance(self.extra_path, str):
651                  self.extra_path = self.extra_path.split(',')
652  
653              if len(self.extra_path) == 1:
654                  path_file = extra_dirs = self.extra_path[0]
655              elif len(self.extra_path) == 2:
656                  path_file, extra_dirs = self.extra_path
657              else:
658                  raise DistutilsOptionError(
659                      "'extra_path' option must be a list, tuple, or "
660                      "comma-separated string with 1 or 2 elements"
661                  )
662  
663              # convert to local form in case Unix notation used (as it
664              # should be in setup scripts)
665              extra_dirs = convert_path(extra_dirs)
666          else:
667              path_file = None
668              extra_dirs = ''
669  
670          # XXX should we warn if path_file and not extra_dirs? (in which
671          # case the path file would be harmless but pointless)
672          self.path_file = path_file
673          self.extra_dirs = extra_dirs
674  
675      def change_roots(self, *names):
676          """Change the install directories pointed by name using root."""
677          for name in names:
678              attr = "install_" + name
679              setattr(self, attr, change_root(self.root, getattr(self, attr)))
680  
681      def create_home_path(self):
682          """Create directories under ~."""
683          if not self.user:
684              return
685          home = convert_path(os.path.expanduser("~"))
686          for _name, path in self.config_vars.items():
687              if str(path).startswith(home) and not os.path.isdir(path):
688                  self.debug_print("os.makedirs('%s', 0o700)" % path)
689                  os.makedirs(path, 0o700)
690  
691      # -- Command execution methods -------------------------------------
692  
693      def run(self):
694          """Runs the command."""
695          # Obviously have to build before we can install
696          if not self.skip_build:
697              self.run_command('build')
698              # If we built for any other platform, we can't install.
699              build_plat = self.distribution.get_command_obj('build').plat_name
700              # check warn_dir - it is a clue that the 'install' is happening
701              # internally, and not to sys.path, so we don't check the platform
702              # matches what we are running.
703              if self.warn_dir and build_plat != get_platform():
704                  raise DistutilsPlatformError("Can't install when cross-compiling")
705  
706          # Run all sub-commands (at least those that need to be run)
707          for cmd_name in self.get_sub_commands():
708              self.run_command(cmd_name)
709  
710          if self.path_file:
711              self.create_path_file()
712  
713          # write list of installed files, if requested.
714          if self.record:
715              outputs = self.get_outputs()
716              if self.root:  # strip any package prefix
717                  root_len = len(self.root)
718                  for counter in range(len(outputs)):
719                      outputs[counter] = outputs[counter][root_len:]
720              self.execute(
721                  write_file,
722                  (self.record, outputs),
723                  "writing list of installed files to '%s'" % self.record,
724              )
725  
726          sys_path = map(os.path.normpath, sys.path)
727          sys_path = map(os.path.normcase, sys_path)
728          install_lib = os.path.normcase(os.path.normpath(self.install_lib))
729          if (
730              self.warn_dir
731              and not (self.path_file and self.install_path_file)
732              and install_lib not in sys_path
733          ):
734              log.debug(
735                  (
736                      "modules installed to '%s', which is not in "
737                      "Python's module search path (sys.path) -- "
738                      "you'll have to change the search path yourself"
739                  ),
740                  self.install_lib,
741              )
742  
743      def create_path_file(self):
744          """Creates the .pth file"""
745          filename = os.path.join(self.install_libbase, self.path_file + ".pth")
746          if self.install_path_file:
747              self.execute(
748                  write_file, (filename, [self.extra_dirs]), "creating %s" % filename
749              )
750          else:
751              self.warn("path file '%s' not created" % filename)
752  
753      # -- Reporting methods ---------------------------------------------
754  
755      def get_outputs(self):
756          """Assembles the outputs of all the sub-commands."""
757          outputs = []
758          for cmd_name in self.get_sub_commands():
759              cmd = self.get_finalized_command(cmd_name)
760              # Add the contents of cmd.get_outputs(), ensuring
761              # that outputs doesn't contain duplicate entries
762              for filename in cmd.get_outputs():
763                  if filename not in outputs:
764                      outputs.append(filename)
765  
766          if self.path_file and self.install_path_file:
767              outputs.append(os.path.join(self.install_libbase, self.path_file + ".pth"))
768  
769          return outputs
770  
771      def get_inputs(self):
772          """Returns the inputs of all the sub-commands"""
773          # XXX gee, this looks familiar ;-(
774          inputs = []
775          for cmd_name in self.get_sub_commands():
776              cmd = self.get_finalized_command(cmd_name)
777              inputs.extend(cmd.get_inputs())
778  
779          return inputs

783      def has_lib(self):
784          """Returns true if the current distribution has any Python 
785          modules to install."""
786          return (
787              self.distribution.has_pure_modules() or self.distribution.has_ext_modules()
788          )

790      def has_headers(self):
791          """Returns true if the current distribution has any headers to

795      def has_scripts(self):
796          """Returns true if the current distribution has any scripts to. 
797          install."""
798          return self.distribution.has_scripts()

803          return self.distribution.has_data_files()
