12   # Starting with v4.3.5, the `ffpyplayer` package attempts to use `site.USER_BASE` in path manipulation functions.
13   # As frozen application runs with disabled `site`, the value of this variable is `None`, and causes path manipulation
14   # functions to raise an error. As a work-around, we set `site.USER_BASE` to an empty string, which is also what the
15   # fake `site` module available in PyInstaller prior to v5.5 did.
16   import site
17   
18   if site.USER_BASE is None:
19       site.USER_BASE = ''
