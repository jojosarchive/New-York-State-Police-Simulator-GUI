12   # The path to Qt's components may not default to the wheel layout for self-compiled PySide6 installations. Mandate the
13   # wheel layout. See ``utils/hooks/qt.py`` for more details.
14   
15   
16   def _pyi_rthook():
17       import os
18       import sys
19   
20       from _pyi_rth_utils import is_macos_app_bundle, prepend_path_to_environment_variable
21       from _pyi_rth_utils import qt as qt_rth_utils
22   
23       # Ensure this is the only Qt bindings package in the application.
24       qt_rth_utils.ensure_single_qt_bindings_package("PySide6")
25   
26       if sys.platform.startswith('win'):
27           pyqt_path = os.path.join(sys._MEIPASS, 'PySide6')
28       else:
29           pyqt_path = os.path.join(sys._MEIPASS, 'PySide6', 'Qt')
30   
31       os.environ['QT_PLUGIN_PATH'] = os.path.join(pyqt_path, 'plugins')
32   
33       if is_macos_app_bundle:
34           # Special handling for macOS .app bundles. To satisfy codesign requirements, we are forced to split `qml`
35           # directory into two parts; one that keeps only binaries (rooted in `Contents/Frameworks`) and one that keeps
36           # only data files (rooted in `Contents/Resources), with files from one directory tree being symlinked to the
37           # other to maintain illusion of a single mixed-content directory. As Qt seems to compute the identifier of its
38           # QML components based on location of the `qmldir` file w.r.t. the registered QML import paths, we need to
39           # register both paths, because the `qmldir` file for a component could be reached via either directory tree.
40           pyqt_path_res = os.path.normpath(
41               os.path.join(sys._MEIPASS, '..', 'Resources', os.path.relpath(pyqt_path, sys._MEIPASS))
42           )
43           os.environ['QML2_IMPORT_PATH'] = os.pathsep.join([
44               os.path.join(pyqt_path_res, 'qml'),
45               os.path.join(pyqt_path, 'qml'),
46           ])
47       else:
48           os.environ['QML2_IMPORT_PATH'] = os.path.join(pyqt_path, 'qml')
49   
50       # Add `sys._MEIPASS` to `PATH` in order to ensure that `QtNetwork` can discover OpenSSL DLLs that might have been
51       # collected there (i.e., when they were not shipped with the package, and were collected from an external location).
52       if sys.platform.startswith('win'):
53           prepend_path_to_environment_variable(sys._MEIPASS, 'PATH')
54   
55       # For macOS POSIX builds, we need to add `sys._MEIPASS` to `DYLD_LIBRARY_PATH` so that QtNetwork can discover
56       # OpenSSL dynamic libraries for its `openssl` TLS backend. This also prevents fallback to external locations, such
57       # as Homebrew. For .app bundles, this is unnecessary because `QtNetwork` explicitly searches `Contents/Frameworks`.
58       if sys.platform == 'darwin' and not is_macos_app_bundle:
59           prepend_path_to_environment_variable(sys._MEIPASS, 'DYLD_LIBRARY_PATH')
60   
61       # Qt bindings package installed via PyPI wheels typically ensures that its bundled Qt is relocatable, by creating
62       # embedded `qt.conf` file during its initialization. This run-time generated qt.conf dynamically sets the Qt prefix
63       # path to the package's Qt directory. For bindings packages that do not create embedded `qt.conf` during their
64       # initialization (for example, conda-installed packages), try to perform this step ourselves.
65       qt_rth_utils.create_embedded_qt_conf("PySide6", pyqt_path)
66   
67   
68   _pyi_rthook()
69   del _pyi_rthook
