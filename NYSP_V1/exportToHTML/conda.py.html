13   """ 
14   Additional helper methods for working specifically with Anaconda distributions are found at 
15   :mod:`PyInstaller.utils.hooks.conda_support<PyInstaller.utils.hooks.conda>` 
16   which is designed to mimic (albeit loosely) the `importlib.metadata`_ package. These functions find and parse the 
17   distribution metadata from json files located in the ``conda-meta`` directory. 
18    
19   .. versionadded:: 4.2.0 
20    
21   This module is available only if run inside a Conda environment. Usage of this module should therefore be wrapped in 
22   a conditional clause:: 
23    
24       from PyInstaller.compat import is_pure_conda 
25    
26       if is_pure_conda: 
27           from PyInstaller.utils.hooks import conda_support 
28    
29           # Code goes here. e.g. 
30           binaries = conda_support.collect_dynamic_libs("numpy") 
31           ... 
32    
33   Packages are all referenced by the *distribution name* you use to install it, rather than the *package name* you import 
34   it with. I.e., use ``distribution("pillow")`` instead of ``distribution("PIL")`` or use ``package_distribution("PIL")``. 
35   """
36   from __future__ import annotations
37   
38   import fnmatch
39   import json
40   import sys
41   from pathlib import Path
42   from typing import Iterable, List
43   from importlib.metadata import PackagePath as _PackagePath
44   
45   from PyInstaller import compat
46   from PyInstaller.log import logger
47   
48   # Conda virtual environments each get their own copy of `conda-meta` so the use of `sys.prefix` instead of
49   # `sys.base_prefix`, `sys.real_prefix` or anything from our `compat` module is intentional.
50   CONDA_ROOT = Path(sys.prefix)
51   CONDA_META_DIR = CONDA_ROOT / "conda-meta"
52   
53   # Find all paths in `sys.path` that are inside Conda root.
54   PYTHONPATH_PREFIXES = []
55   for _path in sys.path:
56       _path = Path(_path)
57       try:
58           PYTHONPATH_PREFIXES.append(_path.relative_to(sys.prefix))
59       except ValueError:
60           pass
61   
62   PYTHONPATH_PREFIXES.sort(key=lambda p: len(p.parts), reverse=True)

65   class Distribution:
66       """ 
67       A bucket class representation of a Conda distribution. 
68    
69       This bucket exports the following attributes: 
70    
71       :ivar name: The distribution's name. 
72       :ivar version: Its version. 
73       :ivar files: All filenames as :meth:`PackagePath`\\ s included with this distribution. 
74       :ivar dependencies: Names of other distributions that this distribution depends on (with version constraints 
75                           removed). 
76       :ivar packages: Names of importable packages included in this distribution. 
77    
78       This class is not intended to be constructed directly by users. Rather use :meth:`distribution` or 
79       :meth:`package_distribution` to provide one for you. 
80       """
81       def __init__(self, json_path: str):
82           try:
83               self._json_path = Path(json_path)
84               assert self._json_path.exists()
85           except (TypeError, AssertionError):
86               raise TypeError(
87                   "Distribution requires a path to a conda-meta json. Perhaps you want "
88                   "`distribution({})` instead?".format(repr(json_path))
89               )
90   
91           # Everything we need (including this distribution's name) is kept in the metadata json.
92           self.raw: dict = json.loads(self._json_path.read_text())
93   
94           # Unpack the more useful contents of the json.
95           self.name: str = self.raw["name"]
96           self.version: str = self.raw["version"]
97           self.files = [PackagePath(i) for i in self.raw["files"]]
98           self.dependencies = self._init_dependencies()
99           self.packages = self._init_package_names()
100  
101      def __repr__(self):
102          return "{}(name=\"{}\", packages={})".format(type(self).__name__, self.name, self.packages)
103  
104      def _init_dependencies(self):
105          """ 
106          Read dependencies from ``self.raw["depends"]``. 
107   
108          :return: Dependent distribution names. 
109          :rtype: list 
110   
111          The names in ``self.raw["depends"]`` come with extra version constraint information which must be stripped. 
112          """
113          dependencies = []
114          # For each dependency:
115          for dependency in self.raw["depends"]:
116              # ``dependency`` is a string of the form: "[name] [version constraints]"
117              name, *version_constraints = dependency.split(maxsplit=1)
118              dependencies.append(name)
119          return dependencies
120  
121      def _init_package_names(self):
122          """ 
123          Search ``self.files`` for package names shipped by this distribution. 
124   
125          :return: Package names. 
126          :rtype: list 
127   
128          These are names you would ``import`` rather than names you would install. 
129          """
130          packages = []
131          for file in self.files:
132              package = _get_package_name(file)
133              if package is not None:
134                  packages.append(package)
135          return packages
136  
137      @classmethod
138      def from_name(cls, name: str):
139          """ 
140          Get distribution information for a given distribution **name** (i.e., something you would ``conda install``). 
141   
142          :rtype: :class:`Distribution` 
143          """
144          if name in distributions:
145              return distributions[name]
146          raise ModuleNotFoundError(
147              "Distribution {} is either not installed or was not installed using Conda.".format(name)
148          )
149  
150      @classmethod
151      def from_package_name(cls, name: str):
152          """ 
153          Get distribution information for a **package** (i.e., something you would import). 
154   
155          :rtype: :class:`Distribution` 
156   
157          For example, the package ``pkg_resources`` belongs to the distribution ``setuptools``, which contains three 
158          packages. 
159   
160          >>> package_distribution("pkg_resources") 
161          Distribution(name="setuptools", 
162                       packages=['easy_install', 'pkg_resources', 'setuptools']) 
163          """
164          if name in distributions_by_package:
165              return distributions_by_package[name]
166          raise ModuleNotFoundError("Package {} is either not installed or was not installed using Conda.".format(name))
167  
168  
169  distribution = Distribution.from_name
170  package_distribution = Distribution.from_package_name

173  class PackagePath(_PackagePath):
174      """ 
175      A filename relative to Conda's root (``sys.prefix``). 
176   
177      This class inherits from :class:`pathlib.PurePosixPath` even on non-Posix OSs. To convert to a :class:`pathlib.Path` 
178      pointing to the real file, use the :meth:`locate` method. 
179      """
180      def locate(self):
181          """ 
182          Return a path-like object for this path pointing to the file's true location. 
183          """
184          return Path(sys.prefix) / self

187  def walk_dependency_tree(initial: str, excludes: Iterable[str] | None = None):
188      """ 
189      Collect a :class:`Distribution` and all direct and indirect dependencies of that distribution. 
190   
191      Arguments: 
192          initial: 
193              Distribution name to collect from. 
194          excludes: 
195              Distributions to exclude. 
196      Returns: 
197          A ``{name: distribution}`` mapping where ``distribution`` is the output of 
198          :func:`conda_support.distribution(name) <distribution>`. 
199      """
200      if excludes is not None:
201          excludes = set(excludes)
202  
203      # Rather than use true recursion, mimic it with a to-do queue.
204      from collections import deque
205      done = {}
206      names_to_do = deque([initial])
207  
208      while names_to_do:
209          # Grab a distribution name from the to-do list.
210          name = names_to_do.pop()
211          try:
212              # Collect and save it's metadata.
213              done[name] = distribution = Distribution.from_name(name)
214              logger.debug("Collected Conda distribution '%s', a dependency of '%s'.", name, initial)
215          except ModuleNotFoundError:
216              logger.warning(
217                  "Conda distribution '%s', dependency of '%s', was not found. "
218                  "If you installed this distribution with pip then you may ignore this warning.", name, initial
219              )
220              continue
221          # For each dependency:
222          for _name in distribution.dependencies:
223              if _name in done:
224                  # Skip anything already done.
225                  continue
226              if _name == name:
227                  # Avoid infinite recursion if a distribution depends on itself. This will probably never happen but I
228                  # certainly would not chance it.
229                  continue
230              if excludes is not None and _name in excludes:
231                  # Do not recurse to excluded dependencies.
232                  continue
233              names_to_do.append(_name)
234      return done
235

237  def _iter_distributions(name, dependencies, excludes):
238      if dependencies:
239          return walk_dependency_tree(name, excludes).values()
240      else:
241          return [Distribution.from_name(name)]

244  def requires(name: str, strip_versions: bool = False) -> List[str]:
245      """ 
246      List requirements of a distribution. 
247   
248      Arguments: 
249          name: 
250              The name of the distribution. 
251          strip_versions: 
252              List only their names, not their version constraints. 
253      Returns: 
254          A list of distribution names. 
255      """
256      if strip_versions:
257          return distribution(name).dependencies
258      return distribution(name).raw["depends"]

261  def files(name: str, dependencies: bool = False, excludes: list | None = None) -> List[PackagePath]:
262      """ 
263      List all files belonging to a distribution. 
264   
265      Arguments: 
266          name: 
267              The name of the distribution. 
268          dependencies: 
269              Recursively collect files of dependencies too. 
270          excludes: 
271              Distributions to ignore if **dependencies** is true. 
272      Returns: 
273          All filenames belonging to the given distribution. 
274   
275      With ``dependencies=False``, this is just a shortcut for:: 
276   
277          conda_support.distribution(name).files 
278      """
279      return [file for dist in _iter_distributions(name, dependencies, excludes) for file in dist.files]

288  def collect_dynamic_libs(name: str, dest: str = ".", dependencies: bool = True, excludes: Iterable[str] | None = None):
289      """ 
290      Collect DLLs for distribution **name**. 
291   
292      Arguments: 
293          name: 
294              The distribution's project-name. 
295          dest: 
296              Target destination, defaults to ``'.'``. 
297          dependencies: 
298              Recursively collect libs for dependent distributions (recommended). 
299          excludes: 
300              Dependent distributions to skip, defaults to ``None``. 
301      Returns: 
302          List of DLLs in PyInstaller's ``(source, dest)`` format. 
303   
304      This collects libraries only from Conda's shared ``lib`` (Unix) or ``Library/bin`` (Windows) folders. To collect 
305      from inside a distribution's installation use the regular :func:`PyInstaller.utils.hooks.collect_dynamic_libs`. 
306      """
307      DLL_SUFFIXES = ("*.dll", "*.dylib", "*.so", "*.so.*")
308      _files = []
309      for file in files(name, dependencies, excludes):
310          # A file is classified as a dynamic library if:
311          #  1) it lives inside the dedicated ``lib_dir`` DLL folder
312          if file.parent != lib_dir:
313              continue
314          #  2) it is a file (and not a directory or a symbolic link pointing to a directory)
315          resolved_file = file.locate()
316          if not resolved_file.is_file():
317              continue
318          #  3) has a correct suffix
319          if not any([resolved_file.match(suffix) for suffix in DLL_SUFFIXES]):
320              continue
321  
322          _files.append((str(resolved_file), dest))
323      return _files

329  def _get_package_name(file: PackagePath):
330      """ 
331      Determine the package name of a Python file in :data:`sys.path`. 
332   
333      Arguments: 
334          file: 
335              A Python filename relative to Conda root (sys.prefix). 
336      Returns: 
337          Package name or None. 
338   
339      This function only considers single file packages e.g. ``foo.py`` or top level ``foo/__init__.py``\\ s. 
340      Anything else is ignored (returning ``None``). 
341      """
342      file = Path(file)
343      # TODO: Handle PEP 420 namespace packages (which are missing `__init__` module). No such Conda PEP 420 namespace
344      # packages are known.
345  
346      # Get top-level folders by finding parents of `__init__.xyz`s
347      if file.stem == "__init__" and file.suffix in compat.ALL_SUFFIXES:
348          file = file.parent
349      elif file.suffix not in compat.ALL_SUFFIXES:
350          # Keep single-file packages but skip DLLs, data and junk files.
351          return
352  
353      # Check if this file/folder's parent is in ``sys.path`` i.e. it's directly importable. This intentionally excludes
354      # submodules which would cause confusion because ``sys.prefix`` is in ``sys.path``, meaning that every file in an
355      # Conda installation is a submodule.
356      for prefix in PYTHONPATH_PREFIXES:
357          if len(file.parts) != len(prefix.parts) + 1:
358              # This check is redundant but speeds it up quite a bit.
359              continue
360          # There are no wildcards involved here. The use of ``fnmatch`` is simply to handle the `if case-insensitive
361          # file system: use case-insensitive string matching.`
362          if fnmatch.fnmatch(str(file.parent), str(prefix)):
363              return file.stem

366  # All the information we want is organised the wrong way.
367  
368  # We want to look up distribution based on package names, but we can only search for packages using distribution names.
369  # And we would like to search for a distribution's json file, but, due to the noisy filenames of the jsons, we can only
370  # find a json's distribution rather than a distribution's json.
371  
372  # So we have to read everything, then regroup distributions in the ways we want them grouped. This will likely be a
373  # spectacular bottleneck on full-blown Conda (non miniconda) with 250+ packages by default at several GiBs. I suppose we
374  # could cache this on a per-json basis if it gets too much.
375  
376  
377  def _init_distributions():
378      distributions = {}
379      for path in CONDA_META_DIR.glob("*.json"):
380          dist = Distribution(path)
381          distributions[dist.name] = dist
382      return distributions
383  
384  
385  distributions = _init_distributions()
386  
387  
388  def _init_packages():
389      distributions_by_package = {}
390      for distribution in distributions.values():
391          for package in distribution.packages:
392              distributions_by_package[package] = distribution
393      return distributions_by_package

396  distributions_by_package = _init_packages()
