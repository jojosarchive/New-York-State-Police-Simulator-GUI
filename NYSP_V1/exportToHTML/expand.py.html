1    """Utility functions to expand configuration directives or special values 
2    (such glob patterns). 
3     
4    We can split the process of interpreting configuration files into 2 steps: 
5     
6    1. The parsing the file contents from strings to value objects 
7       that can be understand by Python (for example a string with a comma 
8       separated list of keywords into an actual Python list of strings). 
9     
10   2. The expansion (or post-processing) of these values according to the 
11      semantics ``setuptools`` assign to them (for example a configuration field 
12      with the ``file:`` directive should be expanded from a list of file paths to 
13      a single string with the contents of those files concatenated) 
14    
15   This module focus on the second step, and therefore allow sharing the expansion 
16   functions among several configuration file formats. 
17    
18   **PRIVATE MODULE**: API reserved for setuptools internal usage only. 
19   """

62   class StaticModule:
63       """Proxy to a module object that avoids executing arbitrary code."""
64   
65       def __init__(self, name: str, spec: ModuleSpec):
66           module = ast.parse(pathlib.Path(spec.origin).read_bytes())  # type: ignore[arg-type] # Let it raise an error on None
67           vars(self).update(locals())
68           del self.self
69   
70       def _find_assignments(self) -> Iterator[Tuple[ast.AST, ast.AST]]:
71           for statement in self.module.body:
72               if isinstance(statement, ast.Assign):
73                   yield from ((target, statement.value) for target in statement.targets)
74               elif isinstance(statement, ast.AnnAssign) and statement.value:
75                   yield (statement.target, statement.value)
76   
77       def __getattr__(self, attr):
78           """Attempt to load an attribute "statically", via :func:`ast.literal_eval`."""
79           try:
80               return next(
81                   ast.literal_eval(value)
82                   for target, value in self._find_assignments()
83                   if isinstance(target, ast.Name) and target.id == attr
84               )
85           except Exception as e:
86               raise AttributeError(f"{self.name} has no attribute {attr}") from e

89   def glob_relative(
90       patterns: Iterable[str], root_dir: Optional[StrPath] = None
91   ) -> List[str]:
92       """Expand the list of glob patterns, but preserving relative paths. 
93    
94       :param list[str] patterns: List of glob patterns 
95       :param str root_dir: Path to which globs should be relative 
96                            (current directory by default) 
97       :rtype: list 
98       """
99       glob_characters = {'*', '?', '[', ']', '{', '}'}
100      expanded_values = []
101      root_dir = root_dir or os.getcwd()
102      for value in patterns:
103          # Has globby characters?
104          if any(char in value for char in glob_characters):
105              # then expand the glob pattern while keeping paths *relative*:
106              glob_path = os.path.abspath(os.path.join(root_dir, value))
107              expanded_values.extend(
108                  sorted(
109                      os.path.relpath(path, root_dir).replace(os.sep, "/")
110                      for path in iglob(glob_path, recursive=True)
111                  )
112              )
113  
114          else:
115              # take the value as-is
116              path = os.path.relpath(value, root_dir).replace(os.sep, "/")
117              expanded_values.append(path)
118  
119      return expanded_values

122  def read_files(filepaths: Union[str, bytes, Iterable[StrPath]], root_dir=None) -> str:
123      """Return the content of the files concatenated using ``\n`` as str 
124   
125      This function is sandboxed and won't reach anything outside ``root_dir`` 
126   
127      (By default ``root_dir`` is the current directory). 
128      """
129      from setuptools.extern.more_itertools import always_iterable
130  
131      root_dir = os.path.abspath(root_dir or os.getcwd())
132      _filepaths = (os.path.join(root_dir, path) for path in always_iterable(filepaths))
133      return '\n'.join(
134          _read_file(path)
135          for path in _filter_existing_files(_filepaths)
136          if _assert_local(path, root_dir)
137      )
138  
139  
140  def _filter_existing_files(filepaths: Iterable[StrPath]) -> Iterator[StrPath]:
141      for path in filepaths:
142          if os.path.isfile(path):
143              yield path
144          else:
145              SetuptoolsWarning.emit(f"File {path!r} cannot be found")
146

148  def _read_file(filepath: Union[bytes, StrPath]) -> str:
149      with open(filepath, encoding='utf-8') as f:
150          return f.read()
151

153  def _assert_local(filepath: StrPath, root_dir: str):
154      if Path(os.path.abspath(root_dir)) not in Path(os.path.abspath(filepath)).parents:
155          msg = f"Cannot access {filepath!r} (or anything outside {root_dir!r})"
156          raise DistutilsOptionError(msg)
157  
158      return True

161  def read_attr(
162      attr_desc: str,
163      package_dir: Optional[Mapping[str, str]] = None,
164      root_dir: Optional[StrPath] = None,
165  ):
166      """Reads the value of an attribute from a module. 
167   
168      This function will try to read the attributed statically first 
169      (via :func:`ast.literal_eval`), and only evaluate the module if it fails. 
170   
171      Examples: 
172          read_attr("package.attr") 
173          read_attr("package.module.attr") 
174   
175      :param str attr_desc: Dot-separated string describing how to reach the 
176          attribute (see examples above) 
177      :param dict[str, str] package_dir: Mapping of package names to their 
178          location in disk (represented by paths relative to ``root_dir``). 
179      :param str root_dir: Path to directory containing all the packages in 
180          ``package_dir`` (current directory by default). 
181      :rtype: str 
182      """
183      root_dir = root_dir or os.getcwd()
184      attrs_path = attr_desc.strip().split('.')
185      attr_name = attrs_path.pop()
186      module_name = '.'.join(attrs_path)
187      module_name = module_name or '__init__'
188      _parent_path, path, module_name = _find_module(module_name, package_dir, root_dir)
189      spec = _find_spec(module_name, path)
190  
191      try:
192          return getattr(StaticModule(module_name, spec), attr_name)
193      except Exception:
194          # fallback to evaluate module
195          module = _load_spec(spec, module_name)
196          return getattr(module, attr_name)
197

199  def _find_spec(module_name: str, module_path: Optional[StrPath]) -> ModuleSpec:
200      spec = importlib.util.spec_from_file_location(module_name, module_path)
201      spec = spec or importlib.util.find_spec(module_name)
202  
203      if spec is None:
204          raise ModuleNotFoundError(module_name)
205  
206      return spec
207  
208  
209  def _load_spec(spec: ModuleSpec, module_name: str) -> ModuleType:
210      name = getattr(spec, "__name__", module_name)
211      if name in sys.modules:
212          return sys.modules[name]
213      module = importlib.util.module_from_spec(spec)
214      sys.modules[name] = module  # cache (it also ensures `==` works on loaded items)
215      spec.loader.exec_module(module)  # type: ignore
216      return module

219  def _find_module(
220      module_name: str, package_dir: Optional[Mapping[str, str]], root_dir: StrPath
221  ) -> Tuple[StrPath, Optional[str], str]:
222      """Given a module (that could normally be imported by ``module_name`` 
223      after the build is complete), find the path to the parent directory where 
224      it is contained and the canonical name that could be used to import it 
225      considering the ``package_dir`` in the build configuration and ``root_dir`` 
226      """
227      parent_path = root_dir
228      module_parts = module_name.split('.')
229      if package_dir:
230          if module_parts[0] in package_dir:
231              # A custom path was specified for the module we want to import
232              custom_path = package_dir[module_parts[0]]
233              parts = custom_path.rsplit('/', 1)
234              if len(parts) > 1:
235                  parent_path = os.path.join(root_dir, parts[0])
236                  parent_module = parts[1]
237              else:
238                  parent_module = custom_path
239              module_name = ".".join([parent_module, *module_parts[1:]])
240          elif '' in package_dir:
241              # A custom parent directory was specified for all root modules
242              parent_path = os.path.join(root_dir, package_dir[''])
243  
244      path_start = os.path.join(parent_path, *module_name.split("."))
245      candidates = chain(
246          (f"{path_start}.py", os.path.join(path_start, "__init__.py")),
247          iglob(f"{path_start}.*"),
248      )
249      module_path = next((x for x in candidates if os.path.isfile(x)), None)
250      return parent_path, module_path, module_name

253  def resolve_class(
254      qualified_class_name: str,
255      package_dir: Optional[Mapping[str, str]] = None,
256      root_dir: Optional[StrPath] = None,
257  ) -> Callable:
258      """Given a qualified class name, return the associated class object"""
259      root_dir = root_dir or os.getcwd()
260      idx = qualified_class_name.rfind('.')
261      class_name = qualified_class_name[idx + 1 :]
262      pkg_name = qualified_class_name[:idx]
263  
264      _parent_path, path, module_name = _find_module(pkg_name, package_dir, root_dir)
265      module = _load_spec(_find_spec(module_name, path), module_name)
266      return getattr(module, class_name)

270      values: Dict[str, str],
271      package_dir: Optional[Mapping[str, str]] = None,
272      root_dir: Optional[StrPath] = None,
273  ) -> Dict[str, Callable]:
274      """Given a dictionary mapping command names to strings for qualified class

277      return {k: resolve_class(v, package_dir, root_dir) for k, v in values.items()}

280  def find_packages(
281      *,
282      namespaces=True,
283      fill_package_dir: Optional[Dict[str, str]] = None,
284      root_dir: Optional[StrPath] = None,
285      **kwargs,
286  ) -> List[str]:
287      """Works similarly to :func:`setuptools.find_packages`, but with all 
288      arguments given as keyword arguments. Moreover, ``where`` can be given 
289      as a list (the results will be simply concatenated). 
290   
291      When the additional keyword argument ``namespaces`` is ``True``, it will 
292      behave like :func:`setuptools.find_namespace_packages`` (i.e. include 
293      implicit namespaces as per :pep:`420`). 
294   
295      The ``where`` argument will be considered relative to ``root_dir`` (or the current 
296      working directory when ``root_dir`` is not given). 
297   
298      If the ``fill_package_dir`` argument is passed, this function will consider it as a 
299      similar data structure to the ``package_dir`` configuration parameter add fill-in 
300      any missing package location. 
301   
302      :rtype: list 
303      """
304      from setuptools.discovery import construct_package_dir
305      from setuptools.extern.more_itertools import unique_everseen, always_iterable
306  
307      if namespaces:
308          from setuptools.discovery import PEP420PackageFinder as PackageFinder
309      else:
310          from setuptools.discovery import PackageFinder  # type: ignore
311  
312      root_dir = root_dir or os.curdir
313      where = kwargs.pop('where', ['.'])
314      packages: List[str] = []
315      fill_package_dir = {} if fill_package_dir is None else fill_package_dir
316      search = list(unique_everseen(always_iterable(where)))
317  
318      if len(search) == 1 and all(not _same_path(search[0], x) for x in (".", root_dir)):
319          fill_package_dir.setdefault("", search[0])
320  
321      for path in search:
322          package_path = _nest_path(root_dir, path)
323          pkgs = PackageFinder.find(package_path, **kwargs)
324          packages.extend(pkgs)
325          if pkgs and not (
326              fill_package_dir.get("") == path or os.path.samefile(package_path, root_dir)
327          ):
328              fill_package_dir.update(construct_package_dir(pkgs, path))
329  
330      return packages
331

333  def _nest_path(parent: StrPath, path: StrPath) -> str:
334      path = parent if path in {".", ""} else os.path.join(parent, path)
335      return os.path.normpath(path)

338  def version(value: Union[Callable, Iterable[Union[str, int]], str]) -> str:
339      """When getting the version directly from an attribute, 
340      it should be normalised to string. 
341      """
342      if callable(value):
343          value = value()
344  
345      value = cast(Iterable[Union[str, int]], value)
346  
347      if not isinstance(value, str):
348          if hasattr(value, '__iter__'):
349              value = '.'.join(map(str, value))
350          else:
351              value = '%s' % value
352  
353      return value

362  def canonic_data_files(
363      data_files: Union[list, dict], root_dir: Optional[StrPath] = None
364  ) -> List[Tuple[str, List[str]]]:
365      """For compatibility with ``setup.py``, ``data_files`` should be a list 
366      of pairs instead of a dict. 
367   
368      This function also expands glob patterns. 
369      """
370      if isinstance(data_files, list):
371          return data_files
372  
373      return [
374          (dest, glob_relative(patterns, root_dir))
375          for dest, patterns in data_files.items()
376      ]

379  def entry_points(text: str, text_source="entry-points") -> Dict[str, dict]:
380      """Given the contents of entry-points file, 
381      process it into a 2-level dictionary (``dict[str, dict[str, str]]``). 
382      The first level keys are entry-point groups, the second level keys are 
383      entry-point names, and the second level values are references to objects 
384      (that correspond to the entry-point value). 
385      """
386      parser = ConfigParser(default_section=None, delimiters=("=",))  # type: ignore
387      parser.optionxform = str  # case sensitive
388      parser.read_string(text, text_source)
389      groups = {k: dict(v.items()) for k, v in parser.items()}
390      groups.pop(parser.default_section, None)
391      return groups

394  class EnsurePackagesDiscovered:
395      """Some expand functions require all the packages to already be discovered before 
396      they run, e.g. :func:`read_attr`, :func:`resolve_class`, :func:`cmdclass`. 
397   
398      Therefore in some cases we will need to run autodiscovery during the evaluation of 
399      the configuration. However, it is better to postpone calling package discovery as 
400      much as possible, because some parameters can influence it (e.g. ``package_dir``), 
401      and those might not have been processed yet. 
402      """
403  
404      def __init__(self, distribution: "Distribution"):
405          self._dist = distribution
406          self._called = False
407  
408      def __call__(self):
409          """Trigger the automatic package discovery, if it is still necessary."""
410          if not self._called:
411              self._called = True
412              self._dist.set_defaults(name=False)  # Skip name, we can still be parsing
413  
414      def __enter__(self):
415          return self
416  
417      def __exit__(self, _exc_type, _exc_value, _traceback):
418          if self._called:
419              self._dist.set_defaults.analyse_name()  # Now we can set a default name
420  
421      def _get_package_dir(self) -> Mapping[str, str]:
422          self()
423          pkg_dir = self._dist.package_dir
424          return {} if pkg_dir is None else pkg_dir
425  
426      @property
427      def package_dir(self) -> Mapping[str, str]:
428          """Proxy to ``package_dir`` that may trigger auto-discovery when used."""
429          return LazyMappingProxy(self._get_package_dir)

432  class LazyMappingProxy(Mapping[_K, _V]):
433      """Mapping proxy that delays resolving the target object, until really needed. 
434   
435      >>> def obtain_mapping(): 
436      ...     print("Running expensive function!") 
437      ...     return {"key": "value", "other key": "other value"} 
438      >>> mapping = LazyMappingProxy(obtain_mapping) 
439      >>> mapping["key"] 
440      Running expensive function! 
441      'value' 
442      >>> mapping["other key"] 
443      'other value' 
444      """
445  
446      def __init__(self, obtain_mapping_value: Callable[[], Mapping[_K, _V]]):
447          self._obtain = obtain_mapping_value
448          self._value: Optional[Mapping[_K, _V]] = None
449  
450      def _target(self) -> Mapping[_K, _V]:
451          if self._value is None:
452              self._value = self._obtain()
453          return self._value
454  
455      def __getitem__(self, key: _K) -> _V:
456          return self._target()[key]
457  
458      def __len__(self) -> int:
459          return len(self._target())
460  
461      def __iter__(self) -> Iterator[_K]:
462          return iter(self._target())
