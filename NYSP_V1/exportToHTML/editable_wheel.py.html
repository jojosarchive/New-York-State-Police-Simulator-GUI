68   class _EditableMode(Enum):
69       """ 
70       Possible editable installation modes: 
71       `lenient` (new files automatically added to the package - DEFAULT); 
72       `strict` (requires a new installation when files are added/removed); or 
73       `compat` (attempts to emulate `python setup.py develop` - DEPRECATED). 
74       """
75   
76       STRICT = "strict"
77       LENIENT = "lenient"
78       COMPAT = "compat"  # TODO: Remove `compat` after Dec/2022.
79   
80       @classmethod
81       def convert(cls, mode: Optional[str]) -> "_EditableMode":
82           if not mode:
83               return _EditableMode.LENIENT  # default
84   
85           _mode = mode.upper()
86           if _mode not in _EditableMode.__members__:
87               raise errors.OptionError(f"Invalid editable mode: {mode!r}. Try: 'strict'.")
88   
89           if _mode == "COMPAT":
90               SetuptoolsDeprecationWarning.emit(
91                   "Compat editable installs",
92                   """ 
93                   The 'compat' editable mode is transitional and will be removed 
94                   in future versions of `setuptools`. 
95                   Please adapt your code accordingly to use either the 'strict' or the 
96                   'lenient' modes. 
97                   """,
98                   see_docs="userguide/development_mode.html",
99                   # TODO: define due_date
100                  # There is a series of shortcomings with the available editable install
101                  # methods, and they are very controversial. This is something that still
102                  # needs work.
103                  # Moreover, `pip` is still hiding this warning, so users are not aware.
104              )
105  
106          return _EditableMode[_mode]

119  class editable_wheel(Command):
120      """Build 'editable' wheel for development. 
121      This command is private and reserved for internal use of setuptools, 
122      users should rely on ``setuptools.build_meta`` APIs. 
123      """
124  
125      description = "DO NOT CALL DIRECTLY, INTERNAL ONLY: create PEP 660 editable wheel"
126  
127      user_options = [
128          ("dist-dir=", "d", "directory to put final built distributions in"),
129          ("dist-info-dir=", "I", "path to a pre-build .dist-info directory"),
130          ("mode=", None, cleandoc(_EditableMode.__doc__ or "")),
131      ]
132  
133      def initialize_options(self):
134          self.dist_dir = None
135          self.dist_info_dir = None
136          self.project_dir = None
137          self.mode = None

139      def finalize_options(self):
140          dist = self.distribution
141          self.project_dir = dist.src_root or os.curdir
142          self.package_dir = dist.package_dir or {}
143          self.dist_dir = Path(self.dist_dir or os.path.join(self.project_dir, "dist"))

145      def run(self):
146          try:
147              self.dist_dir.mkdir(exist_ok=True)
148              self._ensure_dist_info()
149  
150              # Add missing dist_info files
151              self.reinitialize_command("bdist_wheel")
152              bdist_wheel = self.get_finalized_command("bdist_wheel")
153              bdist_wheel.write_wheelfile(self.dist_info_dir)
154  
155              self._create_wheel_file(bdist_wheel)
156          except Exception:
157              traceback.print_exc()
158              project = self.distribution.name or self.distribution.get_name()
159              _DebuggingTips.emit(project=project)
160              raise

162      def _ensure_dist_info(self):
163          if self.dist_info_dir is None:
164              dist_info = cast(dist_info_cls, self.reinitialize_command("dist_info"))
165              dist_info.output_dir = self.dist_dir
166              dist_info.ensure_finalized()
167              dist_info.run()
168              self.dist_info_dir = dist_info.dist_info_dir
169          else:
170              assert str(self.dist_info_dir).endswith(".dist-info")
171              assert Path(self.dist_info_dir, "METADATA").exists()

173      def _install_namespaces(self, installation_dir, pth_prefix):
174          # XXX: Only required to support the deprecated namespace practice
175          dist = self.distribution
176          if not dist.namespace_packages:
177              return
178  
179          src_root = Path(self.project_dir, self.package_dir.get("", ".")).resolve()
180          installer = _NamespaceInstaller(dist, installation_dir, pth_prefix, src_root)
181          installer.install_namespaces()

183      def _find_egg_info_dir(self) -> Optional[str]:
184          parent_dir = Path(self.dist_info_dir).parent if self.dist_info_dir else Path()
185          candidates = map(str, parent_dir.glob("*.egg-info"))
186          return next(candidates, None)

188      def _configure_build(
189          self, name: str, unpacked_wheel: StrPath, build_lib: StrPath, tmp_dir: StrPath
190      ):
191          """Configure commands to behave in the following ways: 
192   
193          - Build commands can write to ``build_lib`` if they really want to... 
194            (but this folder is expected to be ignored and modules are expected to live 
195            in the project directory...) 
196          - Binary extensions should be built in-place (editable_mode = True) 
197          - Data/header/script files are not part of the "editable" specification 
198            so they are written directly to the unpacked_wheel directory. 
199          """
200          # Non-editable files (data, headers, scripts) are written directly to the
201          # unpacked_wheel
202  
203          dist = self.distribution
204          wheel = str(unpacked_wheel)
205          build_lib = str(build_lib)
206          data = str(Path(unpacked_wheel, f"{name}.data", "data"))
207          headers = str(Path(unpacked_wheel, f"{name}.data", "headers"))
208          scripts = str(Path(unpacked_wheel, f"{name}.data", "scripts"))
209  
210          # egg-info may be generated again to create a manifest (used for package data)
211          egg_info = cast(
212              egg_info_cls, dist.reinitialize_command("egg_info", reinit_subcommands=True)
213          )
214          egg_info.egg_base = str(tmp_dir)
215          egg_info.ignore_egg_info_in_manifest = True
216  
217          build = cast(
218              build_cls, dist.reinitialize_command("build", reinit_subcommands=True)
219          )
220          install = cast(
221              install_cls, dist.reinitialize_command("install", reinit_subcommands=True)
222          )
223  
224          build.build_platlib = build.build_purelib = build.build_lib = build_lib
225          install.install_purelib = install.install_platlib = install.install_lib = wheel
226          install.install_scripts = build.build_scripts = scripts
227          install.install_headers = headers
228          install.install_data = data
229  
230          install_scripts = cast(
231              install_scripts_cls, dist.get_command_obj("install_scripts")
232          )
233          install_scripts.no_ep = True
234  
235          build.build_temp = str(tmp_dir)
236  
237          build_py = cast(build_py_cls, dist.get_command_obj("build_py"))
238          build_py.compile = False
239          build_py.existing_egg_info_dir = self._find_egg_info_dir()
240  
241          self._set_editable_mode()
242  
243          build.ensure_finalized()
244          install.ensure_finalized()

246      def _set_editable_mode(self):
247          """Set the ``editable_mode`` flag in the build sub-commands"""
248          dist = self.distribution
249          build = dist.get_command_obj("build")
250          # TODO: Update typeshed distutils stubs to overload non-None return type by default
251          for cmd_name in build.get_sub_commands():
252              cmd = dist.get_command_obj(cmd_name)
253              if hasattr(cmd, "editable_mode"):
254                  cmd.editable_mode = True
255              elif hasattr(cmd, "inplace"):
256                  cmd.inplace = True  # backward compatibility with distutils

258      def _collect_build_outputs(self) -> Tuple[List[str], Dict[str, str]]:
259          files: List[str] = []
260          mapping: Dict[str, str] = {}
261          build = self.get_finalized_command("build")
262  
263          for cmd_name in build.get_sub_commands():
264              cmd = self.get_finalized_command(cmd_name)
265              if hasattr(cmd, "get_outputs"):
266                  files.extend(cmd.get_outputs() or [])
267              if hasattr(cmd, "get_output_mapping"):
268                  mapping.update(cmd.get_output_mapping() or {})
269  
270          return files, mapping

272      def _run_build_commands(
273          self,
274          dist_name: str,
275          unpacked_wheel: StrPath,
276          build_lib: StrPath,
277          tmp_dir: StrPath,
278      ) -> Tuple[List[str], Dict[str, str]]:
279          self._configure_build(dist_name, unpacked_wheel, build_lib, tmp_dir)
280          self._run_build_subcommands()
281          files, mapping = self._collect_build_outputs()
282          self._run_install("headers")
283          self._run_install("scripts")
284          self._run_install("data")
285          return files, mapping

287      def _run_build_subcommands(self) -> None:
288          """ 
289          Issue #3501 indicates that some plugins/customizations might rely on: 
290   
291          1. ``build_py`` not running 
292          2. ``build_py`` always copying files to ``build_lib`` 
293   
294          However both these assumptions may be false in editable_wheel. 
295          This method implements a temporary workaround to support the ecosystem 
296          while the implementations catch up. 
297          """
298          # TODO: Once plugins/customisations had the chance to catch up, replace
299          #       `self._run_build_subcommands()` with `self.run_command("build")`.
300          #       Also remove _safely_run, TestCustomBuildPy. Suggested date: Aug/2023.
301          build = self.get_finalized_command("build")
302          for name in build.get_sub_commands():
303              cmd = self.get_finalized_command(name)
304              if name == "build_py" and type(cmd) != build_py_cls:
305                  self._safely_run(name)
306              else:
307                  self.run_command(name)

309      def _safely_run(self, cmd_name: str):
310          try:
311              return self.run_command(cmd_name)
312          except Exception:
313              SetuptoolsDeprecationWarning.emit(
314                  "Customization incompatible with editable install",
315                  f"""
316                  {traceback.format_exc()}
317  
318                  If you are seeing this warning it is very likely that a setuptools
319                  plugin or customization overrides the `{cmd_name}` command, without
320                  taking into consideration how editable installs run build steps
321                  starting from setuptools v64.0.0.
322  
323                  Plugin authors and developers relying on custom build steps are
324                  encouraged to update their `{cmd_name}` implementation considering the
325                  information about editable installs in
326                  https://setuptools.pypa.io/en/latest/userguide/extension.html.
327  
328                  For the time being `setuptools` will silence this error and ignore
329                  the faulty command, but this behaviour will change in future versions.
330                  """,
331                  # TODO: define due_date
332                  # There is a series of shortcomings with the available editable install
333                  # methods, and they are very controversial. This is something that still
334                  # needs work.
335              )

337      def _create_wheel_file(self, bdist_wheel):
338          from wheel.wheelfile import WheelFile
339  
340          dist_info = self.get_finalized_command("dist_info")
341          dist_name = dist_info.name
342          tag = "-".join(bdist_wheel.get_tag())
343          build_tag = "0.editable"  # According to PEP 427 needs to start with digit
344          archive_name = f"{dist_name}-{build_tag}-{tag}.whl"
345          wheel_path = Path(self.dist_dir, archive_name)
346          if wheel_path.exists():
347              wheel_path.unlink()
348  
349          unpacked_wheel = TemporaryDirectory(suffix=archive_name)
350          build_lib = TemporaryDirectory(suffix=".build-lib")
351          build_tmp = TemporaryDirectory(suffix=".build-temp")
352  
353          with unpacked_wheel as unpacked, build_lib as lib, build_tmp as tmp:
354              unpacked_dist_info = Path(unpacked, Path(self.dist_info_dir).name)
355              shutil.copytree(self.dist_info_dir, unpacked_dist_info)
356              self._install_namespaces(unpacked, dist_name)
357              files, mapping = self._run_build_commands(dist_name, unpacked, lib, tmp)
358              strategy = self._select_strategy(dist_name, tag, lib)
359              with strategy, WheelFile(wheel_path, "w") as wheel_obj:
360                  strategy(wheel_obj, files, mapping)
361                  wheel_obj.write_files(unpacked)
362  
363          return wheel_path
364  
365      def _run_install(self, category: str):
366          has_category = getattr(self.distribution, f"has_{category}", None)
367          if has_category and has_category():
368              _logger.info(f"Installing {category} as non editable")
369              self.run_command(f"install_{category}")

371      def _select_strategy(
372          self,
373          name: str,
374          tag: str,
375          build_lib: StrPath,
376      ) -> "EditableStrategy":
377          """Decides which strategy to use to implement an editable installation."""
378          build_name = f"__editable__.{name}-{tag}"
379          project_dir = Path(self.project_dir)
380          mode = _EditableMode.convert(self.mode)
381  
382          if mode is _EditableMode.STRICT:
383              auxiliary_dir = _empty_dir(Path(self.project_dir, "build", build_name))
384              return _LinkTree(self.distribution, name, auxiliary_dir, build_lib)
385  
386          packages = _find_packages(self.distribution)
387          has_simple_layout = _simple_layout(packages, self.package_dir, project_dir)
388          is_compat_mode = mode is _EditableMode.COMPAT
389          if set(self.package_dir) == {""} and has_simple_layout or is_compat_mode:
390              # src-layout(ish) is relatively safe for a simple pth file
391              src_dir = self.package_dir.get("", ".")
392              return _StaticPth(self.distribution, name, [Path(project_dir, src_dir)])
393  
394          # Use a MetaPathFinder to avoid adding accidental top-level packages/modules
395          return _TopLevelFinder(self.distribution, name)

408  class _StaticPth:
409      def __init__(self, dist: Distribution, name: str, path_entries: List[Path]):
410          self.dist = dist
411          self.name = name
412          self.path_entries = path_entries
413  
414      def __call__(self, wheel: "WheelFile", files: List[str], mapping: Dict[str, str]):
415          entries = "\n".join(str(p.resolve()) for p in self.path_entries)
416          contents = _encode_pth(f"{entries}\n")
417          wheel.writestr(f"__editable__.{self.name}.pth", contents)
418  
419      def __enter__(self):
420          msg = f"""
421          Editable install will be performed using .pth file to extend `sys.path` with:
422          {list(map(os.fspath, self.path_entries))!r}
423          """
424          _logger.warning(msg + _LENIENT_WARNING)
425          return self
426  
427      def __exit__(self, _exc_type, _exc_value, _traceback): ...

430  class _LinkTree(_StaticPth):
431      """ 
432      Creates a ``.pth`` file that points to a link tree in the ``auxiliary_dir``. 
433   
434      This strategy will only link files (not dirs), so it can be implemented in 
435      any OS, even if that means using hardlinks instead of symlinks. 
436   
437      By collocating ``auxiliary_dir`` and the original source code, limitations 
438      with hardlinks should be avoided. 
439      """
440  
441      def __init__(
442          self,
443          dist: Distribution,
444          name: str,
445          auxiliary_dir: StrPath,
446          build_lib: StrPath,
447      ):
448          self.auxiliary_dir = Path(auxiliary_dir)
449          self.build_lib = Path(build_lib).resolve()
450          # TODO: Update typeshed distutils stubs to overload non-None return type by default
451          self._file = dist.get_command_obj("build_py").copy_file  # type: ignore[union-attr]
452          super().__init__(dist, name, [self.auxiliary_dir])
453  
454      def __call__(self, wheel: "WheelFile", files: List[str], mapping: Dict[str, str]):
455          self._create_links(files, mapping)
456          super().__call__(wheel, files, mapping)
457  
458      def _normalize_output(self, file: str) -> Optional[str]:
459          # Files relative to build_lib will be normalized to None
460          with suppress(ValueError):
461              path = Path(file).resolve().relative_to(self.build_lib)
462              return str(path).replace(os.sep, '/')
463          return None
464  
465      def _create_file(self, relative_output: str, src_file: str, link=None):
466          dest = self.auxiliary_dir / relative_output
467          if not dest.parent.is_dir():
468              dest.parent.mkdir(parents=True)
469          # TODO: Update typeshed distutils stubs so distutils.cmd.Command.copy_file, accepts PathLike
470          # same with methods used by copy_file
471          self._file(src_file, dest, link=link)  # type: ignore[arg-type]
472  
473      def _create_links(self, outputs, output_mapping):
474          self.auxiliary_dir.mkdir(parents=True, exist_ok=True)
475          link_type = "sym" if _can_symlink_files(self.auxiliary_dir) else "hard"
476          mappings = {self._normalize_output(k): v for k, v in output_mapping.items()}
477          mappings.pop(None, None)  # remove files that are not relative to build_lib
478  
479          for output in outputs:
480              relative = self._normalize_output(output)
481              if relative and relative not in mappings:
482                  self._create_file(relative, output)
483  
484          for relative, src in mappings.items():
485              self._create_file(relative, src, link=link_type)
486  
487      def __enter__(self):
488          msg = "Strict editable install will be performed using a link tree.\n"
489          _logger.warning(msg + _STRICT_WARNING)
490          return self
491  
492      def __exit__(self, _exc_type, _exc_value, _traceback):
493          msg = f"""\n
494          Strict editable installation performed using the auxiliary directory:
495              {self.auxiliary_dir}
496  
497          Please be careful to not remove this directory, otherwise you might not be able
498          to import/use your package.
499          """
500          InformationOnly.emit("Editable installation.", msg)

503  class _TopLevelFinder:
504      def __init__(self, dist: Distribution, name: str):
505          self.dist = dist
506          self.name = name
507  
508      def template_vars(self) -> Tuple[str, str, Dict[str, str], Dict[str, List[str]]]:
509          src_root = self.dist.src_root or os.curdir
510          top_level = chain(_find_packages(self.dist), _find_top_level_modules(self.dist))
511          package_dir = self.dist.package_dir or {}
512          roots = _find_package_roots(top_level, package_dir, src_root)
513  
514          namespaces_: Dict[str, List[str]] = dict(
515              chain(
516                  _find_namespaces(self.dist.packages or [], roots),
517                  ((ns, []) for ns in _find_virtual_namespaces(roots)),
518              )
519          )
520  
521          legacy_namespaces = {
522              cast(str, pkg): find_package_path(pkg, roots, self.dist.src_root or "")
523              for pkg in self.dist.namespace_packages or []
524          }
525  
526          mapping = {**roots, **legacy_namespaces}
527          # ^-- We need to explicitly add the legacy_namespaces to the mapping to be
528          #     able to import their modules even if another package sharing the same
529          #     namespace is installed in a conventional (non-editable) way.
530  
531          name = f"__editable__.{self.name}.finder"
532          finder = _normalization.safe_identifier(name)
533          return finder, name, mapping, namespaces_
534  
535      def get_implementation(self) -> Iterator[Tuple[str, bytes]]:
536          finder, name, mapping, namespaces_ = self.template_vars()
537  
538          content = bytes(_finder_template(name, mapping, namespaces_), "utf-8")
539          yield (f"{finder}.py", content)
540  
541          content = _encode_pth(f"import {finder}; {finder}.install()")
542          yield (f"__editable__.{self.name}.pth", content)
543  
544      def __call__(self, wheel: "WheelFile", files: List[str], mapping: Dict[str, str]):
545          for file, content in self.get_implementation():
546              wheel.writestr(file, content)
547  
548      def __enter__(self):
549          msg = "Editable install will be performed using a meta path finder.\n"
550          _logger.warning(msg + _LENIENT_WARNING)
551          return self
552  
553      def __exit__(self, _exc_type, _exc_value, _traceback):
554          msg = """\n
555          Please be careful with folders in your working directory with the same 
556          name as your package as they may take precedence during imports. 
557          """
558          InformationOnly.emit("Editable installation.", msg)

561  def _encode_pth(content: str) -> bytes:
562      """.pth files are always read with 'locale' encoding, the recommendation 
563      from the cpython core developers is to write them as ``open(path, "w")`` 
564      and ignore warnings (see python/cpython#77102, pypa/setuptools#3937). 
565      This function tries to simulate this behaviour without having to create an 
566      actual file, in a way that supports a range of active Python versions. 
567      (There seems to be some variety in the way different version of Python handle 
568      ``encoding=None``, not all of them use ``locale.getpreferredencoding(False)`` 
569      or ``locale.getencoding()``). 
570      """
571      with io.BytesIO() as buffer:
572          wrapper = io.TextIOWrapper(buffer, encoding=py39.LOCALE_ENCODING)
573          wrapper.write(content)
574          wrapper.flush()
575          buffer.seek(0)
576          return buffer.read()

579  def _can_symlink_files(base_dir: Path) -> bool:
580      with TemporaryDirectory(dir=str(base_dir.resolve())) as tmp:
581          path1, path2 = Path(tmp, "file1.txt"), Path(tmp, "file2.txt")
582          path1.write_text("file1", encoding="utf-8")
583          with suppress(AttributeError, NotImplementedError, OSError):
584              os.symlink(path1, path2)
585              if path2.is_symlink() and path2.read_text(encoding="utf-8") == "file1":
586                  return True
587  
588          try:
589              os.link(path1, path2)  # Ensure hard links can be created
590          except Exception as ex:
591              msg = (
592                  "File system does not seem to support either symlinks or hard links. "
593                  "Strict editable installs require one of them to be supported."
594              )
595              raise LinksNotSupported(msg) from ex
596          return False

599  def _simple_layout(
600      packages: Iterable[str], package_dir: Dict[str, str], project_dir: StrPath
601  ) -> bool:
602      """Return ``True`` if: 
603      - all packages are contained by the same parent directory, **and** 
604      - all packages become importable if the parent directory is added to ``sys.path``. 
605   
606      >>> _simple_layout(['a'], {"": "src"}, "/tmp/myproj") 
607      True 
608      >>> _simple_layout(['a', 'a.b'], {"": "src"}, "/tmp/myproj") 
609      True 
610      >>> _simple_layout(['a', 'a.b'], {}, "/tmp/myproj") 
611      True 
612      >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"": "src"}, "/tmp/myproj") 
613      True 
614      >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "a", "b": "b"}, ".") 
615      True 
616      >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "_a", "b": "_b"}, ".") 
617      False 
618      >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a": "_a"}, "/tmp/myproj") 
619      False 
620      >>> _simple_layout(['a', 'a.a1', 'a.a1.a2', 'b'], {"a.a1.a2": "_a2"}, ".") 
621      False 
622      >>> _simple_layout(['a', 'a.b'], {"": "src", "a.b": "_ab"}, "/tmp/myproj") 
623      False 
624      >>> # Special cases, no packages yet: 
625      >>> _simple_layout([], {"": "src"}, "/tmp/myproj") 
626      True 
627      >>> _simple_layout([], {"a": "_a", "": "src"}, "/tmp/myproj") 
628      False 
629      """
630      layout = {pkg: find_package_path(pkg, package_dir, project_dir) for pkg in packages}
631      if not layout:
632          return set(package_dir) in ({}, {""})
633      parent = os.path.commonpath(starmap(_parent_path, layout.items()))
634      return all(
635          _path.same_path(Path(parent, *key.split('.')), value)
636          for key, value in layout.items()
637      )

640  def _parent_path(pkg, pkg_path):
641      """Infer the parent path containing a package, that if added to ``sys.path`` would 
642      allow importing that package. 
643      When ``pkg`` is directly mapped into a directory with a different name, return its 
644      own path. 
645      >>> _parent_path("a", "src/a") 
646      'src' 
647      >>> _parent_path("b", "src/c") 
648      'src/c' 
649      """
650      parent = pkg_path[: -len(pkg)] if pkg_path.endswith(pkg) else pkg_path
651      return parent.rstrip("/" + os.sep)

654  def _find_packages(dist: Distribution) -> Iterator[str]:
655      yield from iter(dist.packages or [])
656  
657      py_modules = dist.py_modules or []
658      nested_modules = [mod for mod in py_modules if "." in mod]
659      if dist.ext_package:
660          yield dist.ext_package
661      else:
662          ext_modules = dist.ext_modules or []
663          nested_modules += [x.name for x in ext_modules if "." in x.name]
664  
665      for module in nested_modules:
666          package, _, _ = module.rpartition(".")
667          yield package

670  def _find_top_level_modules(dist: Distribution) -> Iterator[str]:
671      py_modules = dist.py_modules or []
672      yield from (mod for mod in py_modules if "." not in mod)
673  
674      if not dist.ext_package:
675          ext_modules = dist.ext_modules or []
676          yield from (x.name for x in ext_modules if "." not in x.name)

679  def _find_package_roots(
680      packages: Iterable[str],
681      package_dir: Mapping[str, str],
682      src_root: StrPath,
683  ) -> Dict[str, str]:
684      pkg_roots: Dict[str, str] = {
685          pkg: _absolute_root(find_package_path(pkg, package_dir, src_root))
686          for pkg in sorted(packages)
687      }
688  
689      return _remove_nested(pkg_roots)

692  def _absolute_root(path: StrPath) -> str:
693      """Works for packages and top-level modules"""
694      path_ = Path(path)
695      parent = path_.parent
696  
697      if path_.exists():
698          return str(path_.resolve())
699      else:
700          return str(parent.resolve() / path_.name)

703  def _find_virtual_namespaces(pkg_roots: Dict[str, str]) -> Iterator[str]:
704      """By carefully designing ``package_dir``, it is possible to implement the logical 
705      structure of PEP 420 in a package without the corresponding directories. 
706   
707      Moreover a parent package can be purposefully/accidentally skipped in the discovery 
708      phase (e.g. ``find_packages(include=["mypkg.*"])``, when ``mypkg.foo`` is included 
709      by ``mypkg`` itself is not). 
710      We consider this case to also be a virtual namespace (ignoring the original 
711      directory) to emulate a non-editable installation. 
712   
713      This function will try to find these kinds of namespaces. 
714      """
715      for pkg in pkg_roots:
716          if "." not in pkg:
717              continue
718          parts = pkg.split(".")
719          for i in range(len(parts) - 1, 0, -1):
720              partial_name = ".".join(parts[:i])
721              path = Path(find_package_path(partial_name, pkg_roots, ""))
722              if not path.exists() or partial_name not in pkg_roots:
723                  # partial_name not in pkg_roots ==> purposefully/accidentally skipped
724                  yield partial_name
725  
726  
727  def _find_namespaces(
728      packages: List[str], pkg_roots: Dict[str, str]
729  ) -> Iterator[Tuple[str, List[str]]]:
730      for pkg in packages:
731          path = find_package_path(pkg, pkg_roots, "")
732          if Path(path).exists() and not Path(path, "__init__.py").exists():
733              yield (pkg, [path])

736  def _remove_nested(pkg_roots: Dict[str, str]) -> Dict[str, str]:
737      output = dict(pkg_roots.copy())
738  
739      for pkg, path in reversed(list(pkg_roots.items())):
740          if any(
741              pkg != other and _is_nested(pkg, path, other, other_path)
742              for other, other_path in pkg_roots.items()
743          ):
744              output.pop(pkg)
745  
746      return output

749  def _is_nested(pkg: str, pkg_path: str, parent: str, parent_path: str) -> bool:
750      """ 
751      Return ``True`` if ``pkg`` is nested inside ``parent`` both logically and in the 
752      file system. 
753      >>> _is_nested("a.b", "path/a/b", "a", "path/a") 
754      True 
755      >>> _is_nested("a.b", "path/a/b", "a", "otherpath/a") 
756      False 
757      >>> _is_nested("a.b", "path/a/b", "c", "path/c") 
758      False 
759      >>> _is_nested("a.a", "path/a/a", "a", "path/a") 
760      True 
761      >>> _is_nested("b.a", "path/b/a", "a", "path/a") 
762      False 
763      """
764      norm_pkg_path = _path.normpath(pkg_path)
765      rest = pkg.replace(parent, "", 1).strip(".").split(".")
766      return pkg.startswith(parent) and norm_pkg_path == _path.normpath(
767          Path(parent_path, *rest)
768      )

796  _FINDER_TEMPLATE = """\ 
797  from __future__ import annotations 
798  import sys 
799  from importlib.machinery import ModuleSpec, PathFinder 
800  from importlib.machinery import all_suffixes as module_suffixes 
801  from importlib.util import spec_from_file_location 
802  from itertools import chain 
803  from pathlib import Path 
804   
805  MAPPING: dict[str, str] = {mapping!r} 
806  NAMESPACES: dict[str, list[str]] = {namespaces!r} 
807  PATH_PLACEHOLDER = {name!r} + ".__path_hook__" 
808   
809   
810  class _EditableFinder:  # MetaPathFinder 
811      @classmethod 
812      def find_spec(cls, fullname: str, _path=None, _target=None) -> ModuleSpec | None: 
813          # Top-level packages and modules (we know these exist in the FS) 
814          if fullname in MAPPING: 
815              pkg_path = MAPPING[fullname] 
816              return cls._find_spec(fullname, Path(pkg_path)) 
817   
818          # Handle immediate children modules (required for namespaces to work) 
819          # To avoid problems with case sensitivity in the file system we delegate 
820          # to the importlib.machinery implementation. 
821          parent, _, child = fullname.rpartition(".") 
822          if parent and parent in MAPPING: 
823              return PathFinder.find_spec(fullname, path=[MAPPING[parent]]) 
824   
825          # Other levels of nesting should be handled automatically by importlib 
826          # using the parent path. 
827          return None 
828   
829      @classmethod 
830      def _find_spec(cls, fullname: str, candidate_path: Path) -> ModuleSpec | None: 
831          init = candidate_path / "__init__.py" 
832          candidates = (candidate_path.with_suffix(x) for x in module_suffixes()) 
833          for candidate in chain([init], candidates): 
834              if candidate.exists(): 
835                  return spec_from_file_location(fullname, candidate) 
836          return None 
837   
838   
839  class _EditableNamespaceFinder:  # PathEntryFinder 
840      @classmethod 
841      def _path_hook(cls, path) -> type[_EditableNamespaceFinder]: 
842          if path == PATH_PLACEHOLDER: 
843              return cls 
844          raise ImportError 
845   
846      @classmethod 
847      def _paths(cls, fullname: str) -> list[str]: 
848          paths = NAMESPACES[fullname] 
849          if not paths and fullname in MAPPING: 
850              paths = [MAPPING[fullname]] 
851          # Always add placeholder, for 2 reasons: 
852          # 1. __path__ cannot be empty for the spec to be considered namespace. 
853          # 2. In the case of nested namespaces, we need to force 
854          #    import machinery to query _EditableNamespaceFinder again. 
855          return [*paths, PATH_PLACEHOLDER] 
856   
857      @classmethod 
858      def find_spec(cls, fullname: str, _target=None) -> ModuleSpec | None: 
859          if fullname in NAMESPACES: 
860              spec = ModuleSpec(fullname, None, is_package=True) 
861              spec.submodule_search_locations = cls._paths(fullname) 
862              return spec 
863          return None 
864   
865      @classmethod 
866      def find_module(cls, _fullname) -> None: 
867          return None 
868   
869   
870  def install(): 
871      if not any(finder == _EditableFinder for finder in sys.meta_path): 
872          sys.meta_path.append(_EditableFinder) 
873   
874      if not NAMESPACES: 
875          return 
876   
877      if not any(hook == _EditableNamespaceFinder._path_hook for hook in sys.path_hooks): 
878          # PathEntryFinder is needed to create NamespaceSpec without private APIS 
879          sys.path_hooks.append(_EditableNamespaceFinder._path_hook) 
880      if PATH_PLACEHOLDER not in sys.path: 
881          sys.path.append(PATH_PLACEHOLDER)  # Used just to trigger the path hook 
882  """
883

885  def _finder_template(
886      name: str, mapping: Mapping[str, str], namespaces: Dict[str, List[str]]
887  ) -> str:
888      """Create a string containing the code for the``MetaPathFinder`` and 
889      ``PathEntryFinder``. 
890      """
891      mapping = dict(sorted(mapping.items(), key=lambda p: p[0]))
892      return _FINDER_TEMPLATE.format(name=name, mapping=mapping, namespaces=namespaces)
