13   def _pyi_rthook():
14       import os
15       import sys
16   
17       # Prepend the frozen application's data dir to XDG_DATA_DIRS. We need to avoid overwriting the existing paths in
18       # order to allow the frozen application to run system-installed applications (for example, launch a web browser via
19       # the webbrowser module on Linux). Should the user desire complete isolation of the frozen application from the
20       # system, they need to clean up XDG_DATA_DIRS at the start of their program (i.e., remove all entries but first).
21       pyi_data_dir = os.path.join(sys._MEIPASS, 'share')
22   
23       xdg_data_dirs = os.environ.get('XDG_DATA_DIRS', None)
24       if xdg_data_dirs:
25           if pyi_data_dir not in xdg_data_dirs:
26               xdg_data_dirs = pyi_data_dir + os.pathsep + xdg_data_dirs
27       else:
28           xdg_data_dirs = pyi_data_dir
29       os.environ['XDG_DATA_DIRS'] = xdg_data_dirs
30   
31       # Cleanup aux variables
32       del xdg_data_dirs
33       del pyi_data_dir
34   
35   
36   _pyi_rthook()
37   del _pyi_rthook
