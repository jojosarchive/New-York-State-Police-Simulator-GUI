12   """ 
13   APScheduler uses entry points to dynamically load executors, job 
14   stores and triggers. 
15   This hook was tested against APScheduler 3.6.3. 
16   """
17   
18   from PyInstaller.utils.hooks import (collect_submodules, copy_metadata,
19                                        is_module_satisfies)
20   
21   if is_module_satisfies("apscheduler < 4"):
22       if is_module_satisfies("pyinstaller >= 4.4"):
23           datas = copy_metadata('APScheduler', recursive=True)
24       else:
25           datas = copy_metadata('APScheduler')
26   
27       hiddenimports = collect_submodules('apscheduler')
