37   def _execfile(filename, globals, locals=None):
38       """ 
39       Python 3 implementation of execfile. 
40       """
41       mode = 'rb'
42       with open(filename, mode) as stream:
43           script = stream.read()
44       if locals is None:
45           locals = globals
46       code = compile(script, filename, 'exec')
47       exec(code, globals, locals)

97   class UnpickleableException(Exception):
98       """ 
99       An exception representing another Exception that could not be pickled. 
100      """
101  
102      @staticmethod
103      def dump(type, exc):
104          """ 
105          Always return a dumped (pickled) type and exc. If exc can't be pickled, 
106          wrap it in UnpickleableException first. 
107          """
108          try:
109              return pickle.dumps(type), pickle.dumps(exc)
110          except Exception:
111              # get UnpickleableException inside the sandbox
112              from setuptools.sandbox import UnpickleableException as cls
113  
114              return cls.dump(cls, cls(repr(exc)))

117  class ExceptionSaver:
118      """ 
119      A Context Manager that will save an exception, serialize, and restore it 
120      later. 
121      """
122  
123      def __enter__(self):
124          return self
125  
126      def __exit__(self, type, exc, tb):
127          if not exc:
128              return False
129  
130          # dump the exception
131          self._saved = UnpickleableException.dump(type, exc)
132          self._tb = tb
133  
134          # suppress the exception
135          return True
136  
137      def resume(self):
138          "restore and re-raise any exception"
139  
140          if '_saved' not in vars(self):
141              return
142  
143          type, exc = map(pickle.loads, self._saved)
144          raise exc.with_traceback(self._tb)

147  @contextlib.contextmanager
148  def save_modules():
149      """ 
150      Context in which imported modules are saved. 
151   
152      Translates exceptions internal to the context into the equivalent exception 
153      outside the context. 
154      """
155      saved = sys.modules.copy()
156      with ExceptionSaver() as saved_exc:
157          yield saved
158  
159      sys.modules.update(saved)
160      # remove any modules imported since
161      del_modules = (
162          mod_name
163          for mod_name in sys.modules
164          if mod_name not in saved
165          # exclude any encodings modules. See #285
166          and not mod_name.startswith('encodings.')
167      )
168      _clear_modules(del_modules)
169  
170      saved_exc.resume()
171

174      for mod_name in list(module_names):
175          del sys.modules[mod_name]

187  @contextlib.contextmanager
188  def setup_context(setup_dir):
189      temp_dir = os.path.join(setup_dir, 'temp')
190      with save_pkg_resources_state():
191          with save_modules():
192              with save_path():
193                  hide_setuptools()
194                  with save_argv():
195                      with override_temp(temp_dir):
196                          with pushd(setup_dir):
197                              # ensure setuptools commands are available
198                              __import__('setuptools')
199                              yield

211  def _needs_hiding(mod_name):
212      """ 
213      >>> _needs_hiding('setuptools') 
214      True 
215      >>> _needs_hiding('pkg_resources') 
216      True 
217      >>> _needs_hiding('setuptools_plugin') 
218      False 
219      >>> _needs_hiding('setuptools.__init__') 
220      True 
221      >>> _needs_hiding('distutils') 
222      True 
223      >>> _needs_hiding('os') 
224      False 
225      >>> _needs_hiding('Cython') 
226      True 
227      """
228      base_module = mod_name.split('.', 1)[0]
229      return base_module in _MODULES_TO_HIDE

232  def hide_setuptools():
233      """ 
234      Remove references to setuptools' modules from sys.modules to allow the 
235      invocation to import the most appropriate setuptools. This technique is 
236      necessary to avoid issues such as #315 where setuptools upgrading itself 
237      would fail to find a function declared in the metadata. 
238      """
239      _distutils_hack = sys.modules.get('_distutils_hack', None)
240      if _distutils_hack is not None:
241          _distutils_hack._remove_shim()
242  
243      modules = filter(_needs_hiding, sys.modules)
244      _clear_modules(modules)

247  def run_setup(setup_script, args):
248      """Run a distutils setup script, sandboxed in its directory"""
249      setup_dir = os.path.abspath(os.path.dirname(setup_script))
250      with setup_context(setup_dir):
251          try:
252              sys.argv[:] = [setup_script] + list(args)
253              sys.path.insert(0, setup_dir)
254              # reset to include setup dir, w/clean callback list
255              working_set.__init__()
256              working_set.callbacks.append(lambda dist: dist.activate())
257  
258              with DirectorySandbox(setup_dir):
259                  ns = dict(__file__=setup_script, __name__='__main__')
260                  _execfile(setup_script, ns)
261          except SystemExit as v:
262              if v.args and v.args[0]:
263                  raise
264              # Normal exit, just return

267  class AbstractSandbox:
268      """Wrap 'os' module and 'open()' builtin for virtualizing setup scripts"""
269  
270      _active = False
271  
272      def __init__(self):
273          self._attrs = [
274              name
275              for name in dir(_os)
276              if not name.startswith('_') and hasattr(self, name)
277          ]
278  
279      def _copy(self, source):
280          for name in self._attrs:
281              setattr(os, name, getattr(source, name))
282  
283      def __enter__(self):
284          self._copy(self)
285          if _file:
286              builtins.file = self._file
287          builtins.open = self._open
288          self._active = True
289  
290      def __exit__(self, exc_type, exc_value, traceback):
291          self._active = False
292          if _file:
293              builtins.file = _file
294          builtins.open = _open
295          self._copy(_os)
296  
297      def run(self, func):
298          """Run 'func' under os sandboxing"""
299          with self:
300              return func()

302      def _mk_dual_path_wrapper(name: str):  # type: ignore[misc] # https://github.com/pypa/setuptools/pull/4099
303          original = getattr(_os, name)
304  
305          def wrap(self, src, dst, *args, **kw):
306              if self._active:
307                  src, dst = self._remap_pair(name, src, dst, *args, **kw)
308              return original(src, dst, *args, **kw)
309  
310          return wrap
311  
312      for __name in ["rename", "link", "symlink"]:
313          if hasattr(_os, __name):
314              locals()[__name] = _mk_dual_path_wrapper(__name)
315  
316      def _mk_single_path_wrapper(name: str, original=None):  # type: ignore[misc] # https://github.com/pypa/setuptools/pull/4099
317          original = original or getattr(_os, name)
318  
319          def wrap(self, path, *args, **kw):
320              if self._active:
321                  path = self._remap_input(name, path, *args, **kw)
322              return original(path, *args, **kw)
323  
324          return wrap
325  
326      if _file:
327          _file = _mk_single_path_wrapper('file', _file)
328      _open = _mk_single_path_wrapper('open', _open)
329      for __name in [
330          "stat",
331          "listdir",
332          "chdir",
333          "open",
334          "chmod",
335          "chown",
336          "mkdir",
337          "remove",
338          "unlink",
339          "rmdir",
340          "utime",
341          "lchown",
342          "chroot",
343          "lstat",
344          "startfile",
345          "mkfifo",
346          "mknod",
347          "pathconf",
348          "access",
349      ]:
350          if hasattr(_os, __name):
351              locals()[__name] = _mk_single_path_wrapper(__name)
352  
353      def _mk_single_with_return(name: str):  # type: ignore[misc] # https://github.com/pypa/setuptools/pull/4099
354          original = getattr(_os, name)
355  
356          def wrap(self, path, *args, **kw):
357              if self._active:
358                  path = self._remap_input(name, path, *args, **kw)
359                  return self._remap_output(name, original(path, *args, **kw))
360              return original(path, *args, **kw)
361  
362          return wrap
363  
364      for __name in ['readlink', 'tempnam']:
365          if hasattr(_os, __name):
366              locals()[__name] = _mk_single_with_return(__name)
367  
368      def _mk_query(name: str):  # type: ignore[misc] # https://github.com/pypa/setuptools/pull/4099
369          original = getattr(_os, name)
370  
371          def wrap(self, *args, **kw):
372              retval = original(*args, **kw)
373              if self._active:
374                  return self._remap_output(name, retval)
375              return retval
376  
377          return wrap
378  
379      for __name in ['getcwd', 'tmpnam']:
380          if hasattr(_os, __name):
381              locals()[__name] = _mk_query(__name)
382  
383      def _validate_path(self, path):
384          """Called to remap or validate any path, whether input or output"""
385          return path
386  
387      def _remap_input(self, operation, path, *args, **kw):
388          """Called for path inputs"""
389          return self._validate_path(path)
390  
391      def _remap_output(self, operation, path):
392          """Called for path outputs"""
393          return self._validate_path(path)
394  
395      def _remap_pair(self, operation, src, dst, *args, **kw):
396          """Called for path pairs like rename, link, and symlink operations"""
397          return (
398              self._remap_input(operation + '-from', src, *args, **kw),
399              self._remap_input(operation + '-to', dst, *args, **kw),
400          )

409  class DirectorySandbox(AbstractSandbox):
410      """Restrict operations to a single subdirectory - pseudo-chroot"""
411  
412      write_ops = dict.fromkeys([
413          "open",
414          "chmod",
415          "chown",
416          "mkdir",
417          "remove",
418          "unlink",
419          "rmdir",
420          "utime",
421          "lchown",
422          "chroot",
423          "mkfifo",
424          "mknod",
425          "tempnam",
426      ])
427  
428      _exception_patterns: List[Union[str, re.Pattern]] = []
429      "exempt writing to paths that match the pattern"
430  
431      def __init__(self, sandbox, exceptions=_EXCEPTIONS):
432          self._sandbox = os.path.normcase(os.path.realpath(sandbox))
433          self._prefix = os.path.join(self._sandbox, '')
434          self._exceptions = [
435              os.path.normcase(os.path.realpath(path)) for path in exceptions
436          ]
437          AbstractSandbox.__init__(self)
438  
439      def _violation(self, operation, *args, **kw):
440          from setuptools.sandbox import SandboxViolation
441  
442          raise SandboxViolation(operation, args, kw)
443  
444      if _file:
445  
446          def _file(self, path, mode='r', *args, **kw):
447              if mode not in ('r', 'rt', 'rb', 'rU', 'U') and not self._ok(path):
448                  self._violation("file", path, mode, *args, **kw)
449              return _file(path, mode, *args, **kw)
450  
451      def _open(self, path, mode='r', *args, **kw):
452          if mode not in ('r', 'rt', 'rb', 'rU', 'U') and not self._ok(path):
453              self._violation("open", path, mode, *args, **kw)
454          return _open(path, mode, *args, **kw)
455  
456      def tmpnam(self):
457          self._violation("tmpnam")
458  
459      def _ok(self, path):
460          active = self._active
461          try:
462              self._active = False
463              realpath = os.path.normcase(os.path.realpath(path))
464              return (
465                  self._exempted(realpath)
466                  or realpath == self._sandbox
467                  or realpath.startswith(self._prefix)
468              )
469          finally:
470              self._active = active
471  
472      def _exempted(self, filepath):
473          start_matches = (
474              filepath.startswith(exception) for exception in self._exceptions
475          )
476          pattern_matches = (
477              re.match(pattern, filepath) for pattern in self._exception_patterns
478          )
479          candidates = itertools.chain(start_matches, pattern_matches)
480          return any(candidates)
481  
482      def _remap_input(self, operation, path, *args, **kw):
483          """Called for path inputs"""
484          if operation in self.write_ops and not self._ok(path):
485              self._violation(operation, os.path.realpath(path), *args, **kw)
486          return path
487  
488      def _remap_pair(self, operation, src, dst, *args, **kw):
489          """Called for path pairs like rename, link, and symlink operations"""
490          if not self._ok(src) or not self._ok(dst):
491              self._violation(operation, src, dst, *args, **kw)
492          return (src, dst)
493  
494      def open(self, file, flags, mode=0o777, *args, **kw):
495          """Called for low-level os.open()"""
496          if flags & WRITE_FLAGS and not self._ok(file):
497              self._violation("os.open", file, flags, mode, *args, **kw)
498          return _os.open(file, flags, mode, *args, **kw)

503      [

505          for a in "O_WRONLY O_RDWR O_APPEND O_CREAT O_TRUNC O_TEMPORARY".split()
506      ],

510  class SandboxViolation(DistutilsError):
511      """A setup script attempted to modify the filesystem outside the sandbox"""
512  
513      tmpl = textwrap.dedent(
514          """ 
515          SandboxViolation: {cmd}{args!r} {kwargs} 
516   
517          The package setup script has attempted to modify files on your system 
518          that are not within the EasyInstall build area, and has been aborted. 
519   
520          This package cannot be safely installed by EasyInstall, and may not 
521          support alternate installation locations even if you run its setup 
522          script by hand.  Please inform the package's author and the EasyInstall 
523          maintainers to find out if a fix or workaround is available. 
524          """
525      ).lstrip()
526  
527      def __str__(self):
528          cmd, args, kwargs = self.args
529          return self.tmpl.format(**locals())
