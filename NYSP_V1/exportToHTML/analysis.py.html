11   """ 
12   Define a modified ModuleGraph that can return its contents as a TOC and in other ways act like the old ImpTracker. 
13   TODO: This class, along with TOC and Tree, should be in a separate module. 
14    
15   For reference, the ModuleGraph node types and their contents: 
16    
17    nodetype         identifier        filename 
18    
19    Script           full path to .py  full path to .py 
20    SourceModule     basename          full path to .py 
21    BuiltinModule    basename          None 
22    CompiledModule   basename          full path to .pyc 
23    Extension        basename          full path to .so 
24    MissingModule    basename          None 
25    Package          basename          full path to __init__.py 
26           packagepath is ['path to package'] 
27           globalnames is set of global names __init__.py defines 
28    ExtensionPackage basename          full path to __init__.{so,dll} 
29           packagepath is ['path to package'] 
30    
31   The main extension here over ModuleGraph is a method to extract nodes from the flattened graph and return them as a 
32   TOC, or added to a TOC. Other added methods look up nodes by identifier and return facts about them, replacing what 
33   the old ImpTracker list could do. 
34   """

62   class PyiModuleGraph(ModuleGraph):
63       """ 
64       Directed graph whose nodes represent modules and edges represent dependencies between these modules. 
65    
66       This high-level subclass wraps the lower-level `ModuleGraph` class with support for graph and runtime hooks. 
67       While each instance of `ModuleGraph` represents a set of disconnected trees, each instance of this class *only* 
68       represents a single connected tree whose root node is the Python script originally passed by the user on the 
69       command line. For that reason, while there may (and typically do) exist more than one `ModuleGraph` instance, 
70       there typically exists only a singleton instance of this class. 
71    
72       Attributes 
73       ---------- 
74       _hooks : ModuleHookCache 
75           Dictionary mapping the fully-qualified names of all modules with normal (post-graph) hooks to the absolute paths 
76           of such hooks. See the the `_find_module_path()` method for details. 
77       _hooks_pre_find_module_path : ModuleHookCache 
78           Dictionary mapping the fully-qualified names of all modules with pre-find module path hooks to the absolute 
79           paths of such hooks. See the the `_find_module_path()` method for details. 
80       _hooks_pre_safe_import_module : ModuleHookCache 
81           Dictionary mapping the fully-qualified names of all modules with pre-safe import module hooks to the absolute 
82           paths of such hooks. See the `_safe_import_module()` method for details. 
83       _user_hook_dirs : list 
84           List of the absolute paths of all directories containing user-defined hooks for the current application. 
85       _excludes : list 
86           List of module names to be excluded when searching for dependencies. 
87       _additional_files_cache : AdditionalFilesCache 
88           Cache of all external dependencies (e.g., binaries, datas) listed in hook scripts for imported modules. 
89       _module_collection_mode : dict 
90           A dictionary of module/package collection mode settings set by hook scripts for their modules. 
91       _base_modules: list 
92           Dependencies for `base_library.zip` (which remain the same for every executable). 
93       """
94   
95       # Note: these levels are completely arbitrary and may be adjusted if needed.
96       LOG_LEVEL_MAPPING = {0: INFO, 1: DEBUG, 2: TRACE, 3: TRACE, 4: TRACE}

98       def __init__(self, pyi_homepath, user_hook_dirs=(), excludes=(), **kwargs):
99           super().__init__(excludes=excludes, **kwargs)
100          # Homepath to the place where is PyInstaller located.
101          self._homepath = pyi_homepath
102          # modulegraph Node for the main python script that is analyzed by PyInstaller.
103          self._top_script_node = None
104  
105          # Absolute paths of all user-defined hook directories.
106          self._excludes = excludes
107          self._reset(user_hook_dirs)
108          self._analyze_base_modules()

110      def _reset(self, user_hook_dirs):
111          """ 
112          Reset for another set of scripts. This is primary required for running the test-suite. 
113          """
114          self._top_script_node = None
115          self._additional_files_cache = AdditionalFilesCache()
116          self._module_collection_mode = dict()
117          # Command line, Entry Point, and then builtin hook dirs.
118          self._user_hook_dirs = [*user_hook_dirs, os.path.join(PACKAGEPATH, 'hooks')]
119          # Hook-specific lookup tables. These need to reset when reusing cached PyiModuleGraph to avoid hooks to refer to
120          # files or data from another test-case.
121          logger.info('Caching module graph hooks...')
122          self._hooks = self._cache_hooks("")
123          self._hooks_pre_safe_import_module = self._cache_hooks('pre_safe_import_module')
124          self._hooks_pre_find_module_path = self._cache_hooks('pre_find_module_path')
125  
126          # Search for run-time hooks in all hook directories.
127          self._available_rthooks = defaultdict(list)
128          for uhd in self._user_hook_dirs:
129              uhd_path = os.path.abspath(os.path.join(uhd, 'rthooks.dat'))
130              try:
131                  with open(uhd_path, 'r', encoding='utf-8') as f:
132                      rthooks = ast.literal_eval(f.read())
133              except FileNotFoundError:
134                  # Ignore if this hook path doesn't have run-time hooks.
135                  continue
136              except Exception as e:
137                  logger.error('Unable to read run-time hooks from %r: %s' % (uhd_path, e))
138                  continue
139  
140              self._merge_rthooks(rthooks, uhd, uhd_path)
141  
142          # Convert back to a standard dict.
143          self._available_rthooks = dict(self._available_rthooks)
144  
145      def _merge_rthooks(self, rthooks, uhd, uhd_path):
146          """ 
147          The expected data structure for a run-time hook file is a Python dictionary of type ``Dict[str, List[str]]``, 
148          where the dictionary keys are module names and the sequence strings are Python file names. 
149   
150          Check then merge this data structure, updating the file names to be absolute. 
151          """
152          # Check that the root element is a dict.
153          assert isinstance(rthooks, dict), 'The root element in %s must be a dict.' % uhd_path
154          for module_name, python_file_name_list in rthooks.items():
155              # Ensure the key is a string.
156              assert isinstance(module_name, str), \
157                  '%s must be a dict whose keys are strings; %s is not a string.' % (uhd_path, module_name)
158              # Ensure the value is a list.
159              assert isinstance(python_file_name_list, list), \
160                  'The value of %s key %s must be a list.' % (uhd_path, module_name)
161              if module_name in self._available_rthooks:
162                  logger.warning(
163                      'Runtime hooks for %s have already been defined. Skipping the runtime hooks for %s that are '
164                      'defined in %s.', module_name, module_name, os.path.join(uhd, 'rthooks')
165                  )
166                  # Skip this module
167                  continue
168              # Merge this with existing run-time hooks.
169              for python_file_name in python_file_name_list:
170                  # Ensure each item in the list is a string.
171                  assert isinstance(python_file_name, str), \
172                      '%s key %s, item %r must be a string.' % (uhd_path, module_name, python_file_name)
173                  # Transform it into an absolute path.
174                  abs_path = os.path.join(uhd, 'rthooks', python_file_name)
175                  # Make sure this file exists.
176                  assert os.path.exists(abs_path), \
177                      'In %s, key %s, the file %r expected to be located at %r does not exist.' % \
178                      (uhd_path, module_name, python_file_name, abs_path)
179                  # Merge it.
180                  self._available_rthooks[module_name].append(abs_path)

182      @staticmethod
183      def _findCaller(*args, **kwargs):
184          # Used to add an additional stack-frame above logger.findCaller. findCaller expects the caller to be three
185          # stack-frames above itself.
186          return logger.findCaller(*args, **kwargs)

188      def msg(self, level, s, *args):
189          """ 
190          Print a debug message with the given level. 
191   
192          1. Map the msg log level to a logger log level. 
193          2. Generate the message format (the same format as ModuleGraph) 
194          3. Find the caller, which findCaller expects three stack-frames above itself: 
195              [3] caller -> [2] msg (here) -> [1] _findCaller -> [0] logger.findCaller 
196          4. Create a logRecord with the caller's information. 
197          5. Handle the logRecord. 
198          """
199          try:
200              level = self.LOG_LEVEL_MAPPING[level]
201          except KeyError:
202              return
203          if not logger.isEnabledFor(level):
204              return
205  
206          msg = "%s %s" % (s, ' '.join(map(repr, args)))
207  
208          try:
209              fn, lno, func, sinfo = self._findCaller()
210          except ValueError:  # pragma: no cover
211              fn, lno, func, sinfo = "(unknown file)", 0, "(unknown function)", None
212          record = logger.makeRecord(logger.name, level, fn, lno, msg, [], None, func, None, sinfo)
213  
214          logger.handle(record)
215  
216      # Set logging methods so that the stack is correctly detected.
217      msgin = msg
218      msgout = msg

220      def _cache_hooks(self, hook_type):
221          """ 
222          Get a cache of all hooks of the passed type. 
223   
224          The cache will include all official hooks defined by the PyInstaller codebase _and_ all unofficial hooks 
225          defined for the current application. 
226   
227          Parameters 
228          ---------- 
229          hook_type : str 
230              Type of hooks to be cached, equivalent to the basename of the subpackage of the `PyInstaller.hooks` 
231              package containing such hooks (e.g., `post_create_package` for post-create package hooks). 
232          """
233          # Cache of this type of hooks.
234          hook_dirs = []
235          for user_hook_dir in self._user_hook_dirs:
236              # Absolute path of the user-defined subdirectory of this hook type. If this directory exists, add it to the
237              # list to be cached.
238              user_hook_type_dir = os.path.join(user_hook_dir, hook_type)
239              if os.path.isdir(user_hook_type_dir):
240                  hook_dirs.append(user_hook_type_dir)
241  
242          return ModuleHookCache(self, hook_dirs)

244      def _analyze_base_modules(self):
245          """ 
246          Analyze dependencies of the the modules in base_library.zip. 
247          """
248          logger.info('Analyzing base_library.zip ...')
249          required_mods = []
250          # Collect submodules from required modules in base_library.zip.
251          for m in PY3_BASE_MODULES:
252              if is_package(m):
253                  required_mods += collect_submodules(m)
254              else:
255                  required_mods.append(m)
256          # Initialize ModuleGraph.
257          self._base_modules = [mod for req in required_mods for mod in self.import_hook(req)]

259      def add_script(self, pathname, caller=None):
260          """ 
261          Wrap the parent's 'run_script' method and create graph from the first script in the analysis, and save its 
262          node to use as the "caller" node for all others. This gives a connected graph rather than a collection of 
263          unrelated trees. 
264          """
265          if self._top_script_node is None:
266              # Remember the node for the first script.
267              try:
268                  self._top_script_node = super().add_script(pathname)
269              except SyntaxError:
270                  print("\nSyntax error in", pathname, file=sys.stderr)
271                  formatted_lines = traceback.format_exc().splitlines(True)
272                  print(*formatted_lines[-4:], file=sys.stderr)
273                  sys.exit(1)
274              # Create references from the top script to the base_modules in graph.
275              for node in self._base_modules:
276                  self.add_edge(self._top_script_node, node)
277              # Return top-level script node.
278              return self._top_script_node
279          else:
280              if not caller:
281                  # Defaults to as any additional script is called from the top-level script.
282                  caller = self._top_script_node
283              return super().add_script(pathname, caller=caller)

285      def process_post_graph_hooks(self, analysis):
286          """ 
287          For each imported module, run this module's post-graph hooks if any. 
288   
289          Parameters 
290          ---------- 
291          analysis: build_main.Analysis 
292              The Analysis that calls the hooks 
293   
294          """
295          # For each iteration of the infinite "while" loop below:
296          #
297          # 1. All hook() functions defined in cached hooks for imported modules are called. This may result in new
298          #    modules being imported (e.g., as hidden imports) that were ignored earlier in the current iteration: if
299          #    this is the case, all hook() functions defined in cached hooks for these modules will be called by the next
300          #    iteration.
301          # 2. All cached hooks whose hook() functions were called are removed from this cache. If this cache is empty, no
302          #    hook() functions will be called by the next iteration and this loop will be terminated.
303          # 3. If no hook() functions were called, this loop is terminated.
304          logger.info('Processing module hooks...')
305          while True:
306              # Set of the names of all imported modules whose post-graph hooks are run by this iteration, preventing the
307              # next iteration from re- running these hooks. If still empty at the end of this iteration, no post-graph
308              # hooks were run; thus, this loop will be terminated.
309              hooked_module_names = set()
310  
311              # For each remaining hookable module and corresponding hooks...
312              for module_name, module_hooks in self._hooks.items():
313                  # Graph node for this module if imported or "None" otherwise.
314                  module_node = self.find_node(module_name, create_nspkg=False)
315  
316                  # If this module has not been imported, temporarily ignore it. This module is retained in the cache, as
317                  # a subsequently run post-graph hook could import this module as a hidden import.
318                  if module_node is None:
319                      continue
320  
321                  # If this module is unimportable, permanently ignore it.
322                  if type(module_node).__name__ not in VALID_MODULE_TYPES:
323                      hooked_module_names.add(module_name)
324                      continue
325  
326                  # For each hook script for this module...
327                  for module_hook in module_hooks:
328                      # Run this script's post-graph hook.
329                      module_hook.post_graph(analysis)
330  
331                      # Cache all external dependencies listed by this script after running this hook, which could add
332                      # dependencies.
333                      self._additional_files_cache.add(module_name, module_hook.binaries, module_hook.datas)
334  
335                      # Update package collection mode settings.
336                      self._module_collection_mode.update(module_hook.module_collection_mode)
337  
338                  # Prevent this module's hooks from being run again.
339                  hooked_module_names.add(module_name)
340  
341              # Prevent all post-graph hooks run above from being run again by the next iteration.
342              self._hooks.remove_modules(*hooked_module_names)
343  
344              # If no post-graph hooks were run, terminate iteration.
345              if not hooked_module_names:
346                  break

348      def _find_all_excluded_imports(self, module_name):
349          """ 
350          Collect excludedimports from the hooks of the specified module and all its parents. 
351          """
352          excluded_imports = set()
353          while module_name:
354              # Gather excluded imports from hook(s) belonging to the module
355              for module_hook in self._hooks.get(module_name, []):
356                  excluded_imports.update(module_hook.excludedimports)
357              # Change module name to the module's parent name
358              module_name = module_name.rpartition('.')[0]
359          return excluded_imports

361      def _safe_import_hook(
362          self, target_module_partname, source_module, target_attr_names, level=DEFAULT_IMPORT_LEVEL, edge_attr=None
363      ):
364          if source_module is not None:
365              # Gather all excluded imports for the referring modules, as well as its parents.
366              # For example, we want the excluded imports specified by hook for PIL to be also applied when the referring
367              # module is its submodule, PIL.Image.
368              excluded_imports = self._find_all_excluded_imports(source_module.identifier)
369  
370              # Apply extra processing only if we have any excluded-imports rules
371              if excluded_imports:
372                  # Resolve the base module name. Level can be ABSOLUTE_IMPORT_LEVEL (= 0) for absolute imports, or an
373                  # integer indicating the relative level. We do not use equality comparison just in case we ever happen
374                  # to get ABSOLUTE_OR_RELATIVE_IMPORT_LEVEL (-1), which is a remnant of python2 days.
375                  if level > ABSOLUTE_IMPORT_LEVEL:
376                      if isinstance(source_module, Package):
377                          # Package
378                          base_module_name = source_module.identifier
379                      else:
380                          # Module in a package; base name must be the parent package name!
381                          base_module_name = '.'.join(source_module.identifier.split('.')[:-1])
382  
383                      if target_module_partname:
384                          base_module_name += '.' + target_module_partname
385  
386                      # Adjust the base module name based on level
387                      if level > 1:
388                          base_module_name = '.'.join(base_module_name.split('.')[:-(level - 1)])
389                  else:
390                      base_module_name = target_module_partname
391  
392                  def _exclude_module(module_name, excluded_imports):
393                      """ 
394                      Helper for checking whether given module should be excluded. 
395                      Returns the name of exclusion rule if module should be excluded, None otherwise. 
396                      """
397                      module_name_parts = module_name.split('.')
398                      for excluded_import in excluded_imports:
399                          excluded_import_parts = excluded_import.split('.')
400                          match = module_name_parts[:len(excluded_import_parts)] == excluded_import_parts
401                          if match:
402                              return excluded_import
403                      return None
404  
405                  # First, check if base module name is to be excluded.
406                  # This covers both basic `import a` and `import a.b.c`, as well as `from d import e, f` where base
407                  # module `d` is excluded.
408                  excluded_import_rule = _exclude_module(base_module_name, excluded_imports)
409                  if excluded_import_rule:
410                      logger.debug(
411                          "Suppressing import of %r from module %r due to excluded import %r specified in a hook for %r "
412                          "(or its parent package(s)).", base_module_name, source_module.identifier, excluded_import_rule,
413                          source_module.identifier
414                      )
415                      return []
416  
417                  # If we have target attribute names, check each of them, and remove excluded ones from the
418                  # `target_attr_names` list.
419                  if target_attr_names:
420                      filtered_target_attr_names = []
421                      for target_attr_name in target_attr_names:
422                          submodule_name = base_module_name + '.' + target_attr_name
423                          excluded_import_rule = _exclude_module(submodule_name, excluded_imports)
424                          if excluded_import_rule:
425                              logger.debug(
426                                  "Suppressing import of %r from module %r due to excluded import %r specified in a hook "
427                                  "for %r (or its parent package(s)).", submodule_name, source_module.identifier,
428                                  excluded_import_rule, source_module.identifier
429                              )
430                          else:
431                              filtered_target_attr_names.append(target_attr_name)
432  
433                      # Swap with filtered target attribute names list; if no elements remain after the filtering, pass
434                      # None...
435                      target_attr_names = filtered_target_attr_names or None
436  
437          ret_modules = super()._safe_import_hook(
438              target_module_partname, source_module, target_attr_names, level, edge_attr
439          )
440  
441          # Ensure that hooks are pre-loaded for returned module(s), in an attempt to ensure that hooks are called in the
442          # order of imports. The hooks are cached, so there should be no downsides to pre-loading hooks early (as opposed
443          # to loading them in post-graph analysis). When modules are imported from other modules, the hooks for those
444          # referring (source) modules and their parent package(s) are loaded by the exclusion mechanism that takes place
445          # before the above `super()._safe_import_hook` call. The code below attempts to complement that, but for the
446          # referred (target) modules and their parent package(s).
447          for ret_module in ret_modules:
448              if type(ret_module).__name__ not in VALID_MODULE_TYPES:
449                  continue
450              # (Ab)use the `_find_all_excluded_imports` helper to load all hooks for the given module and its parent
451              # package(s).
452              self._find_all_excluded_imports(ret_module.identifier)
453  
454          return ret_modules
455  
456      def _safe_import_module(self, module_basename, module_name, parent_package):
457          """ 
458          Create a new graph node for the module with the passed name under the parent package signified by the passed 
459          graph node. 
460   
461          This method wraps the superclass method with support for pre-import module hooks. If such a hook exists for 
462          this module (e.g., a script `PyInstaller.hooks.hook-{module_name}` containing a function 
463          `pre_safe_import_module()`), that hook will be run _before_ the superclass method is called. 
464   
465          Pre-Safe-Import-Hooks are performed just *prior* to importing the module. When running the hook, the modules 
466          parent package has already been imported and ti's `__path__` is set up. But the module is just about to be 
467          imported. 
468   
469          See the superclass method for description of parameters and return value. 
470          """
471          # If this module has pre-safe import module hooks, run these first.
472          if module_name in self._hooks_pre_safe_import_module:
473              # For the absolute path of each such hook...
474              for hook in self._hooks_pre_safe_import_module[module_name]:
475                  # Dynamically import this hook as a fabricated module.
476                  logger.info('Processing pre-safe import module hook %s from %r.', module_name, hook.hook_filename)
477                  hook_module_name = 'PyInstaller_hooks_pre_safe_import_module_' + module_name.replace('.', '_')
478                  hook_module = importlib_load_source(hook_module_name, hook.hook_filename)
479  
480                  # Object communicating changes made by this hook back to us.
481                  hook_api = PreSafeImportModuleAPI(
482                      module_graph=self,
483                      module_basename=module_basename,
484                      module_name=module_name,
485                      parent_package=parent_package,
486                  )
487  
488                  # Run this hook, passed this object.
489                  if not hasattr(hook_module, 'pre_safe_import_module'):
490                      raise NameError('pre_safe_import_module() function not defined by hook %r.' % hook_module)
491                  hook_module.pre_safe_import_module(hook_api)
492  
493                  # Respect method call changes requested by this hook.
494                  module_basename = hook_api.module_basename
495                  module_name = hook_api.module_name
496  
497              # Prevent subsequent calls from rerunning these hooks.
498              del self._hooks_pre_safe_import_module[module_name]
499  
500          # Call the superclass method.
501          return super()._safe_import_module(module_basename, module_name, parent_package)

503      def _find_module_path(self, fullname, module_name, search_dirs):
504          """ 
505          Get a 3-tuple detailing the physical location of the module with the passed name if that module exists _or_ 
506          raise `ImportError` otherwise. 
507   
508          This method wraps the superclass method with support for pre-find module path hooks. If such a hook exists 
509          for this module (e.g., a script `PyInstaller.hooks.hook-{module_name}` containing a function 
510          `pre_find_module_path()`), that hook will be run _before_ the superclass method is called. 
511   
512          See superclass method for parameter and return value descriptions. 
513          """
514          # If this module has pre-find module path hooks, run these first.
515          if fullname in self._hooks_pre_find_module_path:
516              # For the absolute path of each such hook...
517              for hook in self._hooks_pre_find_module_path[fullname]:
518                  # Dynamically import this hook as a fabricated module.
519                  logger.info('Processing pre-find module path hook %s from %r.', fullname, hook.hook_filename)
520                  hook_fullname = 'PyInstaller_hooks_pre_find_module_path_' + fullname.replace('.', '_')
521                  hook_module = importlib_load_source(hook_fullname, hook.hook_filename)
522  
523                  # Object communicating changes made by this hook back to us.
524                  hook_api = PreFindModulePathAPI(
525                      module_graph=self,
526                      module_name=fullname,
527                      search_dirs=search_dirs,
528                  )
529  
530                  # Run this hook, passed this object.
531                  if not hasattr(hook_module, 'pre_find_module_path'):
532                      raise NameError('pre_find_module_path() function not defined by hook %r.' % hook_module)
533                  hook_module.pre_find_module_path(hook_api)
534  
535                  # Respect method call changes requested by this hook.
536                  search_dirs = hook_api.search_dirs
537  
538              # Prevent subsequent calls from rerunning these hooks.

565      def _make_toc(self, typecode=None):
566          """ 
567          Return the name, path and type of selected nodes as a TOC. The selection is determined by the given list 
568          of PyInstaller TOC typecodes. If that list is empty we return the complete flattened graph as a TOC with the 
569          ModuleGraph note types in place of typecodes -- meant for debugging only. Normally we return ModuleGraph 
570          nodes whose types map to the requested PyInstaller typecode(s) as indicated in the MODULE_TYPES_TO_TOC_DICT. 
571   
572          We use the ModuleGraph (really, ObjectGraph) flatten() method to scan all the nodes. This is patterned after 
573          ModuleGraph.report(). 
574          """
575          # Construct regular expression for matching modules that should be excluded because they are bundled in
576          # base_library.zip.
577          #
578          # This expression matches the base module name, optionally followed by a period and then any number of
579          # characters. This matches the module name and the fully qualified names of any of its submodules.
580          regex_str = '(' + '|'.join(PY3_BASE_MODULES) + r')(\.|$)'
581          module_filter = re.compile(regex_str)
582  
583          toc = list()
584          for node in self.iter_graph(start=self._top_script_node):
585              # Skip modules that are in base_library.zip.
586              if module_filter.match(node.identifier):
587                  continue
588              entry = self._node_to_toc(node, typecode)
589              # Append the entry. We do not check for duplicates here; the TOC normalization is left to caller.
590              # However, as entries are obtained from modulegraph, there should not be any duplicates at this stage.
591              if entry is not None:
592                  toc.append(entry)
593          return toc

614      @staticmethod
615      def _node_to_toc(node, typecode=None):
616          # TODO This is terrible. Everything in Python has a type. It is nonsensical to even speak of "nodes [that] are
617          # not typed." How would that even occur? After all, even "None" has a type! (It is "NoneType", for the curious.)
618          # Remove this, please.
619  
620          # Get node type, e.g., Script
621          mg_type = type(node).__name__
622          assert mg_type is not None
623  
624          if typecode and mg_type not in typecode:
625              # Type is not a to be selected one, skip this one
626              return None
627          # Extract the identifier and a path if any.
628          if mg_type == 'Script':
629              # for Script nodes only, identifier is a whole path
630              (name, ext) = os.path.splitext(node.filename)
631              name = os.path.basename(name)
632          elif mg_type == 'ExtensionPackage':
633              # Package with __init__ module being an extension module. This needs to end up as e.g. 'mypkg/__init__.so'.
634              # Convert the packages name ('mypkg') into the module name ('mypkg.__init__') *here* to keep special cases
635              # away elsewhere (where the module name is converted to a filename).
636              name = node.identifier + ".__init__"
637          else:
638              name = node.identifier
639          path = node.filename if node.filename is not None else ''
640          # Ensure name is really 'str'. Module graph might return object type 'modulegraph.Alias' which inherits fromm
641          # 'str'. But 'marshal.dumps()' function is able to marshal only 'str'. Otherwise on Windows PyInstaller might
642          # fail with message like:
643          #   ValueError: unmarshallable object
644          name = str(name)
645          # Translate to the corresponding TOC typecode.
646          toc_type = MODULE_TYPES_TO_TOC_DICT[mg_type]
647          return name, path, toc_type

664      def get_importers(self, name):
665          """ 
666          List all modules importing the module with the passed name. 
667   
668          Returns a list of (identifier, DependencyIinfo)-tuples. If the names module has not yet been imported, this 
669          method returns an empty list. 
670   
671          Parameters 
672          ---------- 
673          name : str 
674              Fully-qualified name of the module to be examined. 
675   
676          Returns 
677          ---------- 
678          list 
679              List of (fully-qualified names, DependencyIinfo)-tuples of all modules importing the module with the passed 
680              fully-qualified name. 
681   
682          """
683          def get_importer_edge_data(importer):
684              edge = self.graph.edge_by_node(importer, name)
685              # edge might be None in case an AliasModule was added.
686              if edge is not None:
687                  return self.graph.edge_data(edge)
688  
689          node = self.find_node(name)
690          if node is None:
691              return []
692          _, importers = self.get_edges(node)
693          importers = (importer.identifier for importer in importers if importer is not None)
694          return [(importer, get_importer_edge_data(importer)) for importer in importers]

697      def analyze_runtime_hooks(self, custom_runhooks):
698          """ 
699          Analyze custom run-time hooks and run-time hooks implied by found modules. 
700   
701          :return : list of Graph nodes. 
702          """
703          rthooks_nodes = []
704          logger.info('Analyzing run-time hooks ...')
705          # Process custom runtime hooks (from --runtime-hook options). The runtime hooks are order dependent. First hooks
706          # in the list are executed first. Put their graph nodes at the head of the priority_scripts list Pyinstaller
707          # defined rthooks and thus they are executed first.
708          if custom_runhooks:
709              for hook_file in custom_runhooks:
710                  logger.info("Including custom run-time hook %r", hook_file)
711                  hook_file = os.path.abspath(hook_file)
712                  # Not using "try" here because the path is supposed to exist, if it does not, the raised error will
713                  # explain.
714                  rthooks_nodes.append(self.add_script(hook_file))
715  
716          # Find runtime hooks that are implied by packages already imported. Get a temporary TOC listing all the scripts
717          # and packages graphed so far. Assuming that runtime hooks apply only to modules and packages.
718          temp_toc = self._make_toc(VALID_MODULE_TYPES)
719          for (mod_name, path, typecode) in temp_toc:
720              # Look if there is any run-time hook for given module.
721              if mod_name in self._available_rthooks:
722                  # There could be several run-time hooks for a module.
723                  for abs_path in self._available_rthooks[mod_name]:
724                      logger.info("Including run-time hook %r", abs_path)
725                      rthooks_nodes.append(self.add_script(abs_path))
726  
727          return rthooks_nodes

729      def add_hiddenimports(self, module_list):
730          """ 
731          Add hidden imports that are either supplied as CLI option --hidden-import=MODULENAME or as dependencies from 
732          some PyInstaller features when enabled (e.g., crypto feature). 
733          """
734          assert self._top_script_node is not None
735          # Analyze the script's hidden imports (named on the command line).
736          for modnm in module_list:
737              node = self.find_node(modnm)
738              if node is not None:
739                  logger.debug('Hidden import %r already found', modnm)
740              else:
741                  logger.info("Analyzing hidden import %r", modnm)
742                  # ModuleGraph throws ImportError if import not found.
743                  try:
744                      nodes = self.import_hook(modnm)
745                      assert len(nodes) == 1
746                      node = nodes[0]
747                  except ImportError:
748                      logger.error("Hidden import %r not found", modnm)
749                      continue
750              # Create references from the top script to the hidden import, even if found otherwise. Do not waste time
751              # checking whether it is actually added by this (test-) script.
752              self.add_edge(self._top_script_node, node)

754      def get_code_using(self, module: str) -> dict:
755          """ 
756          Find modules that import a given **module**. 
757          """
758          co_dict = {}
759          pure_python_module_types = PURE_PYTHON_MODULE_TYPES | {
760              'Script',
761          }
762          node = self.find_node(module)
763          if node:
764              referrers = self.incoming(node)
765              for r in referrers:
766                  # Under python 3.7 and earlier, if `module` is added to hidden imports, one of referrers ends up being
767                  # None, causing #3825. Work around it.
768                  if r is None:
769                      continue
770                  # Ensure that modulegraph objects have 'code' attribute.
771                  if type(r).__name__ not in pure_python_module_types:
772                      continue
773                  identifier = r.identifier
774                  if identifier == module or identifier.startswith(module + '.'):
775                      # Skip self references or references from `modules`'s own submodules.
776                      continue
777                  # The code object may be None if referrer ends up shadowed by eponymous directory that ends up treated
778                  # as a namespace package. See #6873 for an example.
779                  if r.code is None:
780                      continue
781                  co_dict[r.identifier] = r.code
782          return co_dict

809      def _metadata_from(self, package, methods=(), recursive_methods=()) -> set:
810          """ 
811          Collect metadata whose requirements are implied by given function names. 
812   
813          Args: 
814              package: 
815                  The module name that must be imported in a source file to trigger the search. 
816              methods: 
817                  Function names from **package** which take a distribution name as an argument and imply that metadata 
818                  is required for that distribution. 
819              recursive_methods: 
820                  Like **methods** but also implies that a distribution's dependencies' metadata must be collected too. 
821          Returns: 
822              Required metadata in hook data ``(source, dest)`` format as returned by 
823              :func:`PyInstaller.utils.hooks.copy_metadata()`. 
824   
825          Scan all source code to be included for usage of particular *key* functions which imply that that code will 
826          require metadata for some distribution (which may not be its own) at runtime. In the case of a match, 
827          collect the required metadata. 
828          """
829          from PyInstaller.utils.hooks import copy_metadata
830          from PyInstaller.compat import importlib_metadata
831  
832          # Generate sets of possible function names to search for.
833          need_metadata = set()
834          need_recursive_metadata = set()
835          for method in methods:
836              need_metadata.update(bytecode.any_alias(package + "." + method))
837          for method in recursive_methods:
838              need_recursive_metadata.update(bytecode.any_alias(package + "." + method))
839  
840          out = set()
841  
842          for name, code in self.get_code_using(package).items():
843              for calls in bytecode.recursive_function_calls(code).values():
844                  for function_name, args in calls:
845                      # Only consider function calls taking one argument.
846                      if len(args) != 1:
847                          continue
848                      package = args[0]
849                      try:
850                          if function_name in need_metadata:
851                              out.update(copy_metadata(package))
852                          elif function_name in need_recursive_metadata:
853                              out.update(copy_metadata(package, recursive=True))
854  
855                      except importlib_metadata.PackageNotFoundError:
856                          # Currently, we opt to silently skip over missing metadata.
857                          continue
858  
859          return out

899  def initialize_modgraph(excludes=(), user_hook_dirs=()):
900      """ 
901      Create the cached module graph. 
902   
903      This function might appear weird but is necessary for speeding up test runtime because it allows caching basic 
904      ModuleGraph object that gets created for 'base_library.zip'. 
905   
906      Parameters 
907      ---------- 
908      excludes : list 
909          List of the fully-qualified names of all modules to be "excluded" and hence _not_ frozen into the executable. 
910      user_hook_dirs : list 
911          List of the absolute paths of all directories containing user-defined hooks for the current application or 
912          `None` if no such directories were specified. 
913   
914      Returns 
915      ---------- 
916      PyiModuleGraph 
917          Module graph with core dependencies. 
918      """
919      # Normalize parameters to ensure tuples and make comparison work.
920      user_hook_dirs = user_hook_dirs or ()
921      excludes = excludes or ()
922  
923      # Ensure that __main__ is always excluded from the modulegraph, to prevent accidentally pulling PyInstaller itself
924      # into the modulegraph. This seems to happen on Windows, because modulegraph is able to resolve `__main__` as
925      # `.../PyInstaller.exe/__main__.py` and analyze it. The `__main__` has a different meaning during analysis compared
926      # to the program run-time, when it refers to the program's entry-point (which would always be part of the
927      # modulegraph anyway, by virtue of being the starting point of the analysis).
928      if "__main__" not in excludes:
929          excludes += ("__main__",)
930  
931      # If there is a graph cached with the same excludes, reuse it. See ``PyiModulegraph._reset()`` for what is
932      # reset. This cache is used primarily to speed up the test-suite. Fixture `pyi_modgraph` calls this function with
933      # empty excludes, creating a graph suitable for the huge majority of tests.
934      global _cached_module_graph_
935      if _cached_module_graph_ and _cached_module_graph_._excludes == excludes:
936          logger.info('Reusing cached module dependency graph...')
937          graph = deepcopy(_cached_module_graph_)
938          graph._reset(user_hook_dirs)
939          return graph
940  
941      logger.info('Initializing module dependency graph...')
942  
943      # Construct the initial module graph by analyzing all import statements.
944      graph = PyiModuleGraph(
945          HOMEPATH,
946          excludes=excludes,
947          # get_implies() are hidden imports known by modulgraph.
948          implies=get_implies(),
949          user_hook_dirs=user_hook_dirs,
950      )
951  
952      if not _cached_module_graph_:
953          # Only cache the first graph, see above for explanation.
954          logger.info('Caching module dependency graph...')
955          # cache a deep copy of the graph
956          _cached_module_graph_ = deepcopy(graph)
957          # Clear data which does not need to be copied from the cached graph since it will be reset by
958          # ``PyiModulegraph._reset()`` anyway.
959          _cached_module_graph_._hooks = None
960          _cached_module_graph_._hooks_pre_safe_import_module = None
961          _cached_module_graph_._hooks_pre_find_module_path = None
962  
963      return graph

966  def get_bootstrap_modules():
967      """ 
968      Get TOC with the bootstrapping modules and their dependencies. 
969      :return: TOC with modules 
970      """
971      # Import 'struct' modules to get real paths to module file names.
972      mod_struct = __import__('struct')
973      # Basic modules necessary for the bootstrap process.
974      loader_mods = list()
975      loaderpath = os.path.join(HOMEPATH, 'PyInstaller', 'loader')
976      # On some platforms (Windows, Debian/Ubuntu) '_struct' and zlib modules are built-in modules (linked statically)
977      # and thus does not have attribute __file__. 'struct' module is required for reading Python bytecode from
978      # executable. 'zlib' is required to decompress this bytecode.
979      for mod_name in ['_struct', 'zlib']:
980          mod = __import__(mod_name)  # C extension.
981          if hasattr(mod, '__file__'):
982              mod_file = os.path.abspath(mod.__file__)
983              if os.path.basename(os.path.dirname(mod_file)) == 'lib-dynload':
984                  # Divert extensions originating from python's lib-dynload directory, to match behavior of #5604.
985                  mod_name = os.path.join('lib-dynload', mod_name)
986              loader_mods.append(add_suffix_to_extension(mod_name, mod_file, 'EXTENSION'))
987      loader_mods.append(('struct', os.path.abspath(mod_struct.__file__), 'PYMODULE'))
988      # Loader/bootstrap modules.
989      # NOTE: These modules should be kept simple without any complicated dependencies.
990      loader_mods += [
991          ('pyimod01_archive', os.path.join(loaderpath, 'pyimod01_archive.py'), 'PYMODULE'),
992          ('pyimod02_importers', os.path.join(loaderpath, 'pyimod02_importers.py'), 'PYMODULE'),
993          ('pyimod03_ctypes', os.path.join(loaderpath, 'pyimod03_ctypes.py'), 'PYMODULE'),
994      ]
995      if is_win:
996          loader_mods.append(('pyimod04_pywin32', os.path.join(loaderpath, 'pyimod04_pywin32.py'), 'PYMODULE'))
997      # The bootstrap script
998      loader_mods.append(('pyiboot01_bootstrap', os.path.join(loaderpath, 'pyiboot01_bootstrap.py'), 'PYSOURCE'))
999      return loader_mods
