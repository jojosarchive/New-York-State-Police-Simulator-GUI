11   """ 
12   This module holds run-time PyInstaller configuration. 
13    
14   Variable CONF is a dict() with all configuration options that are necessary for the build phase. Build phase is done by 
15   passing .spec file to exec() function. CONF variable is the only way how to pass arguments to exec() and how to avoid 
16   using 'global' variables. 
17    
18   NOTE: Having 'global' variables does not play well with the test suite because it does not provide isolated environments 
19   for tests. Some tests might fail in this case. 
20    
21   NOTE: The 'CONF' dict() is cleaned after building phase to not interfere with any other possible test. 
22    
23   To pass any arguments to build phase, just do: 
24    
25       from PyInstaller.config import CONF 
26       CONF['my_var_name'] = my_value 
27    
28   And to use this variable in the build phase: 
29    
30       from PyInstaller.config import CONF 
31       foo = CONF['my_var_name'] 
32    
33    
34   This is the list of known variables. (Please update it if necessary.) 
35    
36   cachedir 
37   hiddenimports 
38   noconfirm 
39   pathex 
40   ui_admin 
41   ui_access 
42   upx_available 
43   upx_dir 
44   workpath 
45    
46   tests_modgraph  - cached PyiModuleGraph object to speed up tests 
47    
48   code_cache - dictionary associating `Analysis.pure` list instances with code cache dictionaries. Used by PYZ writer. 
49   """
50   
51   # NOTE: Do not import other PyInstaller modules here. Just define constants here.
52   
53   CONF = {
54       # Unit tests require this key to exist.
55       'pathex': [],
56   }
