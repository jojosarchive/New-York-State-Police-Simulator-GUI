11   """ 
12   importlib_metadata is a library to access the metadata for a Python package. This functionality intends to replace most 
13   uses of pkg_resources entry point API and metadata API. 
14   """
15   
16   from PyInstaller.utils.hooks import copy_metadata
17   
18   # Normally, we should never need to use copy_metadata() in a hook since metadata requirements detection is now
19   # automatic. However, that detection first uses `PyiModuleGraph.get_code_using("importlib_metadata")` to find
20   # files which `import importlib_metadata` and `get_code_using()` intentionally excludes internal imports. This
21   # means that importlib_metadata is not scanned for usages of importlib_metadata and therefore when
22   # importlib_metadata uses its own API to get its version, this goes undetected. Therefore, we must collect its
23   # metadata manually.
24   datas = copy_metadata('importlib_metadata')
