13   
14   DIR = os.path.dirname(__file__)

20   def get_test_dirs():
21   
22       dirs = []
23       # For every directory and sub directory (including cwd)
24       for path, _, _ in os.walk(DIR):
25           # Add the norm'd path to dirs
26           dirs.append(os.path.normpath(path))
27   
28       return dirs
