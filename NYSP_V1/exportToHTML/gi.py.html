28   class GiModuleInfo:
29       def __init__(self, module, version, hook_api=None):
30           self.name = module
31           self.version = version
32           self.available = False
33           self.sharedlibs = []
34           self.typelib = None
35           self.dependencies = []
36   
37           # If hook API is available, use it to override the version from hookconfig.
38           if hook_api is not None:
39               module_versions = get_hook_config(hook_api, 'gi', 'module-versions')
40               if module_versions:
41                   self.version = module_versions.get(module, version)
42   
43           logger.debug("Gathering GI module info for %s %s", module, self.version)
44   
45           @isolated.decorate
46           def _get_module_info(module, version):
47               import gi
48               gi.require_version("GIRepository", "2.0")
49               from gi.repository import GIRepository
50   
51               repo = GIRepository.Repository.get_default()
52               repo.require(module, version, GIRepository.RepositoryLoadFlags.IREPOSITORY_LOAD_FLAG_LAZY)
53   
54               # Shared library/libraries
55               # Comma-separated list of paths to shared libraries, or None if none are associated. Convert to list.
56               sharedlibs = repo.get_shared_library(module)
57               sharedlibs = [lib.strip() for lib in sharedlibs.split(",")] if sharedlibs else []
58   
59               # Path to .typelib file
60               typelib = repo.get_typelib_path(module)
61   
62               # Dependencies
63               # GIRepository.Repository.get_immediate_dependencies is available from gobject-introspection v1.44 on
64               if hasattr(repo, 'get_immediate_dependencies'):
65                   dependencies = repo.get_immediate_dependencies(module)
66               else:
67                   dependencies = repo.get_dependencies(module)
68   
69               return {
70                   'sharedlibs': sharedlibs,
71                   'typelib': typelib,
72                   'dependencies': dependencies,
73               }
74   
75           # Try to query information; if this fails, mark module as unavailable.
76           try:
77               info = _get_module_info(module, self.version)
78               self.sharedlibs = info['sharedlibs']
79               self.typelib = info['typelib']
80               self.dependencies = info['dependencies']
81               self.available = True
82           except Exception as e:
83               logger.debug("Failed to query GI module %s %s: %s", module, self.version, e)
84               self.available = False
85   
86       def get_libdir(self):
87           """ 
88           Return the path to shared library used by the module. If no libraries are associated with the typelib, None is 
89           returned. If multiple library names are associated with the typelib, the path to the first resolved shared 
90           library is returned. Raises exception if module is unavailable or none of the shared libraries could be 
91           resolved. 
92           """
93           # Module unavailable
94           if not self.available:
95               raise ValueError(f"Module {self.name} {self.version} is unavailable!")
96           # Module has no associated shared libraries
97           if not self.sharedlibs:
98               return None
99           for lib in self.sharedlibs:
100              path = findSystemLibrary(lib)
101              if path:
102                  return os.path.normpath(os.path.dirname(path))
103          raise ValueError(f"Could not resolve any shared library of {self.name} {self.version}: {self.sharedlibs}!")
104  
105      def collect_typelib_data(self):
106          """ 
107          Return a tuple of (binaries, datas, hiddenimports) to be used by PyGObject related hooks. 
108          """
109          datas = []
110          binaries = []
111          hiddenimports = []
112  
113          logger.debug("Collecting module data for %s %s", self.name, self.version)
114  
115          # Module unavailable
116          if not self.available:
117              raise ValueError(f"Module {self.name} {self.version} is unavailable!")
118  
119          # Find shared libraries
120          resolved_libs = _resolveCtypesImports(self.sharedlibs)
121          for resolved_lib in resolved_libs:
122              logger.debug("Collecting shared library %s at %s", resolved_lib[0], resolved_lib[1])
123              binaries.append((resolved_lib[1], "."))
124  
125          # Find and collect .typelib file. Run it through the `gir_library_path_fix` to fix the library path, if
126          # necessary.
127          typelib_entry = gir_library_path_fix(self.typelib)
128          if typelib_entry:
129              logger.debug('Collecting gir typelib at %s', typelib_entry[0])
130              datas.append(typelib_entry)
131  
132          # Overrides for the module
133          hiddenimports += collect_submodules('gi.overrides', lambda name: name.endswith('.' + self.name))
134  
135          # Module dependencies
136          for dep in self.dependencies:
137              dep_module, _ = dep.rsplit('-', 1)
138              hiddenimports += [f'gi.repository.{dep_module}']
139  
140          return binaries, datas, hiddenimports

150  def get_gi_typelibs(module, version):
151      """ 
152      Return a tuple of (binaries, datas, hiddenimports) to be used by PyGObject related hooks. Searches for and adds 
153      dependencies recursively. 
154   
155      :param module: GI module name, as passed to 'gi.require_version()' 
156      :param version: GI module version, as passed to 'gi.require_version()' 
157      """
158      module_info = GiModuleInfo(module, version)
159      return module_info.collect_typelib_data()

162  def gir_library_path_fix(path):
163      import subprocess
164  
165      # 'PyInstaller.config' cannot be imported as other top-level modules.
166      from PyInstaller.config import CONF
167  
168      path = os.path.abspath(path)
169  
170      # On Mac OS we need to recompile the GIR files to reference the loader path,
171      # but this is not necessary on other platforms.
172      if compat.is_darwin:
173  
174          # If using a virtualenv, the base prefix and the path of the typelib
175          # have really nothing to do with each other, so try to detect that.
176          common_path = os.path.commonprefix([compat.base_prefix, path])
177          if common_path == '/':
178              logger.debug("virtualenv detected? fixing the gir path...")
179              common_path = os.path.abspath(os.path.join(path, '..', '..', '..'))
180  
181          gir_path = os.path.join(common_path, 'share', 'gir-1.0')
182  
183          typelib_name = os.path.basename(path)
184          gir_name = os.path.splitext(typelib_name)[0] + '.gir'
185  
186          gir_file = os.path.join(gir_path, gir_name)
187  
188          if not os.path.exists(gir_path):
189              logger.error(
190                  "Unable to find gir directory: %s.\nTry installing your platform's gobject-introspection package.",
191                  gir_path
192              )
193              return None
194          if not os.path.exists(gir_file):
195              logger.error(
196                  "Unable to find gir file: %s.\nTry installing your platform's gobject-introspection package.", gir_file
197              )
198              return None
199  
200          with open(gir_file, 'r', encoding='utf-8') as f:
201              lines = f.readlines()
202          # GIR files are `XML encoded <https://developer.gnome.org/gi/stable/gi-gir-reference.html>`_,
203          # which means they are by definition encoded using UTF-8.
204          with open(os.path.join(CONF['workpath'], gir_name), 'w', encoding='utf-8') as f:
205              for line in lines:
206                  if 'shared-library' in line:
207                      split = re.split('(=)', line)
208                      files = re.split('(["|,])', split[2])
209                      for count, item in enumerate(files):
210                          if 'lib' in item:
211                              files[count] = '@loader_path/' + os.path.basename(item)
212                      line = ''.join(split[0:2]) + ''.join(files)
213                  f.write(line)
214  
215          # g-ir-compiler expects a file so we cannot just pipe the fixed file to it.
216          command = subprocess.Popen((
217              'g-ir-compiler', os.path.join(CONF['workpath'], gir_name),
218              '-o', os.path.join(CONF['workpath'], typelib_name)
219          ))  # yapf: disable
220          command.wait()
221  
222          return os.path.join(CONF['workpath'], typelib_name), 'gi_typelibs'
223      else:
224          return path, 'gi_typelibs'

235  def get_glib_sysconf_dirs():
236      """ 
237      Try to return the sysconf directories (e.g., /etc). 
238      """
239      if compat.is_win:
240          # On Windows, if you look at gtkwin32.c, sysconfdir is actually relative to the location of the GTK DLL. Since
241          # that is what we are actually interested in (not the user path), we have to do that the hard way...
242          return [os.path.join(get_gi_libdir('GLib', '2.0'), 'etc')]
243  
244      @isolated.call
245      def data_dirs():
246          import gi
247          gi.require_version('GLib', '2.0')
248          from gi.repository import GLib
249          return GLib.get_system_config_dirs()
250  
251      return data_dirs

254  def collect_glib_share_files(*path):
255      """ 
256      Path is relative to the system data directory (e.g., /usr/share). 
257      """
258      glib_data_dirs = get_glib_system_data_dirs()
259      if glib_data_dirs is None:
260          return []
261  
262      destdir = os.path.join('share', *path)
263  
264      # TODO: will this return too much?
265      collected = []
266      for data_dir in glib_data_dirs:
267          p = os.path.join(data_dir, *path)
268          collected += collect_system_data_files(p, destdir=destdir, include_py_files=False)
269  
270      return collected

273  def collect_glib_etc_files(*path):
274      """ 
275      Path is relative to the system config directory (e.g., /etc). 
276      """
277      glib_config_dirs = get_glib_sysconf_dirs()
278      if glib_config_dirs is None:
279          return []
280  
281      destdir = os.path.join('etc', *path)
282  
283      # TODO: will this return too much?
284      collected = []
285      for config_dir in glib_config_dirs:
286          p = os.path.join(config_dir, *path)
287          collected += collect_system_data_files(p, destdir=destdir, include_py_files=False)
288  
289      return collected

295  def collect_glib_translations(prog, lang_list=None):
296      """ 
297      Return a list of translations in the system locale directory whose names equal prog.mo. 
298      """
299      global _glib_translations
300      if _glib_translations is None:
301          if lang_list is not None:
302              trans = []
303              for lang in lang_list:
304                  trans += collect_glib_share_files(os.path.join("locale", lang))
305              _glib_translations = trans
306          else:
307              _glib_translations = collect_glib_share_files('locale')
308  
309      names = [os.sep + prog + '.mo', os.sep + prog + '.po']
310      namelen = len(names[0])
311  
312      return [(src, dst) for src, dst in _glib_translations if src[-namelen:] in names]

315  # Not a hook utility function per-se (used by main Analysis class), but kept here to have all GLib/GObject functions
316  # in one place...
317  def compile_glib_schema_files(datas_toc, workdir, collect_source_files=False):
318      """ 
319      Compile collected GLib schema files. Extracts the list of GLib schema files from the given input datas TOC, copies 
320      them to temporary working directory, and compiles them. The resulting `gschemas.compiled` file is added to the 
321      output TOC, replacing any existing entry with that name. If `collect_source_files` flag is set, the source XML 
322      schema files are also (re)added to the output TOC; by default, they are not. This function is no-op (returns the 
323      original TOC) if no GLib schemas are found in TOC or if `glib-compile-schemas` executable is not found in `PATH`. 
324      """
325      SCHEMA_DEST_DIR = pathlib.PurePath("share/glib-2.0/schemas")
326      workdir = pathlib.Path(workdir)
327  
328      schema_files = []
329      output_toc = []
330      for toc_entry in datas_toc:
331          dest_name, src_name, typecode = toc_entry
332          dest_name = pathlib.PurePath(dest_name)
333          src_name = pathlib.PurePath(src_name)
334  
335          # Pass-through for non-schema files, identified based on the destination directory.
336          if dest_name.parent != SCHEMA_DEST_DIR:
337              output_toc.append(toc_entry)
338              continue
339  
340          # It seems schemas directory contains different files with different suffices:
341          #  - .gschema.xml
342          #  - .schema.override
343          #  - .enums.xml
344          # To avoid omitting anything, simply collect everything into temporary directory.
345          # Exemptions are gschema.dtd (which should be unnecessary) and gschemas.compiled (which we will generate
346          # ourselves in this function).
347          if src_name.name in {"gschema.dtd", "gschemas.compiled"}:
348              continue
349  
350          schema_files.append(src_name)
351  
352      # If there are no schema files available, simply return the input datas TOC.
353      if not schema_files:
354          return datas_toc
355  
356      # Ensure that `glib-compile-schemas` executable is in PATH, just in case...
357      schema_compiler_exe = shutil.which('glib-compile-schemas')
358      if not schema_compiler_exe:
359          logger.warning("GLib schema compiler (glib-compile-schemas) not found! Skipping GLib schema recompilation...")
360          return datas_toc
361  
362      # If `gschemas.compiled` file already exists in the temporary working directory, record its modification time and
363      # hash. This will allow us to restore the modification time on the newly-compiled copy, if the latter turns out
364      # to be identical to the existing old one. Just in case, if the file becomes subject to timestamp-based caching
365      # mechanism.
366      compiled_file = workdir / "gschemas.compiled"
367      old_compiled_file_hash = None
368      old_compiled_file_stat = None
369  
370      if compiled_file.is_file():
371          # Record creation/modification time
372          old_compiled_file_stat = compiled_file.stat()
373          # Compute SHA1 hash; since compiled schema files are relatively small, do it in single step.
374          old_compiled_file_hash = hashlib.sha1(compiled_file.read_bytes()).digest()
375  
376      # Ensure that temporary working directory exists, and is empty.
377      if workdir.exists():
378          shutil.rmtree(workdir)
379      workdir.mkdir(exist_ok=True)
380  
381      # Copy schema (source) files to temporary working directory
382      for schema_file in schema_files:
383          shutil.copy(schema_file, workdir)
384  
385      # Compile. The glib-compile-schema might produce warnings on its own (e.g., schemas using deprecated paths, or
386      # overrides for non-existent keys). Since these are non-actionable, capture and display them only as a DEBUG
387      # message, or as a WARNING one if the command fails.
388      logger.info("Compiling collected GLib schema files in %r...", str(workdir))
389      try:
390          cmd_args = [schema_compiler_exe, str(workdir), '--targetdir', str(workdir)]
391          p = subprocess.run(
392              cmd_args,
393              stdin=subprocess.DEVNULL,
394              stdout=subprocess.PIPE,
395              stderr=subprocess.STDOUT,
396              check=True,
397              errors='ignore',
398              encoding='utf-8',
399          )
400          logger.debug("Output from glib-compile-schemas:\n%s", p.stdout)
401      except subprocess.CalledProcessError as e:
402          # The called glib-compile-schema returned error. Display stdout/stderr, and return original datas TOC to
403          # minimize damage.
404          logger.warning("Failed to recompile GLib schemas! Returning collected files as-is!", exc_info=True)
405          logger.warning("Output from glib-compile-schemas:\n%s", e.stdout)
406          return datas_toc
407      except Exception:
408          # Compilation failed for whatever reason. Return original datas TOC to minimize damage.
409          logger.warning("Failed to recompile GLib schemas! Returning collected files as-is!", exc_info=True)
410          return datas_toc
411  
412      # Compute the checksum of the new compiled file, and if it matches the old checksum, restore the modification time.
413      if old_compiled_file_hash is not None:
414          new_compiled_file_hash = hashlib.sha1(compiled_file.read_bytes()).digest()
415          if new_compiled_file_hash == old_compiled_file_hash:
416              os.utime(compiled_file, ns=(old_compiled_file_stat.st_atime_ns, old_compiled_file_stat.st_mtime_ns))
417  
418      # Add the resulting gschemas.compiled file to the output TOC
419      output_toc.append((str(SCHEMA_DEST_DIR / compiled_file.name), str(compiled_file), "DATA"))
420  
421      # Include source schema files in the output TOC (optional)
422      if collect_source_files:
423          for schema_file in schema_files:
424              output_toc.append((str(SCHEMA_DEST_DIR / schema_file.name), str(schema_file), "DATA"))
425  
426      return output_toc
