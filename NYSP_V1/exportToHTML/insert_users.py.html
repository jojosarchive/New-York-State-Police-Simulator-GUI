54       def add_user(self):

58   
59           # Can this be enough t validate input

62               return
63   
64           # Add user to the database
65           try:
66               conn = sqlite3.connect('../driving_simulator.db')
67               c = conn.cursor()
68               c.execute("INSERT INTO users (badge_number, name) VALUES (?, ?)", (badge_number, name))
69               conn.commit()

119      def populate_user_list(self):
120          try:
121              self.user_listbox.delete(0, tk.END)  # Clear the listbox
122              conn = sqlite3.connect('../driving_simulator.db')
123              c = conn.cursor()
124              c.execute("SELECT badge_number, name FROM users")
125              users = c.fetchall()
126              conn.close()
127  
128              for user in users:
129                  self.user_listbox.insert(tk.END, f"{user[0]} - {user[1]}")
130          except Exception as e:
131              messagebox.showerror("Error", f"Failed to retrieve user list: {e}")
132  
133      def populate_edit_fields(self, event):
134          selected_user = self.user_listbox.get(self.user_listbox.curselection())
135          badge_number = selected_user.split(' - ')[0]
136          self.edit_badge_entry.delete(0, tk.END)
137          self.edit_badge_entry.insert(0, badge_number)
