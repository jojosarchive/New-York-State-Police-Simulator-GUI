74   class Reg:
75       """Helper class to read values from the registry"""
76   
77       def get_value(cls, path, key):
78           for base in HKEYS:
79               d = cls.read_values(base, path)
80               if d and key in d:
81                   return d[key]
82           raise KeyError(key)
83   
84       get_value = classmethod(get_value)
85   
86       def read_keys(cls, base, key):
87           """Return list of registry keys."""
88           try:
89               handle = RegOpenKeyEx(base, key)
90           except RegError:
91               return None
92           L = []
93           i = 0
94           while True:
95               try:
96                   k = RegEnumKey(handle, i)
97               except RegError:
98                   break
99               L.append(k)
100              i += 1
101          return L
102  
103      read_keys = classmethod(read_keys)
104  
105      def read_values(cls, base, key):
106          """Return dict of registry keys and values. 
107   
108          All names are converted to lowercase. 
109          """
110          try:
111              handle = RegOpenKeyEx(base, key)
112          except RegError:
113              return None
114          d = {}
115          i = 0
116          while True:
117              try:
118                  name, value, type = RegEnumValue(handle, i)
119              except RegError:
120                  break
121              name = name.lower()
122              d[cls.convert_mbcs(name)] = cls.convert_mbcs(value)
123              i += 1
124          return d
125  
126      read_values = classmethod(read_values)
127  
128      def convert_mbcs(s):
129          dec = getattr(s, "decode", None)
130          if dec is not None:
131              try:
132                  s = dec("mbcs")
133              except UnicodeError:
134                  pass
135          return s
136  
137      convert_mbcs = staticmethod(convert_mbcs)

140  class MacroExpander:
141      def __init__(self, version):
142          self.macros = {}
143          self.vsbase = VS_BASE % version
144          self.load_macros(version)
145  
146      def set_macro(self, macro, path, key):
147          self.macros["$(%s)" % macro] = Reg.get_value(path, key)
148  
149      def load_macros(self, version):
150          self.set_macro("VCInstallDir", self.vsbase + r"\Setup\VC", "productdir")
151          self.set_macro("VSInstallDir", self.vsbase + r"\Setup\VS", "productdir")
152          self.set_macro("FrameworkDir", NET_BASE, "installroot")
153          try:
154              if version >= 8.0:
155                  self.set_macro("FrameworkSDKDir", NET_BASE, "sdkinstallrootv2.0")
156              else:
157                  raise KeyError("sdkinstallrootv2.0")
158          except KeyError:
159              raise DistutilsPlatformError(
160                  """Python was built with Visual Studio 2008; 
161  extensions must be built with a compiler than can generate compatible binaries. 
162  Visual Studio 2008 was not found on this system. If you have Cygwin installed, 
163  you can try compiling with MingW32, by passing "-c mingw32" to setup.py."""
164              )
165  
166          if version >= 9.0:
167              self.set_macro("FrameworkVersion", self.vsbase, "clr version")
168              self.set_macro("WindowsSdkDir", WINSDK_BASE, "currentinstallfolder")
169          else:
170              p = r"Software\Microsoft\NET Framework Setup\Product"
171              for base in HKEYS:
172                  try:
173                      h = RegOpenKeyEx(base, p)
174                  except RegError:
175                      continue
176                  key = RegEnumKey(h, 0)
177                  d = Reg.get_value(base, rf"{p}\{key}")
178                  self.macros["$(FrameworkVersion)"] = d["version"]
179  
180      def sub(self, s):
181          for k, v in self.macros.items():
182              s = s.replace(k, v)
183          return s

186  def get_build_version():
187      """Return the version of MSVC that was used to build Python. 
188   
189      For Python 2.3 and up, the version number is included in 
190      sys.version.  For earlier versions, assume the compiler is MSVC 6. 
191      """
192      prefix = "MSC v."
193      i = sys.version.find(prefix)
194      if i == -1:
195          return 6
196      i = i + len(prefix)
197      s, rest = sys.version[i:].split(" ", 1)
198      majorVersion = int(s[:-2]) - 6
199      if majorVersion >= 13:
200          # v13 was skipped and should be v14
201          majorVersion += 1
202      minorVersion = int(s[2:3]) / 10.0
203      # I don't think paths are affected by minor version in version 6
204      if majorVersion == 6:
205          minorVersion = 0
206      if majorVersion >= 6:
207          return majorVersion + minorVersion
208      # else we don't know what version of the compiler this is
209      return None

212  def normalize_and_reduce_paths(paths):
213      """Return a list of normalized paths with duplicates removed. 
214   
215      The current order of paths is maintained. 
216      """
217      # Paths are normalized so things like:  /a and /a/ aren't both preserved.
218      reduced_paths = []
219      for p in paths:
220          np = os.path.normpath(p)
221          # XXX(nnorwitz): O(n**2), if reduced_paths gets long perhaps use a set.
222          if np not in reduced_paths:
223              reduced_paths.append(np)
224      return reduced_paths

227  def removeDuplicates(variable):
228      """Remove duplicate values of an environment variable."""
229      oldList = variable.split(os.pathsep)
230      newList = []
231      for i in oldList:
232          if i not in newList:
233              newList.append(i)
234      newVariable = os.pathsep.join(newList)
235      return newVariable

238  def find_vcvarsall(version):
239      """Find the vcvarsall.bat file 
240   
241      At first it tries to find the productdir of VS 2008 in the registry. If 
242      that fails it falls back to the VS90COMNTOOLS env var. 
243      """
244      vsbase = VS_BASE % version
245      try:
246          productdir = Reg.get_value(r"%s\Setup\VC" % vsbase, "productdir")
247      except KeyError:
248          log.debug("Unable to find productdir in registry")
249          productdir = None
250  
251      if not productdir or not os.path.isdir(productdir):
252          toolskey = "VS%0.f0COMNTOOLS" % version
253          toolsdir = os.environ.get(toolskey, None)
254  
255          if toolsdir and os.path.isdir(toolsdir):
256              productdir = os.path.join(toolsdir, os.pardir, os.pardir, "VC")
257              productdir = os.path.abspath(productdir)
258              if not os.path.isdir(productdir):
259                  log.debug("%s is not a valid directory" % productdir)
260                  return None
261          else:
262              log.debug("Env var %s is not set or invalid" % toolskey)
263      if not productdir:
264          log.debug("No productdir found")
265          return None
266      vcvarsall = os.path.join(productdir, "vcvarsall.bat")
267      if os.path.isfile(vcvarsall):
268          return vcvarsall
269      log.debug("Unable to find vcvarsall.bat")
270      return None

273  def query_vcvarsall(version, arch="x86"):
274      """Launch vcvarsall.bat and read the settings from its environment"""
275      vcvarsall = find_vcvarsall(version)
276      interesting = {"include", "lib", "libpath", "path"}
277      result = {}
278  
279      if vcvarsall is None:
280          raise DistutilsPlatformError("Unable to find vcvarsall.bat")
281      log.debug("Calling 'vcvarsall.bat %s' (version=%s)", arch, version)
282      popen = subprocess.Popen(
283          f'"{vcvarsall}" {arch} & set',
284          stdout=subprocess.PIPE,
285          stderr=subprocess.PIPE,
286      )
287      try:
288          stdout, stderr = popen.communicate()
289          if popen.wait() != 0:
290              raise DistutilsPlatformError(stderr.decode("mbcs"))
291  
292          stdout = stdout.decode("mbcs")
293          for line in stdout.split("\n"):
294              line = Reg.convert_mbcs(line)
295              if '=' not in line:
296                  continue
297              line = line.strip()
298              key, value = line.split('=', 1)
299              key = key.lower()
300              if key in interesting:
301                  if value.endswith(os.pathsep):
302                      value = value[:-1]
303                  result[key] = removeDuplicates(value)
304  
305      finally:
306          popen.stdout.close()
307          popen.stderr.close()
308  
309      if len(result) != len(interesting):
310          raise ValueError(str(list(result.keys())))
311  
312      return result

320  class MSVCCompiler(CCompiler):
321      """Concrete class that implements an interface to Microsoft Visual C++, 
322      as defined by the CCompiler abstract class."""
323  
324      compiler_type = 'msvc'
325  
326      # Just set this so CCompiler's constructor doesn't barf.  We currently
327      # don't use the 'set_executables()' bureaucracy provided by CCompiler,
328      # as it really isn't necessary for this sort of single-compiler class.
329      # Would be nice to have a consistent interface with UnixCCompiler,
330      # though, so it's worth thinking about.
331      executables = {}
332  
333      # Private class data (need to distinguish C from C++ source for compiler)
334      _c_extensions = ['.c']
335      _cpp_extensions = ['.cc', '.cpp', '.cxx']
336      _rc_extensions = ['.rc']
337      _mc_extensions = ['.mc']
338  
339      # Needed for the filename generation methods provided by the
340      # base class, CCompiler.
341      src_extensions = _c_extensions + _cpp_extensions + _rc_extensions + _mc_extensions
342      res_extension = '.res'
343      obj_extension = '.obj'
344      static_lib_extension = '.lib'
345      shared_lib_extension = '.dll'
346      static_lib_format = shared_lib_format = '%s%s'
347      exe_extension = '.exe'
348  
349      def __init__(self, verbose=0, dry_run=0, force=0):
350          super().__init__(verbose, dry_run, force)
351          self.__version = VERSION
352          self.__root = r"Software\Microsoft\VisualStudio"
353          # self.__macros = MACROS
354          self.__paths = []
355          # target platform (.plat_name is consistent with 'bdist')
356          self.plat_name = None
357          self.__arch = None  # deprecated name
358          self.initialized = False
359  
360      def initialize(self, plat_name=None):  # noqa: C901
361          # multi-init means we would need to check platform same each time...
362          assert not self.initialized, "don't init multiple times"
363          if self.__version < 8.0:
364              raise DistutilsPlatformError(
365                  "VC %0.1f is not supported by this module" % self.__version
366              )
367          if plat_name is None:
368              plat_name = get_platform()
369          # sanity check for platforms to prevent obscure errors later.
370          ok_plats = 'win32', 'win-amd64'
371          if plat_name not in ok_plats:
372              raise DistutilsPlatformError(f"--plat-name must be one of {ok_plats}")
373  
374          if (
375              "DISTUTILS_USE_SDK" in os.environ
376              and "MSSdk" in os.environ
377              and self.find_exe("cl.exe")
378          ):
379              # Assume that the SDK set up everything alright; don't try to be
380              # smarter
381              self.cc = "cl.exe"
382              self.linker = "link.exe"
383              self.lib = "lib.exe"
384              self.rc = "rc.exe"
385              self.mc = "mc.exe"
386          else:
387              # On x86, 'vcvars32.bat amd64' creates an env that doesn't work;
388              # to cross compile, you use 'x86_amd64'.
389              # On AMD64, 'vcvars32.bat amd64' is a native build env; to cross
390              # compile use 'x86' (ie, it runs the x86 compiler directly)
391              if plat_name in (get_platform(), 'win32'):
392                  # native build or cross-compile to win32
393                  plat_spec = PLAT_TO_VCVARS[plat_name]
394              else:
395                  # cross compile from win32 -> some 64bit
396                  plat_spec = (
397                      PLAT_TO_VCVARS[get_platform()] + '_' + PLAT_TO_VCVARS[plat_name]
398                  )
399  
400              vc_env = query_vcvarsall(VERSION, plat_spec)
401  
402              self.__paths = vc_env['path'].split(os.pathsep)
403              os.environ['lib'] = vc_env['lib']
404              os.environ['include'] = vc_env['include']
405  
406              if len(self.__paths) == 0:
407                  raise DistutilsPlatformError(
408                      "Python was built with %s, "
409                      "and extensions need to be built with the same "
410                      "version of the compiler, but it isn't installed." % self.__product
411                  )
412  
413              self.cc = self.find_exe("cl.exe")
414              self.linker = self.find_exe("link.exe")
415              self.lib = self.find_exe("lib.exe")
416              self.rc = self.find_exe("rc.exe")  # resource compiler
417              self.mc = self.find_exe("mc.exe")  # message compiler
418              # self.set_path_env_var('lib')
419              # self.set_path_env_var('include')
420  
421          # extend the MSVC path with the current path
422          try:
423              for p in os.environ['path'].split(';'):
424                  self.__paths.append(p)
425          except KeyError:
426              pass
427          self.__paths = normalize_and_reduce_paths(self.__paths)
428          os.environ['path'] = ";".join(self.__paths)
429  
430          self.preprocess_options = None
431          if self.__arch == "x86":
432              self.compile_options = ['/nologo', '/O2', '/MD', '/W3', '/DNDEBUG']
433              self.compile_options_debug = [
434                  '/nologo',
435                  '/Od',
436                  '/MDd',
437                  '/W3',
438                  '/Z7',
439                  '/D_DEBUG',
440              ]
441          else:
442              # Win64
443              self.compile_options = ['/nologo', '/O2', '/MD', '/W3', '/GS-', '/DNDEBUG']
444              self.compile_options_debug = [
445                  '/nologo',
446                  '/Od',
447                  '/MDd',
448                  '/W3',
449                  '/GS-',
450                  '/Z7',
451                  '/D_DEBUG',
452              ]
453  
454          self.ldflags_shared = ['/DLL', '/nologo', '/INCREMENTAL:NO']
455          if self.__version >= 7:
456              self.ldflags_shared_debug = ['/DLL', '/nologo', '/INCREMENTAL:no', '/DEBUG']
457          self.ldflags_static = ['/nologo']
458  
459          self.initialized = True

463      def object_filenames(self, source_filenames, strip_dir=0, output_dir=''):
464          # Copied from ccompiler.py, extended to return .res as 'object'-file
465          # for .rc input file
466          if output_dir is None:
467              output_dir = ''
468          obj_names = []
469          for src_name in source_filenames:
470              (base, ext) = os.path.splitext(src_name)
471              base = os.path.splitdrive(base)[1]  # Chop off the drive
472              base = base[os.path.isabs(base) :]  # If abs, chop off leading /
473              if ext not in self.src_extensions:
474                  # Better to raise an exception instead of silently continuing
475                  # and later complain about sources and targets having
476                  # different lengths
477                  raise CompileError("Don't know how to compile %s" % src_name)
478              if strip_dir:
479                  base = os.path.basename(base)
480              if ext in self._rc_extensions:
481                  obj_names.append(os.path.join(output_dir, base + self.res_extension))
482              elif ext in self._mc_extensions:
483                  obj_names.append(os.path.join(output_dir, base + self.res_extension))
484              else:
485                  obj_names.append(os.path.join(output_dir, base + self.obj_extension))
486          return obj_names

488      def compile(  # noqa: C901
489          self,
490          sources,
491          output_dir=None,
492          macros=None,
493          include_dirs=None,
494          debug=0,
495          extra_preargs=None,
496          extra_postargs=None,
497          depends=None,
498      ):
499          if not self.initialized:
500              self.initialize()
501          compile_info = self._setup_compile(
502              output_dir, macros, include_dirs, sources, depends, extra_postargs
503          )
504          macros, objects, extra_postargs, pp_opts, build = compile_info
505  
506          compile_opts = extra_preargs or []
507          compile_opts.append('/c')
508          if debug:
509              compile_opts.extend(self.compile_options_debug)
510          else:
511              compile_opts.extend(self.compile_options)
512  
513          for obj in objects:
514              try:
515                  src, ext = build[obj]
516              except KeyError:
517                  continue
518              if debug:
519                  # pass the full pathname to MSVC in debug mode,
520                  # this allows the debugger to find the source file
521                  # without asking the user to browse for it
522                  src = os.path.abspath(src)
523  
524              if ext in self._c_extensions:
525                  input_opt = "/Tc" + src
526              elif ext in self._cpp_extensions:
527                  input_opt = "/Tp" + src
528              elif ext in self._rc_extensions:
529                  # compile .RC to .RES file
530                  input_opt = src
531                  output_opt = "/fo" + obj
532                  try:
533                      self.spawn([self.rc] + pp_opts + [output_opt] + [input_opt])
534                  except DistutilsExecError as msg:
535                      raise CompileError(msg)
536                  continue
537              elif ext in self._mc_extensions:
538                  # Compile .MC to .RC file to .RES file.
539                  #   * '-h dir' specifies the directory for the
540                  #     generated include file
541                  #   * '-r dir' specifies the target directory of the
542                  #     generated RC file and the binary message resource
543                  #     it includes
544                  #
545                  # For now (since there are no options to change this),
546                  # we use the source-directory for the include file and
547                  # the build directory for the RC file and message
548                  # resources. This works at least for win32all.
549                  h_dir = os.path.dirname(src)
550                  rc_dir = os.path.dirname(obj)
551                  try:
552                      # first compile .MC to .RC and .H file
553                      self.spawn([self.mc] + ['-h', h_dir, '-r', rc_dir] + [src])
554                      base, _ = os.path.splitext(os.path.basename(src))
555                      rc_file = os.path.join(rc_dir, base + '.rc')
556                      # then compile .RC to .RES file
557                      self.spawn([self.rc] + ["/fo" + obj] + [rc_file])
558  
559                  except DistutilsExecError as msg:
560                      raise CompileError(msg)
561                  continue
562              else:
563                  # how to handle this file?
564                  raise CompileError(f"Don't know how to compile {src} to {obj}")
565  
566              output_opt = "/Fo" + obj
567              try:
568                  self.spawn(
569                      [self.cc]
570                      + compile_opts
571                      + pp_opts
572                      + [input_opt, output_opt]
573                      + extra_postargs
574                  )
575              except DistutilsExecError as msg:
576                  raise CompileError(msg)
577  
578          return objects

599      def link(  # noqa: C901
600          self,
601          target_desc,
602          objects,
603          output_filename,
604          output_dir=None,
605          libraries=None,
606          library_dirs=None,
607          runtime_library_dirs=None,
608          export_symbols=None,
609          debug=0,
610          extra_preargs=None,
611          extra_postargs=None,
612          build_temp=None,
613          target_lang=None,
614      ):
615          if not self.initialized:
616              self.initialize()
617          (objects, output_dir) = self._fix_object_args(objects, output_dir)
618          fixed_args = self._fix_lib_args(libraries, library_dirs, runtime_library_dirs)
619          (libraries, library_dirs, runtime_library_dirs) = fixed_args
620  
621          if runtime_library_dirs:
622              self.warn(
623                  "I don't know what to do with 'runtime_library_dirs': "
624                  + str(runtime_library_dirs)
625              )
626  
627          lib_opts = gen_lib_options(self, library_dirs, runtime_library_dirs, libraries)
628          if output_dir is not None:
629              output_filename = os.path.join(output_dir, output_filename)
630  
631          if self._need_link(objects, output_filename):
632              if target_desc == CCompiler.EXECUTABLE:
633                  if debug:
634                      ldflags = self.ldflags_shared_debug[1:]
635                  else:
636                      ldflags = self.ldflags_shared[1:]
637              else:
638                  if debug:
639                      ldflags = self.ldflags_shared_debug
640                  else:
641                      ldflags = self.ldflags_shared
642  
643              export_opts = []
644              for sym in export_symbols or []:
645                  export_opts.append("/EXPORT:" + sym)
646  
647              ld_args = (
648                  ldflags + lib_opts + export_opts + objects + ['/OUT:' + output_filename]
649              )
650  
651              # The MSVC linker generates .lib and .exp files, which cannot be
652              # suppressed by any linker switches. The .lib files may even be
653              # needed! Make sure they are generated in the temporary build
654              # directory. Since they have different names for debug and release
655              # builds, they can go into the same directory.
656              build_temp = os.path.dirname(objects[0])
657              if export_symbols is not None:
658                  (dll_name, dll_ext) = os.path.splitext(
659                      os.path.basename(output_filename)
660                  )
661                  implib_file = os.path.join(build_temp, self.library_filename(dll_name))
662                  ld_args.append('/IMPLIB:' + implib_file)
663  
664              self.manifest_setup_ldargs(output_filename, build_temp, ld_args)
665  
666              if extra_preargs:
667                  ld_args[:0] = extra_preargs
668              if extra_postargs:
669                  ld_args.extend(extra_postargs)
670  
671              self.mkpath(os.path.dirname(output_filename))
672              try:
673                  self.spawn([self.linker] + ld_args)
674              except DistutilsExecError as msg:
675                  raise LinkError(msg)
676  
677              # embed the manifest
678              # XXX - this is somewhat fragile - if mt.exe fails, distutils
679              # will still consider the DLL up-to-date, but it will not have a
680              # manifest.  Maybe we should link to a temp file?  OTOH, that
681              # implies a build environment error that shouldn't go undetected.
682              mfinfo = self.manifest_get_embed_info(target_desc, ld_args)
683              if mfinfo is not None:
684                  mffilename, mfid = mfinfo
685                  out_arg = f'-outputresource:{output_filename};{mfid}'
686                  try:
687                      self.spawn(['mt.exe', '-nologo', '-manifest', mffilename, out_arg])
688                  except DistutilsExecError as msg:
689                      raise LinkError(msg)
690          else:
691              log.debug("skipping %s (up-to-date)", output_filename)
692  
693      def manifest_setup_ldargs(self, output_filename, build_temp, ld_args):
694          # If we need a manifest at all, an embedded manifest is recommended.
695          # See MSDN article titled
696          # "Understanding manifest generation for C/C++ programs"
697          # (currently at https://learn.microsoft.com/en-us/cpp/build/understanding-manifest-generation-for-c-cpp-programs)
698          # Ask the linker to generate the manifest in the temp dir, so
699          # we can check it, and possibly embed it, later.
700          temp_manifest = os.path.join(
701              build_temp, os.path.basename(output_filename) + ".manifest"
702          )
703          ld_args.append('/MANIFESTFILE:' + temp_manifest)
704  
705      def manifest_get_embed_info(self, target_desc, ld_args):
706          # If a manifest should be embedded, return a tuple of
707          # (manifest_filename, resource_id).  Returns None if no manifest
708          # should be embedded.  See https://bugs.python.org/issue7833 for why
709          # we want to avoid any manifest for extension modules if we can)
710          for arg in ld_args:
711              if arg.startswith("/MANIFESTFILE:"):
712                  temp_manifest = arg.split(":", 1)[1]
713                  break
714          else:
715              # no /MANIFESTFILE so nothing to do.
716              return None
717          if target_desc == CCompiler.EXECUTABLE:
718              # by default, executables always get the manifest with the
719              # CRT referenced.
720              mfid = 1
721          else:
722              # Extension modules try and avoid any manifest if possible.
723              mfid = 2
724              temp_manifest = self._remove_visual_c_ref(temp_manifest)
725          if temp_manifest is None:
726              return None
727          return temp_manifest, mfid
728  
729      def _remove_visual_c_ref(self, manifest_file):
730          try:
731              # Remove references to the Visual C runtime, so they will
732              # fall through to the Visual C dependency of Python.exe.
733              # This way, when installed for a restricted user (e.g.
734              # runtimes are not in WinSxS folder, but in Python's own
735              # folder), the runtimes do not need to be in every folder
736              # with .pyd's.
737              # Returns either the filename of the modified manifest or
738              # None if no manifest should be embedded.
739              manifest_f = open(manifest_file)
740              try:
741                  manifest_buf = manifest_f.read()
742              finally:
743                  manifest_f.close()
744              pattern = re.compile(
745                  r"""<assemblyIdentity.*?name=("|')Microsoft\."""
746                  r"""VC\d{2}\.CRT("|').*?(/>|</assemblyIdentity>)""",
747                  re.DOTALL,
748              )
749              manifest_buf = re.sub(pattern, "", manifest_buf)
750              pattern = r"<dependentAssembly>\s*</dependentAssembly>"
751              manifest_buf = re.sub(pattern, "", manifest_buf)
752              # Now see if any other assemblies are referenced - if not, we
753              # don't want a manifest embedded.
754              pattern = re.compile(
755                  r"""<assemblyIdentity.*?name=(?:"|')(.+?)(?:"|')"""
756                  r""".*?(?:/>|</assemblyIdentity>)""",
757                  re.DOTALL,
758              )
759              if re.search(pattern, manifest_buf) is None:
760                  return None
761  
762              manifest_f = open(manifest_file, 'w')
763              try:
764                  manifest_f.write(manifest_buf)
765                  return manifest_file
766              finally:
767                  manifest_f.close()
768          except OSError:
769              pass

786      def find_library_file(self, dirs, lib, debug=0):
787          # Prefer a debugging library if found (and requested), but deal
788          # with it if we don't have one.
789          if debug:
790              try_names = [lib + "_d", lib]
791          else:
792              try_names = [lib]
793          for dir in dirs:
794              for name in try_names:
795                  libfile = os.path.join(dir, self.library_filename(name))
796                  if os.path.exists(libfile):
797                      return libfile
798          else:
799              # Oops, didn't find it in *any* of 'dirs'
800              return None

802      # Helper methods for using the MSVC registry settings
803  
804      def find_exe(self, exe):
805          """Return path to an MSVC executable program. 
806   
807          Tries to find the program in several places: first, one of the 
808          MSVC program search paths from the registry; next, the directories 
809          in the PATH environment variable.  If any of those work, return an 
810          absolute path that is known to exist.  If none of them work, just 
811          return the original program name, 'exe'. 
812          """
813          for p in self.__paths:
814              fn = os.path.join(os.path.abspath(p), exe)
815              if os.path.isfile(fn):
816                  return fn
817  
818          # didn't find it; try existing path
819          for p in os.environ['Path'].split(';'):
820              fn = os.path.join(os.path.abspath(p), exe)
821              if os.path.isfile(fn):
822                  return fn
823  
824          return exe
