23   # Basic import tests for sub-packages of skimage. Run only on demand, and only in onedir mode.
24   @pytest.mark.slow
25   @onedir_only
26   @importorskip('skimage')
27   @pytest.mark.skipif(
28       not is_module_satisfies('scikit_image >= 0.16'),
29       reason='The test supports only scikit-image >= 0.16.',
30   )
31   @pytest.mark.parametrize('submodule', [
32       'color', 'data', 'draw', 'exposure', 'feature', 'filters', 'future',
33       'graph', 'io', 'measure', 'metrics', 'morphology', 'registration',
34       'restoration', 'segmentation', 'transform', 'util'
35   ])
36   def test_skimage(pyi_builder, submodule):
37       pyi_builder.test_source(""" 
38           import skimage.{0} 
39           """.format(submodule))

42   # Test the ORB descriptor, which requires the data file with descriptor sample points.
43   @importorskip('skimage')
44   def test_skimage_feature_orb(pyi_builder):
45       pyi_builder.test_source(""" 
46           import skimage.feature 
47           import numpy as np 
48    
49           # Prepare test images 
50           img1 = np.zeros((100, 100)) 
51           img2 = np.zeros_like(img1) 
52           rng = np.random.default_rng(1984) 
53           square = rng.random((20, 20)) 
54           img1[40:60, 40:60] = square 
55           img2[53:73, 53:73] = square 
56    
57           # ORB detector/descriptor extractor 
58           detector_extractor1 = skimage.feature.ORB(n_keypoints=5) 
59           detector_extractor2 = skimage.feature.ORB(n_keypoints=5) 
60    
61           # Process 
62           detector_extractor1.detect_and_extract(img1) 
63           detector_extractor2.detect_and_extract(img2) 
64    
65           matches = skimage.feature.match_descriptors( 
66               detector_extractor1.descriptors, 
67               detector_extractor2.descriptors, 
68           ) 
69           print(matches) 
70           """)
