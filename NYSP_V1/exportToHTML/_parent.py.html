60   def create_pipe(read_handle_inheritable, write_handle_inheritable):
61       """ 
62       Create a one-way pipe for sending data to child processes. 
63    
64       Args: 
65           read_handle_inheritable: 
66               A boolean flag indicating whether the handle corresponding to the read end-point of the pipe should be 
67               marked as inheritable by subprocesses. 
68           write_handle_inheritable: 
69               A boolean flag indicating whether the handle corresponding to the write end-point of the pipe should be 
70               marked as inheritable by subprocesses. 
71    
72       Returns: 
73           A read/write pair of file descriptors (which are just integers) on posix or system file handles on Windows. 
74    
75       The pipe may be used either by this process or subprocesses of this process but not globally. 
76       """
77       return _create_pipe_impl(read_handle_inheritable, write_handle_inheritable)

81       """ 
82       Close the file descriptor (posix) or handle (Windows) belonging to a pipe. 
83       """
84       return _close_pipe_endpoint_impl(pipe_handle)

89       def _create_pipe_impl(read_handle_inheritable, write_handle_inheritable):
90           # Use WinAPI CreatePipe function to create the pipe. Python's os.pipe() does the same, but wraps the resulting
91           # handles into inheritable file descriptors (https://github.com/python/cpython/issues/77046). Instead, we want
92           # just handles, and will set the inheritable flag on corresponding handle ourselves.
93           read_handle = ctypes.wintypes.HANDLE()
94           write_handle = ctypes.wintypes.HANDLE()
95   
96           # SECURITY_ATTRIBUTES with inherit handle set to True
97           security_attributes = SECURITY_ATTRIBUTES()
98           security_attributes.nLength = ctypes.sizeof(security_attributes)
99           security_attributes.bInheritHandle = True
100          security_attributes.lpSecurityDescriptor = None
101  
102          # CreatePipe()
103          succeeded = CreatePipe(
104              ctypes.byref(read_handle),  # hReadPipe
105              ctypes.byref(write_handle),  # hWritePipe
106              ctypes.byref(security_attributes),  # lpPipeAttributes
107              0,  # nSize
108          )
109          if not succeeded:
110              raise ctypes.WinError()
111  
112          # Set inheritable flags. Instead of binding and using SetHandleInformation WinAPI function, we can use
113          # os.set_handle_inheritable().
114          os.set_handle_inheritable(read_handle.value, read_handle_inheritable)
115          os.set_handle_inheritable(write_handle.value, write_handle_inheritable)
116  
117          return read_handle.value, write_handle.value

119      def _close_pipe_endpoint_impl(pipe_handle):
120          succeeded = CloseHandle(pipe_handle)
121          if not succeeded:
122              raise ctypes.WinError()

125      def _create_pipe_impl(read_fd_inheritable, write_fd_inheritable):
126          # Create pipe, using os.pipe()
127          read_fd, write_fd = os.pipe()
128  
129          # The default behaviour of pipes is that they are process specific. I.e., they can only be used by this
130          # process to talk to itself. Setting inheritable flags means that child processes may also use these pipes.
131          os.set_inheritable(read_fd, read_fd_inheritable)
132          os.set_inheritable(write_fd, write_fd_inheritable)
133  
134          return read_fd, write_fd
135  
136      def _close_pipe_endpoint_impl(pipe_fd):
137          os.close(pipe_fd)

140  def child(read_from_parent: int, write_to_parent: int):
141      """ 
142      Spawn a Python subprocess sending it the two file descriptors it needs to talk back to this parent process. 
143      """
144      if os.name != 'nt':
145          # Explicitly disabling close_fds is a requirement for making file descriptors inheritable by child processes.
146          extra_kwargs = {
147              "env": _subprocess_env(),
148              "close_fds": False,
149          }
150      else:
151          # On Windows, we can use subprocess.STARTUPINFO to explicitly pass the list of file handles to be inherited,
152          # so we can avoid disabling close_fds
153          extra_kwargs = {
154              "env": _subprocess_env(),
155              "close_fds": True,
156              "startupinfo": subprocess.STARTUPINFO(lpAttributeList={"handle_list": [read_from_parent, write_to_parent]})
157          }
158  
159      # Run the _child.py script directly passing it the two file descriptors it needs to talk back to the parent.
160      cmd, options = compat.__wrap_python([str(CHILD_PY), str(read_from_parent), str(write_to_parent)], extra_kwargs)
161  
162      # I'm intentionally leaving stdout and stderr alone so that print() can still be used for emergency debugging and
163      # unhandled errors in the child are still visible.
164      return subprocess.Popen(cmd, **options)

167  def _subprocess_env():
168      """ 
169      Define the environment variables to be readable in a child process. 
170      """
171      from PyInstaller.config import CONF
172      python_path = CONF["pathex"]
173      if "PYTHONPATH" in os.environ:
174          python_path = python_path + [os.environ["PYTHONPATH"]]
175      env = os.environ.copy()
176      env["PYTHONPATH"] = os.pathsep.join(python_path)
177      return env

184  class Python:
185      """ 
186      Start and connect to a separate Python subprocess. 
187   
188      This is the lowest level of public API provided by this module. The advantage of using this class directly is 
189      that it allows multiple functions to be evaluated in a single subprocess, making it faster than multiple calls to 
190      :func:`call`. 
191   
192      The ``strict_mode`` argument controls behavior when the child process fails to shut down; if strict mode is enabled, 
193      an error is raised, otherwise only warning is logged. If the value of ``strict_mode`` is ``None``, the value of 
194      ``PyInstaller.compat.strict_collect_mode`` is used (which in turn is controlled by the 
195      ``PYINSTALLER_STRICT_COLLECT_MODE`` environment variable. 
196   
197      Examples: 
198          To call some predefined functions ``x = foo()``, ``y = bar("numpy")`` and ``z = bazz(some_flag=True)`` all using 
199          the same isolated subprocess use:: 
200   
201              with isolated.Python() as child: 
202                  x = child.call(foo) 
203                  y = child.call(bar, "numpy") 
204                  z = child.call(bazz, some_flag=True) 
205   
206      """
207      def __init__(self, strict_mode=None):
208          self._child = None
209  
210          # Re-use the compat.strict_collect_mode and its PYINSTALLER_STRICT_COLLECT_MODE environment variable for
211          # default strict-mode setting.
212          self._strict_mode = strict_mode if strict_mode is not None else compat.strict_collect_mode
213  
214      def __enter__(self):
215          # We need two pipes. One for the child to send data to the parent. The (write) end-point passed to the
216          # child needs to be marked as inheritable.
217          read_from_child, write_to_parent = create_pipe(False, True)
218          # And one for the parent to send data to the child. The (read) end-point passed to the child needs to be
219          # marked as inheritable.
220          read_from_parent, write_to_child = create_pipe(True, False)
221  
222          # Spawn a Python subprocess sending it the two file descriptors it needs to talk back to this parent process.
223          self._child = child(read_from_parent, write_to_parent)
224  
225          # Close the end-points that were inherited by the child.
226          close_pipe_endpoint(read_from_parent)
227          close_pipe_endpoint(write_to_parent)
228          del read_from_parent
229          del write_to_parent
230  
231          # Open file handles to talk to the child. This should fully transfer ownership of the underlying file
232          # descriptor to the opened handle; so when we close the latter, the former should be closed as well.
233          if os.name == 'nt':
234              # On Windows, we must first open file descriptor on top of the handle using _open_osfhandle (which
235              # python wraps in msvcrt.open_osfhandle). According to MSDN, this transfers the ownership of the
236              # underlying file handle to the file descriptors; i.e., they are both closed when the file descriptor
237              # is closed).
238              self._write_handle = os.fdopen(msvcrt.open_osfhandle(write_to_child, 0), "wb")
239              self._read_handle = os.fdopen(msvcrt.open_osfhandle(read_from_child, 0), "rb")
240          else:
241              self._write_handle = os.fdopen(write_to_child, "wb")
242              self._read_handle = os.fdopen(read_from_child, "rb")
243  
244          self._send(sys.path)
245  
246          return self
247  
248      def __exit__(self, exc_type, exc_val, exc_tb):
249          if exc_type and issubclass(exc_type, SubprocessDiedError):
250              self._write_handle.close()
251              self._read_handle.close()
252              del self._read_handle, self._write_handle
253              self._child = None
254              return
255          # Send the signal (a blank line) to the child to tell it that it's time to stop.
256          self._write_handle.write(b"\n")
257          self._write_handle.flush()
258  
259          # Wait for the child process to exit. The timeout is necessary for corner cases when the sub-process fails to
260          # exit (such as due to dangling non-daemon threads; see #7290). At this point, the subprocess already did all
261          # its work, so it should be safe to terminate. And as we expect it to shut down quickly (or not at all), the
262          # timeout is relatively short.
263          #
264          # In strict build mode, we raise an error when the subprocess fails to exit on its own, but do so only after
265          # we attempt to kill the subprocess, to avoid leaving zombie processes.
266          shutdown_error = False
267  
268          try:
269              self._child.wait(timeout=5)
270          except subprocess.TimeoutExpired:
271              logger.warning("Timed out while waiting for the child process to exit!")
272              shutdown_error = True
273              self._child.kill()
274              try:
275                  self._child.wait(timeout=15)
276              except subprocess.TimeoutExpired:
277                  logger.warning("Timed out while waiting for the child process to be killed!")
278                  # Give up and fall through
279  
280          # Close the handles. This should also close the underlying file descriptors.
281          self._write_handle.close()
282          self._read_handle.close()
283          del self._read_handle, self._write_handle
284  
285          self._child = None
286  
287          # Raise an error in strict mode, after all clean-up has been performed.
288          if shutdown_error and self._strict_mode:
289              raise RuntimeError("Timed out while waiting for the child process to exit!")
290  
291      def call(self, function, *args, **kwargs):
292          """ 
293          Call a function in the child Python. Retrieve its return value. Usage of this method is identical to that 
294          of the :func:`call` function. 
295          """
296          if self._child is None:
297              raise RuntimeError("An isolated.Python object must be used in a 'with' clause.")
298  
299          self._send(function.__code__, function.__defaults__, function.__kwdefaults__, args, kwargs)
300  
301          # Read a single line of output back from the child. This contains if the function worked and either its return
302          # value or a traceback. This will block indefinitely until it receives a '\n' byte.
303          try:
304              ok, output = loads(b64decode(self._read_handle.readline()))
305          except (EOFError, BrokenPipeError):
306              # Subprocess appears to have died in an unhandleable way (e.g. SIGSEV). Raise an error.
307              raise SubprocessDiedError(
308                  f"Child process died calling {function.__name__}() with args={args} and "
309                  f"kwargs={kwargs}. Its exit code was {self._child.wait()}."
310              ) from None
311  
312          # If all went well, then ``output`` is the return value.
313          if ok:
314              return output
315  
316          # Otherwise an error happened and ``output`` is a string-ified stacktrace. Raise an error appending the
317          # stacktrace. Having the output in this order gives a nice fluent transition from parent to child in the stack
318          # trace.
319          raise RuntimeError(f"Child process call to {function.__name__}() failed with:\n" + output)
320  
321      def _send(self, *objects):
322          for object in objects:
323              self._write_handle.write(b64encode(dumps(object)))
324              self._write_handle.write(b"\n")
325          # Flushing is very important. Without it, the data is not sent but forever sits in a buffer so that the child is
326          # forever waiting for its data and the parent in turn is forever waiting for the child's response.
327          self._write_handle.flush()

330  def call(function, *args, **kwargs):
331      r""" 
332      Call a function with arguments in a separate child Python. Retrieve its return value. 
333   
334      Args: 
335          function: 
336              The function to send and invoke. 
337          *args: 
338          **kwargs: 
339              Positional and keyword arguments to send to the function. These must be simple builtin types - not custom 
340              classes. 
341      Returns: 
342          The return value of the function. Again, these must be basic types serialisable by :func:`marshal.dumps`. 
343      Raises: 
344          RuntimeError: 
345              Any exception which happens inside an isolated process is caught and reraised in the parent process. 
346   
347      To use, define a function which returns the information you're looking for. Any imports it requires must happen in 
348      the body of the function. For example, to safely check the output of ``matplotlib.get_data_path()`` use:: 
349   
350          # Define a function to be ran in isolation. 
351          def get_matplotlib_data_path(): 
352              import matplotlib 
353              return matplotlib.get_data_path() 
354   
355          # Call it with isolated.call(). 
356          get_matplotlib_data_path = isolated.call(matplotlib_data_path) 
357   
358      For single use functions taking no arguments like the above you can abuse the decorator syntax slightly to define 
359      and execute a function in one go. :: 
360   
361          >>> @isolated.call 
362          ... def matplotlib_data_dir(): 
363          ...     import matplotlib 
364          ...     return matplotlib.get_data_path() 
365          >>> matplotlib_data_dir 
366          '/home/brenainn/.pyenv/versions/3.9.6/lib/python3.9/site-packages/matplotlib/mpl-data' 
367   
368      Functions may take positional and keyword arguments and return most generic Python data types. :: 
369   
370          >>> def echo_parameters(*args, **kwargs): 
371          ...     return args, kwargs 
372          >>> isolated.call(echo_parameters, 1, 2, 3) 
373          (1, 2, 3), {} 
374          >>> isolated.call(echo_parameters, foo=["bar"]) 
375          (), {'foo': ['bar']} 
376   
377      Notes: 
378          To make a function behave differently if it's isolated, check for the ``__isolated__`` global. :: 
379   
380              if globals().get("__isolated__", False): 
381                  # We're inside a child process. 
382                  ... 
383              else: 
384                  # This is the master process. 
385                  ... 
386   
387      """
388      with Python() as isolated:
389          return isolated.call(function, *args, **kwargs)

392  def decorate(function):
393      """ 
394      Decorate a function so that it is always called in an isolated subprocess. 
395   
396      Examples: 
397   
398          To use, write a function then prepend ``@isolated.decorate``. :: 
399   
400              @isolated.decorate 
401              def add_1(x): 
402                  '''Add 1 to ``x``, displaying the current process ID.''' 
403                  import os 
404                  print(f"Process {os.getpid()}: Adding 1 to {x}.") 
405                  return x + 1 
406   
407          The resultant ``add_1()`` function can now be called as you would a 
408          normal function and it'll automatically use a subprocess. 
409   
410              >>> add_1(4) 
411              Process 4920: Adding 1 to 4. 
412              5 
413              >>> add_1(13.2) 
414              Process 4928: Adding 1 to 13.2. 
415              14.2 
416   
417      """
418      @functools.wraps(function)
419      def wrapped(*args, **kwargs):
420          return call(function, *args, **kwargs)
421  
422      return wrapped
