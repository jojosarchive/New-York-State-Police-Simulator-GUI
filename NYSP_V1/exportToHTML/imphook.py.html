33   class ModuleHookCache(dict):
34       """ 
35       Cache of lazily loadable hook script objects. 
36    
37       This cache is implemented as a `dict` subclass mapping from the fully-qualified names of all modules with at 
38       least one hook script to lists of `ModuleHook` instances encapsulating these scripts. As a `dict` subclass, 
39       all cached module names and hook scripts are accessible via standard dictionary operations. 
40    
41       Attributes 
42       ---------- 
43       module_graph : ModuleGraph 
44           Current module graph. 
45       _hook_module_name_prefix : str 
46           String prefixing the names of all in-memory modules lazily loaded from cached hook scripts. See also the 
47           `hook_module_name_prefix` parameter passed to the `ModuleHook.__init__()` method. 
48       """
49   
50       _cache_id_next = 0
51       """ 
52       0-based identifier unique to the next `ModuleHookCache` to be instantiated. 
53    
54       This identifier is incremented on each instantiation of a new `ModuleHookCache` to isolate in-memory modules of 
55       lazily loaded hook scripts in that cache to the same cache-specific namespace, preventing edge-case collisions 
56       with existing in-memory modules in other caches. 
57    
58       """
59       def __init__(self, module_graph, hook_dirs):
60           """ 
61           Cache all hook scripts in the passed directories. 
62    
63           **Order of caching is significant** with respect to hooks for the same module, as the values of this 
64           dictionary are lists. Hooks for the same module will be run in the order in which they are cached. Previously 
65           cached hooks are always preserved rather than overridden. 
66    
67           By default, official hooks are cached _before_ user-defined hooks. For modules with both official and 
68           user-defined hooks, this implies that the former take priority over and hence will be loaded _before_ the 
69           latter. 
70    
71           Parameters 
72           ---------- 
73           module_graph : ModuleGraph 
74               Current module graph. 
75           hook_dirs : list 
76               List of the absolute or relative paths of all directories containing **hook scripts** (i.e., 
77               Python scripts with filenames matching `hook-{module_name}.py`, where `{module_name}` is the module 
78               hooked by that script) to be cached. 
79           """
80           super().__init__()
81   
82           # To avoid circular references and hence increased memory consumption, a weak rather than strong reference is
83           # stored to the passed graph. Since this graph is guaranteed to live longer than this cache,
84           # this is guaranteed to be safe.
85           self.module_graph = weakref.proxy(module_graph)
86   
87           # String unique to this cache prefixing the names of all in-memory modules lazily loaded from cached hook
88           # scripts, privatized for safety.
89           self._hook_module_name_prefix = '__PyInstaller_hooks_{}_'.format(ModuleHookCache._cache_id_next)
90           ModuleHookCache._cache_id_next += 1
91   
92           # Cache all hook scripts in the passed directories.
93           self._cache_hook_dirs(hook_dirs)
94   
95       def _cache_hook_dirs(self, hook_dirs):
96           """ 
97           Cache all hook scripts in the passed directories. 
98    
99           Parameters 
100          ---------- 
101          hook_dirs : list 
102              List of the absolute or relative paths of all directories containing hook scripts to be cached. 
103          """
104  
105          for hook_dir in hook_dirs:
106              # Canonicalize this directory's path and validate its existence.
107              hook_dir = os.path.abspath(hook_dir)
108              if not os.path.isdir(hook_dir):
109                  raise FileNotFoundError('Hook directory "{}" not found.'.format(hook_dir))
110  
111              # For each hook script in this directory...
112              hook_filenames = glob.glob(os.path.join(hook_dir, 'hook-*.py'))
113              for hook_filename in hook_filenames:
114                  # Fully-qualified name of this hook's corresponding module, constructed by removing the "hook-" prefix
115                  # and ".py" suffix.
116                  module_name = os.path.basename(hook_filename)[5:-3]
117  
118                  # Lazily loadable hook object.
119                  module_hook = ModuleHook(
120                      module_graph=self.module_graph,
121                      module_name=module_name,
122                      hook_filename=hook_filename,
123                      hook_module_name_prefix=self._hook_module_name_prefix,
124                  )
125  
126                  # Add this hook to this module's list of hooks.
127                  module_hooks = self.setdefault(module_name, [])
128                  module_hooks.append(module_hook)
129  
130      def remove_modules(self, *module_names):
131          """ 
132          Remove the passed modules and all hook scripts cached for these modules from this cache. 
133   
134          Parameters 
135          ---------- 
136          module_names : list 
137              List of all fully-qualified module names to be removed. 
138          """
139  
140          for module_name in module_names:
141              # Unload this module's hook script modules from memory. Since these are top-level pure-Python modules cached
142              # only in the "sys.modules" dictionary, popping these modules from this dictionary suffices to garbage
143              # collect these modules.
144              module_hooks = self.get(module_name, [])
145              for module_hook in module_hooks:
146                  sys.modules.pop(module_hook.hook_module_name, None)
147  
148              # Remove this module and its hook script objects from this cache.
149              self.pop(module_name, None)

152  def _module_collection_mode_sanitizer(value):
153      if isinstance(value, dict):
154          # Hook set a dictionary; use it as-is
155          return value
156      elif isinstance(value, str):
157          # Hook set a mode string; convert to a dictionary and assign the string to `None` (= the hooked module).
158          return {None: value}
159  
160      raise ValueError(f"Invalid module collection mode setting value: {value!r}")

163  # Dictionary mapping the names of magic attributes required by the "ModuleHook" class to 2-tuples "(default_type,
164  # sanitizer_func)", where:
165  #
166  # * "default_type" is the type to which that attribute will be initialized when that hook is lazily loaded.
167  # * "sanitizer_func" is the callable sanitizing the original value of that attribute defined by that hook into a
168  #   safer value consumable by "ModuleHook" callers if any or "None" if the original value requires no sanitization.
169  #
170  # To avoid subtleties in the ModuleHook.__getattr__() method, this dictionary is declared as a module rather than a
171  # class attribute. If declared as a class attribute and then undefined (...for whatever reason), attempting to access
172  # this attribute from that method would produce infinite recursion.
173  _MAGIC_MODULE_HOOK_ATTRS = {
174      # Collections in which order is insignificant. This includes:
175      #
176      # * "datas", sanitized from hook-style 2-tuple lists defined by hooks into TOC-style 2-tuple sets consumable by
177      #   "ModuleHook" callers.
178      # * "binaries", sanitized in the same way.
179      'datas': (set, format_binaries_and_datas),
180      'binaries': (set, format_binaries_and_datas),
181      'excludedimports': (set, None),
182  
183      # Collections in which order is significant. This includes:
184      #
185      # * "hiddenimports", as order of importation is significant. On module importation, hook scripts are loaded and hook
186      #   functions declared by these scripts are called. As these scripts and functions can have side effects dependent
187      #   on module importation order, module importation itself can have side effects dependent on this order!
188      'hiddenimports': (list, None),
189  
190      # Flags
191      'warn_on_missing_hiddenimports': (lambda: True, bool),
192  
193      # Package/module collection mode dictionary.
194      'module_collection_mode': (dict, _module_collection_mode_sanitizer),
195  }

198  class ModuleHook:
199      """ 
200      Cached object encapsulating a lazy loadable hook script. 
201   
202      This object exposes public attributes (e.g., `datas`) of the underlying hook script as attributes of the same 
203      name of this object. On the first access of any such attribute, this hook script is lazily loaded into an 
204      in-memory private module reused on subsequent accesses. These dynamic attributes are referred to as "magic." All 
205      other static attributes of this object (e.g., `hook_module_name`) are referred to as "non-magic." 
206   
207      Attributes (Magic) 
208      ---------- 
209      datas : set 
210          Set of `TOC`-style 2-tuples `(target_file, source_file)` for all external non-executable files required by 
211          the module being hooked, converted from the `datas` list of hook-style 2-tuples `(source_dir_or_glob, 
212          target_dir)` defined by this hook script. 
213      binaries : set 
214          Set of `TOC`-style 2-tuples `(target_file, source_file)` for all external executable files required by the 
215          module being hooked, converted from the `binaries` list of hook-style 2-tuples `(source_dir_or_glob, 
216          target_dir)` defined by this hook script. 
217      excludedimports : set 
218          Set of the fully-qualified names of all modules imported by the module being hooked to be ignored rather than 
219          imported from that module, converted from the `excludedimports` list defined by this hook script. These 
220          modules will only be "locally" rather than "globally" ignored. These modules will remain importable from all 
221          modules other than the module being hooked. 
222      hiddenimports : set 
223          Set of the fully-qualified names of all modules imported by the module being hooked that are _not_ 
224          automatically detectable by PyInstaller (usually due to being dynamically imported in that module), 
225          converted from the `hiddenimports` list defined by this hook script. 
226      warn_on_missing_hiddenimports : bool 
227          Boolean flag indicating whether missing hidden imports from the hook should generate warnings or not. This 
228          behavior is enabled by default, but individual hooks can opt out of it. 
229      module_collection_mode : dict 
230          A dictionary of package/module names and their corresponding collection mode strings ('pyz', 'pyc', 'py', 
231          'pyz+py', 'py+pyz'). 
232   
233      Attributes (Non-magic) 
234      ---------- 
235      module_graph : ModuleGraph 
236          Current module graph. 
237      module_name : str 
238          Name of the module hooked by this hook script. 
239      hook_filename : str 
240          Absolute or relative path of this hook script. 
241      hook_module_name : str 
242          Name of the in-memory module of this hook script's interpreted contents. 
243      _hook_module : module 
244          In-memory module of this hook script's interpreted contents, lazily loaded on the first call to the 
245          `_load_hook_module()` method _or_ `None` if this method has yet to be accessed. 
246      """
247  
248      #-- Magic --
249  
250      def __init__(self, module_graph, module_name, hook_filename, hook_module_name_prefix):
251          """ 
252          Initialize this metadata. 
253   
254          Parameters 
255          ---------- 
256          module_graph : ModuleGraph 
257              Current module graph. 
258          module_name : str 
259              Name of the module hooked by this hook script. 
260          hook_filename : str 
261              Absolute or relative path of this hook script. 
262          hook_module_name_prefix : str 
263              String prefixing the name of the in-memory module for this hook script. To avoid namespace clashes with 
264              similar modules created by other `ModuleHook` objects in other `ModuleHookCache` containers, this string 
265              _must_ be unique to the `ModuleHookCache` container containing this `ModuleHook` object. If this string 
266              is non-unique, an existing in-memory module will be erroneously reused when lazily loading this hook 
267              script, thus erroneously resanitizing previously sanitized hook script attributes (e.g., `datas`) with 
268              the `format_binaries_and_datas()` helper. 
269   
270          """
271          # Note that the passed module graph is already a weak reference, avoiding circular reference issues. See
272          # ModuleHookCache.__init__(). TODO: Add a failure message
273          assert isinstance(module_graph, weakref.ProxyTypes)
274          self.module_graph = module_graph
275          self.module_name = module_name
276          self.hook_filename = hook_filename
277  
278          # Name of the in-memory module fabricated to refer to this hook script.
279          self.hook_module_name = hook_module_name_prefix + self.module_name.replace('.', '_')
280  
281          # Safety check, see above
282          global HOOKS_MODULE_NAMES
283          if self.hook_module_name in HOOKS_MODULE_NAMES:
284              # When self._shallow is true, this class never loads the hook and sets the attributes to empty values
285              self._shallow = True
286          else:
287              self._shallow = False
288              HOOKS_MODULE_NAMES.add(self.hook_module_name)
289  
290          # Attributes subsequently defined by the _load_hook_module() method.
291          self._loaded = False
292          self._has_hook_function = False
293          self._hook_module = None
294  
295      def __getattr__(self, attr_name):
296          """ 
297          Get the magic attribute with the passed name (e.g., `datas`) from this lazily loaded hook script if any _or_ 
298          raise `AttributeError` otherwise. 
299   
300          This special method is called only for attributes _not_ already defined by this object. This includes 
301          undefined attributes and the first attempt to access magic attributes. 
302   
303          This special method is _not_ called for subsequent attempts to access magic attributes. The first attempt to 
304          access magic attributes defines corresponding instance variables accessible via the `self.__dict__` instance 
305          dictionary (e.g., as `self.datas`) without calling this method. This approach also allows magic attributes to 
306          be deleted from this object _without_ defining the `__delattr__()` special method. 
307   
308          See Also 
309          ---------- 
310          Class docstring for supported magic attributes. 
311          """
312  
313          # If this is a magic attribute, initialize this attribute by lazy loading this hook script and then return
314          # this attribute.
315          if attr_name in _MAGIC_MODULE_HOOK_ATTRS and not self._loaded:
316              self._load_hook_module()
317              return getattr(self, attr_name)
318          # Else, this is an undefined attribute. Raise an exception.
319          else:
320              raise AttributeError(attr_name)
321  
322      def __setattr__(self, attr_name, attr_value):
323          """ 
324          Set the attribute with the passed name to the passed value. 
325   
326          If this is a magic attribute, this hook script will be lazily loaded before setting this attribute. Unlike 
327          `__getattr__()`, this special method is called to set _any_ attribute -- including magic, non-magic, 
328          and undefined attributes. 
329   
330          See Also 
331          ---------- 
332          Class docstring for supported magic attributes. 
333          """
334  
335          # If this is a magic attribute, initialize this attribute by lazy loading this hook script before overwriting
336          # this attribute.
337          if attr_name in _MAGIC_MODULE_HOOK_ATTRS:
338              self._load_hook_module()
339  
340          # Set this attribute to the passed value. To avoid recursion, the superclass method rather than setattr() is
341          # called.
342          return super().__setattr__(attr_name, attr_value)
343  
344      #-- Loading --
345  
346      def _load_hook_module(self, keep_module_ref=False):
347          """ 
348          Lazily load this hook script into an in-memory private module. 
349   
350          This method (and, indeed, this class) preserves all attributes and functions defined by this hook script as 
351          is, ensuring sane behaviour in hook functions _not_ expecting unplanned external modification. Instead, 
352          this method copies public attributes defined by this hook script (e.g., `binaries`) into private attributes 
353          of this object, which the special `__getattr__()` and `__setattr__()` methods safely expose to external 
354          callers. For public attributes _not_ defined by this hook script, the corresponding private attributes will 
355          be assigned sane defaults. For some public attributes defined by this hook script, the corresponding private 
356          attributes will be transformed into objects more readily and safely consumed elsewhere by external callers. 
357   
358          See Also 
359          ---------- 
360          Class docstring for supported attributes. 
361          """
362  
363          # If this hook script module has already been loaded, or we are _shallow, noop.
364          if (self._loaded and (self._hook_module is not None or not keep_module_ref)) or self._shallow:
365              if self._shallow:
366                  self._loaded = True
367                  self._hook_module = True  # Not None
368                  # Inform the user
369                  logger.debug(
370                      'Skipping module hook %r from %r because a hook for %s has already been loaded.',
371                      *os.path.split(self.hook_filename)[::-1], self.module_name
372                  )
373                  # Set the default attributes to empty instances of the type.
374                  for attr_name, (attr_type, _) in _MAGIC_MODULE_HOOK_ATTRS.items():
375                      super().__setattr__(attr_name, attr_type())
376              return
377  
378          # Load and execute the hook script. Even if mechanisms from the import machinery are used, this does not import
379          # the hook as the module.
380          head, tail = os.path.split(self.hook_filename)
381          logger.info('Loading module hook %r from %r...', tail, head)
382          try:
383              self._hook_module = importlib_load_source(self.hook_module_name, self.hook_filename)
384          except ImportError:
385              logger.debug("Hook failed with:", exc_info=True)
386              raise ImportErrorWhenRunningHook(self.hook_module_name, self.hook_filename)
387  
388          # Mark as loaded
389          self._loaded = True
390  
391          # Check if module has hook() function.
392          self._has_hook_function = hasattr(self._hook_module, 'hook')
393  
394          # Copy hook script attributes into magic attributes exposed as instance variables of the current "ModuleHook"
395          # instance.
396          for attr_name, (default_type, sanitizer_func) in _MAGIC_MODULE_HOOK_ATTRS.items():
397              # Unsanitized value of this attribute.
398              attr_value = getattr(self._hook_module, attr_name, None)
399  
400              # If this attribute is undefined, expose a sane default instead.
401              if attr_value is None:
402                  attr_value = default_type()
403              # Else if this attribute requires sanitization, do so.
404              elif sanitizer_func is not None:
405                  attr_value = sanitizer_func(attr_value)
406              # Else, expose the unsanitized value of this attribute.
407  
408              # Expose this attribute as an instance variable of the same name.
409              setattr(self, attr_name, attr_value)
410  
411          # If module_collection_mode has an entry with None key, reassign it to the hooked module's name.
412          setattr(
413              self, 'module_collection_mode', {
414                  key if key is not None else self.module_name: value
415                  for key, value in getattr(self, 'module_collection_mode').items()
416              }
417          )
418  
419          # Release the module if we do not need the reference. This is the case when hook is loaded during the analysis
420          # rather as part of the post-graph operations.
421          if not keep_module_ref:
422              self._hook_module = None
423

426      def post_graph(self, analysis):
427          """ 
428          Call the **post-graph hook** (i.e., `hook()` function) defined by this hook script, if any. 
429   
430          Parameters 
431          ---------- 
432          analysis: build_main.Analysis 
433              Analysis that calls the hook 
434   
435          This method is intended to be called _after_ the module graph for this application is constructed. 
436          """
437  
438          # Lazily load this hook script into an in-memory module.
439          # The script might have been loaded before during modulegraph analysis; in that case, it needs to be reloaded
440          # only if it provides a hook() function.
441          if not self._loaded or self._has_hook_function:
442              # Keep module reference when loading the hook, so we can call its hook function!
443              self._load_hook_module(keep_module_ref=True)
444  
445              # Call this hook script's hook() function, which modifies attributes accessed by subsequent methods and
446              # hence must be called first.
447              self._process_hook_func(analysis)
448  
449          # Order is insignificant here.
450          self._process_hidden_imports()
451  
452      def _process_hook_func(self, analysis):
453          """ 
454          Call this hook's `hook()` function if defined. 
455   
456          Parameters 
457          ---------- 
458          analysis: build_main.Analysis 
459              Analysis that calls the hook 
460          """
461  
462          # If this hook script defines no hook() function, noop.
463          if not hasattr(self._hook_module, 'hook'):
464              return
465  
466          # Call this hook() function.
467          hook_api = PostGraphAPI(module_name=self.module_name, module_graph=self.module_graph, analysis=analysis)
468          try:
469              self._hook_module.hook(hook_api)
470          except ImportError:
471              logger.debug("Hook failed with:", exc_info=True)
472              raise ImportErrorWhenRunningHook(self.hook_module_name, self.hook_filename)
473  
474          # Update all magic attributes modified by the prior call.
475          self.datas.update(set(hook_api._added_datas))
476          self.binaries.update(set(hook_api._added_binaries))
477          self.hiddenimports.extend(hook_api._added_imports)
478          self.module_collection_mode.update(hook_api._module_collection_mode)
479  
480          # FIXME: `hook_api._deleted_imports` should be appended to `self.excludedimports` and used to suppress module
481          # import during the modulegraph construction rather than handled here. However, for that to work, the `hook()`
482          # function needs to be ran during modulegraph construction instead of in post-processing (and this in turn
483          # requires additional code refactoring in order to be able to pass `analysis` to `PostGraphAPI` object at
484          # that point). So once the modulegraph rewrite is complete, remove the code block below.
485          for deleted_module_name in hook_api._deleted_imports:
486              # Remove the graph link between the hooked module and item. This removes the 'item' node from the graph if
487              # no other links go to it (no other modules import it)
488              self.module_graph.removeReference(hook_api.node, deleted_module_name)
489  
490      def _process_hidden_imports(self):
491          """ 
492          Add all imports listed in this hook script's `hiddenimports` attribute to the module graph as if directly 
493          imported by this hooked module. 
494   
495          These imports are typically _not_ implicitly detectable by PyInstaller and hence must be explicitly defined 
496          by hook scripts. 
497          """
498  
499          # For each hidden import required by the module being hooked...
500          for import_module_name in self.hiddenimports:
501              try:
502                  # Graph node for this module. Do not implicitly create namespace packages for non-existent packages.
503                  caller = self.module_graph.find_node(self.module_name, create_nspkg=False)
504  
505                  # Manually import this hidden import from this module.
506                  self.module_graph.import_hook(import_module_name, caller)
507              # If this hidden import is unimportable, print a non-fatal warning. Hidden imports often become
508              # desynchronized from upstream packages and hence are only "soft" recommendations.
509              except ImportError:
510                  if self.warn_on_missing_hiddenimports:
511                      logger.warning('Hidden import "%s" not found!', import_module_name)

514  class AdditionalFilesCache:
515      """ 
516      Cache for storing what binaries and datas were pushed by what modules when import hooks were processed. 
517      """
518      def __init__(self):
519          self._binaries = {}
520          self._datas = {}
521  
522      def add(self, modname, binaries, datas):
523  
524          self._binaries.setdefault(modname, [])
525          self._binaries[modname].extend(binaries or [])
526          self._datas.setdefault(modname, [])
527          self._datas[modname].extend(datas or [])
528  
529      def __contains__(self, name):
530          return name in self._binaries or name in self._datas
531  
532      def binaries(self, modname):
533          """ 
534          Return list of binaries for given module name. 
535          """
536          return self._binaries.get(modname, [])
537  
538      def datas(self, modname):
539          """ 
540          Return list of datas for given module name. 
541          """
542          return self._datas.get(modname, [])
