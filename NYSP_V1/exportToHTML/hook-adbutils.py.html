13   from PyInstaller.utils.hooks import collect_data_files, is_module_satisfies
14   
15   # adb.exe is not automatically collected by collect_dynamic_libs()
16   datas = collect_data_files("adbutils", subdir="binaries", includes=["adb*"])
17   
18   # adbutils v2.2.2 replaced `pkg_resources` with `importlib.resources`, and now uses the following code to determine the
19   # path to the `adbutils.binaries` sub-package directory:
20   # https://github.com/openatx/adbutils/blob/2.2.2/adbutils/_utils.py#L78-L87
21   # As `adbutils.binaries` is not directly imported anywhere, we need a hidden import.
22   if is_module_satisfies('adbutils >= 2.2.2'):
23       hiddenimports = ['adbutils.binaries']
