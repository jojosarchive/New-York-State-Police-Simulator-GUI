26   @isolated.decorate
27   def _get_tcl_tk_info():
28       """ 
29       Isolated-subprocess helper to retrieve the basic Tcl/Tk information: 
30        - tcl_dir = path to the Tcl library/data directory. 
31        - tcl_version = Tcl version 
32        - tk_version = Tk version 
33        - tcl_theaded = boolean indicating whether Tcl/Tk is built with multi-threading support. 
34       """
35       try:
36           import tkinter
37           from _tkinter import TCL_VERSION, TK_VERSION
38       except ImportError:
39           # tkinter unavailable
40           return None, None, None, False
41       try:
42           tcl = tkinter.Tcl()
43       except tkinter.TclError:  # e.g. "Can't find a usable init.tcl in the following directories: ..."
44           return None, None, None, False
45   
46       # Query the location of Tcl library/data directory.
47       tcl_dir = tcl.eval("info library")
48   
49       # Check if Tcl/Tk is built with multi-threaded support (built with --enable-threads), as indicated by the presence
50       # of optional `threaded` member in `tcl_platform` array.
51       try:
52           tcl.getvar("tcl_platform(threaded)")  # Ignore the actual value.
53           tcl_threaded = True
54       except tkinter.TclError:
55           tcl_threaded = False
56   
57       return tcl_dir, TCL_VERSION, TK_VERSION, tcl_threaded
58   
59   
60   # Populate the variables. If `tkinter` is unavailable, the values are set to `None` or `False`.
61   (
62       tcl_dir,
63       tcl_version,
64       tk_version,
65       tcl_threaded,
66   ) = _get_tcl_tk_info()

69   def _warn_if_activetcl_or_teapot_installed(tcl_root, tcltree):
70       """ 
71       If the current Tcl installation is a Teapot-distributed version of ActiveTcl *and* the current platform is macOS, 
72       log a non-fatal warning that the resulting executable will (probably) fail to run on non-host systems. 
73    
74       PyInstaller does *not* freeze all ActiveTcl dependencies -- including Teapot, which is typically ignorable. Since 
75       Teapot is *not* ignorable in this case, this function warns of impending failure. 
76    
77       See Also 
78       ------- 
79       https://github.com/pyinstaller/pyinstaller/issues/621 
80       """
81       import macholib.util
82   
83       # System libraries do not experience this problem.
84       if macholib.util.in_system_path(tcl_root):
85           return
86   
87       # Absolute path of the "init.tcl" script.
88       try:
89           init_resource = [r[1] for r in tcltree if r[1].endswith('init.tcl')][0]
90       except IndexError:
91           # If such script could not be found, silently return.
92           return
93   
94       mentions_activetcl = False
95       mentions_teapot = False
96       # TCL/TK reads files using the system encoding:
97       # https://www.tcl.tk/doc/howto/i18n.html#system_encoding
98       # On macOS, system encoding is UTF-8
99       with open(init_resource, 'r', encoding='utf8') as init_file:
100          for line in init_file.readlines():
101              line = line.strip().lower()
102              if line.startswith('#'):
103                  continue
104              if 'activetcl' in line:
105                  mentions_activetcl = True
106              if 'teapot' in line:
107                  mentions_teapot = True
108              if mentions_activetcl and mentions_teapot:
109                  break
110  
111      if mentions_activetcl and mentions_teapot:
112          logger.warning(
113              """ 
114  You appear to be using an ActiveTcl build of Tcl/Tk, which PyInstaller has 
115  difficulty freezing. To fix this, comment out all references to "teapot" in: 
116   
117       %s 
118   
119  See https://github.com/pyinstaller/pyinstaller/issues/621 for more information. 
120              """ % init_resource
121          )

124  def find_tcl_tk_shared_libs(tkinter_ext_file):
125      """ 
126      Find Tcl and Tk shared libraries against which the _tkinter module is linked. 
127   
128      Returns 
129      ------- 
130      list 
131          list containing two tuples, one for Tcl and one for Tk library, where each tuple contains the library name and 
132          its full path, i.e., [(tcl_lib, tcl_libpath), (tk_lib, tk_libpath)]. If a library is not found, the 
133          corresponding tuple elements are set to None. 
134      """
135      tcl_lib = None
136      tcl_libpath = None
137      tk_lib = None
138      tk_libpath = None
139  
140      for _, lib_path in bindepend.get_imports(tkinter_ext_file):  # (name, fullpath) tuple
141          if lib_path is None:
142              continue  # Skip unresolved entries
143  
144          # For comparison, take basename of lib_path. On macOS, lib_name returned by get_imports is in fact referenced
145          # name, which is not necessarily just a basename.
146          lib_name = os.path.basename(lib_path)
147          lib_name_lower = lib_name.lower()  # lower-case for comparisons
148  
149          if 'tcl' in lib_name_lower:
150              tcl_lib = lib_name
151              tcl_libpath = lib_path
152          elif 'tk' in lib_name_lower:
153              tk_lib = lib_name
154              tk_libpath = lib_path
155  
156      return [(tcl_lib, tcl_libpath), (tk_lib, tk_libpath)]

159  def _find_tcl_tk(tkinter_ext_file):
160      """ 
161      Get a platform-specific 2-tuple of the absolute paths of the top-level external data directories for both 
162      Tcl and Tk, respectively. 
163   
164      Returns 
165      ------- 
166      list 
167          2-tuple that contains the values of `${TCL_LIBRARY}` and `${TK_LIBRARY}`, respectively. 
168      """
169      if compat.is_darwin:
170          # On macOS, _tkinter extension is linked either against the system Tcl/Tk framework (older homebrew python,
171          # python3 from XCode tools) or against bundled Tcl/Tk library (recent python.org builds, recent homebrew
172          # python with python-tk). PyInstaller does not bundle data from system frameworks (as it does not not collect
173          # shared libraries from them, either), so we need to determine what kind of Tcl/Tk we are dealing with.
174          libs = find_tcl_tk_shared_libs(tkinter_ext_file)
175  
176          # Check the full path to the Tcl library.
177          path_to_tcl = libs[0][1]
178  
179          # Starting with macOS 11, system libraries are hidden (unless both Python and PyInstaller's bootloader are built
180          # against MacOS 11.x SDK). Therefore, libs may end up empty; but that implicitly indicates that the system
181          # framework is used, so return (None, None) to inform the caller.
182          if path_to_tcl is None:
183              return None, None
184  
185          # Check if the path corresponds to the system framework, i.e., [/System]/Library/Frameworks/Tcl.framework/Tcl
186          if 'Library/Frameworks/Tcl.framework' in path_to_tcl:
187              return None, None  # Do not collect system framework's data.
188  
189          # Bundled copy of Tcl/Tk; in this case, the dynamic library is
190          # /Library/Frameworks/Python.framework/Versions/3.x/lib/libtcl8.6.dylib
191          # and the data directories have standard layout that is handled by code below.
192      else:
193          # On Windows and linux, data directories have standard layout that is handled by code below.
194          pass
195  
196      # The Tcl library location is already stored in `tcl_dir` global variable. The Tk library is in the same prefix, so
197      # construct the path using `tk_version` global variable.
198      tk_dir = os.path.join(os.path.dirname(tcl_dir), f"tk{tk_version}")
199      return tcl_dir, tk_dir

202  def _collect_tcl_modules(tcl_root):
203      """ 
204      Get a list of TOC-style 3-tuples describing Tcl modules. The modules directory is separate from the library/data 
205      one, and is located at $tcl_root/../tclX, where X is the major Tcl version. 
206   
207      Returns 
208      ------- 
209      Tree 
210          Such list, if the modules directory exists. 
211      """
212  
213      # Obtain Tcl major version.
214      tcl_major_version = tcl_version.split('.')[0]
215  
216      modules_dirname = f"tcl{tcl_major_version}"
217      modules_path = os.path.join(tcl_root, '..', modules_dirname)
218  
219      if not os.path.isdir(modules_path):
220          logger.warning('Tcl modules directory %s does not exist.', modules_path)
221          return []
222  
223      return Tree(modules_path, prefix=modules_dirname)

226  def collect_tcl_tk_files(tkinter_ext_file):
227      """ 
228      Get a list of TOC-style 3-tuples describing all external Tcl/Tk data files. 
229   
230      Returns 
231      ------- 
232      Tree 
233          Such list. 
234      """
235      # Find Tcl and Tk data directory by analyzing the _tkinter extension.
236      tcl_root, tk_root = _find_tcl_tk(tkinter_ext_file)
237  
238      # On macOS, we do not collect system libraries. Therefore, if system Tcl/Tk framework is used, it makes no sense to
239      # collect its data, either. In this case, _find_tcl_tk() will return None, None - either deliberately (we found the
240      # data paths, but ignore them) or not (starting with macOS 11, the data path cannot be found until shared library
241      # discovery is fixed).
242      if compat.is_darwin and not tcl_root and not tk_root:
243          logger.info(
244              "Not collecting Tcl/Tk data - either python is using macOS\' system Tcl/Tk framework, or Tcl/Tk data "
245              "directories could not be found."
246          )
247          return []
248  
249      # TODO Shouldn't these be fatal exceptions?
250      if not tcl_root:
251          logger.error('Tcl/Tk improperly installed on this system.')
252          return []
253      if not os.path.isdir(tcl_root):
254          logger.error('Tcl data directory "%s" not found.', tcl_root)
255          return []
256      if not os.path.isdir(tk_root):
257          logger.error('Tk data directory "%s" not found.', tk_root)
258          return []
259  
260      # Collect Tcl and Tk scripts from their corresponding library/data directories. In contrast to source directories,
261      # which are typically versioned (tcl8.6, tk8.6), the target directories are unversioned (tcl, tk); they are added
262      # to the Tcl/Tk search path via runtime hook for _tkinter, which sets the `TCL_LIBRARY` and `TK_LIBRARY` environment
263      # variables.
264      tcltree = Tree(tcl_root, prefix=TCL_ROOTNAME, excludes=['demos', '*.lib', 'tclConfig.sh'])
265      tktree = Tree(tk_root, prefix=TK_ROOTNAME, excludes=['demos', '*.lib', 'tkConfig.sh'])
266  
267      # If the current Tcl installation is a Teapot-distributed version of ActiveTcl and the current platform is Mac OS,
268      # warn that this is bad.
269      if compat.is_darwin:
270          _warn_if_activetcl_or_teapot_installed(tcl_root, tcltree)
271  
272      # Collect Tcl modules.
273      tclmodulestree = _collect_tcl_modules(tcl_root)
274  
275      return tcltree + tktree + tclmodulestree
