12   # Enchant hook test. Tested with PyEnchant 1.6.6.
13   
14   import sys
15   import enchant
16   
17   print(80 * '-')
18   print('PYTHONPATH: %s' % sys.path)
19   
20   # At least one backend should be available
21   backends = [x.name for x in enchant.Broker().describe()]
22   if len(backends) < 1:
23       raise SystemExit('Error: No dictionary backend available')
24   print(80 * '-')
25   print('Backends: ' + ', '.join(backends))
26   
27   # Usually en_US dictionary should be bundled.
28   languages = enchant.list_languages()
29   dicts = [x[0] for x in enchant.list_dicts()]
30   if len(dicts) < 1:
31       raise SystemExit('No dictionary available')
32   print(80 * '-')
33   print('Languages: %s' % ', '.join(languages))
34   print('Dictionaries: %s' % dicts)
35   print(80 * '-')
36   
37   # Try spell checking if English is availale
38   language = 'en_US'
39   if language in languages:
40       d = enchant.Dict(language)
41       print('d.check("hallo") %s' % d.check('hallo'))
42       print('d.check("halllo") %s' % d.check('halllo'))
43       print('d.suggest("halllo") %s' % d.suggest('halllo'))
