27   class ZlibArchiveWriter:
28       """ 
29       Writer for PyInstaller's PYZ (ZlibArchive) archive. The archive is used to store collected byte-compiled Python 
30       modules, as individually-compressed entries. 
31       """
32       _PYZ_MAGIC_PATTERN = b'PYZ\0'
33       _HEADER_LENGTH = 12 + 5
34       _COMPRESSION_LEVEL = 6  # zlib compression level
35   
36       def __init__(self, filename, entries, code_dict=None):
37           """ 
38           filename 
39               Target filename of the archive. 
40           entries 
41               An iterable containing entries in the form of tuples: (name, src_path, typecode), where `name` is the name 
42               under which the resource is stored (e.g., python module name, without suffix), `src_path` is name of the 
43               file from which the resource is read, and `typecode` is the Analysis-level TOC typecode (`PYMODULE`). 
44           code_dict 
45               Optional code dictionary containing code objects for analyzed/collected python modules. 
46           """
47           code_dict = code_dict or {}
48   
49           with open(filename, "wb") as fp:
50               # Reserve space for the header.
51               fp.write(b'\0' * self._HEADER_LENGTH)
52   
53               # Write entries' data and collect TOC entries
54               toc = []
55               for entry in entries:
56                   toc_entry = self._write_entry(fp, entry, code_dict)
57                   toc.append(toc_entry)
58   
59               # Write TOC
60               toc_offset = fp.tell()
61               toc_data = marshal.dumps(toc)
62               fp.write(toc_data)
63   
64               # Write header:
65               #  - PYZ magic pattern (4 bytes)
66               #  - python bytecode magic pattern (4 bytes)
67               #  - TOC offset (32-bit int, 4 bytes)
68               #  - 4 unused bytes
69               fp.seek(0, os.SEEK_SET)
70   
71               fp.write(self._PYZ_MAGIC_PATTERN)
72               fp.write(BYTECODE_MAGIC)
73               fp.write(struct.pack('!i', toc_offset))
74   
75       @classmethod
76       def _write_entry(cls, fp, entry, code_dict):
77           name, src_path, typecode = entry
78           assert typecode in {'PYMODULE', 'PYMODULE-1', 'PYMODULE-2'}
79   
80           typecode = PYZ_ITEM_MODULE
81           if src_path in ('-', None):
82               # This is a NamespacePackage, modulegraph marks them by using the filename '-'. (But wants to use None,
83               # so check for None, too, to be forward-compatible.)
84               typecode = PYZ_ITEM_NSPKG
85           else:
86               src_basename, _ = os.path.splitext(os.path.basename(src_path))
87               if src_basename == '__init__':
88                   typecode = PYZ_ITEM_PKG
89           data = marshal.dumps(code_dict[name])
90   
91           # First compress, then encrypt.
92           obj = zlib.compress(data, cls._COMPRESSION_LEVEL)
93   
94           # Create TOC entry
95           toc_entry = (name, (typecode, fp.tell(), len(obj)))
96   
97           # Write data blob
98           fp.write(obj)
99   
100          return toc_entry

103  class CArchiveWriter:
104      """ 
105      Writer for PyInstaller's CArchive (PKG) archive. 
106   
107      This archive contains all files that are bundled within an executable; a PYZ (ZlibArchive), DLLs, Python C 
108      extensions, and other data files that are bundled in onefile mode. 
109   
110      The archive can be read from either C (bootloader code at application's run-time) or Python (for debug purposes). 
111      """
112      _COOKIE_MAGIC_PATTERN = b'MEI\014\013\012\013\016'
113  
114      # For cookie and TOC entry structure, see `PyInstaller.archive.readers.CArchiveReader`.
115      _COOKIE_FORMAT = '!8sIIii64s'
116      _COOKIE_LENGTH = struct.calcsize(_COOKIE_FORMAT)
117  
118      _TOC_ENTRY_FORMAT = '!iIIIBB'
119      _TOC_ENTRY_LENGTH = struct.calcsize(_TOC_ENTRY_FORMAT)
120  
121      _COMPRESSION_LEVEL = 9  # zlib compression level
122  
123      def __init__(self, filename, entries, pylib_name):
124          """ 
125          filename 
126              Target filename of the archive. 
127          entries 
128              An iterable containing entries in the form of tuples: (dest_name, src_name, compress, typecode), where 
129              `dest_name` is the name under which the resource is stored in the archive (and name under which it is 
130              extracted at runtime), `src_name` is name of the file from which the resouce is read, `compress` is a 
131              boolean compression flag, and `typecode` is the Analysis-level TOC typecode. 
132          pylib_name 
133              Name of the python shared library. 
134          """
135          self._collected_names = set()  # Track collected names for strict package mode.
136  
137          with open(filename, "wb") as fp:
138              # Write entries' data and collect TOC entries
139              toc = []
140              for entry in entries:
141                  toc_entry = self._write_entry(fp, entry)
142                  toc.append(toc_entry)
143  
144              # Write TOC
145              toc_offset = fp.tell()
146              toc_data = self._serialize_toc(toc)
147              toc_length = len(toc_data)
148  
149              fp.write(toc_data)
150  
151              # Write cookie
152              archive_length = toc_offset + toc_length + self._COOKIE_LENGTH
153              pyvers = sys.version_info[0] * 100 + sys.version_info[1]
154              cookie_data = struct.pack(
155                  self._COOKIE_FORMAT,
156                  self._COOKIE_MAGIC_PATTERN,
157                  archive_length,
158                  toc_offset,
159                  toc_length,
160                  pyvers,
161                  pylib_name.encode('ascii'),
162              )
163  
164              fp.write(cookie_data)
165  
166      def _write_entry(self, fp, entry):
167          dest_name, src_name, compress, typecode = entry
168  
169          # Write OPTION entries as-is, without normalizing them. This also exempts them from duplication check,
170          # allowing them to be specified multiple times.
171          if typecode == 'o':
172              return self._write_blob(fp, b"", dest_name, typecode)
173  
174          # Ensure forward slashes in paths are on Windows converted to back slashes '\\', as on Windows the bootloader
175          # works only with back slashes.
176          dest_name = os.path.normpath(dest_name)
177          if is_win and os.path.sep == '/':
178              # When building under MSYS, the above path normalization uses Unix-style separators, so replace them
179              # manually.
180              dest_name = dest_name.replace(os.path.sep, '\\')
181  
182          # Strict pack/collect mode: keep track of the destination names, and raise an error if we try to add a duplicate
183          # (a file with same destination name, subject to OS case normalization rules).
184          if strict_collect_mode:
185              normalized_dest = None
186              if typecode in {'s', 's1', 's2', 'm', 'M'}:
187                  # Exempt python source scripts and modules from the check.
188                  pass
189              else:
190                  # Everything else; normalize the case
191                  normalized_dest = os.path.normcase(dest_name)
192              # Check for existing entry, if applicable
193              if normalized_dest:
194                  if normalized_dest in self._collected_names:
195                      raise ValueError(
196                          f"Attempting to collect a duplicated file into CArchive: {normalized_dest} (type: {typecode})"
197                      )
198                  self._collected_names.add(normalized_dest)
199  
200          if typecode == 'd':
201              # Dependency; merge src_name (= reference path prefix) and dest_name (= name) into single-string format that
202              # is parsed by bootloader.
203              return self._write_blob(fp, b"", f"{src_name}:{dest_name}", typecode)
204          elif typecode in {'s', 's1', 's2'}:
205              # If it is a source code file, compile it to a code object and marshal the object, so it can be unmarshalled
206              # by the bootloader. For that, we need to know target optimization level, which is stored in typecode.
207              optim_level = {'s': 0, 's1': 1, 's2': 2}[typecode]
208              code = get_code_object(dest_name, src_name, optimize=optim_level)
209              code = strip_paths_in_code(code)
210              return self._write_blob(fp, marshal.dumps(code), dest_name, 's', compress=compress)
211          elif typecode in ('m', 'M'):
212              # Read the PYC file. We do not perform compilation here (in contrast to script files in the above branch),
213              # so typecode does not contain optimization level information.
214              with open(src_name, "rb") as in_fp:
215                  data = in_fp.read()
216              assert data[:4] == BYTECODE_MAGIC
217              # Skip the PYC header, load the code object.
218              code = marshal.loads(data[16:])
219              code = strip_paths_in_code(code)
220              # These module entries are loaded and executed within the bootloader, which requires only the code
221              # object, without the PYC header.
222              return self._write_blob(fp, marshal.dumps(code), dest_name, typecode, compress=compress)
223          elif typecode == 'n':
224              # Symbolic link; store target name (as NULL-terminated string)
225              data = src_name.encode('utf-8') + b'\x00'
226              return self._write_blob(fp, data, dest_name, typecode, compress=compress)
227          else:
228              return self._write_file(fp, src_name, dest_name, typecode, compress=compress)
229  
230      def _write_blob(self, out_fp, blob: bytes, dest_name, typecode, compress=False):
231          """ 
232          Write the binary contents (**blob**) of a small file to the archive and return the corresponding CArchive TOC 
233          entry. 
234          """
235          data_offset = out_fp.tell()
236          data_length = len(blob)
237          if compress:
238              blob = zlib.compress(blob, level=self._COMPRESSION_LEVEL)
239          out_fp.write(blob)
240  
241          return (data_offset, len(blob), data_length, int(compress), typecode, dest_name)
242  
243      def _write_file(self, out_fp, src_name, dest_name, typecode, compress=False):
244          """ 
245          Stream copy a large file into the archive and return the corresponding CArchive TOC entry. 
246          """
247          data_offset = out_fp.tell()
248          data_length = os.stat(src_name).st_size
249          with open(src_name, 'rb') as in_fp:
250              if compress:
251                  tmp_buffer = bytearray(16 * 1024)
252                  compressor = zlib.compressobj(self._COMPRESSION_LEVEL)
253                  while True:
254                      num_read = in_fp.readinto(tmp_buffer)
255                      if not num_read:
256                          break
257                      out_fp.write(compressor.compress(tmp_buffer[:num_read]))
258                  out_fp.write(compressor.flush())
259              else:
260                  shutil.copyfileobj(in_fp, out_fp)
261  
262          return (data_offset, out_fp.tell() - data_offset, data_length, int(compress), typecode, dest_name)
263  
264      @classmethod
265      def _serialize_toc(cls, toc):
266          serialized_toc = []
267          for toc_entry in toc:
268              data_offset, compressed_length, data_length, compress, typecode, name = toc_entry
269  
270              # Encode names as UTF-8. This should be safe as standard python modules only contain ASCII-characters (and
271              # standard shared libraries should have the same), and thus the C-code still can handle this correctly.
272              name = name.encode('utf-8')
273              name_length = len(name) + 1  # Add 1 for string-terminating zero byte.
274  
275              # Ensure TOC entries are aligned on 16-byte boundary, so they can be read by bootloader (C code) on
276              # platforms with strict data alignment requirements (for example linux on `armhf`/`armv7`, such as 32-bit
277              # Debian Buster on Raspberry Pi).
278              entry_length = cls._TOC_ENTRY_LENGTH + name_length
279              if entry_length % 16 != 0:
280                  padding_length = 16 - (entry_length % 16)
281                  name_length += padding_length
282  
283              # Serialize
284              serialized_entry = struct.pack(
285                  cls._TOC_ENTRY_FORMAT + f"{name_length}s",  # "Ns" format automatically pads the string with zero bytes.
286                  cls._TOC_ENTRY_LENGTH + name_length,
287                  data_offset,
288                  compressed_length,
289                  data_length,
290                  compress,
291                  ord(typecode),
292                  name,
293              )
294              serialized_toc.append(serialized_entry)
295  
296          return b''.join(serialized_toc)

299  class SplashWriter:
300      """ 
301      Writer for the splash screen resources archive. 
302   
303      The resulting archive is added as an entry into the CArchive with the typecode PKG_ITEM_SPLASH. 
304      """
305      # This struct describes the splash resources as it will be in an buffer inside the bootloader. All necessary parts
306      # are bundled, the *_len and *_offset fields describe the data beyond this header definition.
307      # Whereas script and image fields are binary data, the requirements fields describe an array of strings. Each string
308      # is null-terminated in order to easily iterate over this list from within C.
309      #
310      #   typedef struct _splash_data_header {
311      #       char tcl_libname[16];  /* Name of tcl library, e.g. tcl86t.dll */
312      #       char tk_libname[16];   /* Name of tk library, e.g. tk86t.dll */
313      #       char tk_lib[16];       /* Tk Library generic, e.g. "tk/" */
314      #       char rundir[16];       /* temp folder inside extraction path in
315      #                               * which the dependencies are extracted */
316      #
317      #       int script_len;        /* Length of the script */
318      #       int script_offset;     /* Offset (rel to start) of the script */
319      #
320      #       int image_len;         /* Length of the image data */
321      #       int image_offset;      /* Offset (rel to start) of the image */
322      #
323      #       int requirements_len;
324      #       int requirements_offset;
325      #
326      #   } SPLASH_DATA_HEADER;
327      #
328      _HEADER_FORMAT = '!16s 16s 16s 16s ii ii ii'
329      _HEADER_LENGTH = struct.calcsize(_HEADER_FORMAT)
330  
331      # The created archive is compressed by the CArchive, so no need to compress the data here.
332  
333      def __init__(self, filename, name_list, tcl_libname, tk_libname, tklib, rundir, image, script):
334          """ 
335          Writer for splash screen resources that are bundled into the CArchive as a single archive/entry. 
336   
337          :param filename: The filename of the archive to create 
338          :param name_list: List of filenames for the requirements array 
339          :param str tcl_libname: Name of the tcl shared library file 
340          :param str tk_libname: Name of the tk shared library file 
341          :param str tklib: Root of tk library (e.g. tk/) 
342          :param str rundir: Unique path to extract requirements to 
343          :param Union[str, bytes] image: Image like object 
344          :param str script: The tcl/tk script to execute to create the screen. 
345          """
346  
347          # Ensure forward slashes in dependency names are on Windows converted to back slashes '\\', as on Windows the
348          # bootloader works only with back slashes.
349          def _normalize_filename(filename):
350              filename = os.path.normpath(filename)
351              if is_win and os.path.sep == '/':
352                  # When building under MSYS, the above path normalization uses Unix-style separators, so replace them
353                  # manually.
354                  filename = filename.replace(os.path.sep, '\\')
355              return filename
356  
357          name_list = [_normalize_filename(name) for name in name_list]
358  
359          with open(filename, "wb") as fp:
360              # Reserve space for the header.
361              fp.write(b'\0' * self._HEADER_LENGTH)
362  
363              # Serialize the requirements list. This list (more an array) contains the names of all files the bootloader
364              # needs to extract before the splash screen can be started. The implementation terminates every name with a
365              # null-byte, that keeps the list short memory wise and makes it iterable from C.
366              requirements_len = 0
367              requirements_offset = fp.tell()
368              for name in name_list:
369                  name = name.encode('utf-8') + b'\0'
370                  fp.write(name)
371                  requirements_len += len(name)
372  
373              # Write splash script
374              script_offset = fp.tell()
375              script_len = len(script)
376              fp.write(script.encode("utf-8"))
377  
378              # Write splash image. If image is a bytes buffer, it is written directly into the archive. Otherwise, it
379              # is assumed to be a path and the file is copied into the archive.
380              image_offset = fp.tell()
381              if isinstance(image, bytes):
382                  # Image was converted by PIL/Pillow and is already in buffer
383                  image_len = len(image)
384                  fp.write(image)
385              else:
386                  # Read image into buffer
387                  with open(image, 'rb') as image_fp:
388                      image_data = image_fp.read()
389                  image_len = len(image_data)
390                  fp.write(image_data)
391                  del image_data
392  
393              # Write header
394              header_data = struct.pack(
395                  self._HEADER_FORMAT,
396                  tcl_libname.encode("utf-8"),
397                  tk_libname.encode("utf-8"),
398                  tklib.encode("utf-8"),
399                  rundir.encode("utf-8"),
400                  script_len,
401                  script_offset,
402                  image_len,
403                  image_offset,
404                  requirements_len,
405                  requirements_offset,
406              )
407  
408              fp.seek(0, os.SEEK_SET)
409              fp.write(header_data)
