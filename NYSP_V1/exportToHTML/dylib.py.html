253  class ExcludeList:
254      def __init__(self):
255          self.regex = re.compile('|'.join(_excludes), re.I)
256  
257      def search(self, libname):
258          # Running re.search() on '' regex never returns None.
259          if _excludes:
260              return self.regex.match(os.path.basename(libname))
261          else:
262              return False
263

266      def __init__(self):
267          self.regex = re.compile('|'.join(_includes), re.I)

269      def search(self, libname):
270          # Running re.search() on '' regex never returns None.
271          if _includes:
272              return self.regex.match(os.path.basename(libname))

274              return False

280  if compat.is_darwin:
281      # On Mac use macholib to decide if a binary is a system one.
282      from macholib import util
283  
284      class MacExcludeList:
285          def __init__(self, global_exclude_list):
286              # Wraps the global 'exclude_list' before it is overridden by this class.
287              self._exclude_list = global_exclude_list
288  
289          def search(self, libname):
290              # First try global exclude list. If it matches, return its result; otherwise continue with other check.
291              result = self._exclude_list.search(libname)
292              if result:
293                  return result
294              else:
295                  return util.in_system_path(libname)
296  
297      exclude_list = MacExcludeList(exclude_list)

299  elif compat.is_win:
300  
301      class WinExcludeList:
302          def __init__(self, global_exclude_list):
303              self._exclude_list = global_exclude_list
304              # use normpath because msys2 uses / instead of \
305              self._windows_dir = os.path.normpath(winutils.get_windows_dir().lower())
306  
307          def search(self, libname):
308              libname = libname.lower()
309              result = self._exclude_list.search(libname)
310              if result:
311                  return result
312              else:
313                  # Exclude everything from the Windows directory by default.
314                  # .. sometimes realpath changes the case of libname, lower it
315                  # .. use normpath because msys2 uses / instead of \
316                  fn = os.path.normpath(os.path.realpath(libname).lower())
317                  return fn.startswith(self._windows_dir)
318  
319      exclude_list = WinExcludeList(exclude_list)

321  _seen_wine_dlls = set()  # Used for warning tracking in include_library()
322  
323  
324  def include_library(libname):
325      """ 
326      Check if the dynamic library should be included with application or not. 
327      """
328      if exclude_list:
329          if exclude_list.search(libname) and not include_list.search(libname):
330              # Library is excluded and is not overridden by include list. It should be excluded.
331              return False
332  
333      # If we are running under Wine and the library is a Wine built-in DLL, ensure that it is always excluded. Typically,
334      # excluding a DLL leads to an incomplete bundle and run-time errors when the said DLL is not installed on the target
335      # system. However, having Wine built-in DLLs collected is even more detrimental, as they usually provide Wine's
336      # implementation of low-level functionality, and therefore cannot be used on actual Windows (i.e., system libraries
337      # from the C:\Windows\system32 directory that might end up collected due to ``_win_includes`` list; a prominent
338      # example are VC runtime DLLs, for which Wine provides their own implementation, unless user explicitly installs
339      # Microsoft's VC redistributable package in their Wine environment). Therefore, excluding the Wine built-in DLLs
340      # actually improves the chances of the bundle running on Windows, or at least makes the issue easier to debug by
341      # turning it into the "standard" missing DLL problem. Exclusion should not affect the bundle's ability to run under
342      #  Wine itself, as the excluded DLLs are available there.
343      if compat.is_win_wine and compat.is_wine_dll(libname):
344          if libname not in _seen_wine_dlls:
345              logger.warning("Excluding Wine built-in DLL: %s", libname)  # displayed only if DLL would have been included
346              _seen_wine_dlls.add(libname)  # display only once for each DLL
347          return False
348  
349      return True

352  # Patterns for suppressing warnings about missing dynamically linked libraries
353  _warning_suppressions = []
354  
355  # On some systems (e.g., openwrt), libc.so might point to ldd. Suppress warnings about it.
356  if compat.is_linux:
357      _warning_suppressions.append(r'ldd')
358  
359  # Suppress warnings about unresolvable UCRT DLLs (see issue #1566) on Windows 10 and 11.
360  if compat.is_win_10 or compat.is_win_11:
361      _warning_suppressions.append(r'api-ms-win-.*\.dll')

364  class MissingLibWarningSuppressionList:
365      def __init__(self):
366          self.regex = re.compile('|'.join(_warning_suppressions), re.I)
367  
368      def search(self, libname):
369          # Running re.search() on '' regex never returns None.
370          if _warning_suppressions:
371              return self.regex.match(os.path.basename(libname))
372          else:
373              return False

376  missing_lib_warning_suppression_list = MissingLibWarningSuppressionList()

379  def warn_missing_lib(libname):
380      """ 
381      Check if a missing-library warning should be displayed for the given library name (or full path). 
382      """
383      return not missing_lib_warning_suppression_list.search(libname)
