11   """ 
12   `importlib_resources` is a backport of the 3.9+ module `importlib.resources` 
13   """
14   
15   from PyInstaller.utils.hooks import check_requirement, collect_data_files
16   
17   # Prior to v1.2.0, a `version.txt` file is used to set __version__. Later versions use `importlib.metadata`.
18   if check_requirement("importlib_resources < 1.2.0"):
19       datas = collect_data_files("importlib_resources", includes=["version.txt"])
20   
21   if check_requirement("importlib_resources >= 1.3.1"):
22       hiddenimports = ['importlib_resources.trees']
