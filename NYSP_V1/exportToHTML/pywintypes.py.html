8    """ A module which supports common Windows types. """
9    import contextlib
10   
11   
12   class error(Exception):
13       def __init__(self, *args, **kw):
14           nargs = len(args)
15           if nargs > 0:
16               self.winerror = args[0]
17           else:
18               self.winerror = None
19           if nargs > 1:
20               self.funcname = args[1]
21           else:
22               self.funcname = None
23           if nargs > 2:
24               self.strerror = args[2]
25           else:
26               self.strerror = None
27           Exception.__init__(self, *args, **kw)
28   
29   
30   @contextlib.contextmanager
31   def pywin32error():
32       try:
33           yield
34       except WindowsError as exception:
35           if not hasattr(exception, 'function'):
36               exception.function = 'unknown'
37           raise error(exception.winerror, exception.function, exception.strerror)
