62   def _find_elements_by_tag(root, tag):
63       """ 
64       Find all elements with given tag under the given root element. 
65       """
66       return [node for node in root.childNodes if node.nodeType == xml.dom.Node.ELEMENT_NODE and node.tagName == tag]
67   
68   
69   def _find_element_by_tag(root, tag):
70       """ 
71       Attempt to find a single element with given tag under the given root element, and return None if no such element 
72       is found. Raises an error if multiple elements are found. 
73       """
74       elements = _find_elements_by_tag(root, tag)
75       if len(elements) > 1:
76           raise ValueError(f"Expected a single {tag!r} element, found {len(elements)} element(s)!")
77       if not elements:
78           return None
79       return elements[0]

85   def _set_execution_level(manifest_dom, root_element, uac_admin=False, uac_uiaccess=False):
86       """ 
87       Find <security> -> <requestedPrivileges> -> <requestedExecutionLevel> element, and set its `level` and `uiAccess` 
88       attributes based on supplied arguments. Create the XML elements if necessary, as they are optional. 
89       """
90   
91       # <trustInfo xmlns="urn:schemas-microsoft-com:asm.v3">
92       trust_info_element = _find_element_by_tag(root_element, "trustInfo")
93       if not trust_info_element:
94           trust_info_element = manifest_dom.createElement("trustInfo")
95           trust_info_element.setAttribute("xmlns", "urn:schemas-microsoft-com:asm.v3")
96           root_element.appendChild(trust_info_element)
97   
98       # <security>
99       security_element = _find_element_by_tag(trust_info_element, "security")
100      if not security_element:
101          security_element = manifest_dom.createElement("security")
102          trust_info_element.appendChild(security_element)
103  
104      # <requestedPrivileges>
105      requested_privileges_element = _find_element_by_tag(security_element, "requestedPrivileges")
106      if not requested_privileges_element:
107          requested_privileges_element = manifest_dom.createElement("requestedPrivileges")
108          security_element.appendChild(requested_privileges_element)
109  
110      # <requestedExecutionLevel>
111      requested_execution_level_element = _find_element_by_tag(requested_privileges_element, "requestedExecutionLevel")
112      if not requested_execution_level_element:
113          requested_execution_level_element = manifest_dom.createElement("requestedExecutionLevel")
114          requested_privileges_element.appendChild(requested_execution_level_element)
115  
116      requested_execution_level_element.setAttribute("level", "requireAdministrator" if uac_admin else "asInvoker")
117      requested_execution_level_element.setAttribute("uiAccess", "true" if uac_uiaccess else "false")

120  def _ensure_common_controls_dependency(manifest_dom, root_element):
121      """ 
122      Scan <dependency> elements for the one whose <<dependentAssembly> -> <assemblyIdentity> corresponds to the 
123      `Microsoft.Windows.Common-Controls`. If found, overwrite its properties. If not, create new <dependency> 
124      element with corresponding sub-elements and attributes. 
125      """
126  
127      # <dependency>
128      dependency_elements = _find_elements_by_tag(root_element, "dependency")
129      for dependency_element in dependency_elements:
130          # <dependentAssembly>
131          dependent_assembly_element = _find_element_by_tag(dependency_element, "dependentAssembly")
132          # <assemblyIdentity>
133          assembly_identity_element = _find_element_by_tag(dependent_assembly_element, "assemblyIdentity")
134          # Check the name attribute
135          if assembly_identity_element.attributes["name"].value == "Microsoft.Windows.Common-Controls":
136              common_controls_element = assembly_identity_element
137              break
138      else:
139          # Create <dependency>
140          dependency_element = manifest_dom.createElement("dependency")
141          root_element.appendChild(dependency_element)
142          # Create <dependentAssembly>
143          dependent_assembly_element = manifest_dom.createElement("dependentAssembly")
144          dependency_element.appendChild(dependent_assembly_element)
145          # Create <assemblyIdentity>
146          common_controls_element = manifest_dom.createElement("assemblyIdentity")
147          dependent_assembly_element.appendChild(common_controls_element)
148  
149      common_controls_element.setAttribute("type", "win32")
150      common_controls_element.setAttribute("name", "Microsoft.Windows.Common-Controls")
151      common_controls_element.setAttribute("version", "6.0.0.0")
152      common_controls_element.setAttribute("processorArchitecture", "*")
153      common_controls_element.setAttribute("publicKeyToken", "6595b64144ccf1df")
154      common_controls_element.setAttribute("language", "*")

157  def create_application_manifest(manifest_xml=None, uac_admin=False, uac_uiaccess=False):
158      """ 
159      Create application manifest, from built-in or custom manifest XML template. If provided, `manifest_xml` must be 
160      a string or byte string containing XML source. The returned manifest is a byte string, encoded in UTF-8. 
161   
162      This function sets the attributes of `requestedExecutionLevel` based on provided `uac_admin` and `auc_uiacces` 
163      arguments (creating the parent elements in the XML, if necessary). It also scans `dependency` elements for the 
164      entry corresponding to `Microsoft.Windows.Common-Controls` and creates or modifies it as necessary. 
165      """
166  
167      if manifest_xml is None:
168          manifest_xml = _DEFAULT_MANIFEST_XML
169  
170      with xml.dom.minidom.parseString(manifest_xml) as manifest_dom:
171          root_element = manifest_dom.documentElement
172  
173          # Validate root element - must be <assembly>
174          assert root_element.tagName == "assembly"
175          assert root_element.namespaceURI == "urn:schemas-microsoft-com:asm.v1"
176          assert root_element.attributes["manifestVersion"].value == "1.0"
177  
178          # Modify the manifest
179          _set_execution_level(manifest_dom, root_element, uac_admin, uac_uiaccess)
180          _ensure_common_controls_dependency(manifest_dom, root_element)
181  
182          # Create output XML
183          output = manifest_dom.toprettyxml(indent="  ", encoding="UTF-8")
184  
185      # Strip extra newlines
186      output = [line for line in output.splitlines() if line.strip()]
187  
188      # Replace: `<?xml version="1.0" encoding="UTF-8"?>` with `<?xml version="1.0" encoding="UTF-8" standalone="yes"?>`.
189      # Support for `standalone` was added to `toprettyxml` in python 3.9, so do a manual work around.
190      output[0] = b"""<?xml version="1.0" encoding="UTF-8" standalone="yes"?>"""
191  
192      output = b"\n".join(output)
193  
194      return output

197  def write_manifest_to_executable(filename, manifest_xml):
198      """ 
199      Write the given manifest XML to the given executable's RT_MANIFEST resource. 
200      """
201      from PyInstaller.utils.win32 import winresource
202  
203      # CREATEPROCESS_MANIFEST_RESOURCE_ID is used for manifest resource in executables.
204      # ISOLATIONAWARE_MANIFEST_RESOURCE_ID is used for manifest resources in DLLs.
205      names = [CREATEPROCESS_MANIFEST_RESOURCE_ID]
206  
207      # Ensure LANG_NEUTRAL is updated, and also update any other present languages.
208      languages = [LANG_NEUTRAL, "*"]
209  
210      winresource.add_or_update_resource(filename, manifest_xml, RT_MANIFEST, names, languages)

213  def read_manifest_from_executable(filename):
214      """ 
215      Read manifest from the given executable." 
216      """
217      from PyInstaller.utils.win32 import winresource
218  
219      resources = winresource.get_resources(filename, [RT_MANIFEST])
220  
221      # `resources` is a three-level dictionary:
222      #  - level 1: resource type (RT_MANIFEST)
223      #  - level 2: resource name (CREATEPROCESS_MANIFEST_RESOURCE_ID)
224      #  - level 3: resource language (LANG_NEUTRAL)
225  
226      # Level 1
227      if RT_MANIFEST not in resources:
228          raise ValueError(f"No RT_MANIFEST resources found in {filename!r}.")
229      resources = resources[RT_MANIFEST]
230  
231      # Level 2
232      if CREATEPROCESS_MANIFEST_RESOURCE_ID not in resources:
233          raise ValueError(f"No RT_MANIFEST resource named CREATEPROCESS_MANIFEST_RESOURCE_ID found in {filename!r}.")
234      resources = resources[CREATEPROCESS_MANIFEST_RESOURCE_ID]
235  
236      # Level 3
237      # We prefer LANG_NEUTRAL, but allow fall back to the first available entry.
238      if LANG_NEUTRAL in resources:
239          resources = resources[LANG_NEUTRAL]
240      else:
241          resources = next(iter(resources.items()))
242  
243      manifest_xml = resources
244      return manifest_xml
