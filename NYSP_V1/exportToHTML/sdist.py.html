120      def initialize_options(self):
121          # 'template' and 'manifest' are, respectively, the names of
122          # the manifest template and manifest file.
123          self.template = None
124          self.manifest = None
125  
126          # 'use_defaults': if true, we will include the default file set
127          # in the manifest
128          self.use_defaults = 1
129          self.prune = 1
130  
131          self.manifest_only = 0
132          self.force_manifest = 0
133  
134          self.formats = ['gztar']
135          self.keep_temp = 0
136          self.dist_dir = None
137  
138          self.archive_files = None
139          self.metadata_check = 1
140          self.owner = None
141          self.group = None

143      def finalize_options(self):
144          if self.manifest is None:
145              self.manifest = "MANIFEST"
146          if self.template is None:
147              self.template = "MANIFEST.in"
148  
149          self.ensure_string_list('formats')
150  
151          bad_format = archive_util.check_archive_formats(self.formats)
152          if bad_format:
153              raise DistutilsOptionError("unknown archive format '%s'" % bad_format)
154  
155          if self.dist_dir is None:
156              self.dist_dir = "dist"

158      def run(self):
159          # 'filelist' contains the list of files that will make up the
160          # manifest
161          self.filelist = FileList()
162  
163          # Run sub commands
164          for cmd_name in self.get_sub_commands():
165              self.run_command(cmd_name)
166  
167          # Do whatever it takes to get the list of files to process
168          # (process the manifest template, read an existing manifest,
169          # whatever).  File list is accumulated in 'self.filelist'.
170          self.get_file_list()
171  
172          # If user just wanted us to regenerate the manifest, stop now.
173          if self.manifest_only:
174              return
175  
176          # Otherwise, go ahead and create the source distribution tarball,
177          # or zipfile, or whatever.
178          self.make_distribution()

191      def get_file_list(self):
192          """Figure out the list of files to include in the source 
193          distribution, and put it in 'self.filelist'.  This might involve 
194          reading the manifest template (and writing the manifest), or just 
195          reading the manifest, or just using the default file set -- it all 
196          depends on the user's options. 
197          """
198          # new behavior when using a template:
199          # the file list is recalculated every time because
200          # even if MANIFEST.in or setup.py are not changed
201          # the user might have added some files in the tree that
202          # need to be included.
203          #
204          #  This makes --force the default and only behavior with templates.
205          template_exists = os.path.isfile(self.template)
206          if not template_exists and self._manifest_is_not_generated():
207              self.read_manifest()
208              self.filelist.sort()
209              self.filelist.remove_duplicates()
210              return
211  
212          if not template_exists:
213              self.warn(
214                  ("manifest template '%s' does not exist " + "(using default file list)")
215                  % self.template
216              )
217          self.filelist.findall()
218  
219          if self.use_defaults:
220              self.add_defaults()
221  
222          if template_exists:
223              self.read_template()
224  
225          if self.prune:
226              self.prune_file_list()
227  
228          self.filelist.sort()
229          self.filelist.remove_duplicates()
230          self.write_manifest()

232      def add_defaults(self):
233          """Add all the default files to self.filelist: 
234            - README or README.txt 
235            - setup.py 
236            - tests/test*.py and test/test*.py 
237            - all pure Python modules mentioned in setup script 
238            - all files pointed by package_data (build_py) 
239            - all files defined in data_files. 
240            - all files defined as scripts. 
241            - all C sources listed as part of extensions or C libraries 
242              in the setup script (doesn't catch C headers!) 
243          Warns if (README or README.txt) or setup.py are missing; everything 
244          else is optional. 
245          """
246          self._add_defaults_standards()
247          self._add_defaults_optional()
248          self._add_defaults_python()
249          self._add_defaults_data_files()
250          self._add_defaults_ext()
251          self._add_defaults_c_libs()
252          self._add_defaults_scripts()
253  
254      @staticmethod
255      def _cs_path_exists(fspath):
256          """ 
257          Case-sensitive path existence check 
258   
259          >>> sdist._cs_path_exists(__file__) 
260          True 
261          >>> sdist._cs_path_exists(__file__.upper()) 
262          False 
263          """
264          if not os.path.exists(fspath):
265              return False
266          # make absolute so we always have a directory
267          abspath = os.path.abspath(fspath)
268          directory, filename = os.path.split(abspath)
269          return filename in os.listdir(directory)

271      def _add_defaults_standards(self):
272          standards = [self.READMES, self.distribution.script_name]
273          for fn in standards:
274              if isinstance(fn, tuple):
275                  alts = fn
276                  got_it = False
277                  for fn in alts:
278                      if self._cs_path_exists(fn):
279                          got_it = True
280                          self.filelist.append(fn)
281                          break
282  
283                  if not got_it:
284                      self.warn(
285                          "standard file not found: should have one of " + ', '.join(alts)
286                      )
287              else:
288                  if self._cs_path_exists(fn):
289                      self.filelist.append(fn)
290                  else:
291                      self.warn("standard file '%s' not found" % fn)
292  
293      def _add_defaults_optional(self):
294          optional = ['tests/test*.py', 'test/test*.py', 'setup.cfg']
295          for pattern in optional:
296              files = filter(os.path.isfile, glob(pattern))
297              self.filelist.extend(files)

299      def _add_defaults_python(self):
300          # build_py is used to get:
301          #  - python modules
302          #  - files defined in package_data
303          build_py = self.get_finalized_command('build_py')
304  
305          # getting python files
306          if self.distribution.has_pure_modules():
307              self.filelist.extend(build_py.get_source_files())
308  
309          # getting package_data files
310          # (computed in build_py.data_files by build_py.finalize_options)
311          for _pkg, src_dir, _build_dir, filenames in build_py.data_files:
312              for filename in filenames:
313                  self.filelist.append(os.path.join(src_dir, filename))
314  
315      def _add_defaults_data_files(self):
316          # getting distribution.data_files
317          if self.distribution.has_data_files():
318              for item in self.distribution.data_files:
319                  if isinstance(item, str):
320                      # plain file
321                      item = convert_path(item)
322                      if os.path.isfile(item):
323                          self.filelist.append(item)
324                  else:
325                      # a (dirname, filenames) tuple
326                      dirname, filenames = item
327                      for f in filenames:
328                          f = convert_path(f)
329                          if os.path.isfile(f):
330                              self.filelist.append(f)
331  
332      def _add_defaults_ext(self):
333          if self.distribution.has_ext_modules():
334              build_ext = self.get_finalized_command('build_ext')
335              self.filelist.extend(build_ext.get_source_files())

347      def read_template(self):
348          """Read and parse manifest template file named by self.template. 
349   
350          (usually "MANIFEST.in") The parsing and processing is done by 
351          'self.filelist', which updates itself accordingly. 
352          """
353          log.info("reading manifest template '%s'", self.template)
354          template = TextFile(
355              self.template,
356              strip_comments=1,
357              skip_blanks=1,
358              join_lines=1,
359              lstrip_ws=1,
360              rstrip_ws=1,
361              collapse_join=1,
362          )
363  
364          try:
365              while True:
366                  line = template.readline()
367                  if line is None:  # end of file
368                      break
369  
370                  try:
371                      self.filelist.process_template_line(line)
372                  # the call above can raise a DistutilsTemplateError for
373                  # malformed lines, or a ValueError from the lower-level
374                  # convert_path function
375                  except (DistutilsTemplateError, ValueError) as msg:
376                      self.warn(
377                          "%s, line %d: %s"
378                          % (template.filename, template.current_line, msg)
379                      )
380          finally:
381              template.close()

383      def prune_file_list(self):
384          """Prune off branches that might slip into the file list as created 
385          by 'read_template()', but really don't belong there: 
386            * the build tree (typically "build") 
387            * the release tree itself (only an issue if we ran "sdist" 
388              previously with --keep-temp, or it aborted) 
389            * any RCS, CVS, .svn, .hg, .git, .bzr, _darcs directories 
390          """
391          build = self.get_finalized_command('build')
392          base_dir = self.distribution.get_fullname()
393  
394          self.filelist.exclude_pattern(None, prefix=build.build_base)
395          self.filelist.exclude_pattern(None, prefix=base_dir)
396  
397          if sys.platform == 'win32':
398              seps = r'/|\\'
399          else:
400              seps = '/'
401  
402          vcs_dirs = ['RCS', 'CVS', r'\.svn', r'\.hg', r'\.git', r'\.bzr', '_darcs']
403          vcs_ptrn = r'(^|{})({})({}).*'.format(seps, '|'.join(vcs_dirs), seps)
404          self.filelist.exclude_pattern(vcs_ptrn, is_regex=1)
405  
406      def write_manifest(self):
407          """Write the file list in 'self.filelist' (presumably as filled in 
408          by 'add_defaults()' and 'read_template()') to the manifest file 
409          named by 'self.manifest'. 
410          """
411          if self._manifest_is_not_generated():
412              log.info(
413                  "not writing to manually maintained "
414                  "manifest file '%s'" % self.manifest
415              )
416              return
417  
418          content = self.filelist.files[:]
419          content.insert(0, '# file GENERATED by distutils, do NOT edit')
420          self.execute(
421              file_util.write_file,
422              (self.manifest, content),
423              "writing manifest file '%s'" % self.manifest,
424          )
425  
426      def _manifest_is_not_generated(self):
427          # check for special comment used in 3.1.3 and higher
428          if not os.path.isfile(self.manifest):
429              return False
430  
431          with open(self.manifest, encoding='utf-8') as fp:
432              first_line = next(fp)
433          return first_line != '# file GENERATED by distutils, do NOT edit\n'

435      def read_manifest(self):
436          """Read the manifest file (named by 'self.manifest') and use it to 
437          fill in 'self.filelist', the list of files to include in the source 
438          distribution. 
439          """
440          log.info("reading manifest file '%s'", self.manifest)
441          with open(self.manifest, encoding='utf-8') as lines:
442              self.filelist.extend(
443                  # ignore comments and blank lines
444                  filter(None, filterfalse(is_comment, map(str.strip, lines)))
445              )

447      def make_release_tree(self, base_dir, files):
448          """Create the directory tree that will become the source 
449          distribution archive.  All directories implied by the filenames in 
450          'files' are created under 'base_dir', and then we hard link or copy 
451          (if hard linking is unavailable) those files into place. 
452          Essentially, this duplicates the developer's source tree, but in a 
453          directory named after the distribution, containing only the files 
454          to be distributed. 
455          """
456          # Create all the directories under 'base_dir' necessary to
457          # put 'files' there; the 'mkpath()' is just so we don't die
458          # if the manifest happens to be empty.
459          self.mkpath(base_dir)
460          dir_util.create_tree(base_dir, files, dry_run=self.dry_run)
461  
462          # And walk over the list of files, either making a hard link (if
463          # os.link exists) to each one that doesn't already exist in its
464          # corresponding location under 'base_dir', or copying each file
465          # that's out-of-date in 'base_dir'.  (Usually, all files will be
466          # out-of-date, because by default we blow away 'base_dir' when
467          # we're done making the distribution archives.)
468  
469          if hasattr(os, 'link'):  # can make hard links on this system
470              link = 'hard'
471              msg = "making hard links in %s..." % base_dir
472          else:  # nope, have to copy
473              link = None
474              msg = "copying files to %s..." % base_dir
475  
476          if not files:
477              log.warning("no files to distribute -- empty manifest?")
478          else:
479              log.info(msg)
480          for file in files:
481              if not os.path.isfile(file):
482                  log.warning("'%s' not a regular file -- skipping", file)
483              else:
484                  dest = os.path.join(base_dir, file)
485                  self.copy_file(file, dest, link=link)
486  
487          self.distribution.metadata.write_pkg_info(base_dir)
488  
489      def make_distribution(self):
490          """Create the source distribution(s).  First, we create the release 
491          tree with 'make_release_tree()'; then, we create all required 
492          archive files (according to 'self.formats') from the release tree. 
493          Finally, we clean up by blowing away the release tree (unless 
494          'self.keep_temp' is true).  The list of archive files created is 
495          stored so it can be retrieved later by 'get_archive_files()'. 
496          """
497          # Don't warn about missing meta-data here -- should be (and is!)
498          # done elsewhere.
499          base_dir = self.distribution.get_fullname()
500          base_name = os.path.join(self.dist_dir, base_dir)
501  
502          self.make_release_tree(base_dir, self.filelist.files)
503          archive_files = []  # remember names of files we create
504          # tar archive must be created last to avoid overwrite and remove
505          if 'tar' in self.formats:
506              self.formats.append(self.formats.pop(self.formats.index('tar')))
507  
508          for fmt in self.formats:
509              file = self.make_archive(
510                  base_name, fmt, base_dir=base_dir, owner=self.owner, group=self.group
511              )
512              archive_files.append(file)
513              self.distribution.dist_files.append(('sdist', '', file))
514  
515          self.archive_files = archive_files
516  
517          if not self.keep_temp:
518              dir_util.remove_tree(base_dir, dry_run=self.dry_run)
519  
520      def get_archive_files(self):
521          """Return the list of archive files created when the command 
522          was run, or None if the command hasn't run yet. 
523          """
524          return self.archive_files
