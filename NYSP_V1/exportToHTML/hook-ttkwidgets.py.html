12   """ 
13   Hook for use with the ttkwidgets package 
14    
15   ttkwidgets provides a set of cross-platform widgets for Tkinter/ttk, 
16   some of which depend on image files in order to function properly. 
17    
18   These images files are all provided in the `ttkwidgets/assets` folder, 
19   which has to be copied by PyInstaller. 
20    
21   This hook has been tested on Ubuntu 18.04 (Python 3.6.8 venv) and 
22   Windows 7 (Python 3.5.4 system-wide). 
23    
24   >>> import tkinter as tk 
25   >>> from ttkwidgets import CheckboxTreeview 
26   >>> 
27   >>> window = tk.Tk() 
28   >>> tree = CheckboxTreeview(window) 
29   >>> tree.insert("", tk.END, "test", text="Hello World!") 
30   >>> tree.insert("test", tk.END, "test2", text="Hello World again!") 
31   >>> tree.insert("test", tk.END, "test3", text="Hello World again again!") 
32   >>> tree.pack() 
33   >>> window.mainloop() 
34   """
35   
36   from PyInstaller.utils.hooks import collect_data_files
37   
38   datas = collect_data_files("ttkwidgets")
