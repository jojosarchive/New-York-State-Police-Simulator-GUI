51   class Splash(Target):
52       """ 
53       Bundles the required resources for the splash screen into a file, which will be included in the CArchive. 
54    
55       A Splash has two outputs, one is itself and one is stored in splash.binaries. Both need to be passed to other 
56       build targets in order to enable the splash screen. 
57       """
58       def __init__(self, image_file, binaries, datas, **kwargs):
59           """ 
60           :param str image_file: 
61               A path-like object to the image to be used. Only the PNG file format is supported. 
62    
63               .. note:: If a different file format is supplied and PIL (Pillow) is installed, the file will be converted 
64                   automatically. 
65    
66               .. note:: *Windows*: The color ``'magenta'`` / ``'#ff00ff'`` must not be used in the image or text, as it is 
67                   used by splash screen to indicate transparent areas. Use a similar color (e.g., ``'#ff00fe'``) instead. 
68    
69               .. note:: If PIL (Pillow) is installed and the image is bigger than max_img_size, the image will be resized 
70                   to fit into the specified area. 
71           :param list binaries: 
72               The TOC list of binaries the Analysis build target found. This TOC includes all extension modules and their 
73               binary dependencies. This is required to determine whether the user's program uses `tkinter`. 
74           :param list datas: 
75               The TOC list of data the Analysis build target found. This TOC includes all data-file dependencies of the 
76               modules. This is required to check if all splash screen requirements can be bundled. 
77    
78           :keyword text_pos: 
79               An optional two-integer tuple that represents the origin of the text on the splash screen image. The 
80               origin of the text is its lower left corner. A unit in the respective coordinate system is a pixel of the 
81               image, its origin lies in the top left corner of the image. This parameter also acts like a switch for 
82               the text feature. If omitted, no text will be displayed on the splash screen. This text will be used to 
83               show textual progress in onefile mode. 
84           :type text_pos: Tuple[int, int] 
85           :keyword text_size: 
86               The desired size of the font. If the size argument is a positive number, it is interpreted as a size in 
87               points. If size is a negative number, its absolute value is interpreted as a size in pixels. Default: ``12`` 
88           :type text_size: int 
89           :keyword text_font: 
90               An optional name of a font for the text. This font must be installed on the user system, otherwise the 
91               system default font is used. If this parameter is omitted, the default font is also used. 
92           :keyword text_color: 
93               An optional color for the text. HTML color codes (``'#40e0d0'``) and color names (``'turquoise'``) are 
94               supported. Default: ``'black'`` 
95               (Windows: the color ``'magenta'`` / ``'#ff00ff'`` is used to indicate transparency, and should not be used) 
96           :type text_color: str 
97           :keyword text_default: 
98               The default text which will be displayed before the extraction starts. Default: ``"Initializing"`` 
99           :type text_default: str 
100          :keyword full_tk: 
101              By default Splash bundles only the necessary files for the splash screen (some tk components). This 
102              options enables adding full tk and making it a requirement, meaning all tk files will be unpacked before 
103              the splash screen can be started. This is useful during development of the splash screen script. 
104              Default: ``False`` 
105          :type full_tk: bool 
106          :keyword minify_script: 
107              The splash screen is created by executing an Tcl/Tk script. This option enables minimizing the script, 
108              meaning removing all non essential parts from the script. Default: ``True`` 
109          :keyword rundir: 
110              The folder name in which tcl/tk will be extracted at runtime. There should be no matching folder in your 
111              application to avoid conflicts. Default:  ``'__splash'`` 
112          :type rundir: str 
113          :keyword name: 
114              An optional alternative filename for the .res file. If not specified, a name is generated. 
115          :type name: str 
116          :keyword script_name: 
117              An optional alternative filename for the Tcl script, that will be generated. If not specified, a name is 
118              generated. 
119          :type script_name: str 
120          :keyword max_img_size: 
121              Maximum size of the splash screen image as a tuple. If the supplied image exceeds this limit, it will be 
122              resized to fit the maximum width (to keep the original aspect ratio). This option can be disabled by 
123              setting it to None. Default: ``(760, 480)`` 
124          :type max_img_size: Tuple[int, int] 
125          :keyword always_on_top: 
126              Force the splashscreen to be always on top of other windows. If disabled, other windows (e.g., from other 
127              applications) can cover the splash screen by user bringing them to front. This might be useful for 
128              frozen applications with long startup times. Default: ``True`` 
129          :type always_on_top: bool 
130          """
131          from ..config import CONF
132          Target.__init__(self)
133  
134          # Splash screen is not supported on macOS. It operates in a secondary thread and macOS disallows UI operations
135          # in any thread other than main.
136          if is_darwin:
137              raise SystemExit("Splash screen is not supported on macOS.")
138  
139          # Make image path relative to .spec file
140          if not os.path.isabs(image_file):
141              image_file = os.path.join(CONF['specpath'], image_file)
142          image_file = os.path.normpath(image_file)
143          if not os.path.exists(image_file):
144              raise ValueError("Image file '%s' not found" % image_file)
145  
146          # Copy all arguments
147          self.image_file = image_file
148          self.full_tk = kwargs.get("full_tk", False)
149          self.name = kwargs.get("name", None)
150          self.script_name = kwargs.get("script_name", None)
151          self.minify_script = kwargs.get("minify_script", True)
152          self.rundir = kwargs.get("rundir", None)
153          self.max_img_size = kwargs.get("max_img_size", (760, 480))
154  
155          # text options
156          self.text_pos = kwargs.get("text_pos", None)
157          self.text_size = kwargs.get("text_size", 12)
158          self.text_font = kwargs.get("text_font", "TkDefaultFont")
159          self.text_color = kwargs.get("text_color", "black")
160          self.text_default = kwargs.get("text_default", "Initializing")
161  
162          # always-on-top behavior
163          self.always_on_top = kwargs.get("always_on_top", True)
164  
165          # Save the generated file separately so that it is not necessary to generate the data again and again
166          root = os.path.splitext(self.tocfilename)[0]
167          if self.name is None:
168              self.name = root + '.res'
169          if self.script_name is None:
170              self.script_name = root + '_script.tcl'
171  
172          if self.rundir is None:
173              self.rundir = self._find_rundir(binaries + datas)
174  
175          # Internal variables
176          try:
177              # Do not import _tkinter at the toplevel, because on some systems _tkinter will fail to load, since it is
178              # not installed. This would cause a runtime error in PyInstaller, since this module is imported from
179              # build_main.py, instead we just want to inform the user that the splash screen feature is not supported on
180              # his platform
181              import _tkinter
182              self._tkinter_module = _tkinter
183              self._tkinter_file = self._tkinter_module.__file__
184          except ModuleNotFoundError:
185              raise SystemExit(
186                  "Your platform does not support the splash screen feature, since tkinter is not installed. Please "
187                  "install tkinter and try again."
188              )
189  
190          # Calculated / analysed values
191          self.uses_tkinter = self._uses_tkinter(self._tkinter_file, binaries)
192          logger.debug("Program uses tkinter: %r", self.uses_tkinter)
193          self.script = self.generate_script()
194          self.tcl_lib, self.tk_lib = tcltk_utils.find_tcl_tk_shared_libs(self._tkinter_file)
195          if is_darwin:
196              # Outdated Tcl/Tk 8.5 system framework is not supported. Depending on macOS version, the library path will
197              # come up empty (hidden system libraries on Big Sur), or will be
198              # [/System]/Library/Frameworks/Tcl.framework/Tcl
199              if self.tcl_lib[1] is None or 'Library/Frameworks/Tcl.framework' in self.tcl_lib[1]:
200                  raise SystemExit("The splash screen feature does not support macOS system framework version of Tcl/Tk.")
201          # Check if tcl/tk was found
202          assert all(self.tcl_lib)
203          assert all(self.tk_lib)
204          logger.debug("Use Tcl Library from %s and Tk From %s", self.tcl_lib, self.tk_lib)
205          self.splash_requirements = set([self.tcl_lib[0], self.tk_lib[0]] + splash_requirements)
206  
207          logger.info("Collect tcl/tk binaries for the splash screen")
208          tcltk_tree = tcltk_utils.collect_tcl_tk_files(self._tkinter_file)
209          if self.full_tk:
210              # The user wants a full copy of tk, so make all tk files a requirement.
211              self.splash_requirements.update(entry[0] for entry in tcltk_tree)
212  
213          # Scan for binary dependencies of the Tcl/Tk shared libraries, and add them to `binaries` TOC list (which
214          # should really be called `dependencies` as it is not limited to binaries. But it is too late now, and
215          # existing spec files depend on this naming). We specify these binary dependencies (which include the
216          # Tcl and Tk shared libaries themselves) even if the user's program uses tkinter and they would be collected
217          # anyway; let the collection mechanism deal with potential duplicates.
218          tcltk_libs = [(dest_name, src_name, 'BINARY') for dest_name, src_name in (self.tcl_lib, self.tk_lib)]
219          self.binaries = bindepend.binary_dependency_analysis(tcltk_libs)
220  
221          # Put all shared library dependencies in `splash_requirements`, so they are made available in onefile mode.
222          self.splash_requirements.update(entry[0] for entry in self.binaries)
223  
224          # If the user's program does not use tkinter, add resources from Tcl/Tk tree to the dependencies list.
225          # Do so only for the resources that are part of splash requirements.
226          if not self.uses_tkinter:
227              self.binaries.extend(entry for entry in tcltk_tree if entry[0] in self.splash_requirements)
228  
229          # Check if all requirements were found.
230          collected_files = set(entry[0] for entry in (binaries + datas + self.binaries))
231  
232          def _filter_requirement(filename):
233              if filename not in collected_files:
234                  # Item is not bundled, so warn the user about it. This actually may happen on some tkinter installations
235                  # that are missing the license.terms file.
236                  logger.warning(
237                      "The local Tcl/Tk installation is missing the file %s. The behavior of the splash screen is "
238                      "therefore undefined and may be unsupported.", filename
239                  )
240                  return False
241              return True
242  
243          # Remove all files which were not found.
244          self.splash_requirements = set(filter(_filter_requirement, self.splash_requirements))
245  
246          # Test if the tcl/tk version is supported by the bootloader.
247          self.test_tk_version()
248  
249          logger.debug("Splash Requirements: %s", self.splash_requirements)
250  
251          self.__postinit__()

259          ('text_size', _check_guts_eq),
260          ('text_font', _check_guts_eq),
261          ('text_color', _check_guts_eq),
262          ('text_default', _check_guts_eq),
263          ('always_on_top', _check_guts_eq),

277          # installations of the same version.
278          ('_tkinter_file', _check_guts_eq),
279      )

281      def _check_guts(self, data, last_build):
282          if Target._check_guts(self, data, last_build):
283              return True
284  
285          # Check if the image has been modified.
286          if misc.mtime(self.image_file) > last_build:
287              logger.info("Building %s because file %s changed", self.tocbasename, self.image_file)
288              return True
289  
290          return False

292      def assemble(self):
293          logger.info("Building Splash %s", self.name)
294  
295          # Function to resize a given image to fit into the area defined by max_img_size.
296          def _resize_image(_image, _orig_size):
297              if PILImage:
298                  _w, _h = _orig_size
299                  _ratio_w = self.max_img_size[0] / _w
300                  if _ratio_w < 1:
301                      # Image width exceeds limit
302                      _h = int(_h * _ratio_w)
303                      _w = self.max_img_size[0]
304  
305                  _ratio_h = self.max_img_size[1] / _h
306                  if _ratio_h < 1:
307                      # Image height exceeds limit
308                      _w = int(_w * _ratio_h)
309                      _h = self.max_img_size[1]
310  
311                  # If a file is given it will be open
312                  if isinstance(_image, PILImage.Image):
313                      _img = _image
314                  else:
315                      _img = PILImage.open(_image)
316                  _img_resized = _img.resize((_w, _h))
317  
318                  # Save image into a stream
319                  _image_stream = io.BytesIO()
320                  _img_resized.save(_image_stream, format='PNG')
321                  _img.close()
322                  _img_resized.close()
323                  _image_data = _image_stream.getvalue()
324                  logger.info("Resized image %s from dimensions %s to (%d, %d)", self.image_file, str(_orig_size), _w, _h)
325                  return _image_data
326              else:
327                  raise ValueError(
328                      "The splash image dimensions (w: %d, h: %d) exceed max_img_size (w: %d, h:%d), but the image "
329                      "cannot be resized due to missing PIL.Image! Either install the Pillow package, adjust the "
330                      "max_img_size, or use an image of compatible dimensions.", _orig_size[0], _orig_size[1],
331                      self.max_img_size[0], self.max_img_size[1]
332                  )
333  
334          # Open image file
335          image_file = open(self.image_file, 'rb')
336  
337          # Check header of the file to identify it
338          if image_file.read(8) == b'\x89PNG\r\n\x1a\n':
339              # self.image_file is a PNG file
340              image_file.seek(16)
341              img_size = (struct.unpack("!I", image_file.read(4))[0], struct.unpack("!I", image_file.read(4))[0])
342  
343              if img_size > self.max_img_size:
344                  # The image exceeds the maximum image size, so resize it
345                  image = _resize_image(self.image_file, img_size)
346              else:
347                  image = os.path.abspath(self.image_file)
348          elif PILImage:
349              # Pillow is installed, meaning the image can be converted automatically
350              img = PILImage.open(self.image_file, mode='r')
351  
352              if img.size > self.max_img_size:
353                  image = _resize_image(img, img.size)
354              else:
355                  image_data = io.BytesIO()
356                  img.save(image_data, format='PNG')
357                  img.close()
358                  image = image_data.getvalue()
359              logger.info("Converted image %s to PNG format", self.image_file)
360          else:
361              raise ValueError(
362                  "The image %s needs to be converted to a PNG file, but PIL.Image is not available! Either install the "
363                  "Pillow package, or use a PNG image for you splash screen.", self.image_file
364              )
365  
366          image_file.close()
367  
368          SplashWriter(
369              self.name,
370              self.splash_requirements,
371              self.tcl_lib[0],  # tcl86t.dll
372              self.tk_lib[0],  # tk86t.dll
373              tcltk_utils.TK_ROOTNAME,
374              self.rundir,
375              image,
376              self.script
377          )

379      def test_tk_version(self):
380          tcl_version = float(self._tkinter_module.TCL_VERSION)
381          tk_version = float(self._tkinter_module.TK_VERSION)
382  
383          # Test if tcl/tk version is supported
384          if tcl_version < 8.6 or tk_version < 8.6:
385              logger.warning(
386                  "The installed Tcl/Tk (%s/%s) version might not work with the splash screen feature of the bootloader. "
387                  "The bootloader is tested against Tcl/Tk 8.6", self._tkinter_module.TCL_VERSION,
388                  self._tkinter_module.TK_VERSION
389              )
390  
391          # This should be impossible, since tcl/tk is released together with the same version number, but just in case
392          if tcl_version != tk_version:
393              logger.warning(
394                  "The installed version of Tcl (%s) and Tk (%s) do not match. PyInstaller is tested against matching "
395                  "versions", self._tkinter_module.TCL_VERSION, self._tkinter_module.TK_VERSION
396              )
397  
398          # Ensure that Tcl is built with multi-threading support.
399          if not tcltk_utils.tcl_threaded:
400              # This is a feature breaking problem, so exit.
401              raise SystemExit(
402                  "The installed tcl version is not threaded. PyInstaller only supports the splash screen "
403                  "using threaded tcl."
404              )

406      def generate_script(self):
407          """ 
408          Generate the script for the splash screen. 
409   
410          If minify_script is True, all unnecessary parts will be removed. 
411          """
412          d = {}
413          if self.text_pos is not None:
414              logger.debug("Add text support to splash screen")
415              d.update({
416                  'pad_x': self.text_pos[0],
417                  'pad_y': self.text_pos[1],
418                  'color': self.text_color,
419                  'font': self.text_font,
420                  'font_size': self.text_size,
421                  'default_text': self.text_default,
422              })
423          script = splash_templates.build_script(text_options=d, always_on_top=self.always_on_top)
424  
425          if self.minify_script:
426              # Remove any documentation, empty lines and unnecessary spaces
427              script = '\n'.join(
428                  line for line in map(lambda line: line.strip(), script.splitlines())
429                  if not line.startswith('#')  # documentation
430                  and line  # empty lines
431              )
432              # Remove unnecessary spaces
433              script = re.sub(' +', ' ', script)
434  
435          # Write script to disk, so that it is transparent to the user what script is executed.
436          with open(self.script_name, "w", encoding="utf-8") as script_file:
437              script_file.write(script)
438          return script

440      @staticmethod
441      def _uses_tkinter(tkinter_file, binaries):
442          # Test for _tkinter extension instead of tkinter module, because user might use a different wrapping library for
443          # Tk. Use `pathlib.PurePath˙ in comparisons to account for case normalization and separator normalization.
444          tkinter_file = pathlib.PurePath(tkinter_file)
445          for dest_name, src_name, typecode in binaries:
446              if pathlib.PurePath(src_name) == tkinter_file:
447                  return True
448          return False

450      @staticmethod
451      def _find_rundir(structure):
452          # First try a name the user could understand, if one would find the directory.
453          rundir = '__splash%s'
454          candidate = rundir % ""
455          counter = 0
456  
457          # Run this loop as long as a folder exist named like rundir. In most cases __splash will be sufficient and this
458          # loop won't enter.
459          while any(e[0].startswith(candidate + os.sep) for e in structure):
460              # just append to rundir a counter
461              candidate = rundir % str(counter)
462              counter += 1
463  
464              # The SPLASH_DATA_HEADER structure limits the name to be 16 bytes at maximum. So if we exceed the limit
465              # raise an error. This will never happen, since there are 10^8 different possibilities, but just in case.
466              assert len(candidate) <= 16
467  
468          return candidate
