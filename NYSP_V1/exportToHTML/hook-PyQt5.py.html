12   from PyInstaller.utils.hooks.qt import pyqt5_library_info, ensure_single_qt_bindings_package
13   
14   # Allow only one Qt bindings package to be collected in frozen application.
15   ensure_single_qt_bindings_package("PyQt5")
16   
17   # Only proceed if PyQt5 can be imported.
18   if pyqt5_library_info.version is not None:
19       hiddenimports = [
20           # PyQt5.10 and earlier uses sip in an separate package;
21           'sip',
22           # PyQt5.11 and later provides SIP in a private package. Support both.
23           'PyQt5.sip',
24           # Imported via __import__ in PyQt5/__init__.py
25           'pkgutil',
26       ]
27   
28       # Collect required Qt binaries.
29       binaries = pyqt5_library_info.collect_extra_binaries()
