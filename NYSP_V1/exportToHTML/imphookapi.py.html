11   """ 
12   Classes facilitating communication between PyInstaller and import hooks. 
13    
14   PyInstaller passes instances of classes defined by this module to corresponding functions defined by external import 
15   hooks, which commonly modify the contents of these instances before returning. PyInstaller then detects and converts 
16   these modifications into appropriate operations on the current `PyiModuleGraph` instance, thus modifying which 
17   modules will be frozen into the executable. 
18   """
19   
20   from PyInstaller.building.utils import format_binaries_and_datas
21   from PyInstaller.lib.modulegraph.modulegraph import (RuntimeModule, RuntimePackage)

24   class PreSafeImportModuleAPI:
25       """ 
26       Metadata communicating changes made by the current **pre-safe import module hook** (i.e., hook run immediately 
27       _before_ a call to `ModuleGraph._safe_import_module()` recursively adding the hooked module, package, 
28       or C extension and all transitive imports thereof to the module graph) back to PyInstaller. 
29    
30       Pre-safe import module hooks _must_ define a `pre_safe_import_module()` function accepting an instance of this 
31       class, whose attributes describe the subsequent `ModuleGraph._safe_import_module()` call creating the hooked 
32       module's graph node. 
33    
34       Each pre-safe import module hook is run _only_ on the first attempt to create the hooked module's graph node and 
35       then subsequently ignored. If this hook successfully creates that graph node, the subsequent 
36       `ModuleGraph._safe_import_module()` call will observe this fact and silently return without attempting to 
37       recreate that graph node. 
38    
39       Pre-safe import module hooks are typically used to create graph nodes for **runtime modules** (i.e., 
40       modules dynamically defined at runtime). Most modules are physically defined in external `.py`-suffixed scripts. 
41       Some modules, however, are dynamically defined at runtime (e.g., `six.moves`, dynamically defined by the 
42       physically defined `six.py` module). However, `ModuleGraph` only parses `import` statements residing in external 
43       scripts. `ModuleGraph` is _not_ a full-fledged, Turing-complete Python interpreter and hence has no means of 
44       parsing `import` statements performed by runtime modules existing only in-memory. 
45    
46       'With great power comes great responsibility.' 
47    
48    
49       Attributes (Immutable) 
50       ---------------------------- 
51       The following attributes are **immutable** (i.e., read-only). For safety, any attempts to change these attributes 
52       _will_ result in a raised exception: 
53    
54       module_graph : PyiModuleGraph 
55           Current module graph. 
56       parent_package : Package 
57           Graph node for the package providing this module _or_ `None` if this module is a top-level module. 
58    
59       Attributes (Mutable) 
60       ----------------------------- 
61       The following attributes are editable. 
62    
63       module_basename : str 
64           Unqualified name of the module to be imported (e.g., `text`). 
65       module_name : str 
66           Fully-qualified name of this module (e.g., `email.mime.text`). 
67       """
68       def __init__(self, module_graph, module_basename, module_name, parent_package):
69           self._module_graph = module_graph
70           self.module_basename = module_basename
71           self.module_name = module_name
72           self._parent_package = parent_package
73   
74       # Immutable properties. No corresponding setters are defined.
75       @property
76       def module_graph(self):
77           """ 
78           Current module graph. 
79           """
80           return self._module_graph
81   
82       @property
83       def parent_package(self):
84           """ 
85           Parent Package of this node. 
86           """
87           return self._parent_package
88   
89       def add_runtime_module(self, module_name):
90           """ 
91           Add a graph node representing a non-package Python module with the passed name dynamically defined at runtime. 
92    
93           Most modules are statically defined on-disk as standard Python files. Some modules, however, are dynamically 
94           defined in-memory at runtime (e.g., `gi.repository.Gst`, dynamically defined by the statically defined 
95           `gi.repository.__init__` module). 
96    
97           This method adds a graph node representing such a runtime module. Since this module is _not_ a package, 
98           all attempts to import submodules from this module in `from`-style import statements (e.g., the `queue` 
99           submodule in `from six.moves import queue`) will be silently ignored. To circumvent this, simply call 
100          `add_runtime_package()` instead. 
101   
102          Parameters 
103          ---------- 
104          module_name : str 
105              Fully-qualified name of this module (e.g., `gi.repository.Gst`). 
106   
107          Examples 
108          ---------- 
109          This method is typically called by `pre_safe_import_module()` hooks, e.g.: 
110   
111              def pre_safe_import_module(api): 
112                  api.add_runtime_module(api.module_name) 
113          """
114  
115          self._module_graph.add_module(RuntimeModule(module_name))
116  
117      def add_runtime_package(self, package_name):
118          """ 
119          Add a graph node representing a non-namespace Python package with the passed name dynamically defined at 
120          runtime. 
121   
122          Most packages are statically defined on-disk as standard subdirectories containing `__init__.py` files. Some 
123          packages, however, are dynamically defined in-memory at runtime (e.g., `six.moves`, dynamically defined by 
124          the statically defined `six` module). 
125   
126          This method adds a graph node representing such a runtime package. All attributes imported from this package 
127          in `from`-style import statements that are submodules of this package (e.g., the `queue` submodule in `from 
128          six.moves import queue`) will be imported rather than ignored. 
129   
130          Parameters 
131          ---------- 
132          package_name : str 
133              Fully-qualified name of this package (e.g., `six.moves`). 
134   
135          Examples 
136          ---------- 
137          This method is typically called by `pre_safe_import_module()` hooks, e.g.: 
138   
139              def pre_safe_import_module(api): 
140                  api.add_runtime_package(api.module_name) 
141          """
142  
143          self._module_graph.add_module(RuntimePackage(package_name))
144  
145      def add_alias_module(self, real_module_name, alias_module_name):
146          """ 
147          Alias the source module to the target module with the passed names. 
148   
149          This method ensures that the next call to findNode() given the target module name will resolve this alias. 
150          This includes importing and adding a graph node for the source module if needed as well as adding a reference 
151          from the target to the source module. 
152   
153          Parameters 
154          ---------- 
155          real_module_name : str 
156              Fully-qualified name of the **existing module** (i.e., the module being aliased). 
157          alias_module_name : str 
158              Fully-qualified name of the **non-existent module** (i.e., the alias to be created). 
159          """
160  
161          self._module_graph.alias_module(real_module_name, alias_module_name)
162  
163      def append_package_path(self, directory):
164          """ 
165          Modulegraph does a good job at simulating Python's, but it cannot handle packagepath `__path__` modifications 
166          packages make at runtime. 
167   
168          Therefore there is a mechanism whereby you can register extra paths in this map for a package, and it will be 
169          honored. 
170   
171          Parameters 
172          ---------- 
173          directory : str 
174              Absolute or relative path of the directory to be appended to this package's `__path__` attribute. 
175          """
176  
177          self._module_graph.append_package_path(self.module_name, directory)

180  class PreFindModulePathAPI:
181      """ 
182      Metadata communicating changes made by the current **pre-find module path hook** (i.e., hook run immediately 
183      _before_ a call to `ModuleGraph._find_module_path()` finding the hooked module's absolute path) back to PyInstaller. 
184   
185      Pre-find module path hooks _must_ define a `pre_find_module_path()` function accepting an instance of this class, 
186      whose attributes describe the subsequent `ModuleGraph._find_module_path()` call to be performed. 
187   
188      Pre-find module path hooks are typically used to change the absolute path from which a module will be 
189      subsequently imported and thus frozen into the executable. To do so, hooks may overwrite the default 
190      `search_dirs` list of the absolute paths of all directories to be searched for that module: e.g., 
191   
192          def pre_find_module_path(api): 
193              api.search_dirs = ['/the/one/true/package/providing/this/module'] 
194   
195      Each pre-find module path hook is run _only_ on the first call to `ModuleGraph._find_module_path()` for the 
196      corresponding module. 
197   
198      Attributes 
199      ---------- 
200      The following attributes are **mutable** (i.e., modifiable). All changes to these attributes will be immediately 
201      respected by PyInstaller: 
202   
203      search_dirs : list 
204          List of the absolute paths of all directories to be searched for this module (in order). Searching will halt 
205          at the first directory containing this module. 
206   
207      Attributes (Immutable) 
208      ---------- 
209      The following attributes are **immutable** (i.e., read-only). For safety, any attempts to change these attributes 
210      _will_ result in a raised exception: 
211   
212      module_name : str 
213          Fully-qualified name of this module. 
214      module_graph : PyiModuleGraph 
215          Current module graph. For efficiency, this attribute is technically mutable. To preserve graph integrity, 
216          this attribute should nonetheless _never_ be modified. While read-only `PyiModuleGraph` methods (e.g., 
217          `findNode()`) are safely callable from within pre-find module path hooks, methods modifying the graph are 
218          _not_. If graph modifications are required, consider an alternative type of hook (e.g., pre-import module 
219          hooks). 
220      """
221      def __init__(
222          self,
223          module_graph,
224          module_name,
225          search_dirs,
226      ):
227          # Mutable attributes.
228          self.search_dirs = search_dirs
229  
230          # Immutable attributes.
231          self._module_graph = module_graph
232          self._module_name = module_name
233  
234      # Immutable properties. No corresponding setters are defined.
235      @property
236      def module_graph(self):
237          """ 
238          Current module graph. 
239          """
240          return self._module_graph
241  
242      @property
243      def module_name(self):
244          """ 
245          Fully-qualified name of this module. 
246          """
247          return self._module_name

250  class PostGraphAPI:
251      """ 
252      Metadata communicating changes made by the current **post-graph hook** (i.e., hook run for a specific module 
253      transitively imported by the current application _after_ the module graph of all `import` statements performed by 
254      this application has been constructed) back to PyInstaller. 
255   
256      Post-graph hooks may optionally define a `post_graph()` function accepting an instance of this class, 
257      whose attributes describe the current state of the module graph and the hooked module's graph node. 
258   
259      Attributes (Mutable) 
260      ---------- 
261      The following attributes are **mutable** (i.e., modifiable). All changes to these attributes will be immediately 
262      respected by PyInstaller: 
263   
264      module_graph : PyiModuleGraph 
265          Current module graph. 
266      module : Node 
267          Graph node for the currently hooked module. 
268   
269      'With great power comes great responsibility.' 
270   
271      Attributes (Immutable) 
272      ---------- 
273      The following attributes are **immutable** (i.e., read-only). For safety, any attempts to change these attributes 
274      _will_ result in a raised exception: 
275   
276      __name__ : str 
277          Fully-qualified name of this module (e.g., `six.moves.tkinter`). 
278      __file__ : str 
279          Absolute path of this module. If this module is: 
280          * A standard (rather than namespace) package, this is the absolute path of this package's directory. 
281          * A namespace (rather than standard) package, this is the abstract placeholder `-`. (Don't ask. Don't tell.) 
282          * A non-package module or C extension, this is the absolute path of the corresponding file. 
283      __path__ : list 
284          List of the absolute paths of all directories comprising this package if this module is a package _or_ `None` 
285          otherwise. If this module is a standard (rather than namespace) package, this list contains only the absolute 
286          path of this package's directory. 
287      co : code 
288          Code object compiled from the contents of `__file__` (e.g., via the `compile()` builtin). 
289      analysis: build_main.Analysis 
290          The Analysis that load the hook. 
291   
292      Attributes (Private) 
293      ---------- 
294      The following attributes are technically mutable but private, and hence should _never_ be externally accessed or 
295      modified by hooks. Call the corresponding public methods instead: 
296   
297      _added_datas : list 
298          List of the `(name, path)` 2-tuples or TOC objects of all external data files required by the current hook, 
299          defaulting to the empty list. This is equivalent to the global `datas` hook attribute. 
300      _added_imports : list 
301          List of the fully-qualified names of all modules imported by the current hook, defaulting to the empty list. 
302          This is equivalent to the global `hiddenimports` hook attribute. 
303      _added_binaries : list 
304          List of the `(name, path)` 2-tuples or TOC objects of all external C extensions imported by the current hook, 
305          defaulting to the empty list. This is equivalent to the global `binaries` hook attribute. 
306      _module_collection_mode : dict 
307          Dictionary of package/module names and their corresponding collection mode strings. This is equivalent to the 
308          global `module_collection_mode` hook attribute. 
309      """
310      def __init__(self, module_name, module_graph, analysis):
311          # Mutable attributes.
312          self.module_graph = module_graph
313          self.module = module_graph.find_node(module_name)
314          assert self.module is not None  # should not occur
315  
316          # Immutable attributes.
317          self.___name__ = module_name
318          self.___file__ = self.module.filename
319          self._co = self.module.code
320          self._analysis = analysis
321  
322          # To enforce immutability, convert this module's package path if any into an immutable tuple.
323          self.___path__ = tuple(self.module.packagepath) \
324              if self.module.packagepath is not None else None
325  
326          #FIXME: Refactor "_added_datas", "_added_binaries", and "_deleted_imports" into sets. Since order of
327          #import is important, "_added_imports" must remain a list.
328  
329          # Private attributes.
330          self._added_binaries = []
331          self._added_datas = []
332          self._added_imports = []
333          self._deleted_imports = []
334          self._module_collection_mode = {}
335  
336      # Immutable properties. No corresponding setters are defined.
337      @property
338      def __file__(self):
339          """ 
340          Absolute path of this module's file. 
341          """
342          return self.___file__
343  
344      @property
345      def __path__(self):
346          """ 
347          List of the absolute paths of all directories comprising this package if this module is a package _or_ `None` 
348          otherwise. If this module is a standard (rather than namespace) package, this list contains only the absolute 
349          path of this package's directory. 
350          """
351          return self.___path__
352  
353      @property
354      def __name__(self):
355          """ 
356          Fully-qualified name of this module (e.g., `six.moves.tkinter`). 
357          """
358          return self.___name__
359  
360      @property
361      def co(self):
362          """ 
363          Code object compiled from the contents of `__file__` (e.g., via the `compile()` builtin). 
364          """
365          return self._co
366  
367      @property
368      def analysis(self):
369          """ 
370          build_main.Analysis that calls the hook. 
371          """
372          return self._analysis
373  
374      # Obsolete immutable properties provided to preserve backward compatibility.
375      @property
376      def name(self):
377          """ 
378          Fully-qualified name of this module (e.g., `six.moves.tkinter`). 
379   
380          **This property has been deprecated by the `__name__` property.**

396          Graph node for the currently hooked module.

413      def add_imports(self, *module_names):
414          """ 
415          Add all Python modules whose fully-qualified names are in the passed list as "hidden imports" upon which the 
416          current module depends. 
417   
418          This is equivalent to appending such names to the hook-specific `hiddenimports` attribute. 
419          """
420          # Append such names to the current list of all such names.
421          self._added_imports.extend(module_names)
422  
423      def del_imports(self, *module_names):
424          """ 
425          Remove the named fully-qualified modules from the set of imports (either hidden or visible) upon which the 
426          current module depends. 
427   
428          This is equivalent to appending such names to the hook-specific `excludedimports` attribute. 
429          """
430          self._deleted_imports.extend(module_names)
431  
432      def add_binaries(self, binaries):
433          """ 
434          Add all external dynamic libraries in the passed list of `(src_name, dest_name)` 2-tuples as dependencies of the 
435          current module. This is equivalent to adding to the global `binaries` hook attribute. 
436   
437          For convenience, the `binaries` may also be a list of TOC-style 3-tuples `(dest_name, src_name, typecode)`. 
438          """
439  
440          # Detect TOC 3-tuple list by checking the length of the first entry
441          if binaries and len(binaries[0]) == 3:
442              self._added_binaries.extend(entry[:2] for entry in binaries)
443          else:
444              # NOTE: `format_binaries_and_datas` changes tuples from input format `(src_name, dest_name)` to output
445              # format `(dest_name, src_name)`.
446              self._added_binaries.extend(format_binaries_and_datas(binaries))
447  
448      def add_datas(self, datas):
449          """ 
450          Add all external data files in the passed list of `(src_name, dest_name)` 2-tuples as dependencies of the 
451          current module. This is equivalent to adding to the global `datas` hook attribute. 
452   
453          For convenience, the `datas` may also be a list of TOC-style 3-tuples `(dest_name, src_name, typecode)`. 
454          """
455  
456          # Detect TOC 3-tuple list by checking the length of the first entry
457          if datas and len(datas[0]) == 3:
458              self._added_datas.extend(entry[:2] for entry in datas)
459          else:
460              # NOTE: `format_binaries_and_datas` changes tuples from input format `(src_name, dest_name)` to output
461              # format `(dest_name, src_name)`.
462              self._added_datas.extend(format_binaries_and_datas(datas))

464      def set_module_collection_mode(self, name, mode):
465          """" 
466          Set the package/module collection mode for the specified module name. If `name` is `None`, the hooked 
467          module/package name is used. `mode` can be one of valid mode strings (`'pyz'`, `'pyc'`, ˙'py'˙, `'pyz+py'`, 
468          ˙'py+pyz'`) or `None`, which clears the setting for the module/package - but only  within this hook's context! 
469          """
470          if name is None:
471              name = self.__name__
472          if mode is None:
473              self._module_collection_mode.pop(name)
474          else:
475              self._module_collection_mode[name] = mode
