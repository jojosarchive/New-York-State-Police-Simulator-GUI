13   from PyInstaller.compat import is_win, is_cygwin
14   from PyInstaller.utils.hooks import collect_dynamic_libs
15   
16   # The clr-loader is used by pythonnet 3.x to load CLR (.NET) runtime.
17   # On Windows, the default runtime is the .NET Framework, and its corresponding
18   # loader requires DLLs from clr_loader\ffi\dlls to be collected. This runtime
19   # is supported only on Windows, so we do not have to worry about it on other
20   # OSes (where Mono or .NET Core are supported).
21   if is_win or is_cygwin:
22       binaries = collect_dynamic_libs("clr_loader")
